    36  // GenerateCloudMetadataRetriever tries to identify the current cloud provider
    37  // the instance is running in by pinging the metadata endpoints. It tries to be
    38  // smart about this by pinging the endpoints in order of provider adoption so that
    39  // the most used provider is first. Unless there is need for creating a retriever
    40  // for a specific provider, this function should be used to generate a new metadata
    41  // retriever instead of directly instanciating a new metadata retriever.
    42  func GenerateCloudMetadataRetriever() error {
    43  	httpClient := http.Client{
    44  		Timeout: 1 * time.Second,
    45  	}
    46  
    47  	// The current number of providers supported.
    48  	// This is the number of responses we expect on
    49  	// the provider select.
    50  	const supportedProviders = 2
    51  
    52  	var (
    53  		awsChan = make(chan error)
    54  		gcpChan = make(chan error)
    55  	)
    56  
    57  	// Try to ping the AWS metadata endpoint
    58  	go func() {
    59  		resourceToPing := "instance-id"
    60  		url, err := url.Parse(aws.EndpointBase)
    61  		if err != nil {
    62  			awsChan <- utils.MakeError("failed to parse metadata endpoint URL: %s", err)
    63  		}
    64  		url.Path = path.Join("latest", "meta-data", resourceToPing)
    65  		resp, err := httpClient.Get(url.String())
    66  		if err != nil {
    67  			awsChan <- err
    68  			return
    69  		}
    70  		if resp.StatusCode != http.StatusOK {
    71  			err = utils.MakeError("did not get a 200 status code")
    72  		}
    73  
    74  		awsChan <- err
    75  	}()
    76  
    77  	// Try to ping the GCP metadata endpoint
    78  	go func() {
    79  		resourceToPing := "id"
    80  		url, err := url.Parse(gcp.EndpointBase)
    81  		if err != nil {
    82  			gcpChan <- utils.MakeError("failed to parse metadata endpoint URL: %s", err)
    83  		}
    84  		url.Path = path.Join(url.Path, resourceToPing)
    85  		req, err := http.NewRequest(http.MethodGet, url.String(), nil)
    86  		if err != nil {
    87  			gcpChan <- utils.MakeError("failed to create request for GCP endpoint: %s", err)
    88  			return
    89  		}
    90  
    91  		// Add necessary headers for GCP and send the request
    92  		req.Header.Add("Metadata-Flavor", "Google")
    93  		resp, err := httpClient.Do(req)
    94  		if err != nil {
    95  			gcpChan <- err
    96  			return
    97  		}
    98  
    99  		if resp.StatusCode != http.StatusOK {
   100  			err = utils.MakeError("did not get a 200 status code")
   101  		}
   102  
   103  		gcpChan <- err
   104  	}()
   105  
   106  	// TODO: Once we add support for more cloud providers, ping their metadata
   107  	// endpoint (or similar feature) here so that the host service can decide
   108  	// which metadata retriever to use. Each ping should happen on a separate
   109  	// goroutine so that the select works as intended.
   110  
   111  	// This select will block until any request to the cloud providers
   112  	// metadata endpoint (or an error) completes. This way we the host
   113  	// service can "know" on which cloud provider its running and start
   114  	// querying for the relevant metadata.
   115  
   116  	for i := 0; i < supportedProviders; i++ {
   117  		select {
   118  		case err := <-awsChan:
   119  			if err == nil {
   120  				CloudMetadata = &aws.Metadata{}
   121  			}
   122  		case err := <-gcpChan:
   123  			if err == nil {
   124  				CloudMetadata = &gcp.Metadata{}
   125  			}
   126  		}
   127  	}
   128  
   129  	return nil
   130  }
