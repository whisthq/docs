   151  func (r *resolver) broadcastHelloSaid() {
   152  	subscribers := map[string]*helloSaidSubscriber{}
   153  	unsubscribe := make(chan string)
   154  
   155  	// NOTE: subscribing and sending events are at odds.
   156  	for {
   157  		select {
   158  		case id := <-unsubscribe:
   159  			delete(subscribers, id)
   160  		case s := <-r.helloSaidSubscriber:
   161  			subscribers[randomID()] = s
   162  		case e := <-r.helloSaidEvents:
   163  			for id, s := range subscribers {
   164  				go func(id string, s *helloSaidSubscriber) {
   165  					select {
   166  					case <-s.stop:
   167  						unsubscribe <- id
   168  						return
   169  					default:
   170  					}
   171  
   172  					select {
   173  					case <-s.stop:
   174  						unsubscribe <- id
   175  					case s.events <- e:
   176  					case <-time.After(time.Second):
   177  					}
   178  				}(id, s)
   179  			}
   180  		}
   181  	}
   182  }
