    56  func (sc *StripeClient) getSubscription() (*stripe.Subscription, error) {
    57  	subscriptionsList := sub.List(&stripe.SubscriptionListParams{
    58  		Customer: sc.customerID,
    59  		CurrentPeriodEndRange: &stripe.RangeQueryParams{
    60  			GreaterThan: time.Now().Unix(),
    61  		},
    62  	})
    63  
    64  	var subscription *stripe.Subscription
    65  	for subscriptionsList.Next() {
    66  		subscription = subscriptionsList.Subscription()
    67  	}
    68  	if subscriptionsList.Err() != nil || subscription == nil {
    69  		return nil, utils.MakeError("Failed to obtain subscription for customer %v", sc.customerID)
    70  	}
    71  
    72  	return subscription, nil
    73  }

   100  // findPrice will search for a Stripe price matching the desired amount and
   101  // will return the price ID. If it fails to find it, then creates a new price
   102  // in Stripe and returns its ID.
   103  func (sc *StripeClient) findPrice() (string, error) {
   104  	var priceID string
   105  
   106  	priceList := price.List(&stripe.PriceListParams{
   107  		Active: stripe.Bool(true),
   108  	})
   109  
   110  	// Try to find a Stripe Price with the desired monthly price.
   111  	for priceList.Next() {
   112  		currentPrice := priceList.Current().(*stripe.Price)
   113  		if currentPrice.UnitAmount == sc.monthlyPriceInCents {
   114  			priceID = currentPrice.ID
   115  			break
   116  		}
   117  	}
   118  
   119  	if priceList.Err() != nil || priceID == "" {
   120  		logger.Warningf("Failed to find price in Stripe. Creating new price.")
   121  	}
   122  
   123  	// If a Stripe Price was not found, create one.
   124  	if priceID != "" {
   125  		logger.Infof("Found price of %v in Stripe.", sc.monthlyPriceInCents)
   126  	} else {
   127  		//Create new price
   128  		newPrice, err := sc.createPrice(sc.monthlyPriceInCents, "Whist", "month")
   129  		if err != nil {
   130  			return "", utils.MakeError("failed to create new Stripe price. Err: %v", err)
   131  		}
   132  		priceID = newPrice.ID
   133  	}
   134  
   135  	return priceID, nil
   136  }

   139  func (sc *StripeClient) createCheckoutSession(withTrialPeriod bool) (string, error) {
   140  	// Set how many subscriptions we want started, and what number of
   141  	// days we offer as a trial period.
   142  	const (
   143  		subscriptionQuantity = 1  // The number of subscriptions that will be created
   144  		trialPeriodDays      = 14 // The number of days offered as a free trial period
   145  	)
   146  
   147  	// Get a Stripe price with the desired amount
   148  	priceID, err := sc.findPrice()
   149  	if err != nil {
   150  		return "", utils.MakeError("failed to find or create Stripe price for amount %v. Err: %v", sc.monthlyPriceInCents, err)
   151  	}
   152  
   153  	// Create subscription params depending if we are offering a free trial or not
   154  	var subscriptionParams *stripe.CheckoutSessionSubscriptionDataParams
   155  	if withTrialPeriod {
   156  		subscriptionParams = &stripe.CheckoutSessionSubscriptionDataParams{
   157  			TrialPeriodDays: stripe.Int64(trialPeriodDays),
   158  		}
   159  	} else {
   160  		subscriptionParams = &stripe.CheckoutSessionSubscriptionDataParams{}
   161  	}
   162  
   163  	params := &stripe.CheckoutSessionParams{
   164  		Customer: stripe.String(sc.customerID),
   165  		LineItems: []*stripe.CheckoutSessionLineItemParams{
   166  			{
   167  				Price:    stripe.String(priceID),
   168  				Quantity: stripe.Int64(int64(subscriptionQuantity)),
   169  			},
   170  		},
   171  		SubscriptionData: subscriptionParams,
   172  		PaymentMethodTypes: []*string{
   173  			stripe.String("card"),
   174  		},
   175  		Mode:       stripe.String(string(stripe.CheckoutSessionModeSubscription)),
   176  		SuccessURL: stripe.String("http://localhost/callback/payment?success=true"),
   177  		CancelURL:  stripe.String("http://localhost/callback/payment?success=false"),
   178  	}
   179  
   180  	s, err := checkout.New(params)
   181  	if err != nil {
   182  		return "", utils.MakeError("failed to create checkout session. Err: %v", err)
   183  	}
   184  
   185  	return s.URL, nil
   186  }
