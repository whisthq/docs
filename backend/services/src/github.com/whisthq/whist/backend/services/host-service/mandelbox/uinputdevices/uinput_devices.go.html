   123  // SendDeviceFDsOverSocket blocks until it can send the provided devices' file
   124  // descriptors over a socket at the provided path. If the given context is
   125  // cancelled, it also aborts (therefore letting us not leak goroutines if a
   126  // mandelbox dies or fails during creation after this function is called).
   127  func SendDeviceFDsOverSocket(baseCtx context.Context, goroutineTracker *sync.WaitGroup, devices *UinputDevices, socketPath string) error {
   128  	// Create our own context so we can safely cancel it.
   129  	ctx, cancel := context.WithCancel(baseCtx)
   130  	defer cancel()
   131  
   132  	dir, err := filepath.Abs(filepath.Dir(socketPath))
   133  	if err != nil {
   134  		return utils.MakeError("Unable to calculate parent directory of given socketPath %s. Error: %s", socketPath, err)
   135  	}
   136  
   137  	os.MkdirAll(dir, 0777)
   138  	os.RemoveAll(socketPath)
   139  
   140  	var lc net.ListenConfig
   141  	listener, err := lc.Listen(ctx, "unix", socketPath)
   142  	if err != nil {
   143  		return utils.MakeError("Could not create unix socket at %s: %s", socketPath, err)
   144  	}
   145  	// Instead of running `defer listener.Close()` we ran `defer cancel()` above,
   146  	// which will cause the following goroutine to always close `listener`.
   147  	// Normally, listener.Accept() blocks until the protocol connects. If the
   148  	// context is closed, though, we want to unblock.
   149  	goroutineTracker.Add(1)
   150  	go func() {
   151  		defer goroutineTracker.Done()
   152  		<-ctx.Done()
   153  		listener.Close()
   154  	}()
   155  
   156  	logger.Infof("Successfully created unix socket at: %s", socketPath)
   157  
   158  	// listener.Accept() blocks until the protocol connects
   159  	client, err := listener.Accept()
   160  	if err != nil {
   161  		return utils.MakeError("Could not connect to client over unix socket at %s: %s", socketPath, err)
   162  	}
   163  	defer client.Close()
   164  
   165  	connf, err := client.(*net.UnixConn).File()
   166  	if err != nil {
   167  		return utils.MakeError("Could not get file corresponding to client connection for unix socket at %s: %s", socketPath, err)
   168  	}
   169  	defer connf.Close()
   170  
   171  	connfd := int(connf.Fd())
   172  	fds := [3]int{
   173  		int(devices.absmouse.DeviceFile().Fd()),
   174  		int(devices.relmouse.DeviceFile().Fd()),
   175  		int(devices.keyboard.DeviceFile().Fd()),
   176  	}
   177  	rights := syscall.UnixRights(fds[:]...)
   178  	err = syscall.Sendmsg(connfd, nil, rights, nil, 0)
   179  	if err != nil {
   180  		return utils.MakeError("Error sending uinput file descriptors over socket: %s", err)
   181  	}
   182  
   183  	logger.Infof("Sent uinput file descriptors to socket at %s", socketPath)
   184  	return nil
   185  }

   187  // getDeviceFilePath returns the file path (e.g. /dev/input/event3) given a
   188  // file descriptor corresponding to a virtual device created with uinput
   189  func getDeviceFilePath(fd *os.File) (string, error) {
   190  	const maxlen uintptr = 32
   191  	bsysname := make([]byte, maxlen)
   192  	_, _, errno := syscall.Syscall(syscall.SYS_IOCTL, uintptr(fd.Fd()), linuxUIGetSysName(maxlen), uintptr(unsafe.Pointer(&bsysname[0])))
   193  	if errno != 0 {
   194  		return "", utils.MakeError("ioctl to get sysname failed with errno %d", errno)
   195  	}
   196  	sysname := string(bytes.Trim(bsysname, "\x00"))
   197  	syspath := utils.Sprintf("/sys/devices/virtual/input/%s", sysname)
   198  	sysdir, err := os.Open(syspath)
   199  	if err != nil {
   200  		return "", utils.MakeError("could not open directory %s: %s", syspath, err)
   201  	}
   202  	names, err := sysdir.Readdirnames(0)
   203  	if err != nil {
   204  		return "", utils.MakeError("Readdirnames for directory %s failed: %s", syspath, err)
   205  	}
   206  	for _, name := range names {
   207  		if strings.HasPrefix(name, "event") {
   208  			return "/dev/input/" + name, nil
   209  		}
   210  	}
   211  	return "", utils.MakeError("did not find file in %s with prefix 'event'", syspath)
   212  }

   214  // In summary, this function computes the argument that gets passed to the
   215  // ioctl system call. For more details, and an explanation of the argument
   216  // names to linuxIoc, see: ttps://www.kernel.org/doc/html/latest/driver-api/ioctl.html
   217  func linuxUIGetSysName(len uintptr) uintptr {
   218  	// from ioctl.h and uinput.h
   219  	const (
   220  		iocDirshift  = 30
   221  		iocSizeshift = 16
   222  		iocTypeshift = 8
   223  		iocNrshift   = 0
   224  		iocRead      = 2
   225  		uiIoctlBase  = 85 // 'U'
   226  	)
   227  
   228  	linuxIoc := func(dir uintptr, requestType uintptr, nr uintptr, size uintptr) uintptr {
   229  		return (dir << iocDirshift) + (requestType << iocTypeshift) + (nr << iocNrshift) + (size << iocSizeshift)
   230  	}
   231  
   232  	return linuxIoc(iocRead, uiIoctlBase, 44, len)
   233  }
