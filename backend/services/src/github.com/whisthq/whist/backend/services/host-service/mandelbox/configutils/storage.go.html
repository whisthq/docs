    64  func GetMostRecentMatchingKey(client *s3.Client, bucket, prefix, suffix string) (*s3types.Object, error) {
    65  	var curMatch *s3types.Object = nil
    66  
    67  	// We deal with the case where there are more than 1000 subkeys for the
    68  	// user's app config with pagination. God forbid this ever happen, but we
    69  	// gotta write resilient software.
    70  	for paginator := s3.NewListObjectsV2Paginator(client, &s3.ListObjectsV2Input{
    71  		Bucket: aws.String(bucket),
    72  		Prefix: aws.String(prefix),
    73  	}); paginator.HasMorePages(); {
    74  		output, err := paginator.NextPage(context.Background())
    75  		if err != nil {
    76  			return curMatch, err
    77  		}
    78  
    79  		for i, v := range output.Contents {
    80  			if (curMatch == nil || curMatch.LastModified.Before(*v.LastModified)) && strings.HasSuffix(*v.Key, suffix) {
    81  				// DO NOT use `&v` instead of `&output.Contents[i]`, since Go reuses
    82  				// memory addresses for loop variables (see the link below). I
    83  				// re-learned this the hard way.
    84  				// https://www.evanjones.ca/go-gotcha-loop-variables.html
    85  				curMatch = &output.Contents[i]
    86  			}
    87  		}
    88  	}
    89
