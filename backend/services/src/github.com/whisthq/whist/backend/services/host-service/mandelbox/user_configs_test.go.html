   159  // TestUserConfigChecksum tests the checksum verification functonality of
   160  // the upload and download user config code.
   161  func TestUserConfigChecksum(t *testing.T) {
   162  	testFile1 := []byte("this is a test file")
   163  	testFile1Hash := configutils.GetMD5Hash(testFile1)
   164  
   165  	// Test upload
   166  	s3Client, err := configutils.NewS3Client("us-east-1")
   167  	if err != nil {
   168  		t.Fatalf("failed to create s3 client: %v", err)
   169  	}
   170  
   171  	_, err = configutils.UploadFileToBucket(s3Client, configutils.GetAccessPoint(), "user_config_test_user/checksum_test", testFile1)
   172  	if err != nil {
   173  		t.Fatalf("failed to upload test file: %v", err)
   174  	}
   175  
   176  	// Check that checksum is present and correct
   177  	head, err := configutils.GetHeadObject(s3Client, configutils.GetAccessPoint(), "user_config_test_user/checksum_test")
   178  	if err != nil {
   179  		t.Fatalf("failed to get head object: %v", err)
   180  	}
   181  
   182  	metadata := head.Metadata
   183  	downloadHash, ok := metadata["md5"]
   184  	if !ok {
   185  		t.Fatalf("md5 metadata not present on head object")
   186  	}
   187  
   188  	if downloadHash != testFile1Hash {
   189  		t.Fatalf("md5 metadata on head object does not match file hash")
   190  	}
   191  
   192  	// Test download happy path
   193  	mandelbox := mandelboxData{ID: types.MandelboxID(utils.PlaceholderTestUUID())}
   194  	data, err := mandelbox.downloadUserConfig(s3Client, "user_config_test_user/checksum_test", head)
   195  	if err != nil {
   196  		t.Errorf("failed to download file: %v", err)
   197  	}
   198  
   199  	if !bytes.Equal(data, testFile1) {
   200  		t.Errorf("downloaded file does not match original file")
   201  	}
   202  
   203  	// Test download with bad checksum
   204  	head.Metadata["md5"] = "bad_hash"
   205  	_, err = mandelbox.downloadUserConfig(s3Client, "user_config_test_user/checksum_test", head)
   206  	if err == nil || !strings.HasPrefix(err.Error(), "could not download object (due to MD5 mismatch)") {
   207  		t.Errorf("download with bad checksum did not fail")
   208  	}
   209  }
