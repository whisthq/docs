    22  // EncryptAES256GCM takes a password and plaintext data and encrypts the data using GCM.
    23  // AES-GCM is an AEAD (authenticated encryption with associated data) method that
    24  // no only provides data confidentiality but also a way to verify authenticity/integrity
    25  // of the data. GCM is also highly performant and fully-parallelizable.
    26  //
    27  // See: https://en.wikipedia.org/wiki/Galois/Counter_Mode
    28  // Based on example code from: https://golang.org/src/crypto/cipher/example_test.go
    29  func EncryptAES256GCM(password string, data []byte) ([]byte, error) {
    30  	// Generate a random salt to use for key generation
    31  	salt, err := generateRandomBytes(defaultSaltLength)
    32  	if err != nil {
    33  		return nil, utils.MakeError("could not generate salt: %v", err)
    34  	}
    35  
    36  	// We need to generate a cryptographically secure key from the password
    37  	key := generateKeyFromPasswordAndSalt(password, salt, aes256KeyLength)
    38  
    39  	block, err := aes.NewCipher(key)
    40  	if err != nil {
    41  		return nil, utils.MakeError("could not create aes cipher block: %v", err)
    42  	}
    43  
    44  	gcm, err := cipher.NewGCM(block)
    45  	if err != nil {
    46  		return nil, utils.MakeError("could not create gcm cipher mode: %v", err)
    47  	}
    48  
    49  	// Generate a random nonce to use for encryption
    50  	nonce, err := generateRandomBytes(gcm.NonceSize())
    51  	if err != nil {
    52  		return nil, utils.MakeError("could not generate nonce: %v", err)
    53  	}
    54  
    55  	// We want to store the salt and nonce as a prefix to the encrypted data
    56  	prefix := append([]byte(saltHeader), salt...)
    57  	prefix = append(prefix, []byte(nonceHeader)...)
    58  	prefix = append(prefix, nonce...)
    59  
    60  	// Encrypt the data and append it to the prefix
    61  	encryptedData := gcm.Seal(prefix, nonce, data, nil)
    62  	return encryptedData, nil

    98  // attempts to extract the salt and nonce used to encrypt the file.
    99  func getSaltNonceAndDataFromEncryptedFile(file []byte, saltLength, nonceLength int) (salt, nonce, data []byte, err error) {
   100  	// The salt and nonce should be prefixed to the data
   101  	saltHeaderLength := len(saltHeader)
   102  	nonceHeaderLength := len(nonceHeader)
   103  
   104  	totalSaltLength := saltLength + saltHeaderLength
   105  	totalNonceLength := nonceLength + nonceHeaderLength
   106  
   107  	prefixLength := totalSaltLength + totalNonceLength
   108  
   109  	if len(file) < prefixLength {
   110  		return nil, nil, nil, utils.MakeError("encrypted file is too short")
   111  	}
   112  
   113  	// Verify salt header and extract salt
   114  	extractedSaltHeader := string(file[:saltHeaderLength])
   115  	if extractedSaltHeader != saltHeader {
   116  		return nil, nil, nil, utils.MakeError("encrypted file does not have a valid salt header")
   117  	}
   118  	salt = file[saltHeaderLength:totalSaltLength]
   119  
   120  	// Verify nonce header and extract nonce
   121  	extractedNonceHeader := string(file[totalSaltLength : totalSaltLength+nonceHeaderLength])
   122  	if extractedNonceHeader != nonceHeader {
   123  		return nil, nil, nil, utils.MakeError("encrypted file does not have a valid nonce header")
   124  	}
   125  	nonce = file[totalSaltLength+nonceHeaderLength : prefixLength]
   126  
   127  	// The remaining is encrypted data
   128  	data = file[prefixLength:]
   129  	return
   130  }
