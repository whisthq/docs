   165  	if err != nil {
   166  		return nil, fmt.Errorf("could not create relative axis input device: %v", err)

   169  	err = registerDevice(deviceFile, uintptr(evKey))

   176  	for _, event := range []int{evBtnLeft, evBtnRight, evBtnMiddle, evBtnSide, evBtnExtra} {
   177  		err = ioctl(deviceFile, uiSetKeyBit, uintptr(event))

   179  			deviceFile.Close()

   181  		}
   182  	}
   183

   186  		deviceFile.Close()
   187  		return nil, fmt.Errorf("failed to register relative axis input device: %v", err)
   188  	}
   189  
   190  	// register relative events
   191  	for _, event := range []int{relX, relY, relWheel, relHWheel, relWheelHiRes, relHWheelHiRes} {
   192  		err = ioctl(deviceFile, uiSetRelBit, uintptr(event))
   193  		if err != nil {
   194  			deviceFile.Close()
   195  			return nil, fmt.Errorf("failed to register relative event %v: %v", event, err)
   196  		}
   197  	}
