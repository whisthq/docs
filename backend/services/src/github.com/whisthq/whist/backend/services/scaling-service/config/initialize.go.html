    93  // getFrontendVersion returns the current version of the frontend, which is initially
    94  // populated from the config database. This value is used by the scaling algorithm to
    95  // determine if the incoming requests come from an outdated frontend, and is part of
    96  // common configuration values shared by the scaling algorithms. Its necessary to grab
    97  // a lock because multiple scaling algorithms read and update it.
    98  func getFrontendVersion(dbVersion subscriptions.FrontendVersion, version *string) {
    99  	if dbVersion == (subscriptions.FrontendVersion{}) {
   100  		*version = "0.0.0"
   101  		logger.Warningf("Got an empty frontend version, falling back to %s", version)
   102  	}
   103  
   104  	*version = dbVersion.String()
   105  
   106  	logger.Infof("Frontend version: %v", *version)
   107  }

   110  func getFreeMandelboxes(db map[string]string, regions []string, dest *map[string]int) {
   111  	mboxes := make(map[string]int)
   112  
   113  	for _, region := range regions {
   114  		var n int
   115  		suffix := strings.ToUpper(strings.ReplaceAll(region, "-", "_"))
   116  		key := fmt.Sprintf("DESIRED_FREE_MANDELBOXES_%s", suffix)
   117  		count, ok := db[key]
   118  		fallback := 2
   119  
   120  		if !ok {
   121  			n = fallback
   122  			logger.Errorf("no value specified for configuration key '%s'. Using %d "+
   123  				"by default.", key, n)
   124  		} else if k, err := strconv.Atoi(count); err != nil {
   125  			n = fallback
   126  			logger.Errorf("Failed to parse value for configuration key '%s': %s", key,
   127  				err)
   128  		} else {
   129  			n = k
   130  		}
   131  
   132  		mboxes[region] = n
   133  	}
   134  
   135  	*dest = mboxes
   136  }

   138  // initialize populates the configuration singleton with values from the
   139  // configuration database.
   140  func initialize(ctx context.Context, client subscriptions.WhistGraphQLClient) error {
   141  	rw.Lock()
   142  	defer rw.Unlock()
   143  
   144  	// Copy the existing configuration after acquiring the write lock so we can
   145  	// perform the update atomically.
   146  	newConfig := config
   147  
   148  	db, err := getConfigFromDB(ctx, client)
   149  
   150  	if err != nil {
   151  		return err
   152  	}
   153  
   154  	if err := getEnabledRegions(db, &newConfig.enabledRegions); err != nil {
   155  		return err
   156  	}
   157  
   158  	if err := getMandelboxLimit(db, &newConfig.mandelboxLimitPerUser); err != nil {
   159  		return err
   160  	}
   161  
   162  	// Get the most recent frontend version from the config database
   163  	dbVersion, err := dbclient.GetFrontendVersion(ctx, client)
   164  	if err != nil {
   165  		logger.Error(err)
   166  	}
   167  
   168  	getFrontendVersion(dbVersion, &newConfig.frontendVersion)
   169  	getFreeMandelboxes(db, newConfig.enabledRegions,
   170  		&newConfig.targetFreeMandelboxes)
   171  
   172  	config = newConfig
   173  
   174  	return nil
   175  }

   177  // initializeLocal populates the global configuration singleton with static
   178  // data.
   179  func initializeLocal(_ context.Context, _ subscriptions.WhistGraphQLClient) error {
   180  	// When APP_ENV has localenv, this function is the Initialize function's
   181  	// implementation. Notice that when this function is the Initialize function's
   182  	// implementation, there are no opportunities to patch the Initialize
   183  	// function's return value. APP_ENV gets localenv when we run the scaling
   184  	// service's automated test suite, so this function is the unpatchable
   185  	// implementation of the Initialize function when we are running tests. There
   186  	// are a lot of existing tests that rely on a region called test-region being
   187  	// enabled. Since this function is unpatchable, we always have to make it
   188  	// return that the region called test-region is enabled.
   189  	config.enabledRegions = []string{"us-east-1", "test-region"}
   190  	config.mandelboxLimitPerUser = 3
   191  	config.targetFreeMandelboxes = make(map[string]int)
   192  
   193  	for _, region := range config.enabledRegions {
   194  		config.targetFreeMandelboxes[region] = 2
   195  	}
   196  
   197  	logger.Warningf("Scaling service local build not fetching configuration " +
   198  		"values from the config database. Using static configuration instead.")
   199  
   200  	return nil
   201  }
