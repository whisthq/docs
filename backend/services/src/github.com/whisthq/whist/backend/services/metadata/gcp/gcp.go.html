   118  func metadataRetriever(resource string) (string, error) {
   119  	httpClient := http.Client{
   120  		Timeout: 1 * time.Second,
   121  	}
   122  	u, err := url.Parse(EndpointBase)
   123  	if err != nil {
   124  		return "", utils.MakeError("failed to parse metadata endpoint URL: %s", err)
   125  	}
   126  
   127  	u.Path = path.Join(u.Path, resource)
   128  	req, err := http.NewRequest(http.MethodGet, u.String(), nil)
   129  	if err != nil {
   130  		return "", utils.MakeError("failed to create request for GCP endpoint: %s", err)
   131  	}
   132  
   133  	req.Header.Add("Metadata-Flavor", "Google")
   134  	resp, err := httpClient.Do(req)
   135  	if err != nil {
   136  		return "", utils.MakeError("error retrieving data from URL %s: %v. Is the service running on a GCP VM instance?", u.String(), err)
   137  	}
   138  	defer resp.Body.Close()
   139  
   140  	body, err := io.ReadAll(resp.Body)
   141  	if err != nil {
   142  		return "", utils.MakeError("error reading response body from URL %s: %v", u.String(), err)
   143  	}
   144  	if resp.StatusCode != 200 {
   145  		return "", utils.MakeError("got non-200 response from URL %s: %s", u.String(), resp.Status)
   146  	}
   147  	return string(body), nil
