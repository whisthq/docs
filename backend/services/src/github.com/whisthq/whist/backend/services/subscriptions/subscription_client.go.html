   114  // Subscribe creates the subscriptions according to the received queries and conditions.
   115  // It passes results through the received channel if the received `conditionFn` is true.
   116  func (wc *SubscriptionClient) Subscribe(query GraphQLQuery, variables map[string]interface{}, result SubscriptionEvent,
   117  	conditionFn Handlerfn, subscriptionEvents chan SubscriptionEvent) (string, error) {
   118  
   119  	id, err := wc.Hasura.Subscribe(query, variables, func(data *json.RawMessage, err error) error {
   120  		if err != nil {
   121  			return utils.MakeError("error receiving subscription event from Hasura: %v", err)
   122  		}
   123  
   124  		// Note: this switch is necessary to unmarshal the result into the appropiate
   125  		// event type. Otherwise it will be unmarshalled as a map[string]interface{}
   126  		switch result := result.(type) {
   127  		case InstanceEvent:
   128  			err = json.Unmarshal(*data, &result)
   129  			if err != nil {
   130  				return utils.MakeError("failed to unmarshal subscription event: %v", err)
   131  			}
   132  			if conditionFn(result, variables) {
   133  				// We notify via the subscriptionsEvent channel
   134  				subscriptionEvents <- &result
   135  			}
   136  		case MandelboxEvent:
   137  			err = json.Unmarshal(*data, &result)
   138  			if err != nil {
   139  				return utils.MakeError("failed to unmarshal subscription event: %v", err)
   140  			}
   141  			if conditionFn(result, variables) {
   142  				// We notify via the subscriptionsEvent channel
   143  				subscriptionEvents <- &result
   144  			}
   145  		case ClientAppVersionEvent:
   146  			err = json.Unmarshal(*data, &result)
   147  			if err != nil {
   148  				return utils.MakeError("failed to unmarshal subscription event: %v", err)
   149  			}
   150  			if conditionFn(result, variables) {
   151  				// We notify via the subscriptionsEvent channel
   152  				subscriptionEvents <- &result
   153  			}
   154  		}
   155  
   156  		return nil
   157  	})
   158  
   159  	if err != nil {
   160  		return "", err
   161  	}
   162  
   163  	return id, nil
   164  }

   177  // Close manages all the logic to unsubscribe to every subscription and close the connection
   178  // to the Hasura server correctly.
   179  func (wc *SubscriptionClient) Close() error {
   180  	// We have to ensure we unsubscribe to every subscription
   181  	// before closing the client, otherwise it will result in a deadlock!
   182  	logger.Infof("Closing Hasura subscriptions...")
   183  	for _, id := range wc.GetSubscriptionIDs() {
   184  		// This is safe to do because the Unsubscribe method
   185  		// acquires a lock when closing the connection.
   186  		err := wc.Hasura.Unsubscribe(id)
   187  
   188  		if err != nil {
   189  			// Only use a warning instead of an error because failure to unsubscribe
   190  			// is not fatal, as we have already started the host service
   191  			// shut down process and the client will get cleaned up.
   192  			logger.Warningf("Failed to unsubscribe from:%v, %v", id, err)
   193  		}
   194  	}
   195  
   196  	var err error
   197  	// Once we have successfully unsubscribed, close the connection to the
   198  	// Hasura server. Only do so if `wc.Hasura` is nil, otherwise the close
   199  	// function will attempt to do so with a nil pointer.
   200  	logger.Infof("Closing connection to Hasura server...")
   201  	if wc.Hasura != nil {
   202  		err = wc.Hasura.Close()
   203  	}
   204  
   205  	if err != nil {
   206  		// Only use a warning instead of an error because failure to close the
   207  		// Hasura server is not fatal, as we have already started the host service
   208  		// shut down process and the client will get cleaned up.
   209  		logger.Warningf("Error closing connection with Hasura server: %v", err)
   210  	}
   211  
   212  	// Once successfully close, set the Hasura field to nil
   213  	wc.Hasura = nil
   214  	return nil
   215  }
