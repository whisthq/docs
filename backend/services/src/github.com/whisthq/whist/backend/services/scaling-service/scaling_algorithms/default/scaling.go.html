    17  // lingering instances, respecting the buffer defined for each region. Free instances will be
    18  // marked as draining, and lingering instances will be terminated and removed from the database.

    21  		zap.String("id", event.ID),
    22  		zap.Any("type", event.Type),
    23  		zap.String("region", event.Region),
    24  	}
    25  	logger.Infow("Starting scale down action.", contextFields)
    26  	defer logger.Infow("Finished scale down action.", contextFields)
    27  
    28  	// We want to verify if we have the desired capacity after scaling down.
    29  	defer func() {
    30  		err := s.VerifyCapacity(scalingCtx, event)
    31  		if err != nil {
    32  			logger.Errorf("error verifying capacity when scaling down: %s", err)
    33  		}
    34  	}()

    36  	// Acceptable time for an instance to drain, if an instance takes longer it will be considered as lingering.
    37  	const expectedDrainTime time.Duration = 10 * time.Minute
    38  
    39  	var (
    40  		// freeInstances is a list of instances with full-capacity available that have no running mandelboxes assigned to them.
    41  		freeInstances []subscriptions.Instance
    42  
    43  		// lingeringInstances is a list of instances that have been draining for more time than expected.
    44  		lingeringInstances []subscriptions.Instance
    45  
    46  		// drainingInstances is a list of instances that are currently draining, within the expected time.
    47  		drainingInstances []subscriptions.Instance
    48  
    49  		// helper list to store lingering instance IDs.
    50  		lingeringIDs []string
    51  
    52  		err error
    53  	)
    54  
    55  	// TODO: set different cloud provider when doing multi-cloud
    56  	latestImage, err := s.DBClient.QueryLatestImage(scalingCtx, s.GraphQLClient, "AWS", event.Region)
    57  	if err != nil {
    58  		return err
    59  	}
    60  
    61  	// Query database for all active instances (status ACTIVE) without mandelboxes
    62  	allActive, err := s.DBClient.QueryInstancesByStatusOnRegion(scalingCtx, s.GraphQLClient, "ACTIVE", event.Region)
    63  	if err != nil {
    64  		return utils.MakeError("failed to query database for active instances: %s", err)
    65  	}
    66  
    67  	// Active instances with space to run mandelboxes
    68  	mandelboxCapacity := helpers.ComputeRealMandelboxCapacity(latestImage.ImageID, allActive)
    69  
    70  	// Extra capacity is considered once we have a full instance worth of capacity
    71  	// more than the desired free mandelboxes. TODO: Set the instance type once we
    72  	// have support for more instance types. For now default to `g4dn.2xlarge`.
    73  	targetFreeMandelboxes := config.GetTargetFreeMandelboxes(event.Region)
    74  	extraCapacity := targetFreeMandelboxes + (defaultInstanceBuffer * instanceCapacity["g4dn.2xlarge"])
    75  
    76  	// Acquire lock on protected from scale down map
    77  	s.protectedMapLock.Lock()
    78  	defer s.protectedMapLock.Unlock()
    79  
    80  	// Create a list of instances that can be scaled down from the active instances list.
    81  	// For this, we have to consider the following conditions:
    82  	// 1. Does the instance have any running mandelboxes? If so, don't scale down.
    83  	// 2. Does the instance have the latest image, corresponding to the latest entry on the database?
    84  	// If so, check if we have more than the desired number of mandelbox capacity. In case we do, add the instance
    85  	// to the list that will be scaled down.
    86  	// 3. If the instance does not have the latest image, and is not running any mandelboxes, add to the
    87  	// list that will be scaled down.
    88  	for _, instance := range allActive {
    89  		// Compute how many mandelboxes are running on the instance. We use the current remaining capacity
    90  		// and the total capacity of the instance type to check if it has running mandelboxes.
    91  		usage := instanceCapacity[string(instance.Type)] - instance.RemainingCapacity
    92  		if usage > 0 {
    93  			// Don't scale down any instance that has running
    94  			// mandelboxes, regardless of the image it uses
    95  			logger.Infow(utils.Sprintf("Not scaling down instance %s because it has %d mandelboxes running.", instance.ID, usage), contextFields)
    96  			continue
    97  		}
    98  
    99  		_, protected := s.protectedFromScaleDown[instance.ImageID]
   100  		if protected {
   101  			// Don't scale down instances with a protected image id. A protected
   102  			// image id refers to the image id that was built and passed by the
   103  			// `build-and-deploy` workflow, that is waiting for the config database
   104  			// to update commit hashes. For this reason, we don't scale down the
   105  			// instance buffer created for this image, instead we "protect" it until
   106  			// its ready to use, to avoid downtimes and to create the buffer only once.
   107  			logger.Infow(utils.Sprintf("Not scaling down instance %s because it has an image id that is protected from scale down.", instance.ID), contextFields)
   108  			continue
   109  		}
   110  
   111  		if instance.ImageID == latestImage.ImageID {
   112  			// Current instances
   113  			// If we have more than one instance worth of extra mandelbox capacity, scale down
   114  			if mandelboxCapacity >= extraCapacity {
   115  				logger.Infow(utils.Sprintf("Scaling down instance %s because we have more mandelbox capacity of %d than desired %d.", instance.ID, mandelboxCapacity, extraCapacity), contextFields)
   116  				freeInstances = append(freeInstances, instance)
   117  				mandelboxCapacity -= instance.RemainingCapacity
   118  			}
   119  		} else {
   120  			// Old instances
   121  			logger.Infow(utils.Sprintf("Scaling down instance %s because it has an old image id %s.", instance.ID, instance.ImageID), contextFields)
   122  			freeInstances = append(freeInstances, instance)
   123  		}
   124  	}
   125  
   126  	// Check database for draining instances without running mandelboxes
   127  	drainingInstances, err = s.DBClient.QueryInstancesByStatusOnRegion(scalingCtx, s.GraphQLClient, "DRAINING", event.Region)
   128  	if err != nil {
   129  		return utils.MakeError("failed to query database for lingering instances: %s", err)
   130  	}
   131  
   132  	for _, drainingInstance := range drainingInstances {
   133  		// Check if lingering instance has any running mandelboxes
   134  		if drainingInstance.RemainingCapacity == 0 {
   135  			// If not, notify, could be a stuck mandelbox (check if mandelbox is > day old?)
   136  			logger.Warningf("Instance %s has associated mandelboxes and is marked as Draining.", drainingInstance.ID)
   137  		} else if time.Since(drainingInstance.UpdatedAt) > expectedDrainTime {
   138  			// Mark instance as lingering only if it has taken more than 10 minutes to shut down.
   139  			lingeringInstances = append(lingeringInstances, drainingInstance)
   140  			lingeringIDs = append(lingeringIDs, string(drainingInstance.ID))
   141  		}
   142  	}
   143  
   144  	// If there are any lingering instances, try to shut them down on AWS. This is a fallback measure for when an instance
   145  	// fails to receive a drain and shutdown event from the pubsub (i.e. the Hasura server restarted or went down).
   146  	if len(lingeringInstances) > 0 && !metadata.IsLocalEnv() {
   147  		logger.Infow(utils.Sprintf("Terminating %d lingering instances in %s", len(lingeringInstances), event.Region), contextFields)
   148  
   149  		err := s.Host.SpinDownInstances(scalingCtx, lingeringIDs)
   150  		if err != nil {
   151  			return utils.MakeError("failed to terminate lingering instances: %s", err)
   152  		}
   153  
   154  		// Verify that each instance is terminated correctly and is removed from the database
   155  		for _, lingeringInstance := range lingeringInstances {
   156  			err = s.VerifyInstanceRemoval(scalingCtx, lingeringInstance)
   157  			if err != nil {
   158  				return err
   159  			}
   160  		}
   161  
   162  	} else if !metadata.IsLocalEnv() {
   163  		logger.Infow(utils.Sprintf("There are no lingering instances in %s.", event.Region), contextFields)
   164  	} else {
   165  		logger.Infow("Running on localdev so not spinning down lingering instances.", contextFields)
   166  
   167  		// Verify that each instance is terminated correctly and is removed from the database
   168  		for _, lingeringInstance := range lingeringInstances {
   169  			err = s.VerifyInstanceRemoval(scalingCtx, lingeringInstance)
   170  			if err != nil {
   171  				return err
   172  			}
   173  		}
   174  	}
   175  
   176  	// Verify if there are free instances that can be scaled down
   177  	if len(freeInstances) == 0 {
   178  		logger.Infow(utils.Sprintf("There are no free instances to scale down in %s.", event.Region), contextFields)
   179  		return nil
   180  	}
   181  
   182  	logger.Infow(utils.Sprintf("Scaling down %d free instances in %s.", len(freeInstances), event.Region), contextFields)
   183  
   184  	for _, instance := range freeInstances {
   185  		logger.Infow(utils.Sprintf("Scaling down instance %s.", instance.ID), contextFields)
   186  
   187  		instance.Status = "DRAINING"
   188  		_, err = s.DBClient.UpdateInstance(scalingCtx, s.GraphQLClient, instance)
   189  		if err != nil {
   190  			return utils.MakeError("failed to mark instance %s as draining: %s", instance.ID, err)
   191  		}
   192  
   193  		logger.Infow(utils.Sprintf("Marked instance %s as draining on database.", instance.ID), contextFields)
   194  	}
   195  
   196  	return err

   199  // ScaleUpIfNecessary is a scaling action that launched the received number of instances on
   200  // the cloud provider and registers them on the database with the initial values.
   201  func (s *DefaultScalingAlgorithm) ScaleUpIfNecessary(instancesToScale int, scalingCtx context.Context, event ScalingEvent, image subscriptions.Image) error {
   202  	contextFields := []interface{}{
   203  		zap.String("id", event.ID),
   204  		zap.Any("type", event.Type),
   205  		zap.String("region", event.Region),
   206  	}
   207  	logger.Infow("Starting scale up action.", contextFields)
   208  	defer logger.Infow("Finished scale up action.", contextFields)
   209  
   210  	var (
   211  		// Slice that will hold the instances and pass them to the dbclient
   212  		instancesForDb []subscriptions.Instance
   213  		// Slice that will hold fake mandelboxes and pass them to the dbclient
   214  		fakeMandelboxesForDb []subscriptions.Mandelbox
   215  		err                  error
   216  	)
   217  
   218  	// If we are running on a local or testing environment, spinup "fake" instances to avoid
   219  	// creating them on a cloud provider. In any other case we call the host handler to create
   220  	// them on the cloud provider for us.
   221  	if metadata.IsLocalEnv() && !metadata.IsRunningInCI() {
   222  		logger.Infow("Running on localdev so scaling up fake instances.", contextFields)
   223  		instancesForDb, fakeMandelboxesForDb = helpers.SpinUpFakeInstances(instancesToScale, image.ImageID, event.Region)
   224  	} else {
   225  		// Call the host handler to handle the instance spinup in the cloud provider
   226  		instancesForDb, err = s.Host.SpinUpInstances(scalingCtx, int32(instancesToScale), maxWaitTimeReady, image)
   227  		if err != nil {
   228  			return utils.MakeError("failed to spin up instances: %s", err)
   229  		}
   230  
   231  		// Check if we could create the desired number of instances
   232  		if len(instancesForDb) != instancesToScale {
   233  			return utils.MakeError("could not scale up %d instances, only scaled up %d", instancesToScale, len(instancesForDb))
   234  		}
   235  
   236  		for i := 0; i < len(instancesForDb); i++ {
   237  			instancesForDb[i].RemainingCapacity = instanceCapacity[instancesForDb[i].Type]
   238  			logger.Infow(utils.Sprintf("Created tagged instance with ID %s", instancesForDb[i].ID), contextFields)
   239  		}
   240  	}
   241  
   242  	logger.Infow("Inserting newly created instances to database.", contextFields)
   243  
   244  	// If successful, write to database
   245  	affectedRows, err := s.DBClient.InsertInstances(scalingCtx, s.GraphQLClient, instancesForDb)
   246  	if err != nil {
   247  		return utils.MakeError("failed to insert instances into database: %s", err)
   248  	}
   249  
   250  	logger.Infow(utils.Sprintf("Inserted %d rows to database.", affectedRows), contextFields)
   251  
   252  	// This is to acommodate localdev testing flows in which we need to register fake mandelboxes
   253  	// on the database.
   254  	if len(fakeMandelboxesForDb) > 0 {
   255  		affectedRows, err = s.DBClient.InsertMandelboxes(scalingCtx, s.GraphQLClient, fakeMandelboxesForDb)
   256  		if err != nil {
   257  			return utils.MakeError("failed to insert fake mandelboxes to database: %s", err)
   258  		}
   259  		logger.Infow(utils.Sprintf("Inserted %d rows to database.", affectedRows), contextFields)
   260  	}
   261  
   262  	return nil
   263  }
