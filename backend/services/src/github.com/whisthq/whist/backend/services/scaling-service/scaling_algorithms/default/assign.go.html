    26  	}

    28  	defer logger.Infow("Finished mandelbox assign action.", contextFields)
    29  
    30  	// We want to verify if we have the desired capacity after assigning a mandelbox
    31  	defer func() {
    32  		err := s.VerifyCapacity(scalingCtx, event)
    33  		if err != nil {
    34  			logger.Errorf("error verifying capacity when assigning mandelbox: %s", err)
    35  		}
    36  	}()
    37

    39

    42  	var serviceUnavailable bool = true
    43  	defer func() {
    44  		if serviceUnavailable {
    45  			mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
    46  				Error: SERVICE_UNAVAILABLE,
    47  			}, nil)
    48  		}
    49  	}()
    50  
    51  	// Note: we receive the email from the client, so its value should
    52  	// not be trusted for anything else other than logging since
    53  	// it can be spoofed. We sanitize the email before using to help mitigate
    54  	// potential attacks.
    55  	unsafeEmail, err := helpers.SanitizeEmail(mandelboxRequest.UserEmail)
    56  	if err != nil {
    57  		return err
    58  	}
    59  	logger.Infof("Frontend reported email %s, this value might not be accurate and is untrusted.", unsafeEmail)
    60  	// Append user email to logging context for better debugging.
    61  	contextFields = append(contextFields, zap.String("user", unsafeEmail))
    62  
    63  	var (
    64  		requestedRegions   = mandelboxRequest.Regions // This is a list of the regions requested by the frontend, in order of proximity.
    65  		availableRegions   []string                   // List of regions that are enabled and match the regions in the request.
    66  		unavailableRegions []string                   // List of regions that are not enabled but are present in the request.
    67  		assignedInstance   subscriptions.Instance     // The instance that is assigned to the user.
    68  	)
    69  
    70  	// Populate availableRegions
    71  	for _, requestedRegion := range requestedRegions {
    72  		var regionFound bool
    73  		for _, enabledRegion := range GetEnabledRegions() {
    74  			if enabledRegion == requestedRegion {
    75  				availableRegions = append(availableRegions, requestedRegion)
    76  				regionFound = true
    77  			}
    78  		}
    79  
    80  		if !regionFound {
    81  			unavailableRegions = append(unavailableRegions, requestedRegion)
    82  		}
    83  	}
    84  
    85  	// This means that the user has requested access to some regions that are not yet enabled,
    86  	// but could still be allocated to a region that is relatively close.
    87  	if len(unavailableRegions) != 0 && len(unavailableRegions) != len(requestedRegions) {
    88  		if metadata.GetAppEnvironment() == metadata.EnvProd {
    89  			logger.Errorf("user %s requested access to the following unavailable regions: %s", unsafeEmail, utils.PrintSlice(unavailableRegions))
    90  		}
    91  	}
    92  
    93  	// The user requested access to only unavailable regions. The last resort is to default to us-east-1.
    94  	if len(unavailableRegions) == len(requestedRegions) {
    95  		if metadata.GetAppEnvironment() == metadata.EnvProd {
    96  			logger.Errorf("user %s requested access to only unavailable regions: %s", unsafeEmail, utils.PrintSlice(unavailableRegions))
    97  		}
    98  		availableRegions = []string{"us-east-1"}
    99  	}
   100  
   101  	if len(availableRegions) == 0 {
   102  		serviceUnavailable = false
   103  		err := utils.MakeError("")
   104  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   105  			Error: REGION_NOT_ENABLED,
   106  		}, err)
   107  		return err
   108  	}
   109  
   110  	// This condition is to accomodate the worflow for developers of the Whist frontend
   111  	// to test their changes without needing to update the development database with
   112  	// commit_hashes on their local machines.
   113  	if metadata.IsLocalEnv() || mandelboxRequest.CommitHash == CLIENT_COMMIT_HASH_DEV_OVERRIDE {
   114  		// Query for the latest image id
   115  		imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region) // TODO: set different provider when doing multi-cloud.
   116  		if err != nil {
   117  			return utils.MakeError("failed to query database for current image: %s", err)
   118  		}
   119  
   120  		if len(imageResult) == 0 {
   121  			return utils.MakeError("image not found in %s", event.Region)
   122  		}
   123  
   124  		mandelboxRequest.CommitHash = string(imageResult[0].ClientSHA)
   125  	}
   126  
   127  	// This is the "main" loop that does all the work and tries to find an instance for a user. first, it will iterate
   128  	// over the list of regions provided on the request, and will query the database on each to return the list of
   129  	// instances with capacity on the current region. Once it gets the instances, it will iterate over them and try
   130  	// to find an instance with a matching commit hash. If it fails to do so, move on to the next region.
   131  	for _, region := range availableRegions {
   132  		logger.Infow(utils.Sprintf("Trying to find instance in region %s, with commit hash %s.",
   133  			region, mandelboxRequest.CommitHash), contextFields)
   134  
   135  		instanceResult, err := s.DBClient.QueryInstanceWithCapacity(scalingCtx, s.GraphQLClient, region)
   136  		if err != nil {
   137  			return utils.MakeError("failed to query for instance with capacity: %s", err)
   138  		}
   139  
   140  		if len(instanceResult) == 0 {
   141  			logger.Warningf("Failed to find an instance in %s for commit hash %s. Trying on next region.", region, mandelboxRequest.CommitHash)
   142  			continue
   143  		}
   144  
   145  		var instanceFound bool
   146  
   147  		// Iterate over available instances, try to find one with a matching commit hash
   148  		for i := range instanceResult {
   149  			assignedInstance = instanceResult[i]
   150  
   151  			if assignedInstance.ClientSHA == mandelboxRequest.CommitHash {
   152  				logger.Infow(utils.Sprintf("Found instance %s with commit hash %s.", assignedInstance.ID, assignedInstance.ClientSHA), contextFields)
   153  				instanceFound = true
   154  				break
   155  			}
   156  		}
   157  
   158  		// Break of outer loop if instance was found. If no instance with
   159  		// matching commit hash was found, move on to the next region.
   160  		if instanceFound {
   161  			break
   162  		}
   163  	}
   164  
   165  	// No instances with capacity were found
   166  	if assignedInstance == (subscriptions.Instance{}) {
   167  		serviceUnavailable = false
   168  		err := utils.MakeError("did not find an instance with capacity for user %s and commit hash %s.", mandelboxRequest.UserEmail, mandelboxRequest.CommitHash)
   169  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   170  			Error: NO_INSTANCE_AVAILABLE,
   171  		}, err)
   172  		return err
   173  	}
   174  
   175  	var (
   176  		// The parsed version from the config
   177  		parsedFrontendVersion *hashicorp.Version
   178  		// The parsed version from the request
   179  		parsedRequestVersion *hashicorp.Version
   180  		// whether the frontend has an outdated version
   181  		isOutdatedFrontend bool
   182  	)
   183  
   184  	// Get the version we keep locally for comparing the incoming request value.
   185  	frontendVersion := getFrontendVersion()
   186  
   187  	// Parse the version with the `hashicorp/go-version` package so we can compare.
   188  	parsedFrontendVersion, err = hashicorp.NewVersion(frontendVersion)
   189  	if err != nil {
   190  		logger.Errorf("ailed parsing frontend version from scaling algorithm config: %s", err)
   191  	}
   192  
   193  	// Parse the version we got in the request.
   194  	if mandelboxRequest.Version == "" {
   195  		logger.Warningf("Request version is empty, this is caused by frontend testing and development.")
   196  	} else {
   197  		parsedRequestVersion, err = hashicorp.NewVersion(mandelboxRequest.Version)
   198  		if err != nil {
   199  			logger.Errorf("failed parsing frontend version from request: %s", err)
   200  		}
   201  	}
   202  
   203  	// Compare the request version with the one from the config. If the
   204  	// version from the request is less than the one we have locally, it
   205  	// means the request comes from an outdated frontend application.
   206  	if parsedFrontendVersion != nil && parsedRequestVersion != nil {
   207  		logger.Infow(utils.Sprintf("Local version is %s, version received from request is %s.", parsedFrontendVersion.String(), parsedRequestVersion.String()), contextFields)
   208  		isOutdatedFrontend = parsedRequestVersion.LessThan(parsedFrontendVersion)
   209  	}
   210  
   211  	// There are instances with capacity available, but none of them with the desired commit hash.
   212  	// We only consider this error in cases when the frontend has a version greater or equal than
   213  	// the one in the config database. This is because when the client version is lesser (outdated client),
   214  	// it will automatically update itself to the most recent version and send another request.
   215  	if assignedInstance.ClientSHA != mandelboxRequest.CommitHash {
   216  		var msg error
   217  
   218  		// Only log the commit mismatch error when running on prod. This is because we only update the full version
   219  		// (major, minor, micro) on the config database when deploying to prod, so this is only a real error in that case.
   220  		if metadata.GetAppEnvironment() == metadata.EnvProd && !isOutdatedFrontend {
   221  			msg = utils.MakeError("found instance with capacity but different commit hash")
   222  		} else {
   223  			logger.Infow(utils.Sprintf("Did not find instance with commit hash %s, but expect frontend to autoupdate and send another request with commit hash %s.", mandelboxRequest.CommitHash, assignedInstance.ClientSHA), contextFields)
   224  		}
   225  
   226  		// Regardless if we log the error, its necessary to return an appropiate response.
   227  		serviceUnavailable = false
   228  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   229  			Error: COMMIT_HASH_MISMATCH,
   230  		}, msg)
   231  		return msg
   232  	}
   233  
   234  	// Try to find a mandelbox in the WAITING status in the assigned instance.
   235  	mandelboxResult, err := s.DBClient.QueryMandelbox(scalingCtx, s.GraphQLClient, assignedInstance.ID, "WAITING")
   236  	if err != nil {
   237  		return utils.MakeError("failed to query database for mandelbox on instance %s: %s", assignedInstance.ID, err)
   238  	}
   239  
   240  	if len(mandelboxResult) == 0 {
   241  		return utils.MakeError("failed to find a waiting mandelbox even though the instance %s had sufficient capacity", assignedInstance.ID)
   242  	}
   243  
   244  	waitingMandelbox := mandelboxResult[0]
   245  	if err != nil {
   246  		return utils.MakeError("failed to parse mandelbox id for instance %s: %s", assignedInstance.ID, err)
   247  	}
   248  
   249  	mandelboxForDb := subscriptions.Mandelbox{
   250  		ID:         waitingMandelbox.ID,
   251  		App:        string(waitingMandelbox.App),
   252  		InstanceID: assignedInstance.ID,
   253  		UserID:     mandelboxRequest.UserID,
   254  		SessionID:  utils.Sprintf("%v", mandelboxRequest.SessionID),
   255  		Status:     "ALLOCATED",
   256  		CreatedAt:  waitingMandelbox.CreatedAt,
   257  		UpdatedAt:  time.Now(),
   258  	}
   259  
   260  	// Allocate mandelbox on database so the host service can start downloading user configs
   261  	affectedRows, err := s.DBClient.UpdateMandelbox(scalingCtx, s.GraphQLClient, mandelboxForDb)
   262  	if err != nil {
   263  		return utils.MakeError("error while inserting mandelbox to database: %s", err)
   264  	}
   265  
   266  	logger.Infow(utils.Sprintf("Inserted %d rows to database.", affectedRows), contextFields)
   267  
   268  	if assignedInstance.RemainingCapacity <= 0 {
   269  		// This should never happen, but we should consider
   270  		// possible edge cases before updating the database.
   271  		return utils.MakeError("instance with id %s has a remaning capacity less than or equal to 0", assignedInstance.ID)
   272  	}
   273  
   274  	// Subtract 1 from the current instance capacity because we allocated a mandelbox
   275  	updatedCapacity := assignedInstance.RemainingCapacity - 1
   276  	assignedInstance.RemainingCapacity = updatedCapacity
   277  
   278  	affectedRows, err = s.DBClient.UpdateInstance(scalingCtx, s.GraphQLClient, assignedInstance)
   279  	if err != nil {
   280  		return utils.MakeError("error while updating instance capacity on database: %s", err)
   281  	}
   282  
   283  	logger.Infow(utils.Sprintf("Updated %d rows in database.", affectedRows), contextFields)
   284  
   285  	// Parse IP address. The database uses the CIDR notation (192.0.2.0/24)
   286  	// so we need to extract the address and send it to the frontend.
   287  	ip, _, err := net.ParseCIDR(assignedInstance.IPAddress)
   288  	if err != nil {
   289  		return utils.MakeError("failed to parse IP address %s: %s", assignedInstance.IPAddress, err)
   290  	}
   291  
   292  	serviceUnavailable = false
   293  	// Return result to assign request
   294  	mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   295  		IP:          ip.String(),
   296  		MandelboxID: types.MandelboxID(waitingMandelbox.ID),
   297  	}, nil)
   298  
   299  	return nil
