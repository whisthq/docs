    19  // MandelboxAssign is the action responsible for assigning an instance to a user,
    20  // and scaling as necessary to satisfy demand.

    22  	contextFields := []interface{}{
    23  		zap.String("id", event.ID),
    24  		zap.Any("type", event.Type),
    25  		zap.String("region", event.Region),
    26  	}
    27  	logger.Infow("Starting mandelbox assign action.", contextFields)
    28  	defer logger.Infow("Finished mandelbox assign action.", contextFields)
    29  
    30  	// We want to verify if we have the desired capacity after assigning a mandelbox
    31  	defer func() {
    32  		err := s.VerifyCapacity(scalingCtx, event)
    33  		if err != nil {
    34  			logger.Errorf("error verifying capacity when assigning mandelbox: %s", err)
    35  		}
    36  	}()
    37  
    38  	mandelboxRequest := event.Data.(*httputils.MandelboxAssignRequest)
    39  
    40  	// Note: we receive the email from the client, so its value should
    41  	// not be trusted for anything else other than logging since
    42  	// it can be spoofed. We sanitize the email before using to help mitigate
    43  	// potential attacks.
    44  	unsafeEmail, err := helpers.SanitizeEmail(mandelboxRequest.UserEmail)
    45  	if err != nil {
    46  		return err
    47  	}
    48  	logger.Infof("Frontend reported email %s, this value might not be accurate and is untrusted.", unsafeEmail)
    49  	// Append user email to logging context for better debugging.
    50  	contextFields = append(contextFields, zap.String("user", unsafeEmail))
    51  
    52  	var (
    53  		requestedRegions   = mandelboxRequest.Regions // This is a list of the regions requested by the frontend, in order of proximity.
    54  		availableRegions   []string                   // List of regions that are enabled and match the regions in the request.
    55  		unavailableRegions []string                   // List of regions that are not enabled but are present in the request.
    56  		assignedInstance   subscriptions.Instance     // The instance that is assigned to the user.
    57  	)
    58  
    59  	// Populate availableRegions
    60  	for _, requestedRegion := range requestedRegions {
    61  		var regionFound bool
    62  		for _, enabledRegion := range GetEnabledRegions() {
    63  			if enabledRegion == requestedRegion {
    64  				availableRegions = append(availableRegions, requestedRegion)
    65  				regionFound = true
    66  			}
    67  		}
    68  
    69  		if !regionFound {
    70  			unavailableRegions = append(unavailableRegions, requestedRegion)
    71  		}
    72  	}
    73  
    74  	// This means that the user has requested access to some regions that are not yet enabled,
    75  	// but could still be allocated to a region that is relatively close.
    76  	if len(unavailableRegions) != 0 && len(unavailableRegions) != len(requestedRegions) {
    77  		if metadata.GetAppEnvironment() == metadata.EnvProd {
    78  			logger.Errorf("user %s requested access to the following unavailable regions: %s", unsafeEmail, utils.PrintSlice(unavailableRegions))
    79  		}
    80  	}
    81  
    82  	// The user requested access to only unavailable regions. The last resort is to default to us-east-1.
    83  	if len(unavailableRegions) == len(requestedRegions) {
    84  		if metadata.GetAppEnvironment() == metadata.EnvProd {
    85  			logger.Errorf("user %s requested access to only unavailable regions: %s", unsafeEmail, utils.PrintSlice(unavailableRegions))
    86  		}
    87  		availableRegions = []string{"us-east-1"}
    88  	}
    89  
    90  	if len(availableRegions) == 0 {
    91  		err := utils.MakeError("")
    92  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
    93  			Error: REGION_NOT_ENABLED,
    94  		}, err)
    95  		return err
    96  	}
    97  
    98  	// This condition is to accomodate the worflow for developers of the Whist frontend
    99  	// to test their changes without needing to update the development database with
   100  	// commit_hashes on their local machines.
   101  	if metadata.IsLocalEnv() || mandelboxRequest.CommitHash == CLIENT_COMMIT_HASH_DEV_OVERRIDE {
   102  		// Query for the latest image id
   103  		imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region) // TODO: set different provider when doing multi-cloud.
   104  		if err != nil {
   105  			return utils.MakeError("failed to query database for current image: %s", err)
   106  		}
   107  
   108  		if len(imageResult) == 0 {
   109  			return utils.MakeError("image not found in %s", event.Region)
   110  		}
   111  
   112  		mandelboxRequest.CommitHash = string(imageResult[0].ClientSHA)
   113  	}
   114  
   115  	// This is the "main" loop that does all the work and tries to find an instance for a user. first, it will iterate
   116  	// over the list of regions provided on the request, and will query the database on each to return the list of
   117  	// instances with capacity on the current region. Once it gets the instances, it will iterate over them and try
   118  	// to find an instance with a matching commit hash. If it fails to do so, move on to the next region.
   119  	for _, region := range availableRegions {
   120  		logger.Infow(utils.Sprintf("Trying to find instance in region %s, with commit hash %s.",
   121  			region, mandelboxRequest.CommitHash), contextFields)
   122  
   123  		instanceResult, err := s.DBClient.QueryInstanceWithCapacity(scalingCtx, s.GraphQLClient, region)
   124  		if err != nil {
   125  			return utils.MakeError("failed to query for instance with capacity: %s", err)
   126  		}
   127  
   128  		if len(instanceResult) == 0 {
   129  			logger.Warningf("Failed to find an instance in %s for commit hash %s. Trying on next region.", region, mandelboxRequest.CommitHash)
   130  			continue
   131  		}
   132  
   133  		var instanceFound bool
   134  
   135  		// Iterate over available instances, try to find one with a matching commit hash
   136  		for i := range instanceResult {
   137  			assignedInstance = instanceResult[i]
   138  
   139  			if assignedInstance.ClientSHA == mandelboxRequest.CommitHash {
   140  				logger.Infow(utils.Sprintf("Found instance %s with commit hash %s.", assignedInstance.ID, assignedInstance.ClientSHA), contextFields)
   141  				instanceFound = true
   142  				break
   143  			}
   144  		}
   145  
   146  		// Break of outer loop if instance was found. If no instance with
   147  		// matching commit hash was found, move on to the next region.
   148  		if instanceFound {
   149  			break
   150  		}
   151  	}
   152  
   153  	// No instances with capacity were found
   154  	if assignedInstance == (subscriptions.Instance{}) {
   155  		err := utils.MakeError("did not find an instance with capacity for user %s and commit hash %s.", mandelboxRequest.UserEmail, mandelboxRequest.CommitHash)
   156  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   157  			Error: NO_INSTANCE_AVAILABLE,
   158  		}, err)
   159  		return err
   160  	}
   161  
   162  	var (
   163  		// The parsed version from the config
   164  		parsedFrontendVersion *hashicorp.Version
   165  		// The parsed version from the request
   166  		parsedRequestVersion *hashicorp.Version
   167  		// whether the frontend has an outdated version
   168  		isOutdatedFrontend bool
   169  	)
   170  
   171  	// Get the version we keep locally for comparing the incoming request value.
   172  	frontendVersion := getFrontendVersion()
   173  
   174  	// Parse the version with the `hashicorp/go-version` package so we can compare.
   175  	parsedFrontendVersion, err = hashicorp.NewVersion(frontendVersion)
   176  	if err != nil {
   177  		logger.Errorf("failed parsing frontend version from scaling algorithm config: %s", err)
   178  	}
   179  
   180  	// Parse the version we got in the request.
   181  	parsedRequestVersion, err = hashicorp.NewVersion(mandelboxRequest.Version)
   182  	if err != nil {
   183  		logger.Errorf("failed parsing frontend version from request: %s", err)
   184  	}
   185  
   186  	// Compare the request version with the one from the config. If the
   187  	// version from the request is less than the one we have locally, it
   188  	// means the request comes from an outdated frontend application.
   189  	if parsedFrontendVersion != nil && parsedRequestVersion != nil {
   190  		logger.Infow(utils.Sprintf("Local version is %s, version received from request is %s.", parsedFrontendVersion.String(), parsedRequestVersion.String()), contextFields)
   191  		isOutdatedFrontend = parsedRequestVersion.LessThan(parsedFrontendVersion)
   192  	}
   193  
   194  	// There are instances with capacity available, but none of them with the desired commit hash.
   195  	// We only consider this error in cases when the frontend has a version greater or equal than
   196  	// the one in the config database. This is because when the client version is lesser (outdated client),
   197  	// it will automatically update itself to the most recent version and send another request.
   198  	if assignedInstance.ClientSHA != mandelboxRequest.CommitHash {
   199  		var msg error
   200  
   201  		// Only log the commit mismatch error when running on prod. This is because we only update the full version
   202  		// (major, minor, micro) on the config database when deploying to prod, so this is only a real error in that case.
   203  		if metadata.GetAppEnvironment() == metadata.EnvProd && !isOutdatedFrontend {
   204  			msg = utils.MakeError("found instance with capacity but different commit hash")
   205  		} else {
   206  			logger.Infow(utils.Sprintf("Did not find instance with commit hash %s, but expect frontend to autoupdate and send another request with commit hash %s.", mandelboxRequest.CommitHash, assignedInstance.ClientSHA), contextFields)
   207  		}
   208  
   209  		// Regardless if we log the error, its necessary to return an appropiate response.
   210  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   211  			Error: COMMIT_HASH_MISMATCH,
   212  		}, msg)
   213  		return msg
   214  	}
   215  
   216  	// Try to find a mandelbox in the WAITING status in the assigned instance.
   217  	mandelboxResult, err := s.DBClient.QueryMandelbox(scalingCtx, s.GraphQLClient, assignedInstance.ID, "WAITING")
   218  	if err != nil {
   219  		return utils.MakeError("failed to query database for mandelbox on instance %s: %s", assignedInstance.ID, err)
   220  	}
   221  
   222  	if len(mandelboxResult) == 0 {
   223  		return utils.MakeError("failed to find a waiting mandelbox even though the instance %s had sufficient capacity", assignedInstance.ID)
   224  	}
   225  
   226  	waitingMandelbox := mandelboxResult[0]
   227  	if err != nil {
   228  		return utils.MakeError("failed to parse mandelbox id for instance %s: %s", assignedInstance.ID, err)
   229  	}
   230  
   231  	mandelboxForDb := subscriptions.Mandelbox{
   232  		ID:         waitingMandelbox.ID,
   233  		App:        string(waitingMandelbox.App),
   234  		InstanceID: assignedInstance.ID,
   235  		UserID:     mandelboxRequest.UserID,
   236  		SessionID:  utils.Sprintf("%v", mandelboxRequest.SessionID),
   237  		Status:     "ALLOCATED",
   238  		CreatedAt:  waitingMandelbox.CreatedAt,
   239  		UpdatedAt:  time.Now(),
   240  	}
   241  
   242  	// Allocate mandelbox on database so the host service can start downloading user configs
   243  	affectedRows, err := s.DBClient.UpdateMandelbox(scalingCtx, s.GraphQLClient, mandelboxForDb)
   244  	if err != nil {
   245  		return utils.MakeError("error while inserting mandelbox to database: %s", err)
   246  	}
   247  
   248  	logger.Infow(utils.Sprintf("Inserted %d rows to database.", affectedRows), contextFields)
   249  
   250  	if assignedInstance.RemainingCapacity <= 0 {
   251  		// This should never happen, but we should consider
   252  		// possible edge cases before updating the database.
   253  		return utils.MakeError("instance with id %s has a remaning capacity less than or equal to 0", assignedInstance.ID)
   254  	}
   255  
   256  	// Subtract 1 from the current instance capacity because we allocated a mandelbox
   257  	updatedCapacity := assignedInstance.RemainingCapacity - 1
   258  	assignedInstance.RemainingCapacity = updatedCapacity
   259  
   260  	affectedRows, err = s.DBClient.UpdateInstance(scalingCtx, s.GraphQLClient, assignedInstance)
   261  	if err != nil {
   262  		return utils.MakeError("error while updating instance capacity on database: %s", err)
   263  	}
   264  
   265  	logger.Infow(utils.Sprintf("Updated %d rows in database.", affectedRows), contextFields)
   266  
   267  	// Parse IP address. The database uses the CIDR notation (192.0.2.0/24)
   268  	// so we need to extract the address and send it to the frontend.
   269  	ip, _, err := net.ParseCIDR(assignedInstance.IPAddress)
   270  	if err != nil {
   271  		return utils.MakeError("failed to parse IP address %s: %s", assignedInstance.IPAddress, err)
   272  	}
   273  
   274  	// Return result to assign request
   275  	mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   276  		IP:          ip.String(),
   277  		MandelboxID: types.MandelboxID(waitingMandelbox.ID),
   278  	}, nil)
   279  
   280  	return nil
