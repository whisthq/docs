    20  // and scaling as necessary to satisfy demand.

    23  	defer logger.Infof("Finished mandelbox assign action for event: %v", event)

    25  	// We want to verify if we have the desired capacity after assigning a mandelbox
    26  	defer func() {
    27  		err := s.VerifyCapacity(scalingCtx, event)
    28  		if err != nil {
    29  			logger.Errorf("Error verifying capacity on %v. Err: %v", event.Region, err)
    30  		}
    31  	}()
    32  
    33  	mandelboxRequest := event.Data.(*httputils.MandelboxAssignRequest)
    34  
    35  	// Note: we receive the email from the client, so its value should
    36  	// not be trusted for anything else other than logging since
    37  	// it can be spoofed. We sanitize the email before using to help mitigate
    38  	// potential attacks.
    39  	unsafeEmail, err := helpers.SanitizeEmail(mandelboxRequest.UserEmail)
    40  	if err != nil {
    41  		// err is already wrapped here
    42  		return err
    43  	}
    44  
    45  	var (
    46  		requestedRegions   = mandelboxRequest.Regions // This is a list of the regions requested by the frontend, in order of proximity.
    47  		availableRegions   []string                   // List of regions that are enabled and match the regions in the request.
    48  		unavailableRegions []string                   // List of regions that are not enabled but are present in the request.
    49  		assignedInstance   subscriptions.Instance     // The instance that is assigned to the user.
    50  	)
    51  
    52  	// Populate availableRegions
    53  	for _, requestedRegion := range requestedRegions {
    54  		var regionFound bool
    55  		for _, enabledRegion := range GetEnabledRegions() {
    56  			if enabledRegion == requestedRegion {
    57  				availableRegions = append(availableRegions, requestedRegion)
    58  				regionFound = true
    59  			}
    60  		}
    61  
    62  		if !regionFound {
    63  			unavailableRegions = append(unavailableRegions, requestedRegion)
    64  		}
    65  	}
    66  
    67  	// This means that the user has requested access to some regions that are not yet enabled,
    68  	// but could still be allocated to a region that is relatively close.
    69  	if len(unavailableRegions) != 0 && len(unavailableRegions) != len(requestedRegions) {
    70  		if metadata.GetAppEnvironment() == metadata.EnvProd {
    71  			logger.Errorf("User %s requested access to the following unavailable regions: %s. Trying to find instance on remaining available regions.", unsafeEmail, utils.PrintSlice(unavailableRegions))
    72  		}
    73  	}
    74  
    75  	// The user requested access to only unavailable regions. The last resort is to default to us-east-1.
    76  	if len(unavailableRegions) == len(requestedRegions) {
    77  		if metadata.GetAppEnvironment() == metadata.EnvProd {
    78  			logger.Errorf("User %s requested access to only unavailable regions: %s. Defaulting to us-east-1.", unsafeEmail, utils.PrintSlice(unavailableRegions))
    79  		}
    80  		availableRegions = []string{"us-east-1"}
    81  	}
    82  
    83  	if len(availableRegions) == 0 {
    84  		err := utils.MakeError("")
    85  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
    86  			Error: REGION_NOT_ENABLED,
    87  		}, err)
    88  		return err
    89  	}
    90  
    91  	// This condition is to accomodate the worflow for developers of client_apps
    92  	// to test their changes without needing to update the development database with
    93  	// commit_hashes on their local machines.
    94  	if metadata.IsLocalEnv() || mandelboxRequest.CommitHash == CLIENT_COMMIT_HASH_DEV_OVERRIDE {
    95  		// Query for the latest image id
    96  		imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region) // TODO: set different provider when doing multi-cloud.
    97  		if err != nil {
    98  			return utils.MakeError("failed to query database for current image. Err: %v", err)
    99  		}
   100  
   101  		if len(imageResult) == 0 {
   102  			return utils.MakeError("Image not found on %v.", event.Region)
   103  		}
   104  
   105  		mandelboxRequest.CommitHash = string(imageResult[0].ClientSHA)
   106  	}
   107  
   108  	// This is the "main" loop that does all the work and tries to find an instance for a user. first, it will iterate
   109  	// over the list of regions provided on the request, and will query the database on each to return the list of
   110  	// instances with capacity on the current region. Once it gets the instances, it will iterate over them and try
   111  	// to find an instance with a matching commit hash. If it fails to do so, move on to the next region.
   112  	for _, region := range availableRegions {
   113  		logger.Infof("Trying to find instance for user %v in region %v, with commit hash %v. (client reported email %v, this value might not be accurate and is untrusted)",
   114  			unsafeEmail, region, mandelboxRequest.CommitHash, unsafeEmail)
   115  
   116  		instanceResult, err := s.DBClient.QueryInstanceWithCapacity(scalingCtx, s.GraphQLClient, region)
   117  		if err != nil {
   118  			return utils.MakeError("failed to query for instance with capacity. Err: %v", err)
   119  		}
   120  
   121  		if len(instanceResult) == 0 {
   122  			logger.Warningf("Failed to find an instance in %v for commit hash %v. Trying on next region.", region, mandelboxRequest.CommitHash)
   123  			continue
   124  		}
   125  
   126  		var instanceFound bool
   127  
   128  		// Iterate over available instances, try to find one with a matching commit hash
   129  		for i := range instanceResult {
   130  			assignedInstance = subscriptions.Instance{
   131  				ID:                string(instanceResult[i].ID),
   132  				IPAddress:         string(instanceResult[i].IPAddress),
   133  				Provider:          string(instanceResult[i].Provider),
   134  				Region:            string(instanceResult[i].Region),
   135  				ImageID:           string(instanceResult[i].ImageID),
   136  				ClientSHA:         string(instanceResult[i].ClientSHA),
   137  				Type:              string(instanceResult[i].Type),
   138  				RemainingCapacity: int64(instanceResult[i].RemainingCapacity),
   139  				Status:            string(instanceResult[i].Status),
   140  				CreatedAt:         instanceResult[i].CreatedAt,
   141  				UpdatedAt:         instanceResult[i].UpdatedAt,
   142  			}
   143  
   144  			if assignedInstance.ClientSHA == mandelboxRequest.CommitHash {
   145  				logger.Infof("Found instance %v for user %v with commit hash %v.", assignedInstance.ID, mandelboxRequest.UserEmail, assignedInstance.ClientSHA)
   146  				instanceFound = true
   147  				break
   148  			}
   149  		}
   150  
   151  		// Break of outer loop if instance was found. If no instance with
   152  		// matching commit hash was found, move on to the next region.
   153  		if instanceFound {
   154  			break
   155  		}
   156  	}
   157  
   158  	// No instances with capacity were found
   159  	if assignedInstance == (subscriptions.Instance{}) {
   160  		err := utils.MakeError("did not find an instance with capacity for user %v and commit hash %v.", mandelboxRequest.UserEmail, mandelboxRequest.CommitHash)
   161  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   162  			Error: NO_INSTANCE_AVAILABLE,
   163  		}, err)
   164  		return err
   165  	}
   166  
   167  	var (
   168  		// The parsed version from the config
   169  		parsedFrontendVersion *hashicorp.Version
   170  		// The parsed version from the request
   171  		parsedRequestVersion *hashicorp.Version
   172  		// whether the client app has an outdated version
   173  		isOutdatedClient bool
   174  	)
   175  
   176  	// Get the version we keep locally for comparing the incoming request value.
   177  	frontendVersion := getFrontendVersion()
   178  
   179  	// Parse the version with the `hashicorp/go-version` package so we can compare.
   180  	parsedFrontendVersion, err = hashicorp.NewVersion(frontendVersion)
   181  	if err != nil {
   182  		logger.Errorf("Failed parsing client app version from scaling algorithm config. Err: %v", err)
   183  	}
   184  
   185  	// Parse the version we got in the request.
   186  	parsedRequestVersion, err = hashicorp.NewVersion(mandelboxRequest.Version)
   187  	if err != nil {
   188  		logger.Errorf("Failed parsing client app version from request. Err: %v", err)
   189  	}
   190  
   191  	// Compare the request version with the one from the config. If the
   192  	// version from the request is less than the one we have locally, it
   193  	// means the request comes from an outdated frontend application.
   194  	if parsedFrontendVersion != nil && parsedRequestVersion != nil {
   195  		logger.Infof("Local version is %s, version received from request is %s.", parsedFrontendVersion.String(), parsedRequestVersion.String())
   196  		isOutdatedClient = parsedRequestVersion.LessThan(parsedFrontendVersion)
   197  	}
   198  
   199  	// There are instances with capacity available, but none of them with the desired commit hash.
   200  	// We only consider this error in cases when the client app has a version greater or equal than
   201  	// the one in the config database. This is because when the client version is lesser (outdated client),
   202  	// it will automatically update itself to the most recent version and send another request.
   203  	if assignedInstance.ClientSHA != mandelboxRequest.CommitHash {
   204  		var msg error
   205  
   206  		// Only log the commit mismatch error when running on prod. This is because we only update the full version
   207  		// (major, minor, micro) on the config database when deploying to prod, so this is only a real error in that case.
   208  		if metadata.GetAppEnvironment() == metadata.EnvProd && !isOutdatedClient {
   209  			msg = utils.MakeError("found instance with capacity but it has a different commit hash %v than frontend with commit hash  %v.", assignedInstance.ClientSHA, mandelboxRequest.CommitHash)
   210  		} else {
   211  			logger.Infof("Did not find instance with commit hash %v, but expect frontend to autoupdate and send another request with commit hash %v.", mandelboxRequest.CommitHash, assignedInstance.ClientSHA)
   212  		}
   213  
   214  		// Regardless if we log the error, its necessary to return an appropiate response.
   215  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   216  			Error: COMMIT_HASH_MISMATCH,
   217  		}, msg)
   218  		return msg
   219  	}
   220  
   221  	// Try to find a mandelbox in the WAITING status in the assigned instance.
   222  	mandelboxResult, err := s.DBClient.QueryMandelbox(scalingCtx, s.GraphQLClient, assignedInstance.ID, "WAITING")
   223  	if err != nil {
   224  		return utils.MakeError("failed to query database for mandelbox on instance %s. Err: %v", assignedInstance.ID, err)
   225  	}
   226  
   227  	if len(mandelboxResult) == 0 {
   228  		return utils.MakeError("failed to find a waiting mandelbox even though the instance %v had sufficient capacity.", assignedInstance.ID)
   229  	}
   230  
   231  	waitingMandelbox := mandelboxResult[0]
   232  	mandelboxID, err := uuid.Parse(string(waitingMandelbox.ID))
   233  	if err != nil {
   234  		return utils.MakeError("failed to parse mandelbox id for instance %v. Err: %v", assignedInstance.ID, err)
   235  	}
   236  
   237  	mandelboxForDb := subscriptions.Mandelbox{
   238  		ID:         types.MandelboxID(mandelboxID),
   239  		App:        string(waitingMandelbox.App),
   240  		InstanceID: assignedInstance.ID,
   241  		UserID:     mandelboxRequest.UserID,
   242  		SessionID:  utils.Sprintf("%v", mandelboxRequest.SessionID),
   243  		Status:     "ALLOCATED",
   244  		CreatedAt:  waitingMandelbox.CreatedAt,
   245  		UpdatedAt:  time.Now(),
   246  	}
   247  
   248  	// Allocate mandelbox on database so the host service can start downloading user configs
   249  	affectedRows, err := s.DBClient.UpdateMandelbox(scalingCtx, s.GraphQLClient, mandelboxForDb)
   250  	if err != nil {
   251  		return utils.MakeError("error while inserting mandelbox to database. Err: %v", err)
   252  	}
   253  
   254  	logger.Infof("Inserted %v rows to database.", affectedRows)
   255  
   256  	if assignedInstance.RemainingCapacity <= 0 {
   257  		// This should never happen, but we should consider
   258  		// possible edge cases before updating the database.
   259  		return utils.MakeError("instance with id %v has a remaning capacity less than or equal to 0.", assignedInstance.ID)
   260  	}
   261  
   262  	// Subtract 1 from the current instance capacity because we allocated a mandelbox
   263  	updatedCapacity := assignedInstance.RemainingCapacity - 1
   264  	assignedInstance.RemainingCapacity = updatedCapacity
   265  
   266  	affectedRows, err = s.DBClient.UpdateInstance(scalingCtx, s.GraphQLClient, assignedInstance)
   267  	if err != nil {
   268  		return utils.MakeError("error while updating instance capacity on database. Err: %v", err)
   269  	}
   270  
   271  	logger.Infof("Updated %v rows in database.", affectedRows)
   272  
   273  	// Parse IP address. The database uses the CIDR notation (192.0.2.0/24)
   274  	// so we need to extract the address and send it to the frontend.
   275  	ip, _, err := net.ParseCIDR(assignedInstance.IPAddress)
   276  	if err != nil {
   277  		return utils.MakeError("failed to parse IP address %v. Err: %v", assignedInstance.IPAddress, err)
   278  	}
   279  
   280  	// Return result to assign request
   281  	mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   282  		IP:          ip.String(),
   283  		MandelboxID: types.MandelboxID(mandelboxID),
   284  	}, nil)
   285  
   286  	return nil
