   128  // loadUserConfigs does the "real work" of LoadUserConfigs. For each
   129  // deviation from the happy path in this function, we need to decide whether it
   130  // is a warning or an error — warnings get logged here, and errors get sent

   132  func (mandelbox *mandelboxData) loadUserConfigs(tokenChan <-chan ConfigEncryptionInfo, errorChan chan<- error) {
   133  	// We don't want to close errorChan (which indicates to the consumer that
   134  	// this function is "done") too early, but it's the last thing we want to do
   135  	// from this function. No matter what, we want to verify the token/encryption
   136  	// info that was passed in). Therefore, we block until the token verification
   137  	// goroutine is done before closing errorChan, and we defer it right here.
   138  	loadingWaitgroup := &sync.WaitGroup{}
   139  	defer func() {
   140  		logger.Infof("loadUserConfigs(): Waiting for loadingWaitgroup to be done...")
   141  		loadingWaitgroup.Wait()
   142  		close(errorChan)
   143  	}()
   144  
   145  	// If the process fails anywhere, we want to at least set up the empty config
   146  	// directories for the mandelbox. We do this by setting `loadFailed` to true
   147  	// until all the steps are done, and then creating the empty config
   148  	// directories on function completion if `loadFailed` is still set to true.
   149  	var loadFailed bool = true
   150  	defer func() {
   151  		if loadFailed {
   152  			err := mandelbox.setupUserConfigDirs()
   153  			if err != nil {
   154  				errorChan <- utils.MakeError("Failed to last-resort setup user config dirs for mandelbox %s: %s", mandelbox.GetID(), err)
   155  				return
   156  			} else {
   157  				logger.Infof("Set up last-resort user config dirs for mandelbox %s because normal loading failed.", mandelbox.GetID())
   158  			}
   159  		}
   160  	}()
   161  
   162  	// We also want to receive and set the config encryption token, even if the
   163  	// process fails before where we would normally block waiting on this info.
   164  	// Therefore, we begin blocking for it immediately and simply pass the
   165  	// message along (after some verification) for the body of loadUserConfigs()
   166  	// to wait on it later.
   167  	// Note that `verifiedTokenChan` needs to be buffered in case the outer function returns before reading from it.
   168  	verifiedTokenChan := make(chan ConfigEncryptionInfo, 1)
   169  	loadingWaitgroup.Add(1)
   170  	go func() {
   171  		defer loadingWaitgroup.Done()
   172  		defer close(verifiedTokenChan)
   173  		encryptionInfo, err := mandelbox.receiveAndSanityCheckEncryptionToken(tokenChan)
   174  		if err != nil {
   175  			errorChan <- err
   176  			// Here we only return from the inner function, but since we close
   177  			// `verifiedTokenChan` the entire outer function will not block.
   178  			return
   179  		}
   180  		verifiedTokenChan <- *encryptionInfo
   181  	}()
   182  
   183  	// If userID is not set, then we don't retrieve configs from s3
   184  	if len(mandelbox.GetUserID()) == 0 {
   185  		logger.Warningf("User ID is not set for mandelbox %s. Skipping config download.", mandelbox.GetID())
   186  		return
   187  	}
   188  
   189  	s3Client, err := configutils.NewS3Client("us-east-1")
   190  	if err != nil {
   191  		errorChan <- utils.MakeError("Could not make new S3 client: %s", err)
   192  		return
   193  	}
   194  
   195  	// At this point, we are still waiting on the client-app to get the user's
   196  	// config encryption token. Instead of waiting, we want to speculatively
   197  	// download a config that we think is the most likely to be the one we need.
   198  	predictedConfigKey, err := mandelbox.predictConfigToDownload(s3Client)
   199  	if err != nil {
   200  		errorChan <- utils.MakeError("Error retrieving existing configs for user %s from S3 for mandelbox %s: %s", mandelbox.GetUserID(), mandelbox.GetID(), err)
   201  		return
   202  	}
   203  	if predictedConfigKey == "" {
   204  		logger.Warningf("There are no existing configs in s3 for user %s for mandelbox %s", mandelbox.GetUserID(), mandelbox.GetID())
   205  		return
   206  	}
   207  
   208  	// Now that we have a prediction for the config file we need, we want to
   209  	// download it into a buffer for decrypting.
   210  	// TODO: Don't wait on this unless we know that the config is indeed the one
   211  	// we need. This is not a big deal now, but will be more important as configs
   212  	// grow in size, if we see users alternating between encryption tokens
   213  	// frequently (which should never happen with the current setup).
   214  	// (https://github.com/whisthq/whist/issues/5287)
   215  	predictedConfigHeadObj, err := configutils.GetHeadObject(s3Client, configutils.GetConfigBucket(), predictedConfigKey)
   216  	if err != nil {
   217  		errorChan <- utils.MakeError("Could not get head object for predicted key %s for user %s for mandelbox %s: %s", predictedConfigKey, mandelbox.GetUserID(), mandelbox.GetID(), err)
   218  		return
   219  	}
   220  	predictedConfigBuf, err := mandelbox.downloadUserConfig(s3Client, predictedConfigKey, predictedConfigHeadObj)
   221  	if err != nil {
   222  		errorChan <- utils.MakeError("Error downloading predicted config %s for user %s and mandelbox %s: %s", predictedConfigKey, mandelbox.GetUserID(), mandelbox.GetID(), err)
   223  		return
   224  	}
   225  
   226  	// What we do want to block on is receiving the configEncryptionToken.
   227  	encryptionInfo, ok := <-verifiedTokenChan
   228  	if !ok {
   229  		// Error already put in errorChan by the inner function above.
   230  		return
   231  	}
   232  
   233  	// Now we compare the config that we speculatively downloaded to the one we
   234  	// know we need based on the token that came back from the client-app.
   235  	correctConfigKey, correctHeadObject, err := mandelbox.determineCorrectConfigKey(s3Client, predictedConfigKey, predictedConfigHeadObj, encryptionInfo)
   236  	if err != nil {
   237  		errorChan <- utils.MakeError("Could not determine correct config key for user %s for mandelbox %s: %s", mandelbox.GetUserID(), mandelbox.GetID(), err)
   238  		return
   239  	}
   240  
   241  	// Check for the frontend's indication that this is a fresh config before
   242  	// we do any more expensive operations.
   243  	if encryptionInfo.IsNewTokenAccordingToClientApp {
   244  		logger.Warningf("Client-app says the hashed config encryption token %s for user %s and mandelbox %s is new. Therefore, we will not try to decrypt any configs.", hash(encryptionInfo.Token), mandelbox.GetUserID(), mandelbox.GetID())
   245  		return
   246  	}
   247  
   248  	// Do some basic sanity checks
   249  	if correctConfigKey != mandelbox.getS3ConfigKey(hash(encryptionInfo.Token)) && correctConfigKey != path.Join(mandelbox.getS3ConfigKeyPrefix(hash(encryptionInfo.Token)), EncryptedArchiveFilename) {
   250  		// If not one of those file paths, then we're about to try to decrypt the wrong file.
   251  		errorChan <- utils.MakeError("Corrected config key %s for user %s for mandelbox %s does not match any expected format (prefix/filename or prefix/token/filename)!", correctConfigKey, mandelbox.GetUserID(), mandelbox.GetID())
   252  		return
   253  	}
   254  
   255  	// If we need to, download the correct config archive from S3
   256  	var correctConfigBuf []byte
   257  	if correctConfigKey != predictedConfigKey {
   258  		correctConfigBuf, err = mandelbox.downloadUserConfig(s3Client, correctConfigKey, correctHeadObject)
   259  		if err != nil {
   260  			errorChan <- utils.MakeError("Error downloading corrected config %s for user %s and mandelbox %s: %s", predictedConfigKey, mandelbox.GetUserID(), mandelbox.GetID(), err)
   261  			return
   262  		}
   263  	} else {
   264  		correctConfigBuf = predictedConfigBuf
   265  	}
   266  
   267  	// Decrypt the downloaded config, then unpack it
   268  	decryptedConfig, err := mandelbox.decryptConfig(correctConfigKey, correctConfigBuf, encryptionInfo)
   269  	if err != nil {
   270  		errorChan <- err
   271  		return
   272  	}
   273  	err = mandelbox.extractConfig(correctConfigKey, decryptedConfig)
   274  	if err != nil {
   275  		errorChan <- err
   276  		return
   277  	}
   278  
   279  	loadFailed = false
   280  }

   303  // downloadUserConfig downloads a given user config from S3 into an in-memory
   304  // buffer. It takes in key and the HeadObjectOutput for the file to be
   305  // downloaded (for size and version info) and returns the byte slice containing
   306  // the object and any errors.
   307  func (mandelbox *mandelboxData) downloadUserConfig(s3Client *s3.Client, key string, headObject *s3.HeadObjectOutput) ([]byte, error) {
   308  	// Log config version
   309  	logger.Infof("Attempting to download user config key %s (version %s) for mandelbox %s", key, *headObject.VersionId, mandelbox.GetID())
   310  
   311  	// Get the md5 hash of the config file
   312  	validateIntegrity := true
   313  	configMetadata := headObject.Metadata
   314  	expectedMD5, ok := configMetadata["md5"]
   315  	if !ok {
   316  		validateIntegrity = false
   317  	}
   318  
   319  	var downloadedFile []byte
   320  	var downloadSuccessful bool
   321  	var numBytesSuccessfullyDownloaded int64
   322  
   323  	// Allow up to 3 retries on download if failure is because of checksum mismatch, this is
   324  	// intended to solve issues with configs sometimes being malformed due to network issues
   325  	for i := 0; i < 3; i++ {
   326  		// Download file into a pre-allocated in-memory buffer
   327  		// This should be okay as we don't expect configs to be very large
   328  		buf := manager.NewWriteAtBuffer(make([]byte, headObject.ContentLength))
   329  		numBytes, err := configutils.DownloadObjectToBuffer(s3Client, configutils.GetConfigBucket(), key, buf)
   330  		if err != nil {
   331  			return nil, utils.MakeError("Could not download object for key %s (version %s) for mandelbox %s: %s", key, *headObject.VersionId, mandelbox.GetID(), err)
   332  		}
   333  
   334  		// Check if the file was downloaded correctly
   335  		downloadedFile = buf.Bytes()
   336  		downloadHash := configutils.GetMD5Hash(downloadedFile)
   337  		if validateIntegrity && downloadHash != expectedMD5 {
   338  			logger.Warningf("MD5 hash mismatch for user config key %s (version %s) for mandelbox %s. Expected %s, got %s", key, *headObject.VersionId, mandelbox.GetID(), expectedMD5, downloadHash)
   339  			continue
   340  		}
   341  
   342  		downloadSuccessful = true
   343  		numBytesSuccessfullyDownloaded = numBytes
   344  		break
   345  	}
   346  
   347  	if !downloadSuccessful {
   348  		return nil, utils.MakeError("Could not download object (due to MD5 mismatch) for key %s (version %s) for mandelbox %s after 3 attempts", key, *headObject.VersionId, mandelbox.GetID())
   349  	}
   350  
   351  	logger.Infof("Downloaded %v bytes for user config key %s (version %s) for mandelbox %s", numBytesSuccessfullyDownloaded, key, *headObject.VersionId, mandelbox.GetID())
   352  
   353  	return downloadedFile, nil
   354  }

   356  // receiveAndSanityCheckEncryptionToken blocks until it receives the encryption
   357  // token info from `uncheckedTokenChan` (or at least until the channel is
   358  // closed first, which it treats as an error). It also performs some basic
   359  // sanity checks on the token (e.g to make sure it's not empty), and if they
   360  // pass, sets the receiving mandelbox's config token to the provided value.
   361  func (mandelbox *mandelboxData) receiveAndSanityCheckEncryptionToken(uncheckedTokenChan <-chan ConfigEncryptionInfo) (*ConfigEncryptionInfo, error) {
   362  	encryptionInfo, gotEncryptionInfo := <-uncheckedTokenChan
   363  	if !gotEncryptionInfo {
   364  		return nil, utils.MakeError("Got no config encryption token for user %s for mandelbox %s, likely because the JSON Transport request never completed.", mandelbox.GetUserID(), mandelbox.GetID())
   365  	}
   366  
   367  	if encryptionInfo.Token == "" {
   368  		return &encryptionInfo, utils.MakeError("Encryption token for user %s for mandelbox %s is empty!", mandelbox.GetUserID(), mandelbox.GetID())
   369  	}
   370  
   371  	if len(encryptionInfo.Token) < 15 {
   372  		return &encryptionInfo, utils.MakeError("Encryption token for user %s for mandelbox %s is way too short (length %v)", mandelbox.GetUserID(), mandelbox.GetID(), len(encryptionInfo.Token))
   373  	}
   374  
   375  	// The encryption token _looks_ reasonable. We set it for the mandelbox and log some info.
   376  	mandelbox.SetConfigEncryptionToken(encryptionInfo.Token)
   377  
   378  	logger.Infof("Got config encryption info that passed sanity checks for user %s and mandelbox %s: %s", mandelbox.GetUserID(), mandelbox.GetID(), encryptionInfo)
   379  
   380  	return &encryptionInfo, nil
   381  }

   383  // determineCorrectConfigKey confirms or corrects our predictions about which
   384  // S3 object to download for configs, given the ConfigEncryptionInfo from the
   385  // client-app.
   386  func (mandelbox *mandelboxData) determineCorrectConfigKey(s3client *s3.Client, predictedKey string, predictedHeadObject *s3.HeadObjectOutput, encryptionInfo ConfigEncryptionInfo) (string, *s3.HeadObjectOutput, error) {
   387  	tokenHash := hash(encryptionInfo.Token)
   388  
   389  	// (happy path) If the predicted config's key contains the hash of the token,
   390  	// then we know we're good to go.
   391  	if strings.Contains(predictedKey, tokenHash) {
   392  		logger.Infof("Predicted key %s to download configs for user %s for mandelbox %s is correct.", predictedKey, mandelbox.GetUserID(), mandelbox.GetID())
   393  		return predictedKey, predictedHeadObject, nil
   394  	}
   395  
   396  	// If there is a config in s3 that contains the token hash, that's the one we desire.
   397  	desiredKey := mandelbox.getS3ConfigKey(tokenHash)
   398  	desiredHead, err := configutils.GetHeadObject(s3client, configutils.GetConfigBucket(), desiredKey)
   399  	if err == nil {
   400  		logger.Warningf("Predicted key %s to download configs for user %s for mandelbox %s were replaced with corrected key %s", predictedKey, mandelbox.GetUserID(), mandelbox.GetID(), desiredKey)
   401  		return desiredKey, desiredHead, nil
   402  	} else if err != nil {
   403  		var apiErr smithy.APIError
   404  		if !(errors.As(err, &apiErr) && apiErr.ErrorCode() == "NotFound") {
   405  			// The desired file exists in S3, but we ran into some error getting the
   406  			// HeadObject. Something went poorly.
   407  			return predictedKey, predictedHeadObject, utils.MakeError("Couldn't get HeadObject for corrected config key %s (from predicted key %s) for user %s for mandelbox %s: %s", desiredKey, predictedKey, mandelbox.GetUserID(), mandelbox.GetID(), err)
   408  		}
   409  	}
   410  
   411  	// There's no upstream key matching desiredKey (which is what we would have
   412  	// expected if we'd have uploaded configs labeled with this token before).
   413  	// Therefore, we just have to roll with our predictions.
   414  	logger.Warningf("There is no key in S3 matching our desired config key %s for user %s for mandelbox %s. Continuing with predicted key %s for user configs.", desiredKey, mandelbox.GetUserID(), mandelbox.GetID(), predictedKey)
   415  	return predictedKey, predictedHeadObject, nil
   416  }

   418  // decryptConfig decrypts the given buffer. This function assumes that
   419  // `receiveAndVerifyEncryptionInfo` has already been called.
   420  func (mandelbox *mandelboxData) decryptConfig(configKey string, encryptedBuf []byte, encryptionInfo ConfigEncryptionInfo) ([]byte, error) {
   421  	if len(encryptionInfo.Token) == 0 {
   422  		return nil, utils.MakeError("Cannot get user configs for user %s for Mandelbox %s since config encryption token is empty", mandelbox.GetUserID(), mandelbox.GetID())
   423  	}
   424  
   425  	if len(encryptedBuf) == 0 {
   426  		return nil, utils.MakeError("Cannot decrypt length-0 buffer for user configs for user %s for Mandelbox %s since config buffer has length 0!", mandelbox.GetUserID(), mandelbox.GetID())
   427  	}
   428  
   429  	logger.Infof("Decrypting config %s for user %s for mandelbox %s with hashed token %s", configKey, mandelbox.GetUserID(), mandelbox.GetID(), hash(encryptionInfo.Token))
   430  
   431  	decryptedBuf, err := configutils.DecryptAES256GCM(string(encryptionInfo.Token), encryptedBuf)
   432  	if err != nil {
   433  		return nil, utils.MakeError("Failed to decrypt config %s for user %s for mandelbox %s with hashed token %s: %s", configKey, mandelbox.GetUserID(), mandelbox.GetID(), hash(encryptionInfo.Token), err)
   434  	}
   435  
   436  	logger.Infof("Successfully decrypted %s for user %s for mandelbox %s with hashed token %s", configKey, mandelbox.GetUserID(), mandelbox.GetID(), hash(encryptionInfo.Token))
   437  
   438  	return decryptedBuf, nil
   439  }

   441  // extractConfig extracts a decrypted config to the relevant directory.
   442  func (mandelbox *mandelboxData) extractConfig(configKey string, decryptedConfig []byte) error {
   443  	logger.Infof("Extracting config %s for user %s for mandelbox %s", configKey, mandelbox.GetUserID(), mandelbox.GetID())
   444  
   445  	// Make directory for user configs
   446  	configDir := mandelbox.GetUserConfigDir()
   447  	unpackedConfigDir := path.Join(configDir, UnpackedConfigsDirectoryName)
   448  	err := mandelbox.setupUserConfigDirs()
   449  	if err != nil {
   450  		return utils.MakeError("Failed to setup user config directories for user %s for mandelbox %s: %s", mandelbox.GetUserID(), mandelbox.GetID(), err)
   451  	}
   452  
   453  	// Once we've extracted everything, we open up permissions for the user
   454  	// config directory so it's accessible by the non-root user in the
   455  	// mandelboxes. We also are okay with setting executable bits here, since
   456  	// it's possible that some apps store executable scripts in their config.
   457  	defer func() {
   458  		cmd := exec.Command("chown", "-R", "ubuntu", configDir)
   459  		cmd.Run()
   460  		cmd = exec.Command("chmod", "-R", "777", configDir)
   461  		cmd.Run()
   462  	}()
   463  
   464  	totalFileSize, err := configutils.ExtractTarLz4(decryptedConfig, unpackedConfigDir)
   465  	if err != nil {
   466  		logger.Errorf("Error extracting tar lz4 file for user %s for mandelbox %s: %s", mandelbox.GetUserID(), mandelbox.GetID(), err)
   467  	}
   468  
   469  	logger.Infof("Untarred config to: %s, total size was %d bytes", unpackedConfigDir, totalFileSize)
   470  
   471  	return nil
   472  }

   478  	configDir := mandelbox.GetUserConfigDir()
