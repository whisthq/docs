   130  // loadUserConfigs does the "real work" of LoadUserConfigs. For each
   131  // deviation from the happy path in this function, we need to decide whether it
   132  // is a warning or an error — warnings get logged here, and errors get sent
   133  // back.
   134  func (mandelbox *mandelboxData) loadUserConfigs(tokenChan <-chan ConfigEncryptionInfo, errorChan chan<- error) {
   135  	// We don't want to close errorChan (which indicates to the consumer that
   136  	// this function is "done") too early, but it's the last thing we want to do
   137  	// from this function. No matter what, we want to verify the token/encryption
   138  	// info that was passed in). Therefore, we block until the token verification
   139  	// goroutine is done before closing errorChan, and we defer it right here.
   140  	loadingWaitgroup := &sync.WaitGroup{}
   141  	defer func() {
   142  		logger.Infof("loadUserConfigs(): Waiting for loadingWaitgroup to be done...")
   143  		loadingWaitgroup.Wait()
   144  		close(errorChan)
   145  	}()
   146  
   147  	// If the process fails anywhere, we want to at least set up the empty config
   148  	// directories for the mandelbox. We do this by setting `loadFailed` to true
   149  	// until all the steps are done, and then creating the empty config
   150  	// directories on function completion if `loadFailed` is still set to true.
   151  	var loadFailed bool = true
   152  	defer func() {
   153  		if loadFailed {
   154  			err := mandelbox.setupUserConfigDirs()
   155  			if err != nil {
   156  				errorChan <- utils.MakeError("Failed to last-resort setup user config dirs for mandelbox %s: %s", mandelbox.GetID(), err)
   157  				return
   158  			} else {
   159  				logger.Infof("Set up last-resort user config dirs for mandelbox %s because normal loading failed.", mandelbox.GetID())
   160  			}
   161  		}
   162  	}()
   163  
   164  	// We also want to receive and set the config encryption token, even if the
   165  	// process fails before where we would normally block waiting on this info.
   166  	// Therefore, we begin blocking for it immediately and simply pass the
   167  	// message along (after some verification) for the body of loadUserConfigs()
   168  	// to wait on it later.
   169  	// Note that `verifiedTokenChan` needs to be buffered in case the outer function returns before reading from it.
   170  	verifiedTokenChan := make(chan ConfigEncryptionInfo, 1)
   171  	loadingWaitgroup.Add(1)
   172  	go func() {
   173  		defer loadingWaitgroup.Done()
   174  		defer close(verifiedTokenChan)
   175  		encryptionInfo, err := mandelbox.receiveAndSanityCheckEncryptionToken(tokenChan)
   176  		if err != nil {
   177  			errorChan <- err
   178  			// Here we only return from the inner function, but since we close
   179  			// `verifiedTokenChan` the entire outer function will not block.
   180  			return
   181  		}
   182  		verifiedTokenChan <- *encryptionInfo
   183  	}()
   184  
   185  	// If userID is not set, then we don't retrieve configs from s3
   186  	if len(mandelbox.GetUserID()) == 0 {
   187  		logger.Warningf("User ID is not set for mandelbox %s. Skipping config download.", mandelbox.GetID())
   188  		return
   189  	}
   190  
   191  	s3Client, err := configutils.NewS3Client("us-east-1")
   192  	if err != nil {
   193  		errorChan <- utils.MakeError("Could not make new S3 client: %s", err)
   194  		return
   195  	}
   196  
   197  	// At this point, we are still waiting on the client-app to get the user's
   198  	// config encryption token. Instead of waiting, we want to speculatively
   199  	// download a config that we think is the most likely to be the one we need.
   200  	predictedConfigObj, err := mandelbox.predictConfigToDownload(s3Client)
   201  	if err != nil {
   202  		errorChan <- utils.MakeError("Error retrieving existing configs for user %s from S3 for mandelbox %s: %s", mandelbox.GetUserID(), mandelbox.GetID(), err)
   203  		return
   204  	}
   205  	if predictedConfigObj == nil {
   206  		logger.Warningf("There are no existing configs in s3 for user %s for mandelbox %s", mandelbox.GetUserID(), mandelbox.GetID())
   207  		return
   208  	}
   209  
   210  	// Now that we have a prediction for the config file we need, we want to
   211  	// download it into a buffer for decrypting.
   212  	// TODO: Don't wait on this unless we know that the config is indeed the one
   213  	// we need. This is not a big deal now, but will be more important as configs
   214  	// grow in size, if we see users alternating between encryption tokens
   215  	// frequently (which should never happen with the current setup).
   216  	// (https://github.com/whisthq/whist/issues/5287)
   217  	predictedConfigHeadObj, err := configutils.GetHeadObject(s3Client, UserConfigS3Bucket, *predictedConfigObj.Key)
   218  	if err != nil {
   219  		errorChan <- utils.MakeError("Could not get head object for predicted key %s for user %s for mandelbox %s: %s", *predictedConfigObj.Key, mandelbox.GetUserID(), mandelbox.GetID(), err)
   220  		return
   221  	}
   222  	predictedConfigBuf, err := mandelbox.downloadUserConfig(s3Client, *predictedConfigObj.Key, predictedConfigHeadObj)
   223  	if err != nil {
   224  		errorChan <- utils.MakeError("Error downloading predicted config %s for user %s and mandelbox %s: %s", *predictedConfigObj.Key, mandelbox.GetUserID(), mandelbox.GetID(), err)
   225  		return
   226  	}
   227  
   228  	// What we do want to block on is receiving the configEncryptionToken.
   229  	encryptionInfo, ok := <-verifiedTokenChan
   230  	if !ok {
   231  		// Error already put in errorChan by the inner function above.
   232  		return
   233  	}
   234  
   235  	// Now we compare the config that we speculatively downloaded to the one we
   236  	// know we need based on the token that came back from the client-app.
   237  	correctConfigKey, correctHeadObject, err := mandelbox.determineCorrectConfigKey(s3Client, *predictedConfigObj.Key, predictedConfigHeadObj, encryptionInfo)
   238  	if err != nil {
   239  		errorChan <- utils.MakeError("Could not determine correct config key for user %s for mandelbox %s: %s", mandelbox.GetUserID(), mandelbox.GetID(), err)
   240  		return
   241  	}
   242  
   243  	// Check for the client app's indication that this is a fresh config before
   244  	// we do any more expensive operations.
   245  	if encryptionInfo.IsNewTokenAccordingToClientApp {
   246  		logger.Warningf("Client-app says the hashed config encryption token %s for user %s and mandelbox %s is new. Therefore, we will not try to decrypt any configs.", hash(encryptionInfo.Token), mandelbox.GetUserID(), mandelbox.GetID())
   247  		return
   248  	}
   249  
   250  	// Do some basic sanity checks.
   251  	if correctConfigKey != mandelbox.getS3ConfigKey(hash(encryptionInfo.Token)) && correctConfigKey != path.Join(mandelbox.getS3ConfigKeyPrefix(), EncryptedArchiveFilename) {
   252  		// If not one of those file paths, then we're about to try to decrypt the wrong file.
   253  		errorChan <- utils.MakeError("Corrected config key %s for user %s for mandelbox %s does not match any expected format (prefix/filename or prefix/token/filename)!", correctConfigKey, mandelbox.GetUserID(), mandelbox.GetID())
   254  		return
   255  	}
   256  
   257  	// If we need to, download the correct config archive from S3.
   258  	var correctConfigBuf []byte
   259  	if correctConfigKey != *predictedConfigObj.Key {
   260  		correctConfigBuf, err = mandelbox.downloadUserConfig(s3Client, correctConfigKey, correctHeadObject)
   261  		if err != nil {
   262  			errorChan <- utils.MakeError("Error downloading corrected config %s for user %s and mandelbox %s: %s", *predictedConfigObj.Key, mandelbox.GetUserID(), mandelbox.GetID(), err)
   263  			return
   264  		}
   265  	} else {
   266  		correctConfigBuf = predictedConfigBuf
   267  	}
   268  
   269  	// Decrypt the downloaded config, then unpack it
   270  	decryptedConfig, err := mandelbox.decryptConfig(correctConfigKey, correctConfigBuf, encryptionInfo)
   271  	if err != nil {
   272  		errorChan <- err
   273  		return
   274  	}
   275  	err = mandelbox.extractConfig(correctConfigKey, decryptedConfig)
   276  	if err != nil {
   277  		errorChan <- err
   278  		return
   279  	}
   280  
   281  	loadFailed = false
   282  }

   326  // receiveAndSanityCheckEncryptionToken blocks until it receives the encryption
   327  // token info from `uncheckedTokenChan` (or at least until the channel is
   328  // closed first, which it treats as an error). It also performs some basic
   329  // sanity checks on the token (e.g to make sure it's not empty), and if they
   330  // pass, sets the receiving mandelbox's config token to the provided value.
   331  func (mandelbox *mandelboxData) receiveAndSanityCheckEncryptionToken(uncheckedTokenChan <-chan ConfigEncryptionInfo) (*ConfigEncryptionInfo, error) {
   332  	encryptionInfo, gotEncryptionInfo := <-uncheckedTokenChan
   333  	if !gotEncryptionInfo {
   334  		return nil, utils.MakeError("Got no config encryption token for user %s for mandelbox %s, likely because the JSON Transport request never completed.", mandelbox.GetUserID(), mandelbox.GetID())
   335  	}
   336  
   337  	if encryptionInfo.Token == "" {
   338  		return &encryptionInfo, utils.MakeError("Encryption token for user %s for mandelbox %s is empty!", mandelbox.GetUserID(), mandelbox.GetID())
   339  	}
   340  
   341  	if len(encryptionInfo.Token) < 15 {
   342  		return &encryptionInfo, utils.MakeError("Encryption token for user %s for mandelbox %s is way too short (length %v)", mandelbox.GetUserID(), mandelbox.GetID(), len(encryptionInfo.Token))
   343  	}
   344  
   345  	// The encryption token _looks_ reasonable. We set it for the mandelbox and log some info.
   346  	mandelbox.SetConfigEncryptionToken(encryptionInfo.Token)
   347  
   348  	logger.Infof("Got config encryption info that passed sanity checks for user %s and mandelbox %s: %s", mandelbox.GetUserID(), mandelbox.GetID(), encryptionInfo)
   349  
   350  	return &encryptionInfo, nil

   353  // determineCorrectConfigKey confirms or corrects our predictions about which
   354  // S3 object to download for configs, given the ConfigEncryptionInfo from the

   356  func (mandelbox *mandelboxData) determineCorrectConfigKey(s3client *s3.Client, predictedKey string, predictedHeadObject *s3.HeadObjectOutput, encryptionInfo ConfigEncryptionInfo) (string, *s3.HeadObjectOutput, error) {
   357  	tokenHash := hash(encryptionInfo.Token)
   358  
   359  	// (happy path) If the predicted config's key contains the hash of the token,
   360  	// then we know we're good to go.
   361  	if strings.Contains(predictedKey, tokenHash) {
   362  		logger.Infof("Predicted key %s to download configs for user %s for mandelbox %s is correct.", predictedKey, mandelbox.GetUserID(), mandelbox.GetID())
   363  		return predictedKey, predictedHeadObject, nil
   364  	}
   365  
   366  	// If there is a config in s3 that contains the token hash, that's the one we desire.
   367  	desiredKey := mandelbox.getS3ConfigKey(tokenHash)
   368  	desiredHead, err := configutils.GetHeadObject(s3client, UserConfigS3Bucket, desiredKey)
   369  	if err == nil {
   370  		logger.Warningf("Predicted key %s to download configs for user %s for mandelbox %s were replaced with corrected key %s", predictedKey, mandelbox.GetUserID(), mandelbox.GetID(), desiredKey)
   371  		return desiredKey, desiredHead, nil
   372  	} else if err != nil {
   373  		var apiErr smithy.APIError
   374  		if !(errors.As(err, &apiErr) && apiErr.ErrorCode() == "NotFound") {
   375  			// The desired file exists in S3, but we ran into some error getting the
   376  			// HeadObject. Something went poorly.
   377  			return predictedKey, predictedHeadObject, utils.MakeError("Couldn't get HeadObject for corrected config key %s (from predicted key %s) for user %s for mandelbox %s: %s", desiredKey, predictedKey, mandelbox.GetUserID(), mandelbox.GetID(), err)
   378  		}
   379  	}
   380  
   381  	// There's no upstream key matching desiredKey (which is what we would have
   382  	// expected if we'd have uploaded configs labeled with this token before).
   383  	// Therefore, we just have to roll with our predictions.
   384  	logger.Warningf("There is no key in S3 matching our desired config key %s for user %s for mandelbox %s. Continuing with predicted key %s for user configs.", desiredKey, mandelbox.GetUserID(), mandelbox.GetID(), predictedKey)
   385  	return predictedKey, predictedHeadObject, nil
   386  }

   421  	}
   422  
   423  	// Once we've extracted everything, we open up permissions for the user
   424  	// config directory so it's accessible by the non-root user in the
   425  	// mandelboxes. We also are okay with setting executable bits here, since
   426  	// it's possible that some apps store executable scripts in their config.
   427  	defer func() {
   428  		cmd := exec.Command("chown", "-R", "ubuntu", configDir)
   429  		cmd.Run()
   430  		cmd = exec.Command("chmod", "-R", "777", configDir)
   431  		cmd.Run()
   432  	}()
   433  
   434  	totalFileSize, err := configutils.ExtractTarLz4(decryptedConfig, unpackedConfigDir)
   435  	if err != nil {
   436  		logger.Errorf("Error extracting tar lz4 file for user %s for mandelbox %s: %s", mandelbox.GetUserID(), mandelbox.GetID(), err)
   437  	}
   438  
   439  	logger.Infof("Untarred config to: %s, total size was %d bytes", unpackedConfigDir, totalFileSize)
   440  
   441  	return nil
   442  }
