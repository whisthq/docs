    58  // VerifyCapacity is a scaling action which checks the database to verify if we have the desired
    59  // capacity (instance buffer). This action is run at the end of the other actions.
    60  func (s *DefaultScalingAlgorithm) VerifyCapacity(scalingCtx context.Context, event ScalingEvent) error {
    61  	contextFields := []interface{}{
    62  		zap.String("id", event.ID),
    63  		zap.Any("type", event.Type),
    64  		zap.String("region", event.Region),
    65  	}
    66  	logger.Infow("Starting verify capacity action.", contextFields)
    67  	defer logger.Infow("Finished verify capacity action.", contextFields)
    68  
    69  	// Query for the latest image id
    70  	imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region) // TODO: set different provider when doing multi-cloud.
    71  	if err != nil {
    72  		return utils.MakeError("failed to query database for current image: %s", err)
    73  	}
    74  
    75  	if len(imageResult) == 0 {
    76  		logger.Warningf("Image not found in %s. Not performing any scaling actions.", event.Region)
    77  		return nil
    78  	}
    79  	latestImage := subscriptions.Image{
    80  		Provider:  string(imageResult[0].Provider),
    81  		Region:    string(imageResult[0].Region),
    82  		ImageID:   string(imageResult[0].ImageID),
    83  		ClientSHA: string(imageResult[0].ClientSHA),
    84  		UpdatedAt: imageResult[0].UpdatedAt,
    85  	}
    86  
    87  	// This query will return all instances with the ACTIVE status
    88  	allActive, err := s.DBClient.QueryInstancesByStatusOnRegion(scalingCtx, s.GraphQLClient, "ACTIVE", event.Region)
    89  	if err != nil {
    90  		return utils.MakeError("failed to query database for active instances: %s", err)
    91  	}
    92  
    93  	// This query will return all instances with the PRE_CONNECTION status
    94  	allStarting, err := s.DBClient.QueryInstancesByStatusOnRegion(scalingCtx, s.GraphQLClient, "PRE_CONNECTION", event.Region)
    95  	if err != nil {
    96  		return utils.MakeError("failed to query database for starting instances: %s", err)
    97  	}
    98  
    99  	mandelboxCapacity := helpers.ComputeExpectedMandelboxCapacity(string(latestImage.ImageID), allActive, allStarting)
   100  
   101  	// We consider the expected mandelbox capacity (active instances + starting instances)
   102  	// to account for warmup time and so that we don't scale up unnecesary instances.
   103  	if mandelboxCapacity < int64(desiredFreeMandelboxesPerRegion[event.Region]) {
   104  		logger.Infow(utils.Sprintf("Current mandelbox capacity of %d is less than desired %d. Scaling up %d instances to satisfy minimum desired capacity.",
   105  			mandelboxCapacity, desiredFreeMandelboxesPerRegion[event.Region], defaultInstanceBuffer), contextFields)
   106  		err = s.ScaleUpIfNecessary(defaultInstanceBuffer, scalingCtx, event, latestImage)
   107  		if err != nil {
   108  			// err is already wrapped here
   109  			return err
   110  		}
   111  	} else {
   112  		logger.Infow(utils.Sprintf("Mandelbox capacity %d in %s is enough to satisfy minimum desired capacity of %d.",
   113  			mandelboxCapacity, event.Region, desiredFreeMandelboxesPerRegion[event.Region]), contextFields)
   114  	}
   115  
   116  	return nil

   119  // VerifyInstanceRemoval is a function to verify that an instance terminates correctly on the cloud provider
   120  // and to make sure it is removed from the database correctly. This keeps the database and cloud provider in
   121  // sync.
   122  func (s *DefaultScalingAlgorithm) VerifyInstanceRemoval(scalingCtx context.Context, instance subscriptions.Instance) error {
   123  	// Wait until the host service terminates the instance in the cloud provider.
   124  	err := s.Host.WaitForInstanceTermination(scalingCtx, maxWaitTimeTerminated, []string{instance.ID})
   125  	if err != nil {
   126  		return err
   127  	}
   128  
   129  	// Once its terminated, verify that it was removed from the database
   130  	instanceResult, err := s.DBClient.QueryInstance(scalingCtx, s.GraphQLClient, instance.ID)
   131  	if err != nil {
   132  		return utils.MakeError("failed to query database for instance %s: %s", instance.ID, err)
   133  	}
   134  
   135  	// Verify that instance removed itself from the database
   136  	if len(instanceResult) == 0 {
   137  		logger.Infof("Instance %s was successfully removed from database.", instance.ID)
   138  		return nil
   139  	}
   140  
   141  	// If instance still exists on the database, delete as it no longer exists on cloud provider.
   142  	// This edge case means something went wrong with shutting down the instance and now the db is
   143  	// out of sync with the cloud provider. To amend this, delete the instance and its mandelboxes
   144  	// from the database.
   145  	logger.Infof("Removing instance %s from database as it no longer exists on cloud provider.", instance.ID)
   146  
   147  	affectedRows, err := s.DBClient.DeleteInstance(scalingCtx, s.GraphQLClient, instance.ID)
   148  	if err != nil {
   149  		return utils.MakeError("failed to delete instance %s from database: %s", instance.ID, err)
   150  	}
   151  
   152  	logger.Infof("Deleted %d rows from database.", affectedRows)
   153  	return nil
   154  }
