   107  // and to make sure it is removed from the database correctly. This keeps the database and cloud provider in
   108  // sync.
   109  func (s *DefaultScalingAlgorithm) VerifyInstanceRemoval(scalingCtx context.Context, instance subscriptions.Instance) error {
   110  	// Wait until the host service terminates the instance in the cloud provider.
   111  	err := s.Host.WaitForInstanceTermination(scalingCtx, maxWaitTimeTerminated, []string{instance.ID})
   112  	if err != nil {
   113  		return err
   114  	}
   115  
   116  	// Once its terminated, verify that it was removed from the database
   117  	instanceResult, err := s.DBClient.QueryInstance(scalingCtx, s.GraphQLClient, instance.ID)
   118  	if err != nil {
   119  		return utils.MakeError("failed to query database for instance %v. Error: %v", instance.ID, err)
   120  	}
   121  
   122  	// Verify that instance removed itself from the database
   123  	if len(instanceResult) == 0 {
   124  		logger.Info("Instance %v was successfully removed from database.", instance.ID)
   125  		return nil
   126  	}
   127  
   128  	// If instance still exists on the database, delete as it no longer exists on cloud provider.
   129  	// This edge case means something went wrong with shutting down the instance and now the db is
   130  	// out of sync with the cloud provider. To amend this, delete the instance and its mandelboxes
   131  	// from the database.
   132  	logger.Info("Removing instance %v from database as it no longer exists on cloud provider.", instance.ID)
   133  
   134  	affectedRows, err := s.DBClient.DeleteInstance(scalingCtx, s.GraphQLClient, instance.ID)
   135  	if err != nil {
   136  		return utils.MakeError("failed to delete instance %v from database. Error: %v", instance.ID, err)
   137  	}
   138  
   139  	logger.Infof("Deleted %v rows from database.", affectedRows)
