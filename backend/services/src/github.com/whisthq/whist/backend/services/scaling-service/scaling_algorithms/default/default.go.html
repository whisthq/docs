    96  	// If on local env, use default configurations
    97  	if metadata.IsLocalEnvWithoutDB() {
    98  		logger.Infof("Running on localdev, using default scaling algorithm configurations...")
    99  		return
   100  	}
   101  
   102  	logger.Infof("Populating config variables from config database...")
   103  	ctx, cancel := context.WithCancel(context.TODO())
   104  	defer cancel()
   105  
   106  	var (
   107  		configs map[string]string
   108  		err     error
   109  	)
   110  
   111  	switch metadata.GetAppEnvironmentLowercase() {
   112  	case string(metadata.EnvDev):
   113  		configs, err = dbclient.GetDevConfigs(ctx, client)
   114  	case string(metadata.EnvStaging):
   115  		configs, err = dbclient.GetStagingConfigs(ctx, client)
   116  	case string(metadata.EnvProd):
   117  		configs, err = dbclient.GetProdConfigs(ctx, client)
   118  	default:
   119  		configs, err = dbclient.GetDevConfigs(ctx, client)
   120  	}
   121  
   122  	if err != nil {
   123  		// Err is already wrapped here
   124  		logger.Error(err)
   125  
   126  		// Something went wrong, use default configuration values
   127  		return
   128  	}
   129  
   130  	// Temporary map to hold all parsed values
   131  	mandelboxRegionMap := make(map[string]int)
   132  
   133  	// Look for each region's entry in the config database values
   134  	// to populate the `desiredFreeMandelboxesPerRegion` map.
   135  	for _, region := range GetEnabledRegions() {
   136  		// Parse the region string to the format used by
   137  		// the config database so we can get the value.
   138  		key := strings.ReplaceAll(region, "-", "_")
   139  		key = strings.ToUpper(key)
   140  
   141  		configMandelboxes, ok := configs[utils.Sprintf("DESIRED_FREE_MANDELBOXES_%s", key)]
   142  		if !ok {
   143  			logger.Errorf("Desired mandelboxes for region %v not found on %v config database. Using default value.", region, metadata.GetAppEnvironmentLowercase())
   144  			// Use default value of 2 if the entry for the specific
   145  			// region was not found
   146  			mandelboxRegionMap[region] = 2
   147  			continue
   148  		}
   149  
   150  		mandelboxInt, err := strconv.Atoi(configMandelboxes)
   151  		if err != nil {
   152  			logger.Errorf("Error parsing desired mandelboxes value. Using default value. Err: %v", err)
   153  			// Use default value of 2 if we failed to convert to int
   154  			mandelboxRegionMap[region] = 2
   155  			continue
   156  		}
   157  
   158  		mandelboxRegionMap[region] = mandelboxInt
   159  	}
   160  
   161  	desiredFreeMandelboxesPerRegion = mandelboxRegionMap
   162  }
