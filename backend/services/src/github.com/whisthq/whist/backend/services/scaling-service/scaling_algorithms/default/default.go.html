     3  /*
     4  Package scaling_algorithms includes the implementations of all the scaling algorithms used by
     5  the scaling service.
     6  
     7  A scaling algorithm is defined as struct that implements the `ScalingAlgorithm` interface
     8  and has a collection of methods that make decisions about how and when to scale instances
     9  in order satisfy demand. Alternative implementations or use cases for a scaling algorithm
    10  could be: a region that has different characteristics such as less instance availability
    11  or user demand, deploying a smarter algorithm on a region with high demand, using different
    12  strategies for detecting user demand, or in general any situation that requires a different
    13  behaviour than the one in the default scaling algorithm.
    14  
    15  The currently implemented algorithms are:
    16  - Default scaling algorithm:
    17  	It handles the scaling of instances by computing mandelbox capacity for each instance
    18  	and determining the overall availability on each region according to a "buffer" set by
    19  	the config database. It applies the same scaling rules to all regions across the world.
    20  */

   120  // GetConfig will query the configuration database and populate the configuration variables
   121  // according to the environment the scaling service is running in. It's necessary to perform
   122  // the query before starting to receive any scaling events.
   123  func (s *DefaultScalingAlgorithm) GetConfig(client subscriptions.WhistGraphQLClient) {
   124  	// If on local env, use default configurations
   125  	if metadata.IsLocalEnvWithoutDB() {
   126  		logger.Infof("Running on localdev, using default scaling algorithm configurations...")
   127  		return
   128  	}
   129  
   130  	logger.Infof("Populating config variables from config database...")
   131  	ctx, cancel := context.WithCancel(context.TODO())
   132  	defer cancel()
   133  
   134  	// Get the most recent frontend version from the config database
   135  	version, err := dbclient.GetFrontendVersion(ctx, client)
   136  	if err != nil {
   137  		logger.Error(err)
   138  	}
   139  
   140  	if version == (subscriptions.FrontendVersion{}) {
   141  		logger.Errorf("got an empty frontend version")
   142  	}
   143  
   144  	// Set the scaling algorithm's internal version to the one received
   145  	// from the config database. The scaling service uses this value for
   146  	// checking if the incoming requests are coming from an outdated frontend.
   147  	setFrontendVersion(version)
   148  
   149  	var configs map[string]string
   150  
   151  	switch metadata.GetAppEnvironmentLowercase() {
   152  	case string(metadata.EnvDev):
   153  		configs, err = dbclient.GetDevConfigs(ctx, client)
   154  	case string(metadata.EnvStaging):
   155  		configs, err = dbclient.GetStagingConfigs(ctx, client)
   156  	case string(metadata.EnvProd):
   157  		configs, err = dbclient.GetProdConfigs(ctx, client)
   158  	default:
   159  		configs, err = dbclient.GetDevConfigs(ctx, client)
   160  	}
   161  
   162  	if err != nil {
   163  		// Err is already wrapped here
   164  		logger.Error(err)
   165  
   166  		// Something went wrong, use default configuration values
   167  		return
   168  	}
   169  
   170  	// Temporary map to hold all parsed values
   171  	mandelboxRegionMap := make(map[string]int)
   172  
   173  	// Look for each region's entry in the config database values
   174  	// to populate the `desiredFreeMandelboxesPerRegion` map.
   175  	for _, region := range GetEnabledRegions() {
   176  		// Parse the region string to the format used by
   177  		// the config database so we can get the value.
   178  		key := strings.ReplaceAll(region, "-", "_")
   179  		key = strings.ToUpper(key)
   180  
   181  		configMandelboxes, ok := configs[utils.Sprintf("DESIRED_FREE_MANDELBOXES_%s", key)]
   182  		if !ok {
   183  			logger.Errorf("desired mandelboxes for region %s not found on %s config database. Using default value.", region, metadata.GetAppEnvironmentLowercase())
   184  			// Use default value of 2 if the entry for the specific
   185  			// region was not found
   186  			mandelboxRegionMap[region] = 2
   187  			continue
   188  		}
   189  
   190  		mandelboxInt, err := strconv.Atoi(configMandelboxes)
   191  		if err != nil {
   192  			logger.Errorf("error parsing desired mandelboxes value: %s", err)
   193  			// Use default value of 2 if we failed to convert to int
   194  			mandelboxRegionMap[region] = 2
   195  			continue
   196  		}
   197  
   198  		mandelboxRegionMap[region] = mandelboxInt
   199  	}
   200  
   201  	desiredFreeMandelboxesPerRegion = mandelboxRegionMap
   202  
   203  }

   205  // ProcessEvents is the main function of the scaling algorithm, it is responsible of processing
   206  // events and executing the appropiate scaling actions. This function is specific for each region
   207  // scaling algorithm to be able to implement different strategies on each region.
   208  func (s *DefaultScalingAlgorithm) ProcessEvents(globalCtx context.Context, goroutineTracker *sync.WaitGroup) {
   209  	if s.Host == nil {
   210  		// TODO when multi-cloud support is introduced, figure out a way to
   211  		// decide which host to use. For now default to AWS.
   212  		handler := &aws.AWSHost{}
   213  		err := handler.Initialize(s.Region)
   214  
   215  		if err != nil {
   216  			logger.Errorf("error starting host in %s: %s", s.Region, err)
   217  		}
   218  
   219  		s.Host = handler
   220  	}
   221  
   222  	// Start algorithm main event loop
   223  	// Track this goroutine so we can wait for it to
   224  	// finish if the global context gets cancelled.
   225  	goroutineTracker.Add(1)
   226  	go func() {
   227  
   228  		defer goroutineTracker.Done()
   229  
   230  		for {
   231  			select {
   232  			case instanceEvent := <-s.InstanceEventChan:
   233  				instance := instanceEvent.Data.(subscriptions.Instance)
   234  
   235  				if instance.Status == "DRAINING" {
   236  					// Track this goroutine so we can wait for it to
   237  					// finish if the global context gets cancelled.
   238  					goroutineTracker.Add(1)
   239  					go func() {
   240  						defer goroutineTracker.Done()
   241  
   242  						// Create context for scaling operation
   243  						scalingCtx, scalingCancel := context.WithCancel(context.Background())
   244  						err := s.VerifyInstanceScaleDown(scalingCtx, instanceEvent, instance)
   245  
   246  						// Cancel context once the operation is done
   247  						scalingCancel()
   248  
   249  						if err != nil {
   250  							contextFields := []interface{}{
   251  								zap.String("id", instanceEvent.ID),
   252  								zap.Any("type", instanceEvent.Type),
   253  								zap.Any("data", instanceEvent.Data),
   254  								zap.String("region", instanceEvent.Region),
   255  							}
   256  							logger.Errorw(utils.Sprintf("error verifying instance scale down: %s", err), contextFields)
   257  						}
   258  					}()
   259  				}
   260  			case versionEvent := <-s.FrontendVersionChan:
   261  				version := versionEvent.Data.(subscriptions.FrontendVersion)
   262  
   263  				// Track this goroutine so we can wait for it to
   264  				// finish if the global context gets cancelled.
   265  				goroutineTracker.Add(1)
   266  				go func() {
   267  					defer goroutineTracker.Done()
   268  
   269  					// Create context for scaling operation
   270  					scalingCtx, scalingCancel := context.WithCancel(context.Background())
   271  
   272  					err := s.SwapOverImages(scalingCtx, versionEvent, version)
   273  
   274  					// Cancel context once the operation is done
   275  					scalingCancel()
   276  
   277  					if err != nil {
   278  						contextFields := []interface{}{
   279  							zap.String("id", versionEvent.ID),
   280  							zap.Any("type", versionEvent.Type),
   281  							zap.Any("data", versionEvent.Data),
   282  							zap.String("region", versionEvent.Region),
   283  						}
   284  						logger.Errorw(utils.Sprintf("error running image swapover: %s", err), contextFields)
   285  					}
   286  				}()
   287  
   288  			case scheduledEvent := <-s.ScheduledEventChan:
   289  				switch scheduledEvent.Type {
   290  				case "SCHEDULED_SCALE_DOWN_EVENT":
   291  					goroutineTracker.Add(1)
   292  					go func() {
   293  						defer goroutineTracker.Done()
   294  
   295  						scalingCtx, scalingCancel := context.WithCancel(context.Background())
   296  						err := s.ScaleDownIfNecessary(scalingCtx, scheduledEvent)
   297  						if err != nil {
   298  							contextFields := []interface{}{
   299  								zap.String("id", scheduledEvent.ID),
   300  								zap.Any("type", scheduledEvent.Type),
   301  								zap.Any("data", scheduledEvent.Data),
   302  								zap.String("region", scheduledEvent.Region),
   303  							}
   304  							logger.Errorw(utils.Sprintf("error running scale down job: %s", err), contextFields)
   305  						}
   306  
   307  						scalingCancel()
   308  					}()
   309  				case "SCHEDULED_IMAGE_UPGRADE_EVENT":
   310  					// Track this goroutine so we can wait for it to
   311  					// finish if the global context gets cancelled.
   312  					goroutineTracker.Add(1)
   313  					go func() {
   314  						defer goroutineTracker.Done()
   315  
   316  						if scheduledEvent.Data == nil {
   317  							logger.Errorf("error running image upgrade, event data is nil.")
   318  							return
   319  						}
   320  
   321  						scalingCtx, scalingCancel := context.WithCancel(context.Background())
   322  
   323  						// Get arguments from scheduled event
   324  						regionImageMap := scheduledEvent.Data.(map[string]interface{})
   325  
   326  						err := s.UpgradeImage(scalingCtx, scheduledEvent, regionImageMap[scheduledEvent.Region])
   327  						if err != nil {
   328  							contextFields := []interface{}{
   329  								zap.String("id", scheduledEvent.ID),
   330  								zap.Any("type", scheduledEvent.Type),
   331  								zap.Any("data", scheduledEvent.Data),
   332  								zap.String("region", scheduledEvent.Region),
   333  							}
   334  							logger.Errorw(utils.Sprintf("error running image upgrade: %s", err), contextFields)
   335  						}
   336  
   337  						scalingCancel()
   338  					}()
   339  				}
   340  			case serverEvent := <-s.ServerEventChan:
   341  				switch serverEvent.Type {
   342  				case "SERVER_MANDELBOX_ASSIGN_EVENT":
   343  					goroutineTracker.Add(1)
   344  					go func() {
   345  						defer goroutineTracker.Done()
   346  						serverEvent.Region = s.Region
   347  
   348  						// Create context for scaling operation
   349  						scalingCtx, scalingCancel := context.WithCancel(context.Background())
   350  
   351  						err := s.MandelboxAssign(scalingCtx, serverEvent)
   352  						if err != nil {
   353  							contextFields := []interface{}{
   354  								zap.String("id", serverEvent.ID),
   355  								zap.Any("type", serverEvent.Type),
   356  								zap.Any("data", serverEvent.Data),
   357  								zap.String("region", serverEvent.Region),
   358  							}
   359  							logger.Errorw(utils.Sprintf("error running mandelbox assign action: %s", err), contextFields)
   360  						}
   361  						// Cancel context once the operation is done
   362  						scalingCancel()
   363  					}()
   364  				}
   365  			case <-globalCtx.Done():
   366  				logger.Info("Global context has been cancelled. Exiting from default scaling algorithm event loop...")
   367  				goroutineTracker.Wait()
   368  				logger.Infof("Finished waiting for all goroutines to finish. Scaling algorithm %s exited.", s.Region)
   369  				return
   370  			}
   371  		}
   372  	}()
