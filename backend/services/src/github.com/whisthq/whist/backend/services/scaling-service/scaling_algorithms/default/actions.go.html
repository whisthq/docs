    79  // VerifyCapacity is a scaling action which checks the database to verify if we have the desired
    80  // capacity (instance buffer). This action is run at the end of the other actions.
    81  func (s *DefaultScalingAlgorithm) VerifyCapacity(scalingCtx context.Context, event ScalingEvent) error {
    82  	logger.Infof("Starting verify capacity action for event: %v", event)
    83  	defer logger.Infof("Finished verify capacity action for event: %v", event)
    84  
    85  	// Query for the latest image id
    86  	imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region) // TODO: set different provider when doing multi-cloud.
    87  	if err != nil {
    88  		return utils.MakeError("failed to query database for current image. Err: %v", err)
    89  	}
    90  
    91  	if len(imageResult) == 0 {
    92  		logger.Warningf("Image not found on %v. Not performing any scaling actions.", event.Region)
    93  		return nil
    94  	}
    95  	latestImageID := string(imageResult[0].ImageID)
    96  
    97  	// This query will return all instances with the ACTIVE status
    98  	allActive, err := s.DBClient.QueryInstancesByStatusOnRegion(scalingCtx, s.GraphQLClient, "ACTIVE", event.Region)
    99  	if err != nil {
   100  		return utils.MakeError("failed to query database for active instances. Err: %v", err)
   101  	}
   102  
   103  	// This query will return all instances with the PRE_CONNECTION status
   104  	allStarting, err := s.DBClient.QueryInstancesByStatusOnRegion(scalingCtx, s.GraphQLClient, "PRE_CONNECTION", event.Region)
   105  	if err != nil {
   106  		return utils.MakeError("failed to query database for starting instances. Err: %v", err)
   107  	}
   108  
   109  	mandelboxCapacity := helpers.ComputeExpectedMandelboxCapacity(latestImageID, allActive, allStarting)
   110  
   111  	// We consider the expected mandelbox capacity (active instances + starting instances)
   112  	// to account for warmup time and so that we don't scale up unnecesary instances.
   113  	if mandelboxCapacity < DESIRED_FREE_MANDELBOXES {
   114  		logger.Infof("Current mandelbox capacity of %v is less than desired %v. Scaling up %v instances to satisfy minimum desired capacity.", mandelboxCapacity, DESIRED_FREE_MANDELBOXES, DEFAULT_INSTANCE_BUFFER)
   115  		err = s.ScaleUpIfNecessary(DEFAULT_INSTANCE_BUFFER, scalingCtx, event, latestImageID)
   116  		if err != nil {
   117  			// err is already wrapped here
   118  			return err
   119  		}
   120  	} else {
   121  		logger.Infof("Mandelbox capacity %v in %v is enough to satisfy minimum desired capacity of %v.", mandelboxCapacity, event.Region, DESIRED_FREE_MANDELBOXES)
   122  	}
   123  
   124  	return nil
   125  }

   127  // ScaleDownIfNecessary is a scaling action which runs every 10 minutes and scales down free and
   128  // lingering instances, respecting the buffer defined for each region. Free instances will be
   129  // marked as draining, and lingering instances will be terminated and removed from the database.
   130  func (s *DefaultScalingAlgorithm) ScaleDownIfNecessary(scalingCtx context.Context, event ScalingEvent) error {
   131  	logger.Infof("Starting scale down action for event: %v", event)
   132  	defer logger.Infof("Finished scale down action for event: %v", event)
   133  
   134  	// We want to verify if we have the desired capacity after scaling down.
   135  	defer func() {
   136  		err := s.VerifyCapacity(scalingCtx, event)
   137  		if err != nil {
   138  			logger.Errorf("Error verifying capacity on %v. Err: %v", event.Region, err)
   139  		}
   140  	}()
   141  
   142  	var (
   143  		freeInstances, lingeringInstances subscriptions.WhistInstances
   144  		lingeringIDs                      []string
   145  		err                               error
   146  	)
   147  
   148  	// Query for the latest image id
   149  	imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region) // TODO: set different provider when doing multi-cloud.
   150  	if err != nil {
   151  		return utils.MakeError("failed to query database for current image. Err: %v", err)
   152  	}
   153  
   154  	if len(imageResult) == 0 {
   155  		logger.Warningf("Image not found on %v. Not performing any scaling actions.", event.Region)
   156  		return nil
   157  	}
   158  	latestImageID := string(imageResult[0].ImageID)
   159  
   160  	// query database for all active instances (status ACTIVE) without mandelboxes
   161  	allActive, err := s.DBClient.QueryInstancesByStatusOnRegion(scalingCtx, s.GraphQLClient, "ACTIVE", event.Region)
   162  	if err != nil {
   163  		return utils.MakeError("failed to query database for active instances. Err: %v", err)
   164  	}
   165  
   166  	// Active instances with space to run mandelboxes
   167  	mandelboxCapacity := helpers.ComputeRealMandelboxCapacity(latestImageID, allActive)
   168  
   169  	// Extra capacity is considered once we have a full instance worth of capacity
   170  	// more than the desired free mandelboxes. TODO: Set the instance type once we
   171  	// have support for more instance types. For now default to `g4dn.2xlarge`.
   172  	extraCapacity := DESIRED_FREE_MANDELBOXES + (DEFAULT_INSTANCE_BUFFER * instanceCapacity["g4dn.2xlarge"])
   173  
   174  	// Create a list of instances that can be scaled down from the active instances list.
   175  	// For this, we have to consider the following conditions:
   176  	// 1. Does the instance have any running mandelboxes? If so, don't scale down.
   177  	// 2. Does the instance have the latest image, corresponding to the latest entry on the database?
   178  	// If so, check if we have more than the desired number of mandelbox capacity. In case we do, add the instance
   179  	// to the list that will be scaled down.
   180  	// 3. If the instance does not have the latest image, and is not running any mandelboxes, add to the
   181  	// list that will be scaled down.
   182  	for _, instance := range allActive {
   183  		if len(instance.Mandelboxes) > 0 {
   184  			// Don't scale down any instance that has running
   185  			// mandelboxes, regardless of the image it uses
   186  			logger.Infof("Not scaling down instance %v because it has %v mandelboxes running.", instance.ID, len(instance.Mandelboxes))
   187  			continue
   188  		}
   189  		if instance.ImageID == graphql.String(latestImageID) {
   190  			// Current instances
   191  			// If we have more than one instance worth of extra mandelbox capacity, scale down
   192  			if mandelboxCapacity >= extraCapacity {
   193  				logger.Infof("Scaling down instance %v because we have more mandelbox capacity of %v than desired %v.", instance.ID, mandelboxCapacity, extraCapacity)
   194  				freeInstances = append(freeInstances, instance)
   195  				mandelboxCapacity -= int(instance.RemainingCapacity)
   196  			}
   197  		} else {
   198  			// Old instances
   199  			logger.Infof("Scaling down instance %v because it has an old image id %v.", instance.ID, instance.ImageID)
   200  			freeInstances = append(freeInstances, instance)
   201  		}
   202  	}
   203  
   204  	// check database for draining instances without running mandelboxes
   205  	drainingInstances, err := s.DBClient.QueryInstancesByStatusOnRegion(scalingCtx, s.GraphQLClient, "DRAINING", event.Region)
   206  	if err != nil {
   207  		return utils.MakeError("failed to query database for lingering instances. Err: %v", err)
   208  	}
   209  
   210  	for _, instance := range drainingInstances {
   211  		// Check if lingering instance is free from mandelboxes
   212  		if len(instance.Mandelboxes) == 0 {
   213  			lingeringInstances = append(lingeringInstances, instance)
   214  			lingeringIDs = append(lingeringIDs, string(instance.ID))
   215  		} else {
   216  			// If not, notify, could be a stuck mandelbox (check if mandelbox is > day old?)
   217  			logger.Warningf("Instance %v has %v associated mandelboxes and is marked as Draining.", instance.ID, len(instance.Mandelboxes))
   218  		}
   219  	}
   220  
   221  	// Verify if there are lingering instances and notify.
   222  	if len(lingeringInstances) > 0 {
   223  		logger.Errorf("There are %v lingering instances on %v. Investigate immediately! Their IDs are %v", len(lingeringInstances), event.Region, lingeringIDs)
   224  
   225  	} else {
   226  		logger.Info("There are no lingering instances in %v.", event.Region)
   227  	}
   228  
   229  	// Verify if there are free instances that can be scaled down
   230  	if len(freeInstances) == 0 {
   231  		logger.Info("There are no free instances to scale down in %v.", event.Region)
   232  		return nil
   233  	}
   234  
   235  	logger.Info("Scaling down %v free instances on %v.", len(freeInstances), event.Region)
   236  
   237  	for _, instance := range freeInstances {
   238  		logger.Infof("Scaling down instance %v.", instance.ID)
   239  		updateParams := map[string]interface{}{
   240  			"id":     instance.ID,
   241  			"status": graphql.String("DRAINING"),
   242  		}
   243  
   244  		_, err = s.DBClient.UpdateInstance(scalingCtx, s.GraphQLClient, updateParams)
   245  		if err != nil {
   246  			logger.Errorf("Failed to mark instance %v as draining. Err: %v", instance, err)
   247  		}
   248  
   249  		logger.Info("Marked instance %v as draining on database.", instance.ID)
   250  	}
   251  
   252  	return err
   253  }

   255  // ScaleUpIfNecessary is a scaling action that launched the received number of instances on
   256  // the cloud provider and registers them on the database with the initial values.
   257  func (s *DefaultScalingAlgorithm) ScaleUpIfNecessary(instancesToScale int, scalingCtx context.Context, event ScalingEvent, imageID string) error {
   258  	logger.Infof("Starting scale up action for event: %v", event)
   259  	defer logger.Infof("Finished scale up action for event: %v", event)
   260  
   261  	// Try scale up in given region
   262  	instanceNum := int32(instancesToScale)
   263  
   264  	createdInstances, err := s.Host.SpinUpInstances(scalingCtx, instanceNum, maxWaitTimeReady, imageID)
   265  	if err != nil {
   266  		return utils.MakeError("Failed to spin up instances, created %v, err: %v", createdInstances, err)
   267  	}
   268  
   269  	// Check if we could create the desired number of instances
   270  	if len(createdInstances) != instancesToScale {
   271  		return utils.MakeError("Could not scale up %v instances, only scaled up %v.", instancesToScale, len(createdInstances))
   272  	}
   273  
   274  	// Create slice with newly created instance ids
   275  	var instancesForDb []subscriptions.Instance
   276  
   277  	for _, instance := range createdInstances {
   278  		instancesForDb = append(instancesForDb, subscriptions.Instance{
   279  			ID:                instance.ID,
   280  			IPAddress:         instance.IPAddress,
   281  			Provider:          instance.Provider,
   282  			Region:            instance.Region,
   283  			ImageID:           instance.ImageID,
   284  			ClientSHA:         instance.ClientSHA,
   285  			Type:              instance.Type,
   286  			RemainingCapacity: int64(instanceCapacity[instance.Type]),
   287  			Status:            instance.Status,
   288  			CreatedAt:         instance.CreatedAt,
   289  			UpdatedAt:         instance.UpdatedAt,
   290  		})
   291  		logger.Infof("Created tagged instance with ID %v", instance.ID)
   292  	}
   293  
   294  	logger.Infof("Inserting newly created instances to database.")
   295  
   296  	// If successful, write to db
   297  	affectedRows, err := s.DBClient.InsertInstances(scalingCtx, s.GraphQLClient, instancesForDb)
   298  	if err != nil {
   299  		return utils.MakeError("Failed to insert instances into database. Error: %v", err)
   300  	}
   301  
   302  	logger.Infof("Inserted %v rows to database.", affectedRows)
   303  
   304  	return nil
   305  }

   307  // UpgradeImage is a scaling action which runs when a new version is deployed. Its responsible of
   308  // starting a buffer of instances with the new image and scaling down instances with the previous
   309  // image.
   310  func (s *DefaultScalingAlgorithm) UpgradeImage(scalingCtx context.Context, event ScalingEvent, imageID interface{}) error {
   311  	logger.Infof("Starting upgrade image action for event: %v", event)
   312  	defer logger.Infof("Finished upgrade image action for event: %v", event)
   313  
   314  	var oldImageID string
   315  
   316  	// Check if we received a valid image before performing more
   317  	// expensive operations.
   318  
   319  	if imageID == nil {
   320  		logger.Warningf("Received an empty image ID on %v. Not performing upgrade.", event.Region)
   321  		return nil
   322  	}
   323  
   324  	newImageID := imageID.(string)
   325  
   326  	// Query for the current image id
   327  	imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region)
   328  	if err != nil {
   329  		return utils.MakeError("failed to query database for current image on %v. Err: %v", event.Region, err)
   330  	}
   331  
   332  	if len(imageResult) == 0 {
   333  		logger.Warningf("Image doesn't exist on %v. Creating a new entry with image %v.", event.Region, newImageID)
   334  	} else {
   335  		// We now consider the "current" image as the "old" image
   336  		oldImageID = string(imageResult[0].ImageID)
   337  	}
   338  
   339  	// create instance buffer with new image
   340  	logger.Infof("Creating new instance buffer for image %v", newImageID)
   341  	bufferInstances, err := s.Host.SpinUpInstances(scalingCtx, DEFAULT_INSTANCE_BUFFER, maxWaitTimeReady, newImageID)
   342  	if err != nil {
   343  		return utils.MakeError("failed to create instance buffer for image %v. Error: %v", newImageID, err)
   344  	}
   345  
   346  	// create slice of newly created instance ids
   347  	var instancesForDb []subscriptions.Instance
   348  
   349  	for _, instance := range bufferInstances {
   350  		instancesForDb = append(instancesForDb, subscriptions.Instance{
   351  			ID:                instance.ID,
   352  			IPAddress:         instance.IPAddress,
   353  			Provider:          instance.Provider,
   354  			Region:            instance.Region,
   355  			ImageID:           instance.ImageID,
   356  			ClientSHA:         instance.ClientSHA,
   357  			Type:              instance.Type,
   358  			RemainingCapacity: int64(instanceCapacity[instance.Type]),
   359  			Status:            instance.Status,
   360  			CreatedAt:         instance.CreatedAt,
   361  			UpdatedAt:         instance.UpdatedAt,
   362  		})
   363  	}
   364  
   365  	logger.Infof("Inserting newly created instances to database.")
   366  
   367  	// If successful, write to db
   368  	affectedRows, err := s.DBClient.InsertInstances(scalingCtx, s.GraphQLClient, instancesForDb)
   369  	if err != nil {
   370  		return utils.MakeError("Failed to insert instances into database. Error: %v", err)
   371  	}
   372  
   373  	logger.Infof("Inserted %v rows to database.", affectedRows)
   374  
   375  	// swapover active image on database
   376  	logger.Infof("Updating old %v image %v to new image %v on database.", event.Region, oldImageID, newImageID)
   377  	updateParams := subscriptions.Image{
   378  		Provider:  "AWS",
   379  		Region:    event.Region,
   380  		ImageID:   newImageID,
   381  		ClientSHA: metadata.GetGitCommit(),
   382  		UpdatedAt: time.Now(),
   383  	}
   384  
   385  	if oldImageID == "" {
   386  		_, err = s.DBClient.InsertImages(scalingCtx, s.GraphQLClient, []subscriptions.Image{updateParams})
   387  		if err != nil {
   388  			return utils.MakeError("Failed to insert image %v into database. Error: %v", newImageID, err)
   389  		}
   390  	} else {
   391  		_, err = s.DBClient.UpdateImage(scalingCtx, s.GraphQLClient, updateParams)
   392  		if err != nil {
   393  			return utils.MakeError("Failed to update image %v to image %v in database. Error: %v", oldImageID, newImageID, err)
   394  		}
   395  	}
   396  
   397  	return nil
   398  }
