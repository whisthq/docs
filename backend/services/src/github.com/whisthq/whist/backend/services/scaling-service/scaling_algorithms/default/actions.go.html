   138  // lingering instances, respecting the buffer defined for each region. Free instances will be
   139  // marked as draining, and lingering instances will be terminated and removed from the database.
   140  func (s *DefaultScalingAlgorithm) ScaleDownIfNecessary(scalingCtx context.Context, event ScalingEvent) error {
   141  	logger.Infof("Starting scale down action for event: %v", event)
   142  	defer logger.Infof("Finished scale down action for event: %v", event)
   143  
   144  	// We want to verify if we have the desired capacity after scaling down.
   145  	defer func() {
   146  		err := s.VerifyCapacity(scalingCtx, event)
   147  		if err != nil {
   148  			logger.Errorf("Error verifying capacity on %v. Err: %v", event.Region, err)
   149  		}
   150  	}()
   151  
   152  	var (
   153  		freeInstances, lingeringInstances []subscriptions.Instance
   154  		lingeringIDs                      []string
   155  		err                               error
   156  	)
   157  
   158  	// Query for the latest image id
   159  	imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region) // TODO: set different provider when doing multi-cloud.
   160  	if err != nil {
   161  		return utils.MakeError("failed to query database for current image. Err: %v", err)
   162  	}
   163  
   164  	if len(imageResult) == 0 {
   165  		logger.Warningf("Image not found on %v. Not performing any scaling actions.", event.Region)
   166  		return nil
   167  	}
   168  	latestImageID := string(imageResult[0].ImageID)
   169  
   170  	// Query database for all active instances (status ACTIVE) without mandelboxes
   171  	allActive, err := s.DBClient.QueryInstancesByStatusOnRegion(scalingCtx, s.GraphQLClient, "ACTIVE", event.Region)
   172  	if err != nil {
   173  		return utils.MakeError("failed to query database for active instances. Err: %v", err)
   174  	}
   175  
   176  	// Active instances with space to run mandelboxes
   177  	mandelboxCapacity := helpers.ComputeRealMandelboxCapacity(latestImageID, allActive)
   178  
   179  	// Extra capacity is considered once we have a full instance worth of capacity
   180  	// more than the desired free mandelboxes. TODO: Set the instance type once we
   181  	// have support for more instance types. For now default to `g4dn.2xlarge`.
   182  	extraCapacity := desiredFreeMandelboxesPerRegion[event.Region] + (defaultInstanceBuffer * instanceCapacity["g4dn.2xlarge"])
   183  
   184  	// Acquire lock on protected from scale down map
   185  	s.protectedMapLock.Lock()
   186  	defer s.protectedMapLock.Unlock()
   187  
   188  	// Create a list of instances that can be scaled down from the active instances list.
   189  	// For this, we have to consider the following conditions:
   190  	// 1. Does the instance have any running mandelboxes? If so, don't scale down.
   191  	// 2. Does the instance have the latest image, corresponding to the latest entry on the database?
   192  	// If so, check if we have more than the desired number of mandelbox capacity. In case we do, add the instance
   193  	// to the list that will be scaled down.
   194  	// 3. If the instance does not have the latest image, and is not running any mandelboxes, add to the
   195  	// list that will be scaled down.
   196  	for _, dbInstance := range allActive {
   197  		if dbInstance.RemainingCapacity == 0 {
   198  			// Don't scale down any instance that has running
   199  			// mandelboxes, regardless of the image it uses
   200  			logger.Infof("Not scaling down instance %v because it has %v mandelboxes running.", dbInstance.ID, dbInstance.RemainingCapacity)
   201  			continue
   202  		}
   203  
   204  		instance := subscriptions.Instance{
   205  			ID:                string(dbInstance.ID),
   206  			Provider:          string(dbInstance.Provider),
   207  			Region:            string(dbInstance.Region),
   208  			ImageID:           string(dbInstance.ImageID),
   209  			ClientSHA:         string(dbInstance.ClientSHA),
   210  			IPAddress:         dbInstance.IPAddress,
   211  			Type:              string(dbInstance.Type),
   212  			RemainingCapacity: int64(dbInstance.RemainingCapacity),
   213  			Status:            string(dbInstance.Status),
   214  			CreatedAt:         dbInstance.CreatedAt,
   215  			UpdatedAt:         dbInstance.UpdatedAt,
   216  		}
   217  
   218  		_, protected := s.protectedFromScaleDown[instance.ImageID]
   219  		if protected {
   220  			// Don't scale down instances with a protected image id. A protected
   221  			// image id refers to the image id that was built and passed by the
   222  			// `build-and-deploy` workflow, that is waiting for the config database
   223  			// to update commit hashes. For this reason, we don't scale down the
   224  			// instance buffer created for this image, instead we "protect" it until
   225  			// its ready to use, to avoid downtimes and to create the buffer only once.
   226  			logger.Infof("Not scaling down instance %v because it has an image id that is protected from scale down.", instance.ID)
   227  			continue
   228  		}
   229  
   230  		if instance.ImageID == latestImageID {
   231  			// Current instances
   232  			// If we have more than one instance worth of extra mandelbox capacity, scale down
   233  			if mandelboxCapacity >= extraCapacity {
   234  				logger.Infof("Scaling down instance %v because we have more mandelbox capacity of %v than desired %v.", instance.ID, mandelboxCapacity, extraCapacity)
   235  				freeInstances = append(freeInstances, instance)
   236  				mandelboxCapacity -= int(instance.RemainingCapacity)
   237  			}
   238  		} else {
   239  			// Old instances
   240  			logger.Infof("Scaling down instance %v because it has an old image id %v.", instance.ID, instance.ImageID)
   241  			freeInstances = append(freeInstances, instance)
   242  		}
   243  	}
   244  
   245  	// Check database for draining instances without running mandelboxes
   246  	drainingInstances, err := s.DBClient.QueryInstancesByStatusOnRegion(scalingCtx, s.GraphQLClient, "DRAINING", event.Region)
   247  	if err != nil {
   248  		return utils.MakeError("failed to query database for lingering instances. Err: %v", err)
   249  	}
   250  
   251  	for _, dbInstance := range drainingInstances {
   252  		instance := subscriptions.Instance{
   253  			ID:                string(dbInstance.ID),
   254  			Provider:          string(dbInstance.Provider),
   255  			Region:            string(dbInstance.Region),
   256  			ImageID:           string(dbInstance.ImageID),
   257  			ClientSHA:         string(dbInstance.ClientSHA),
   258  			IPAddress:         dbInstance.IPAddress,
   259  			Type:              string(dbInstance.Type),
   260  			RemainingCapacity: int64(dbInstance.RemainingCapacity),
   261  			Status:            string(dbInstance.Status),
   262  			CreatedAt:         dbInstance.CreatedAt,
   263  			UpdatedAt:         dbInstance.UpdatedAt,
   264  		}
   265  		// Check if lingering instance has any running mandelboxes
   266  		if dbInstance.RemainingCapacity == 0 {
   267  			// If not, notify, could be a stuck mandelbox (check if mandelbox is > day old?)
   268  			logger.Warningf("Instance %v has associated mandelboxes and is marked as Draining.", instance.ID)
   269  		} else {
   270  			lingeringInstances = append(lingeringInstances, instance)
   271  			lingeringIDs = append(lingeringIDs, string(instance.ID))
   272  		}
   273  	}
   274  
   275  	// Verify if there are lingering instances and notify.
   276  	if len(lingeringInstances) > 0 {
   277  		logger.Errorf("There are %v lingering instances on %v. Investigate immediately! Their IDs are %s", len(lingeringInstances), event.Region, utils.PrintSlice(lingeringIDs))
   278  
   279  	} else {
   280  		logger.Info("There are no lingering instances in %v.", event.Region)
   281  	}
   282  
   283  	// Verify if there are free instances that can be scaled down
   284  	if len(freeInstances) == 0 {
   285  		logger.Info("There are no free instances to scale down in %v.", event.Region)
   286  		return nil
   287  	}
   288  
   289  	logger.Info("Scaling down %v free instances on %v.", len(freeInstances), event.Region)
   290  
   291  	for _, instance := range freeInstances {
   292  		logger.Infof("Scaling down instance %v.", instance.ID)
   293  
   294  		instance.Status = "DRAINING"
   295  		_, err = s.DBClient.UpdateInstance(scalingCtx, s.GraphQLClient, instance)
   296  		if err != nil {
   297  			logger.Errorf("Failed to mark instance %v as draining. Err: %v", instance, err)
   298  		}
   299  
   300  		logger.Info("Marked instance %v as draining on database.", instance.ID)
   301  	}
   302  
   303  	return err
   304  }

   306  // ScaleUpIfNecessary is a scaling action that launched the received number of instances on
   307  // the cloud provider and registers them on the database with the initial values.
   308  func (s *DefaultScalingAlgorithm) ScaleUpIfNecessary(instancesToScale int, scalingCtx context.Context, event ScalingEvent, image subscriptions.Image) error {
   309  	logger.Infof("Starting scale up action for event: %v", event)
   310  	defer logger.Infof("Finished scale up action for event: %v", event)
   311  
   312  	// Try scale up in given region
   313  	instanceNum := int32(instancesToScale)
   314  
   315  	// Slice that will hold the instances and pass them to the dbclient
   316  	var instancesForDb []subscriptions.Instance
   317  
   318  	// If we are running on a local or testing environment, spinup "fake" instances to avoid
   319  	// creating them on a cloud provider. In any other case we call the host handler to create
   320  	// them on the cloud provider for us.
   321  	if metadata.IsLocalEnv() && !metadata.IsRunningInCI() {
   322  		logger.Infof("Running on localdev so scaling up fake instances.")
   323  		instancesForDb = helpers.SpinUpFakeInstances(instancesToScale, image.ImageID, event.Region)
   324  	} else {
   325  		// Call the host handler to handle the instance spinup in the cloud provider
   326  		createdInstances, err := s.Host.SpinUpInstances(scalingCtx, instanceNum, maxWaitTimeReady, image)
   327  		if err != nil {
   328  			return utils.MakeError("Failed to spin up instances, created %v, err: %v", createdInstances, err)
   329  		}
   330  
   331  		// Check if we could create the desired number of instances
   332  		if len(createdInstances) != instancesToScale {
   333  			return utils.MakeError("Could not scale up %v instances, only scaled up %v.", instancesToScale, len(createdInstances))
   334  		}
   335  
   336  		for _, instance := range createdInstances {
   337  			instancesForDb = append(instancesForDb, subscriptions.Instance{
   338  				ID:                instance.ID,
   339  				IPAddress:         instance.IPAddress,
   340  				Provider:          instance.Provider,
   341  				Region:            instance.Region,
   342  				ImageID:           instance.ImageID,
   343  				ClientSHA:         instance.ClientSHA,
   344  				Type:              instance.Type,
   345  				RemainingCapacity: int64(instanceCapacity[instance.Type]),
   346  				Status:            instance.Status,
   347  				CreatedAt:         instance.CreatedAt,
   348  				UpdatedAt:         instance.UpdatedAt,
   349  			})
   350  			logger.Infof("Created tagged instance with ID %v", instance.ID)
   351  		}
   352  	}
   353  
   354  	logger.Infof("Inserting newly created instances to database.")
   355  
   356  	// If successful, write to database
   357  	affectedRows, err := s.DBClient.InsertInstances(scalingCtx, s.GraphQLClient, instancesForDb)
   358  	if err != nil {
   359  		return utils.MakeError("Failed to insert instances into database. Error: %v", err)
   360  	}
   361  
   362  	logger.Infof("Inserted %v rows to database.", affectedRows)
   363  
   364  	return nil
   365  }

   367  // UpgradeImage is a scaling action which runs when a new version is deployed. Its responsible of
   368  // starting a buffer of instances with the new image and scaling down instances with the previous
   369  // image.
   370  func (s *DefaultScalingAlgorithm) UpgradeImage(scalingCtx context.Context, event ScalingEvent, imageID interface{}) error {
   371  	logger.Infof("Starting upgrade image action for event: %v", event)
   372  	defer logger.Infof("Finished upgrade image action for event: %v", event)
   373  
   374  	// Check if we received a valid image before performing more
   375  	// expensive operations.
   376  
   377  	if imageID == nil {
   378  		logger.Warningf("Received an empty image ID on %v. Not performing upgrade.", event.Region)
   379  		return nil
   380  	}
   381  
   382  	newImage := subscriptions.Image{
   383  		Provider:  "AWS",
   384  		Region:    event.Region,
   385  		ImageID:   imageID.(string),
   386  		ClientSHA: metadata.GetGitCommit(),
   387  		UpdatedAt: time.Now(),
   388  	}
   389  
   390  	// Query for the current image id
   391  	imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region)
   392  	if err != nil {
   393  		return utils.MakeError("failed to query database for current image on %v. Err: %v", event.Region, err)
   394  	}
   395  
   396  	logger.Infof("Creating new instance buffer for image %v", newImage.ImageID)
   397  
   398  	// Slice that will hold the instances and pass them to the dbclient
   399  	var instancesForDb []subscriptions.Instance
   400  
   401  	// If we are running on a local or testing environment, spinup "fake" instances to avoid
   402  	// creating them on a cloud provider. In any other case we call the host handler to create
   403  	// them on the cloud provider for us.
   404  	if metadata.IsLocalEnv() && !metadata.IsRunningInCI() {
   405  		logger.Infof("Running on localdev so scaling up fake instances.")
   406  		instancesForDb = helpers.SpinUpFakeInstances(defaultInstanceBuffer, newImage.ImageID, event.Region)
   407  	} else {
   408  		bufferInstances, err := s.Host.SpinUpInstances(scalingCtx, int32(defaultInstanceBuffer), maxWaitTimeReady, newImage)
   409  		if err != nil {
   410  			return utils.MakeError("failed to create instance buffer for image %v. Error: %v", newImage.ImageID, err)
   411  		}
   412  
   413  		for _, instance := range bufferInstances {
   414  			instancesForDb = append(instancesForDb, subscriptions.Instance{
   415  				ID:                instance.ID,
   416  				IPAddress:         instance.IPAddress,
   417  				Provider:          instance.Provider,
   418  				Region:            instance.Region,
   419  				ImageID:           instance.ImageID,
   420  				ClientSHA:         instance.ClientSHA,
   421  				Type:              instance.Type,
   422  				RemainingCapacity: int64(instanceCapacity[instance.Type]),
   423  				Status:            instance.Status,
   424  				CreatedAt:         instance.CreatedAt,
   425  				UpdatedAt:         instance.UpdatedAt,
   426  			})
   427  		}
   428  	}
   429  
   430  	logger.Infof("Inserting newly created instances to database.")
   431  
   432  	// If successful, write to database
   433  	affectedRows, err := s.DBClient.InsertInstances(scalingCtx, s.GraphQLClient, instancesForDb)
   434  	if err != nil {
   435  		return utils.MakeError("Failed to insert instances into database. Error: %v", err)
   436  	}
   437  
   438  	logger.Infof("Inserted %v rows to database.", affectedRows)
   439  
   440  	// Acquire lock on protected from scale down map
   441  	s.protectedMapLock.Lock()
   442  	defer s.protectedMapLock.Unlock()
   443  
   444  	// Protect the new instance buffer from scale down. This is done to avoid any downtimes
   445  	// during deploy, as the active image will be switched until the client app has updated
   446  	// its version on the config database.
   447  	s.protectedFromScaleDown = make(map[string]subscriptions.Image)
   448  	s.protectedFromScaleDown[newImage.ImageID] = newImage
   449  
   450  	// If the region does not have an existing image, insert the new one to the database.
   451  	if len(imageResult) == 0 {
   452  		logger.Warningf("Image doesn't exist on %v. Creating a new entry with image %v.", event.Region, newImage.ImageID)
   453  
   454  		updateParams := subscriptions.Image{
   455  			Provider:  "AWS",
   456  			Region:    event.Region,
   457  			ImageID:   newImage.ImageID,
   458  			ClientSHA: newImage.ClientSHA,
   459  			UpdatedAt: time.Now(),
   460  		}
   461  
   462  		_, err = s.DBClient.InsertImages(scalingCtx, s.GraphQLClient, []subscriptions.Image{updateParams})
   463  		if err != nil {
   464  			return utils.MakeError("Failed to insert image %v into database. Error: %v", newImage.ImageID, err)
   465  		}
   466  	}
   467  
   468  	// Notify through the synchan that the image upgrade is done
   469  	// so that we can continue to swapover images when the config
   470  	// database updates. We time out here in case the frontend build
   471  	// failed to deploy, and if it does we rollback the new version.
   472  	select {
   473  	case s.SyncChan <- true:
   474  		logger.Infof("Finished upgrading image %v in region %v", newImage.ImageID, event.Region)
   475  	case <-time.After(1 * time.Hour):
   476  		// Clear protected map since the client app deploy didn't complete successfully.
   477  		s.protectedFromScaleDown = make(map[string]subscriptions.Image)
   478  
   479  		return utils.MakeError("Timed out waiting for config database to swap versions. Rolling back deploy of new version.")
   480  	}
   481  
   482  	return nil
   483  }

   485  // SwapOverImages is a scaling action that will switch the current image on the given region.
   486  // To the latest one. This is done separately to avoid having downtimes during deploys, since
   487  // we have to wait until the frontend has updated its version on the config database.
   488  func (s *DefaultScalingAlgorithm) SwapOverImages(scalingCtx context.Context, event ScalingEvent, clientVersion interface{}) error {
   489  	// Block until the image upgrade has finished successfully.
   490  	// We time out here in case something went wrong with the
   491  	// upgrade image action, in which case we roll back the new version.
   492  	select {
   493  	case <-s.SyncChan:
   494  		logger.Infof("Got signal that image upgrade action finished correctly.")
   495  	case <-time.After(1 * time.Hour):
   496  		// Clear protected map since the image upgrade didn't complete successfully.
   497  		s.protectedMapLock.Lock()
   498  		s.protectedFromScaleDown = make(map[string]subscriptions.Image)
   499  		s.protectedMapLock.Unlock()
   500  
   501  		return utils.MakeError("Timed out waiting for image upgrade to finish. Rolling back deploy of new version.")
   502  	}
   503  
   504  	logger.Infof("Starting image swapover action for event: %v", event)
   505  	defer logger.Infof("Finished image swapover action for event: %v", event)
   506  
   507  	// version is the entry we receive from the config database
   508  	version := clientVersion.(subscriptions.ClientAppVersion)
   509  
   510  	// Update the internal version with the new one received from the database.
   511  	// This function updates the value inside the config file, so we can keep  track
   512  	// of the current version locally, it does not update the value in the database.
   513  	setFrontendVersion(version)
   514  
   515  	var (
   516  		commitHash string
   517  		newImage   subscriptions.Image
   518  		newImageID string
   519  		oldImageID string
   520  	)
   521  
   522  	// get the commit hash of the environment we are running in
   523  	switch metadata.GetAppEnvironmentLowercase() {
   524  	case string(metadata.EnvDev):
   525  		commitHash = version.DevCommitHash
   526  	case string(metadata.EnvStaging):
   527  		commitHash = version.StagingCommitHash
   528  	case string(metadata.EnvProd):
   529  		commitHash = version.ProdCommitHash
   530  	default:
   531  		commitHash = version.DevCommitHash
   532  	}
   533  
   534  	// Acquire lock on protected from scale down map
   535  	s.protectedMapLock.Lock()
   536  	defer s.protectedMapLock.Unlock()
   537  
   538  	// Find protected image that matches the config db commit hash
   539  	for _, image := range s.protectedFromScaleDown {
   540  		if image.ClientSHA == commitHash {
   541  			newImage = image
   542  			newImageID = image.ImageID
   543  			break
   544  		}
   545  	}
   546  
   547  	if newImage == (subscriptions.Image{}) {
   548  		return utils.MakeError("did not find protected image with commit hash %v. Not performing image swapover.", commitHash)
   549  	}
   550  
   551  	// Query for the current image id
   552  	imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region)
   553  	if err != nil {
   554  		return utils.MakeError("failed to query database for current image on %v. Err: %v", event.Region, err)
   555  	}
   556  
   557  	if len(imageResult) == 0 {
   558  		logger.Warningf("Image doesn't exist on %v. Creating a new entry with image %v.", event.Region, newImageID)
   559  	} else {
   560  		// We now consider the "current" image as the "old" image
   561  		oldImageID = string(imageResult[0].ImageID)
   562  	}
   563  
   564  	// swapover active image on database
   565  	logger.Infof("Updating old %v image %v to new image %v on database.", event.Region, oldImageID, newImageID)
   566  
   567  	if oldImageID == "" {
   568  		_, err = s.DBClient.InsertImages(scalingCtx, s.GraphQLClient, []subscriptions.Image{newImage})
   569  		if err != nil {
   570  			return utils.MakeError("Failed to insert image %v into database. Error: %v", newImageID, err)
   571  		}
   572  	} else {
   573  		_, err = s.DBClient.UpdateImage(scalingCtx, s.GraphQLClient, newImage)
   574  		if err != nil {
   575  			return utils.MakeError("Failed to update image %v to image %v in database. Error: %v", oldImageID, newImageID, err)
   576  		}
   577  	}
   578  
   579  	// Unprotect the image until we have successfully swapped images in database
   580  	delete(s.protectedFromScaleDown, newImage.ImageID)
   581  
   582  	return nil
   583  }

   585  // MandelboxAssign is the action responsible for assigning an instance to a user,
   586  // and scaling as necessary to satisfy demand.
   587  func (s *DefaultScalingAlgorithm) MandelboxAssign(scalingCtx context.Context, event ScalingEvent) error {
   588  	logger.Infof("Starting mandelbox assign action for event: %v", event)
   589  	defer logger.Infof("Finished mandelbox assign action for event: %v", event)
   590  
   591  	// We want to verify if we have the desired capacity after assigning a mandelbox
   592  	defer func() {
   593  		err := s.VerifyCapacity(scalingCtx, event)
   594  		if err != nil {
   595  			logger.Errorf("Error verifying capacity on %v. Err: %v", event.Region, err)
   596  		}
   597  	}()
   598  
   599  	mandelboxRequest := event.Data.(*httputils.MandelboxAssignRequest)
   600  
   601  	// Note: we receive the email from the client, so its value should
   602  	// not be trusted for anything else other than logging since
   603  	// it can be spoofed. We sanitize the email before using to help mitigate
   604  	// potential attacks.
   605  	unsafeEmail, err := helpers.SanitizeEmail(mandelboxRequest.UserEmail)
   606  	if err != nil {
   607  		// err is already wrapped here
   608  		return err
   609  	}
   610  
   611  	var (
   612  		requestedRegions   = mandelboxRequest.Regions // This is a list of the regions requested by the frontend, in order of proximity.
   613  		availableRegions   []string                   // List of regions that are enabled and match the regions in the request.
   614  		unavailableRegions []string                   // List of regions that are not enabled but are present in the request.
   615  		assignedInstance   subscriptions.Instance     // The instance that is assigned to the user.
   616  	)
   617  
   618  	// Populate availableRegions
   619  	for _, requestedRegion := range requestedRegions {
   620  		var regionFound bool
   621  		for _, enabledRegion := range GetEnabledRegions() {
   622  			if enabledRegion == requestedRegion {
   623  				availableRegions = append(availableRegions, requestedRegion)
   624  				regionFound = true
   625  			}
   626  		}
   627  
   628  		if !regionFound {
   629  			unavailableRegions = append(unavailableRegions, requestedRegion)
   630  		}
   631  	}
   632  
   633  	// This means that the user has requested access to some regions that are not yet enabled,
   634  	// but could still be allocated to a region that is relatively close.
   635  	if len(unavailableRegions) != 0 && len(unavailableRegions) != len(requestedRegions) {
   636  		if metadata.GetAppEnvironment() == metadata.EnvProd {
   637  			logger.Errorf("User %s requested access to the following unavailable regions: %s. Trying to find instance on remaining available regions.", unsafeEmail, utils.PrintSlice(unavailableRegions))
   638  		}
   639  	}
   640  
   641  	// The user requested access to only unavailable regions. The last resort is to default to us-east-1.
   642  	if len(unavailableRegions) == len(requestedRegions) {
   643  		if metadata.GetAppEnvironment() == metadata.EnvProd {
   644  			logger.Errorf("User %s requested access to only unavailable regions: %s. Defaulting to us-east-1.", unsafeEmail, utils.PrintSlice(unavailableRegions))
   645  		}
   646  		availableRegions = []string{"us-east-1"}
   647  	}
   648  
   649  	if len(availableRegions) == 0 {
   650  		err := utils.MakeError("")
   651  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   652  			Error: REGION_NOT_ENABLED,
   653  		}, err)
   654  		return err
   655  	}
   656  
   657  	// This condition is to accomodate the worflow for developers of client_apps
   658  	// to test their changes without needing to update the development database with
   659  	// commit_hashes on their local machines.
   660  	if metadata.IsLocalEnv() || mandelboxRequest.CommitHash == CLIENT_COMMIT_HASH_DEV_OVERRIDE {
   661  		// Query for the latest image id
   662  		imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region) // TODO: set different provider when doing multi-cloud.
   663  		if err != nil {
   664  			return utils.MakeError("failed to query database for current image. Err: %v", err)
   665  		}
   666  
   667  		if len(imageResult) == 0 {
   668  			return utils.MakeError("Image not found on %v.", event.Region)
   669  		}
   670  
   671  		mandelboxRequest.CommitHash = string(imageResult[0].ClientSHA)
   672  	}
   673  
   674  	// This is the "main" loop that does all the work and tries to find an instance for a user. first, it will iterate
   675  	// over the list of regions provided on the request, and will query the database on each to return the list of
   676  	// instances with capacity on the current region. Once it gets the instances, it will iterate over them and try
   677  	// to find an instance with a matching commit hash. If it fails to do so, move on to the next region.
   678  	for _, region := range availableRegions {
   679  		logger.Infof("Trying to find instance for user %v in region %v, with commit hash %v. (client reported email %v, this value might not be accurate and is untrusted)",
   680  			unsafeEmail, region, mandelboxRequest.CommitHash, unsafeEmail)
   681  
   682  		instanceResult, err := s.DBClient.QueryInstanceWithCapacity(scalingCtx, s.GraphQLClient, region)
   683  		if err != nil {
   684  			return utils.MakeError("failed to query for instance with capacity. Err: %v", err)
   685  		}
   686  
   687  		if len(instanceResult) == 0 {
   688  			logger.Warningf("Failed to find an instance in %v for commit hash %v. Trying on next region.", region, mandelboxRequest.CommitHash)
   689  			continue
   690  		}
   691  
   692  		var instanceFound bool
   693  
   694  		// Iterate over available instances, try to find one with a matching commit hash
   695  		for i := range instanceResult {
   696  			assignedInstance = subscriptions.Instance{
   697  				ID:                string(instanceResult[i].ID),
   698  				IPAddress:         string(instanceResult[i].IPAddress),
   699  				Provider:          string(instanceResult[i].Provider),
   700  				Region:            string(instanceResult[i].Region),
   701  				ImageID:           string(instanceResult[i].ImageID),
   702  				ClientSHA:         string(instanceResult[i].ClientSHA),
   703  				Type:              string(instanceResult[i].Type),
   704  				RemainingCapacity: int64(instanceResult[i].RemainingCapacity),
   705  				Status:            string(instanceResult[i].Status),
   706  				CreatedAt:         instanceResult[i].CreatedAt,
   707  				UpdatedAt:         instanceResult[i].UpdatedAt,
   708  			}
   709  
   710  			if assignedInstance.ClientSHA == mandelboxRequest.CommitHash {
   711  				logger.Infof("Found instance %v for user %v with commit hash %v.", assignedInstance.ID, mandelboxRequest.UserEmail, assignedInstance.ClientSHA)
   712  				instanceFound = true
   713  				break
   714  			}
   715  		}
   716  
   717  		// Break of outer loop if instance was found. If no instance with
   718  		// matching commit hash was found, move on to the next region.
   719  		if instanceFound {
   720  			break
   721  		}
   722  	}
   723  
   724  	// No instances with capacity were found
   725  	if assignedInstance == (subscriptions.Instance{}) {
   726  		err := utils.MakeError("did not find an instance with capacity for user %v and commit hash %v.", mandelboxRequest.UserEmail, mandelboxRequest.CommitHash)
   727  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   728  			Error: NO_INSTANCE_AVAILABLE,
   729  		}, err)
   730  		return err
   731  	}
   732  
   733  	var (
   734  		// The parsed version from the config
   735  		parsedFrontendVersion *hashicorp.Version
   736  		// The parsed version from the request
   737  		parsedRequestVersion *hashicorp.Version
   738  		// whether the client app has an outdated version
   739  		isOutdatedClient bool
   740  	)
   741  
   742  	// Get the version we keep locally for comparing the incoming request value.
   743  	frontendVersion := getFrontendVersion()
   744  
   745  	// Parse the version with the `hashicorp/go-version` package so we can compare.
   746  	parsedFrontendVersion, err = hashicorp.NewVersion(frontendVersion)
   747  	if err != nil {
   748  		logger.Errorf("Failed parsing client app version from scaling algorithm config. Err: %v", err)
   749  	}
   750  
   751  	// Parse the version we got in the request.
   752  	parsedRequestVersion, err = hashicorp.NewVersion(mandelboxRequest.Version)
   753  	if err != nil {
   754  		logger.Errorf("Failed parsing client app version from request. Err: %v", err)
   755  	}
   756  
   757  	// Compare the request version with the one from the config. If the
   758  	// version from the request is less than the one we have locally, it
   759  	// means the request comes from an outdated frontend application.
   760  	if parsedFrontendVersion != nil && parsedRequestVersion != nil {
   761  		isOutdatedClient = parsedRequestVersion.LessThan(parsedFrontendVersion)
   762  	}
   763  
   764  	// There are instances with capacity available, but none of them with the desired commit hash.
   765  	// We only consider this error in cases when the client app has a version greater or equal than
   766  	// the one in the config database. This is because when the client version is lesser (outdated client),
   767  	// it will automatically update itself to the most recent version and send another request.
   768  	if assignedInstance.ClientSHA != mandelboxRequest.CommitHash {
   769  		var msg error
   770  		if !isOutdatedClient {
   771  			msg = utils.MakeError("found instance with capacity but it has a different commit hash %v than frontend with commit hash  %v", assignedInstance.ClientSHA, mandelboxRequest.CommitHash)
   772  		}
   773  
   774  		// Regardless if we log the error, its necessary to return an appropiate response.
   775  		mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   776  			Error: COMMIT_HASH_MISMATCH,
   777  		}, msg)
   778  		return msg
   779  	}
   780  
   781  	// Try to find a mandelbox in the WAITING status in the assigned instance.
   782  	mandelboxResult, err := s.DBClient.QueryMandelbox(scalingCtx, s.GraphQLClient, assignedInstance.ID, "WAITING")
   783  	if err != nil {
   784  		return utils.MakeError("failed to query database for mandelbox on instance %s. Err: %v", assignedInstance.ID, err)
   785  	}
   786  
   787  	if len(mandelboxResult) == 0 {
   788  		return utils.MakeError("failed to find a waiting mandelbox even though the instance %v had sufficient capacity.", assignedInstance.ID)
   789  	}
   790  
   791  	waitingMandelbox := mandelboxResult[0]
   792  	mandelboxID, err := uuid.Parse(string(waitingMandelbox.ID))
   793  	if err != nil {
   794  		return utils.MakeError("failed to parse mandelbox id for instance %v. Err: %v", assignedInstance.ID, err)
   795  	}
   796  
   797  	mandelboxForDb := subscriptions.Mandelbox{
   798  		ID:         types.MandelboxID(mandelboxID),
   799  		App:        string(waitingMandelbox.App),
   800  		InstanceID: assignedInstance.ID,
   801  		UserID:     mandelboxRequest.UserID,
   802  		SessionID:  utils.Sprintf("%v", mandelboxRequest.SessionID),
   803  		Status:     "ALLOCATED",
   804  		CreatedAt:  waitingMandelbox.CreatedAt,
   805  	}
   806  
   807  	// Allocate mandelbox on database so the host service can start downloading user configs
   808  	affectedRows, err := s.DBClient.UpdateMandelbox(scalingCtx, s.GraphQLClient, mandelboxForDb)
   809  	if err != nil {
   810  		return utils.MakeError("error while inserting mandelbox to database. Err: %v", err)
   811  	}
   812  
   813  	logger.Infof("Inserted %v rows to database.", affectedRows)
   814  
   815  	if assignedInstance.RemainingCapacity <= 0 {
   816  		// This should never happen, but we should consider
   817  		// possible edge cases before updating the database.
   818  		return utils.MakeError("instance with id %v has a remaning capacity less than or equal to 0.", assignedInstance.ID)
   819  	}
   820  
   821  	// Subtract 1 from the current instance capacity because we allocated a mandelbox
   822  	updatedCapacity := assignedInstance.RemainingCapacity - 1
   823  	assignedInstance.RemainingCapacity = updatedCapacity
   824  
   825  	affectedRows, err = s.DBClient.UpdateInstance(scalingCtx, s.GraphQLClient, assignedInstance)
   826  	if err != nil {
   827  		return utils.MakeError("error while updating instance capacity on database. Err: %v", err)
   828  	}
   829  
   830  	logger.Infof("Updated %v rows in database.", affectedRows)
   831  
   832  	// Parse IP address. The database uses the CIDR notation (192.0.2.0/24)
   833  	// so we need to extract the address and send it to the frontend.
   834  	ip, _, err := net.ParseCIDR(assignedInstance.IPAddress)
   835  	if err != nil {
   836  		return utils.MakeError("failed to parse IP address %v. Err: %v", assignedInstance.IPAddress, err)
   837  	}
   838  
   839  	// Return result to assign request
   840  	mandelboxRequest.ReturnResult(httputils.MandelboxAssignRequestResult{
   841  		IP:          ip.String(),
   842  		MandelboxID: types.MandelboxID(mandelboxID),
   843  	}, nil)
   844  
   845  	return nil
   846  }
