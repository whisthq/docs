     1  /*
     2  Package mandelbox provides the abstractions of mandelboxes managed by Whist.
     3  It defines the types and implements lots of functionality associated with
     4  mandelboxes. In `tracker.go`, it also keeps a list of all mandelboxes on this
     5  host at any given point.
     6  
     7  This package, and its children, are meant to be low-level enough that it can
     8  be imported by higher-level host service packages.
     9  
    10  Note: Many of the methods that access the mandelbox data use a lock, please follow the
    11  guidelines outlined below when writing code that directly uses the struct fields:
    12  
    13  1. Are you accessing a mandelbox struct field directly? If so, lock.
    14  2. Are you calling a method? If so, do not lock or you may cause a deadlock.
    15  3. Are you inside a method? If so, assume the lock is unlocked on method entry.
    16  
    17  Additionally, consider the following rules:
    18    - All locking should be done when and where the actual data access occurs.
    19      This means at the point of accessing the actual struct field
    20    - No method should assume that the lock is locked before entry
    21    - Getters and setters should always be used for struct field access when possible
    22  */

   146  	// We create a context for this mandelbox specifically.
   147  	ctx, cancel := context.WithCancel(baseCtx)
   148  
   149  	mandelbox := &mandelboxData{
   150  		ctx:                  ctx,
   151  		cancel:               cancel,
   152  		ID:                   fid,
   153  		uinputDeviceMappings: []dockercontainer.DeviceMapping{},
   154  		otherDeviceMappings:  []dockercontainer.DeviceMapping{},
   155  	}
   156  
   157  	mandelbox.createResourceMappingDir()
   158  
   159  	trackMandelbox(mandelbox)
   160  
   161  	// Mount /dev/fuse for in-mandelbox FUSE filesystems to work
   162  	mandelbox.otherDeviceMappings = append(mandelbox.otherDeviceMappings, dockercontainer.DeviceMapping{
   163  		PathOnHost:        "/dev/fuse",
   164  		PathInContainer:   "/dev/fuse",
   165  		CgroupPermissions: "rwm", // read, write, mknod (the default)
   166  	})
   167  
   168  	// We start a goroutine that cleans up this Mandelbox as soon as its
   169  	// context is cancelled. This ensures that we automatically clean up when
   170  	// either `Close()` is called for this mandelbox, or the global context is
   171  	// cancelled.
   172  	goroutineTracker.Add(1)
   173  	go func() {
   174  		defer goroutineTracker.Done()
   175  
   176  		<-ctx.Done()
   177  
   178  		// Mark mandelbox as dying in the database, but only if it's not a warmup
   179  		if fid != types.MandelboxID(utils.PlaceholderWarmupUUID()) {
   180  			if err := dbdriver.WriteMandelboxStatus(mandelbox.GetID(), dbdriver.MandelboxStatusDying); err != nil {
   181  				logger.Error(err)
   182  			}
   183  		}
   184  
   185  		untrackMandelbox(mandelbox)
   186  		logger.Infof("Successfully untracked mandelbox %s", mandelbox.GetID())
   187  
   188  		mandelbox.rwlock.Lock()
   189  
   190  		// Free port bindings
   191  		portbindings.Free(mandelbox.portBindings)
   192  		mandelbox.portBindings = nil
   193  		logger.Infof("Successfully freed port bindings for mandelbox %s", mandelbox.GetID())
   194  
   195  		// Free uinput devices
   196  		mandelbox.uinputDevices.Close()
   197  		mandelbox.uinputDevices = nil
   198  		mandelbox.uinputDeviceMappings = []dockercontainer.DeviceMapping{}
   199  		logger.Infof("Successfully freed uinput devices for mandelbox %s", mandelbox.GetID())
   200  
   201  		// Free TTY
   202  		ttys.Free(mandelbox.tty)
   203  		logger.Infof("Successfully freed TTY %v for mandelbox %s", mandelbox.tty, mandelbox.GetID())
   204  		mandelbox.tty = 0
   205  
   206  		// CI does not have GPUs
   207  		if !metadata.IsRunningInCI() {
   208  			if err := gpus.Free(mandelbox.gpuIndex, mandelbox.GetID()); err != nil {
   209  				logger.Errorf("Error freeing GPU %v for mandelbox %s: %s", mandelbox.gpuIndex, mandelbox.GetID(), err)
   210  			} else {
   211  				logger.Infof("Successfully freed GPU %v for mandelbox %s", mandelbox.gpuIndex, mandelbox.GetID())
   212  			}
   213  		}
   214  
   215  		mandelbox.rwlock.Unlock()
   216  
   217  		// Clean resource mappings
   218  		mandelbox.cleanResourceMappingDir()
   219  		logger.Infof("Successfully cleaned resource mapping dir for mandelbox %s", mandelbox.GetID())
   220  
   221  		// Backup and clean user config directory.
   222  		err := mandelbox.BackupUserConfigs()
   223  		if err != nil {
   224  			logger.Errorf("Error backing up user configs for MandelboxID %s. Error: %s", mandelbox.GetID(), err)
   225  		} else {
   226  			logger.Infof("Successfully backed up user configs for MandelboxID %s", mandelbox.GetID())
   227  		}
   228  		mandelbox.cleanUserConfigDir()
   229  
   230  		// Remove mandelbox from the database altogether, once again excluding warmups
   231  		if fid != types.MandelboxID(utils.PlaceholderWarmupUUID()) {
   232  			if err := dbdriver.RemoveMandelbox(mandelbox.GetID()); err != nil {
   233  				logger.Error(err)
   234  			}
   235  		}
   236  
   237  		logger.Infof("Cleaned up after Mandelbox %s", mandelbox.GetID())
   238  	}()

   465  // InitializeUinputDevices tries to assign uinput devices to the mandelbox.
   466  func (mandelbox *mandelboxData) InitializeUinputDevices(goroutineTracker *sync.WaitGroup) error {
   467  	devices, mappings, err := uinputdevices.Allocate()
   468  	if err != nil {
   469  		return utils.MakeError("Couldn't allocate uinput devices: %s", err)
   470  	}
   471  
   472  	mandelbox.rwlock.Lock()
   473  	defer mandelbox.rwlock.Unlock()
   474  
   475  	mandelbox.uinputDevices = devices
   476  	mandelbox.uinputDeviceMappings = mappings
   477  
   478  	goroutineTracker.Add(1)
   479  	go func() {
   480  		defer goroutineTracker.Done()
   481  		socketPath := path.Join(utils.TempDir, mandelbox.GetID().String(), "/sockets/uinput.sock")
   482  		err := uinputdevices.SendDeviceFDsOverSocket(mandelbox.ctx, goroutineTracker, devices, socketPath)
   483  		if err != nil {
   484  			placeholderUUID := types.MandelboxID(utils.PlaceholderWarmupUUID())
   485  			if mandelbox.GetID() == placeholderUUID && strings.Contains(err.Error(), "use of closed network connection") {
   486  				logger.Warningf("SendDeviceFDsOverSocket returned for MandelboxID %s with error: %s", mandelbox.GetID(), err)
   487  			} else {
   488  				logger.Errorf("SendDeviceFDsOverSocket returned for MandelboxID %s with error: %s", mandelbox.GetID(), err)
   489  			}
   490  		} else {
   491  			logger.Infof("SendDeviceFDsOverSocket returned successfully for MandelboxID %s", mandelbox.GetID())
   492  		}
   493  	}()
   494  
   495  	return nil
   496  }
