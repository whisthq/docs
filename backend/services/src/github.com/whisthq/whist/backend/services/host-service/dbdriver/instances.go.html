    34  // is not found, then it returns an error. This function also starts the

    36  func RegisterInstance() error {
    37  	if !enabled {
    38  		return nil
    39  	}
    40  	if dbpool == nil {
    41  		return utils.MakeError("registerInstance() called but dbdriver is not initialized!")
    42  	}
    43  
    44  	publicIP4, err := aws.GetPublicIpv4()
    45  	if err != nil {
    46  		return utils.MakeError("Couldn't register instance: couldn't get public IPv4: %s", err)
    47  	}
    48  	imageID, err := aws.GetAmiID()
    49  	if err != nil {
    50  		return utils.MakeError("Couldn't register instance: couldn't get AMI ID: %s", err)
    51  	}
    52  	region, err := aws.GetPlacementRegion()
    53  	if err != nil {
    54  		return utils.MakeError("Couldn't register instance: couldn't get AWS Placement Region: %s", err)
    55  	}
    56  	instanceType, err := aws.GetInstanceType()
    57  	if err != nil {
    58  		return utils.MakeError("Couldn't register instance: couldn't get AWS Instance type: %s", err)
    59  	}
    60  	instanceID, err := aws.GetInstanceID()
    61  	if err != nil {
    62  		return utils.MakeError("Couldn't register instance: couldn't get AWS Instance id: %s", err)
    63  	}
    64  	// Create a transaction to register the instance, since we are querying and
    65  	// writing separately.
    66  	tx, err := dbpool.BeginTx(context.Background(), pgx.TxOptions{IsoLevel: pgx.ReadCommitted})
    67  	if err != nil {
    68  		return utils.MakeError("Couldn't register instance: unable to begin transaction: %s", err)
    69  	}
    70  	// Safe to do even if committed -- see tx.Rollback() docs.
    71  	defer tx.Rollback(context.Background())
    72  
    73  	// Check if there's a row for us in the database already
    74  	q := queries.NewQuerier(tx)
    75  
    76  	rows, err := q.FindInstanceByID(context.Background(), string(instanceID))
    77  	if err != nil {
    78  		return utils.MakeError("RegisterInstance(): Error running query: %s", err)
    79  	}
    80  
    81  	// Since the `instance_name` is the primary key of `whist.instances`,
    82  	// we know that `rows` ought to contain either 0 or 1 results.
    83  	if len(rows) == 0 {
    84  		return utils.MakeError("RegisterInstance(): Existing row for this instance not found in the database.")
    85  	}
    86  
    87  	// Verify that the properties of the existing row are actually as we expect
    88  	if rows[0].ImageID.String != string(imageID) {
    89  		return utils.MakeError(`RegisterInstance(): Existing database row found, but AMI differs. Expected "%s", Got "%s"`, imageID, rows[0].ImageID.String)
    90  	}
    91  	if rows[0].Region.String != string(region) {
    92  		return utils.MakeError(`RegisterInstance(): Existing database row found, but location differs. Expected "%s", Got "%s"`, region, rows[0].Region.String)
    93  	}
    94  	if !(rows[0].ClientSha.Status == pgtype.Present && strings.HasPrefix(metadata.GetGitCommit(), rows[0].ClientSha.String)) {
    95  		// This is the only string where we have to check status, since an empty string is a prefix for anything.
    96  		return utils.MakeError(`RegisterInstance(): Existing database row found, but commit hash differs. Expected "%s", Got "%s"`, metadata.GetGitCommit(), rows[0].ClientSha.String)
    97  	}
    98  	if rows[0].InstanceType.String != string(instanceType) {
    99  		return utils.MakeError(`RegisterInstance(): Existing database row found, but AWS instance type differs. Expected "%s", Got "%s"`, instanceType, rows[0].InstanceType.String)
   100  	}
   101  	if rows[0].Status.String != string(InstanceStatusPreConnection) {
   102  		return utils.MakeError(`RegisterInstance(): Existing database row found, but status differs. Expected "%s", Got "%s"`, InstanceStatusPreConnection, rows[0].Status.String)
   103  	}
   104  
   105  	// There is an existing row in the database for this instance --- we now "take over" and update it with the correct information.
   106  	result, err := q.RegisterInstance(context.Background(), queries.RegisterInstanceParams{
   107  		Provider: pgtype.Varchar{
   108  			String: string("AWS"),
   109  			Status: pgtype.Present,
   110  		},
   111  		Region: pgtype.Varchar{
   112  			String: string(region),
   113  			Status: pgtype.Present,
   114  		},
   115  		ImageID: pgtype.Varchar{
   116  			String: string(imageID),
   117  			Status: pgtype.Present,
   118  		},
   119  		ClientSha: pgtype.Varchar{
   120  			String: string(metadata.GetGitCommit()),
   121  			Status: pgtype.Present,
   122  		},
   123  		IpAddr: pgtype.Inet{
   124  			IPNet: &net.IPNet{
   125  				IP: publicIP4,
   126  			},
   127  			Status: pgtype.Present,
   128  		},
   129  		InstanceType: pgtype.Varchar{
   130  			String: string(instanceType),
   131  			Status: pgtype.Present,
   132  		},
   133  		RemainingCapacity: rows[0].RemainingCapacity,
   134  		Status: pgtype.Varchar{
   135  			String: string(InstanceStatusActive),
   136  			Status: pgtype.Present,
   137  		},
   138  		CreatedAt: pgtype.Timestamptz{
   139  			Time:   time.Now(),
   140  			Status: pgtype.Present,
   141  		},
   142  		UpdatedAt: pgtype.Timestamptz{
   143  			Time:   time.Now(),
   144  			Status: pgtype.Present,
   145  		},
   146  		InstanceID: string(instanceID),
   147  	})
   148  	if err != nil {
   149  		return utils.MakeError("Couldn't register instance: error updating existing row in table `whist.instances`: %s", err)
   150  	} else if result.RowsAffected() == 0 {
   151  		return utils.MakeError("Couldn't register instance in database: row went missing!")
   152  	}
   153  	tx.Commit(context.Background())
   154  
   155  	logger.Infof("Successfully registered %s instance in database.", instanceID)
   156  
   157  	// Initialize the heartbeat goroutine. Note that this goroutine does not
   158  	// listen to the global context, and is not tracked by the global
   159  	// goroutineTracker. This is because we want the heartbeat goroutine to
   160  	// outlive the global context. Instead, we have the explicit Close(), which
   161  	// ends up finishing off the heartbeat goroutine as well.
   162  	go heartbeatGoroutine()

   164  	return nil
