    16  func init() {
    17  	// Define the logic for filtering messages according to their level.
    18  	// For Logzio we want all messages, but for Sentry only the errors
    19  	// are considered. In addition, we disable debug messages (such as metrics)
    20  	// in console output to avoid polluting stdout.
    21  	onlyErrors := zap.LevelEnablerFunc(func(lvl zapcore.Level) bool {
    22  		return lvl >= zapcore.ErrorLevel
    23  	})
    24  	logsAndErrors := zap.LevelEnablerFunc(func(lvl zapcore.Level) bool {
    25  		return lvl >= zapcore.InfoLevel
    26  	})
    27  	allLevels := zap.LevelEnablerFunc(func(lvl zapcore.Level) bool {
    28  		return true
    29  	})
    30  
    31  	// Define a stdout with Locking so that the logging methods are
    32  	// safe for concurrent use.
    33  	consoleDebugging := zapcore.Lock(os.Stdout)
    34  
    35  	// Define configurations for each core.
    36  	sentryEncoderConfig := newSentryEncoderConfig()
    37  	logzEncoderConfig := newLogzioEncoderConfig()
    38  	consoleEncoderConfig := zap.NewDevelopmentEncoderConfig()
    39  
    40  	// Enable colored output on stdout
    41  	consoleEncoderConfig.EncodeLevel = zapcore.CapitalColorLevelEncoder
    42  
    43  	// Create the new encoders for each core.
    44  	sentryEncoder := zapcore.NewJSONEncoder(sentryEncoderConfig)
    45  	logzEncoder := zapcore.NewJSONEncoder(logzEncoderConfig)
    46  	consoleEncoder := zapcore.NewConsoleEncoder(consoleEncoderConfig)
    47  
    48  	logzCore := newLogzioCore(logzEncoder, allLevels)
    49  	sentryCore := newSentryCore(sentryEncoder, onlyErrors)
    50  
    51  	// Join the outputs, encoders, and level-handling functions into
    52  	// zapcore.Cores, then tee the four cores together.
    53  	core := zapcore.NewTee(
    54  		logzCore,
    55  		sentryCore,
    56  		zapcore.NewCore(consoleEncoder, consoleDebugging, logsAndErrors),
    57  	)
    58  
    59  	// Once everything is configured, instanciate the logger.
    60  	logger = zap.New(core)
    61  }

    92  // Panic sends an error to Sentry and "pretends" to panic on it by printing the
    93  // stack trace and calling the provided global context-cancelling function.
    94  // This causes all the goroutines in the program to kill themselves (cleanly).
    95  // This function should not be used except to initiate termination of the
    96  // entire host service. Note that passing in a nil first argument would cause
    97  // this function to _actually_ panic, and if we're gonna panic we might as well
    98  // do so in a useful way. Therefore, passing in a nil `globalCancel` parameter
    99  // will just panic on `err` instead.
   100  func Panic(globalCancel context.CancelFunc, err error) {
   101  	PrintStackTrace()
   102  
   103  	if globalCancel != nil {
   104  		Error(err)
   105  		globalCancel()
   106  	} else {
   107  		// If we're truly trying to panic, let's at least flush our logging queues
   108  		// first so this error actually gets sent.
   109  		logger.Sync()
   110  		logger.Sugar().Panic(err)
   111  	}
   112  }
