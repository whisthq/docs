    19  func init() {
    20  	// Set some options for the Go logging package.
    21  	log.Default().SetFlags(log.Ldate | log.Lmicroseconds | log.LUTC)
    22  
    23  	// The first thing we want to do is to initialize logzio and Sentry so that
    24  	// we can catch any errors that might occur, or logs if we print them.
    25  	// The credentials for Sentry are hardcoded into the host-service,
    26  	// and the credentials for logzio are written to the environment file used
    27  	// by host-service systemd service.
    28  	// It is more likely that logzio initialization would fail, So we initialize
    29  	// Sentry first.
    30  
    31  	// Note that according to the Sentry Go documentation, if we run
    32  	// sentry.CaptureMessage or sentry.CaptureError on separate goroutines, they
    33  	// can overwrite each other's tags. The thread-safe solution is to use
    34  	// goroutine-local hubs, but the way to do that would be to use contexts and
    35  	// add an additional argument to each logging method taking in the current
    36  	// context. This seems like a lot of work, so we just use a set of global
    37  	// tags instead, initializing them in InitializeSentry() and not touching
    38  	// them afterwards. Any mandelbox-specific information (which is what I
    39  	// imagine we would use local tags for) we just add in the text of the
    40  	// respective error message sent to Sentry. Alternatively, we might just be
    41  	// able to use sentry.WithScope(), but that is future work.
    42  }

   215  // Panic sends an error to Sentry and "pretends" to panic on it by printing the
   216  // stack trace and calling the provided global context-cancelling function.
   217  // This causes all the goroutines in the program to kill themselves (cleanly).
   218  // This function should not be used except to initiate termination of the
   219  // entire host service. Note that passing in a nil first argument would cause
   220  // this function to _actually_ panic, and if we're gonna panic we might as well
   221  // do so in a useful way. Therefore, passing in a nil `globalCancel` parameter
   222  // will just panic on `err` instead.
   223  func Panic(globalCancel context.CancelFunc, err error) {
   224  	errstr := fmt.Sprintf("PANIC: %s", err)
   225  	if logzioTransport != nil {
   226  		logzioTransport.send(errstr, logzioTypeError)
   227  	}
   228  	if sentryTransport != nil {
   229  		sentryTransport.send(err)
   230  	}
   231  	PrintStackTrace()
   232  
   233  	if globalCancel != nil {
   234  		log.Print(utils.ColorRed(errstr))
   235  		globalCancel()
   236  	} else {
   237  		// If we're truly trying to panic, let's at least flush our logging queues
   238  		// first so this error actually gets sent.
   239  		FlushLogzio()
   240  		FlushSentry()
   241  		log.Panicf(utils.ColorRed(errstr))
   242  	}
   243  }
