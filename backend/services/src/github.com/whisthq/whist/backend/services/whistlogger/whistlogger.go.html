    16  func init() {
    17  	// Set some options for the Go logging package.
    18  	log.Default().SetFlags(log.Ldate | log.Lmicroseconds | log.LUTC)
    19  
    20  	// The first thing we want to do is to initialize logzio and Sentry so that
    21  	// we can catch any errors that might occur, or logs if we print them.
    22  	// The credentials for Sentry are hardcoded into the host-service,
    23  	// and the credentials for logzio are written to the environment file used
    24  	// by host-service systemd service.
    25  	// It is more likely that logzio initialization would fail, So we initialize
    26  	// Sentry first.
    27  
    28  	// We declare error separately to avoid shadowing sentryTransport.
    29  	var err error
    30  
    31  	// Note that according to the Sentry Go documentation, if we run
    32  	// sentry.CaptureMessage or sentry.CaptureError on separate goroutines, they
    33  	// can overwrite each other's tags. The thread-safe solution is to use
    34  	// goroutine-local hubs, but the way to do that would be to use contexts and
    35  	// add an additional argument to each logging method taking in the current
    36  	// context. This seems like a lot of work, so we just use a set of global
    37  	// tags instead, initializing them in InitializeSentry() and not touching
    38  	// them afterwards. Any mandelbox-specific information (which is what I
    39  	// imagine we would use local tags for) we just add in the text of the
    40  	// respective error message sent to Sentry. Alternatively, we might just be
    41  	// able to use sentry.WithScope(), but that is future work.
    42  	sentryTransport, err = initializeSentry()
    43  	if err != nil {
    44  		// Error, don't Panic, since a Sentry outage should not bring down our
    45  		// entire service.
    46  		Errorf("Failed to initialize Sentry! Error: %s", err)
    47  	}
    48  
    49  	logzioTransport, err = initializeLogzIO()
    50  	if err != nil {
    51  		// Error, don't Panic, since a logzio outage should not bring down our
    52  		// entire service.
    53  		Errorf("Failed to initialize LogzIO! Error: %s", err)
    54  	}
    55  }

   119  // Panic sends an error to Sentry and "pretends" to panic on it by printing the
   120  // stack trace and calling the provided global context-cancelling function.
   121  // This causes all the goroutines in the program to kill themselves (cleanly).
   122  // This function should not be used except to initiate termination of the
   123  // entire host service. Note that passing in a nil first argument would cause
   124  // this function to _actually_ panic, and if we're gonna panic we might as well
   125  // do so in a useful way. Therefore, passing in a nil `globalCancel` parameter
   126  // will just panic on `err` instead.
   127  func Panic(globalCancel context.CancelFunc, err error) {
   128  	errstr := fmt.Sprintf("PANIC: %s", err)
   129  	if logzioTransport != nil {
   130  		logzioTransport.send(errstr, logzioTypeError)
   131  	}
   132  	if sentryTransport != nil {
   133  		sentryTransport.send(err)
   134  	}
   135  	PrintStackTrace()
   136  
   137  	if globalCancel != nil {
   138  		log.Print(utils.ColorRed(errstr))
   139  		globalCancel()
   140  	} else {
   141  		// If we're truly trying to panic, let's at least flush our logging queues
   142  		// first so this error actually gets sent.
   143  		FlushLogzio()
   144  		FlushSentry()
   145  		log.Panicf(utils.ColorRed(errstr))
   146  	}
   147  }
