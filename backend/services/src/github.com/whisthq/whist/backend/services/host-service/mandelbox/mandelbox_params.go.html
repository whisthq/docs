    90  	file, err := os.OpenFile(mandelbox.getResourceMappingDir()+filename, os.O_CREATE|os.O_RDWR|os.O_TRUNC, 0777)
    91  	if err != nil {
    92  		return utils.MakeError("Unable to create file %s to store resource assignment. Error: %v", filename, err)
    93  	}
    94  	// Instead of deferring the close() and sync() of the file, as is
    95  	// conventional, we do it at the end of the function to avoid some annoying
    96  	// linter errors
    97  	_, err = file.WriteString(data)
    98  	if err != nil {
    99  		return utils.MakeError("Couldn't write assignment with data %s to file %s. Error: %v", data, filename, err)
   100  	}
   101  
   102  	err = file.Sync()
   103  	if err != nil {
   104  		return utils.MakeError("Couldn't sync file %s. Error: %v", filename, err)
   105  	}
   106  
   107  	err = file.Close()
   108  	if err != nil {
   109  		return utils.MakeError("Couldn't close file %s. Error: %v", filename, err)
   110  	}
   111  
   112  	logger.Info("Wrote data \"%s\" to file %s\n", data, filename)
