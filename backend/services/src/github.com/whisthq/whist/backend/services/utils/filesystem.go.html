    14  // WaitForFileCreation blocks until the provided filename is created in the
    15  // provided directory, or the timeout duration elapses. If the target file is
    16  // created in time, a nil error is returned. If the timeout elapses, a
    17  // context.DeadlineExceeded error is returned. In any other case, a non-nil
    18  // error is returned explaining what went wrong.
    19  //
    20  // For maximum correctness, we require that any paths passed in are absolute.
    21  // This may not be strictly necessary, but the documentation for `fsnotify` and
    22  // `path/filepath` are just vague enough for me to enforce this rule for the
    23  // callers. This rule may be subject to relaxation in the future.
    24  //
    25  // The function accepts a pointer to a fsnotify watcher. If the caller passes in
    26  // nil then we will create a new watcher and handle the clean up. If a watcher
    27  // is passed by the caller then they are expected to clean up their watcher.
    28  //
    29  // NOTE: Each invocation of this function creates an `inotify` instance and
    30  // holds onto it for the duration of this function call. By default, our
    31  // instances have a limit of 128 watchers per user. Therefore, we bump this
    32  // limit in host-setup scripts to prevent it from being a limiting factor for our
    33  // mandelbox launches.
    34  func WaitForFileCreation(absParentDirectory, fileName string, timeout time.Duration, watcher *fsnotify.Watcher) error {
    35  	if !path.IsAbs(absParentDirectory) {
    36  		return MakeError("Can't pass non-absolute paths into WaitForFileCreation")
    37  	}
    38  	targetFileName := path.Join(absParentDirectory, fileName)
    39  
    40  	var err error
    41  	if watcher == nil {
    42  		watcher, err = fsnotify.NewWatcher()
    43  		if err != nil {
    44  			return MakeError("Couldn't create new fsnotify.Watcher: %s", err)
    45  		}
    46  		defer watcher.Close()
    47  	}
    48  
    49  	// Check if the file has already been created before watcher watches for file
    50  	if _, err := os.Stat(targetFileName); err == nil {
    51  		return nil
    52  	}
    53  
    54  	if err = watcher.Add(absParentDirectory); err != nil {
    55  		return MakeError("Error adding dir %s to fsnotify.Watcher: %s", absParentDirectory, err)
    56  	}
    57  
    58  	// Check if the file has already been created before watcher watches for file
    59  	if _, err := os.Stat(targetFileName); err == nil {
    60  		return nil
    61  	}
    62  
    63  	if err = waitForErrorOrCreation(timeout, targetFileName, watcher.Events, watcher.Errors); err != nil {
    64  		return MakeError("Error waiting for file creation: %v", err)
    65  	}
    66  
    67  	return nil
    68  }

    70  // waitForErrorOrCreation will handle watcher events, errors, and timeouts that occur
    71  func waitForErrorOrCreation(timeout time.Duration, targetFileName string, watcherEvent chan fsnotify.Event, watcherErr chan error) error {
    72  	// Create timer here
    73  	timer := time.NewTimer(timeout)
    74  
    75  	for {
    76  		select {
    77  		case <-timer.C:
    78  			return context.DeadlineExceeded
    79  
    80  		case err, ok := <-watcherErr:
    81  			if !ok {
    82  				return MakeError("fsnotify.Watcher error channel closed with error: %v", err)
    83  			}
    84  			// Note that for us, dropped events _are_ errors, since we should not be
    85  			// having nearly enough filesystem activity to drop any events.
    86  			return MakeError("returned error: %s", err)
    87  
    88  		case ev, ok := <-watcherEvent:
    89  			if !ok {
    90  				return MakeError("fsnotify.Watcher events channel closed.")
    91  			}
    92  			// TODO: remove this log once we're confident this part of the code works
    93  			log.Printf("fsnotify.Watched filesystem event: %+v", ev)
    94  			// Check if it's a creation event that matches the filename we expect
    95  			if ev.Op&fsnotify.Create == fsnotify.Create && ev.Name == targetFileName {
    96  				return nil
    97  			}
    98  		}
    99  	}
   100  }
