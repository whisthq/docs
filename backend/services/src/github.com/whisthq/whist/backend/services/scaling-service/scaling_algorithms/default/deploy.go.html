   139  	// We time out here in case something went wrong with the
   140  	// upgrade image action, in which case we roll back the new version.

   142  	case <-s.SyncChan:

   160  	// This function updates the value inside the config file, so we can keep  track
   161  	// of the current version locally, it does not update the value in the database.
   162  	setFrontendVersion(version)
