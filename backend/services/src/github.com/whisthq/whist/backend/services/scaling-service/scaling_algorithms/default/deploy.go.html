    63  	// creating them on a cloud provider. In any other case we call the host handler to create

    78  		}
    79  	}

    82

    85  	if err != nil {
    86  		return utils.MakeError("failed to insert instances into database: %s", err)
    87  	}
    88

    90

    92  	// during deploy, as the active image will be switched until the frontend has updated

   105  			Region:    event.Region,

   136  // SwapOverImages is a scaling action that will switch the current image on the given region.
   137  // To the latest one. This is done separately to avoid having downtimes during deploys, since
   138  // we have to wait until the frontend has updated its version on the config database.
   139  func (s *DefaultScalingAlgorithm) SwapOverImages(scalingCtx context.Context, event ScalingEvent, clientVersion interface{}) error {
   140  	contextFields := []interface{}{
   141  		zap.String("id", event.ID),
   142  		zap.Any("type", event.Type),
   143  		zap.String("region", event.Region),
   144  	}
   145  	// Block until the image upgrade has finished successfully.
   146  	// We time out here in case something went wrong with the
   147  	// upgrade image action, in which case we roll back the new version.
   148  	select {
   149  	case <-s.SyncChan:
   150  		logger.Infow("Got signal that image upgrade action finished correctly.", contextFields)
   151  	case <-time.After(deployTimeoutInMinutes):
   152  		// Clear protected map since the image upgrade didn't complete successfully.
   153  		s.protectedMapLock.Lock()
   154  		s.protectedFromScaleDown = make(map[string]subscriptions.Image)
   155  		s.protectedMapLock.Unlock()
   156  
   157  		return utils.MakeError("timed out waiting for image upgrade to finish.")
   158  	}

   161  	defer logger.Infow("Finished image swapover action.", contextFields)
   162  
   163  	// version is the entry we receive from the config database
   164  	version := clientVersion.(subscriptions.FrontendVersion)
   165  
   166  	// Update the internal version with the new one received from the database.
   167  	// This function updates the value inside the config file, so we can keep  track
   168  	// of the current version locally, it does not update the value in the database.
   169  	setFrontendVersion(version)
   170

   172  		commitHash string
   173  		newImage   subscriptions.Image
   174  		newImageID string
   175  		oldImageID string
   176  	)
   177

   179  	switch metadata.GetAppEnvironmentLowercase() {
   180  	case string(metadata.EnvDev):
   181  		commitHash = version.DevCommitHash
   182  	case string(metadata.EnvStaging):
   183  		commitHash = version.StagingCommitHash
   184  	case string(metadata.EnvProd):
   185  		commitHash = version.ProdCommitHash
   186  	default:
   187  		commitHash = version.DevCommitHash
   188  	}

   190  	// Acquire lock on protected from scale down map
   191  	s.protectedMapLock.Lock()
   192  	defer s.protectedMapLock.Unlock()
   193

   195  	for _, image := range s.protectedFromScaleDown {
   196  		if image.ClientSHA == commitHash {
   197  			newImage = image
   198  			newImageID = image.ImageID
   199  			break
   200  		}
   201  	}
   202  
   203  	if newImage == (subscriptions.Image{}) {
   204  		return utils.MakeError("did not find protected image with commit hash %s. Not performing image swapover.", commitHash)
   205  	}
   206  
   207  	// Query for the current image id
   208  	imageResult, err := s.DBClient.QueryImage(scalingCtx, s.GraphQLClient, "AWS", event.Region)
   209  	if err != nil {
   210  		return utils.MakeError("failed to query database for current image: %s", err)
   211  	}
   212

   214  		logger.Warningf("Image doesn't exist in %s. Creating a new entry with image %s.", event.Region, newImageID)
   215  	} else {
   216  		// We now consider the "current" image as the "old" image
   217  		oldImageID = string(imageResult[0].ImageID)
   218  	}
   219  
   220  	// swapover active image on database
   221  	logger.Infow(utils.Sprintf("Updating old %s image %s to new image %s on database.", event.Region, oldImageID, newImageID), contextFields)
   222

   224  		_, err = s.DBClient.InsertImages(scalingCtx, s.GraphQLClient, []subscriptions.Image{newImage})
   225  		if err != nil {
   226  			return utils.MakeError("failed to insert image into database: %s", err)
   227  		}
   228  	} else {
   229  		_, err = s.DBClient.UpdateImage(scalingCtx, s.GraphQLClient, newImage)
   230  		if err != nil {
   231  			return utils.MakeError("failed to update image in database: %s", err)
   232  		}
   233  	}
