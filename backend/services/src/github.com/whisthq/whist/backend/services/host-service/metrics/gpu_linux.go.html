    34  // Calculate GPU utilization and memory usage.
    35  func (newMetrics *RuntimeMetrics) collectGPUMetrics() []error {
    36  	if numGPUs, nvmlRet := nvml.DeviceGetCount(); nvmlRet != nvml.SUCCESS {
    37  		newMetrics.NumberOfGPUs = -1
    38  		return []error{utils.MakeError("Error getting GPU count")}
    39  	} else {
    40  		newMetrics.NumberOfGPUs = numGPUs
    41  		errs := make([]error, 0)
    42  
    43  		var totalVramKB uint64 = 0
    44  		var usedVramKB uint64 = 0
    45  		var totalKernelUtilPercent uint32 = 0
    46  		var totalMemBandwidthUtilPercent uint32 = 0
    47  		for i := 0; i < numGPUs; i++ {
    48  			if device, nvmlRet := nvml.DeviceGetHandleByIndex(i); nvmlRet != nvml.SUCCESS {
    49  				errs = append(errs, utils.MakeError("Couldn't get NVIDIA device at index %v", i))
    50  				break
    51  			} else {
    52  				if memInfo, nvmlRet := nvml.DeviceGetMemoryInfo(device); nvmlRet != nvml.SUCCESS {
    53  					errs = append(errs, utils.MakeError("Couldn't get video memory info for device at NVML index %v", i))
    54  					break
    55  				} else {
    56  					totalVramKB += memInfo.Total / 1024
    57  					usedVramKB += memInfo.Used / 1024
    58  					// For some reason, NVML always returns 0 for memInfo.Free, so we do
    59  					// the free calculation ourselves below.
    60  				}
    61  
    62  				if util, nvmlRet := nvml.DeviceGetUtilizationRates(device); nvmlRet != nvml.SUCCESS {
    63  					errs = append(errs, utils.MakeError("Couldn't get utilization info for device at NVML index %v", i))
    64  					break
    65  				} else {
    66  					totalKernelUtilPercent += util.Gpu
    67  					totalMemBandwidthUtilPercent += util.Memory
    68  				}
    69  			}
    70  		}
    71  
    72  		newMetrics.TotalVideoMemoryKB = totalVramKB
    73  		newMetrics.UsedVideoMemoryKB = usedVramKB
    74  		newMetrics.FreeVideoMemoryKB = totalVramKB - usedVramKB
    75  
    76  		newMetrics.GPUKernelUtilizationPercent = float64(totalKernelUtilPercent) / float64(numGPUs)
    77  		newMetrics.GPUMemoryBandwidthUtilizationPercent = float64(totalMemBandwidthUtilPercent) / float64(numGPUs)
    78  		return errs
    79  	}
