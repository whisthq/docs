    25  func (newMetrics *RuntimeMetrics) collectGPUMetrics() []error {
    26  	if numGPUs, nvmlRet := nvml.DeviceGetCount(); nvmlRet != nvml.SUCCESS {
    27  		newMetrics.NumberOfGPUs = -1
    28  		return []error{utils.MakeError("Error getting GPU count")}
    29  	} else {
    30  		newMetrics.NumberOfGPUs = numGPUs
    31  		errs := make([]error, 0)
    32  
    33  		var totalVramKB uint64 = 0
    34  		var usedVramKB uint64 = 0
    35  		var totalKernelUtilPercent uint32 = 0
    36  		var totalMemBandwidthUtilPercent uint32 = 0
    37  		for i := 0; i < numGPUs; i++ {
    38  			if device, nvmlRet := nvml.DeviceGetHandleByIndex(i); nvmlRet != nvml.SUCCESS {
    39  				errs = append(errs, utils.MakeError("Couldn't get NVIDIA device at index %v", i))
    40  				break
    41  			} else {
    42  				if memInfo, nvmlRet := nvml.DeviceGetMemoryInfo(device); nvmlRet != nvml.SUCCESS {
    43  					errs = append(errs, utils.MakeError("Couldn't get video memory info for device at NVML index %v", i))
    44  					break
    45  				} else {
    46  					totalVramKB += memInfo.Total / 1024
    47  					usedVramKB += memInfo.Used / 1024
    48  					// For some reason, NVML always returns 0 for memInfo.Free, so we do
    49  					// the free calculation ourselves below.
    50  				}
    51  
    52  				if util, nvmlRet := nvml.DeviceGetUtilizationRates(device); nvmlRet != nvml.SUCCESS {
    53  					errs = append(errs, utils.MakeError("Couldn't get utilization info for device at NVML index %v", i))
    54  					break
    55  				} else {
    56  					totalKernelUtilPercent += util.Gpu
    57  					totalMemBandwidthUtilPercent += util.Memory
    58  				}
    59  			}
    60  		}
    61  
    62  		newMetrics.TotalVideoMemoryKB = totalVramKB
    63  		newMetrics.UsedVideoMemoryKB = usedVramKB
    64  		newMetrics.FreeVideoMemoryKB = totalVramKB - usedVramKB
    65  
    66  		newMetrics.GPUKernelUtilizationPercent = float64(totalKernelUtilPercent) / float64(numGPUs)
    67  		newMetrics.GPUMemoryBandwidthUtilizationPercent = float64(totalMemBandwidthUtilPercent) / float64(numGPUs)
    68  		return errs
    69  	}
