    95  func (host *AWSHost) SpinUpInstances(scalingCtx context.Context, numInstances int32, maxWaitTimeReady time.Duration, image subscriptions.Image) ([]subscriptions.Instance, error) {
    96  	ctx, cancel := context.WithCancel(scalingCtx)
    97  	defer cancel()
    98  
    99  	// Get base64 encoded user data to start host service at launch
   100  	userData, err := getUserData()
   101  	if err != nil {
   102  		return nil, utils.MakeError("failed to encode userdata for launching instances: %s", err)
   103  	}
   104  
   105  	// Set run input
   106  	input := &ec2.RunInstancesInput{
   107  		MinCount:                          aws.Int32(MIN_INSTANCE_COUNT),
   108  		MaxCount:                          aws.Int32(numInstances),
   109  		ImageId:                           aws.String(image.ImageID),
   110  		InstanceInitiatedShutdownBehavior: ec2Types.ShutdownBehaviorTerminate,
   111  		InstanceType:                      INSTANCE_TYPE,
   112  		IamInstanceProfile: &ec2Types.IamInstanceProfileSpecification{
   113  			Arn: aws.String(host.InstanceProfile),
   114  		},
   115  		SubnetId: host.MainSubnet.SubnetId,
   116  		SecurityGroupIds: []string{
   117  			aws.ToString(host.SecurityGroup.GroupId),
   118  		},
   119  		UserData:     aws.String(userData),
   120  		EbsOptimized: aws.Bool(true), // This has to be set at launch time, otherwise AWS will disable the optimization
   121  	}
   122  
   123  	var result *ec2.RunInstancesOutput
   124  
   125  	for i := 0; i < MAX_RETRY_ATTEMPTS; i++ {
   126  		logger.Infof("Attempt %d to spinup %d instances with image %s", i+1, numInstances, image.ImageID)
   127  		result, err = host.MakeInstances(ctx, input)
   128  
   129  		if err == nil {
   130  			logger.Infof("Done trying to spinup instances")
   131  			break
   132  		} else {
   133  			logger.Warningf("Failed to start instances: %s", err)
   134  			time.Sleep(1 * time.Second)
   135  		}
   136  	}
   137  
   138  	// If the last value of err is not nil, we know the loop failed to
   139  	// spinup instances.
   140  	if err != nil {
   141  		return nil, err
   142  	}
   143  
   144  	// Create slice with instances to write to database
   145  	var (
   146  		outputInstances   []subscriptions.Instance
   147  		outputInstanceIDs []string
   148  	)
   149  
   150  	// Get IDs of all created instances
   151  	for _, outputInstance := range result.Instances {
   152  		outputInstanceIDs = append(outputInstanceIDs, aws.ToString(outputInstance.InstanceId))
   153  	}
   154  
   155  	err = host.WaitForInstanceReady(scalingCtx, maxWaitTimeReady, outputInstanceIDs)
   156  	if err != nil {
   157  		return nil, utils.MakeError("failed to wait for new instances to be ready: %s", err)
   158  	}
   159  
   160  	for _, outputInstance := range result.Instances {
   161  		var (
   162  			ImageID    = aws.ToString(outputInstance.ImageId)
   163  			Type       = string(outputInstance.InstanceType)
   164  			InstanceID = aws.ToString(outputInstance.InstanceId)
   165  			Region     = host.Region
   166  			Status     = "PRE_CONNECTION"
   167  			ClientSHA  = image.ClientSHA
   168  			Name       = host.GenerateName()
   169  		)
   170  
   171  		tagInput := &ec2.CreateTagsInput{
   172  			Resources: []string{InstanceID},
   173  			Tags: []ec2Types.Tag{
   174  				{
   175  					Key:   aws.String("Name"),
   176  					Value: aws.String(Name),
   177  				},
   178  			},
   179  		}
   180  
   181  		_, err = host.MakeTags(ctx, tagInput)
   182  		if err != nil {
   183  			return nil, utils.MakeError("error taging instance: %s", err)
   184  		}
   185  
   186  		// Only append instance to output slice if we have created
   187  		// and tagged successfully.
   188  		outputInstances = append(outputInstances, subscriptions.Instance{
   189  			ID:        InstanceID,
   190  			IPAddress: "0.0.0.0", // Use dummy, will be set by host-service.
   191  			Region:    Region,
   192  			Provider:  "AWS",
   193  			ImageID:   ImageID,
   194  			ClientSHA: ClientSHA,
   195  			Type:      Type,
   196  			Status:    Status,
   197  		})
   198  	}
   199  
   200  	// Verify start output
   201  	desiredInstances := int(numInstances)
   202  	if len(outputInstances) != desiredInstances {
   203  		return outputInstances,
   204  			utils.MakeError("failed to start requested number of instances, only marked as ready %d", len(result.Instances))
   205  	}
   206  
   207  	if err != nil {
   208  		return outputInstances, utils.MakeError("error starting instances: %s", err)
   209  	}
   210  	return outputInstances, nil
   211  }

   281  // getMainSubnet returns the main subnet of the region for the current environment.
   282  func (host *AWSHost) getMainSubnet() (ec2Types.Subnet, error) {
   283  	ctx, cancel := context.WithCancel(context.Background())
   284  	defer cancel()
   285  
   286  	var env string
   287  
   288  	if metadata.IsLocalEnv() {
   289  		env = string(metadata.EnvDev)
   290  	} else {
   291  		env = metadata.GetAppEnvironmentLowercase()
   292  	}
   293  
   294  	// Find the main subnet for the current environment using tags.
   295  	input := &ec2.DescribeSubnetsInput{
   296  		Filters: []ec2Types.Filter{
   297  			{
   298  				Name:   aws.String("tag:Env"),
   299  				Values: []string{env},
   300  			},
   301  			{
   302  				Name:   aws.String("tag:Terraform"),
   303  				Values: []string{"true"},
   304  			},
   305  		},
   306  	}
   307  	output, err := host.EC2.DescribeSubnets(ctx, input)
   308  	if err != nil {
   309  		return ec2Types.Subnet{}, utils.MakeError("failed to get main subnet for env %s: %s", env, err)
   310  	}
   311  
   312  	if len(output.Subnets) == 0 {
   313  		return ec2Types.Subnet{}, utils.MakeError("couldn't find the main subnet for env %s", env)
   314  	}
   315  	return output.Subnets[0], nil
   316  }

   353  // getInstanceProfile returns the arn of the instance profile to use.
   354  func (host *AWSHost) getInstanceProfile() string {
   355  	if metadata.IsLocalEnv() {
   356  		logger.Infof("Running on localdev, using dummy instance profile...")
   357  		return "dummy_instance_profile"
   358  	}
   359  
   360  	switch metadata.GetAppEnvironmentLowercase() {
   361  	case string(metadata.EnvDev):
   362  		return os.Getenv("INSTANCE_PROFILE_DEV")
   363  	case string(metadata.EnvStaging):
   364  		return os.Getenv("INSTANCE_PROFILE_STAGING")
   365  	case string(metadata.EnvProd):
   366  		return os.Getenv("INSTANCE_PROFILE_PROD")
   367  	default:
   368  		return os.Getenv("INSTANCE_PROFILE_DEV")
   369  	}
