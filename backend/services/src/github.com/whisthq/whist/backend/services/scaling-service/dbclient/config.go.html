    15  func GetDevConfigs(ctx context.Context, client subscriptions.WhistGraphQLClient) (map[string]string, error) {

    18  	if err != nil {
    19  		return nil, utils.MakeError("Failed to query config database for configuration values of env %v. Using default values instead. Err: %v", metadata.GetAppEnvironmentLowercase(), err)
    20  	}
    21  
    22  	if len(query.WhistConfigs) == 0 {
    23  		return nil, utils.MakeError("Could not find dev configs on database")

    25  
    26  	// Convert to a map for easier manipulation
    27  	configMap := make(map[string]string)
    28  	for _, entry := range query.WhistConfigs {

    30  	}
    31  
    32  	return configMap, nil
