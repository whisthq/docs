    98  func (whistPayments *PaymentsClient) CreateSession() (string, error) {
    99  	var (
   100  		status     string
   101  		sessionUrl string
   102  	)
   103  
   104  	subscription, err := whistPayments.stripeClient.getSubscription()
   105  	if err != nil {
   106  		logger.Warningf("Failed to get subscription. Defaulting to access token subscription status.")
   107  		status = whistPayments.stripeClient.getSubscriptionStatus()
   108  	} else {
   109  		status = string(subscription.Status)
   110  	}
   111  
   112  	isNewUser := whistPayments.stripeClient.isNewUser()
   113  
   114  	if status == "active" || status == "trialing" {
   115  		// If the authenticated user already has a Whist subscription in a non-terminal state
   116  		// (one of `active` or `trialing`), create a Stripe billing portal that the customer
   117  		// can use to manage their subscription and billing information.
   118  		sessionUrl, err = whistPayments.stripeClient.createBillingPortal()
   119  		if err != nil {
   120  			return "", utils.MakeError("error creating Stripe billing portal: %s", err)
   121  		}
   122  	} else if !isNewUser {
   123  		// The authenticated user has previous Whist subscriptions. This means that the
   124  		// user has already gone through the initial trial period.
   125  		withTrialPeriod := false
   126  		sessionUrl, err = whistPayments.stripeClient.createCheckoutSession(withTrialPeriod)
   127  		if err != nil {
   128  			return "", utils.MakeError("error creating Stripe billing portal: %s", err)
   129  		}
   130  	} else {
   131  		// The authenticated user does not have an active Whist subscription, so we offer
   132  		// a free trial period.
   133  		withTrialPeriod := true
   134  		sessionUrl, err = whistPayments.stripeClient.createCheckoutSession(withTrialPeriod)
   135  		if err != nil {
   136  			return "", utils.MakeError("error creating Stripe checkout session: %s", err)
   137  		}
   138  	}
   139  
   140  	return sessionUrl, nil
