    35  func processJSONDataRequest(w http.ResponseWriter, r *http.Request, queue chan<- httputils.ServerRequest) {
    36  	// Start timer to measure average time processing http requests.
    37  	start := time.Now()
    38  
    39  	// Verify that it is an PUT request
    40  	if httputils.VerifyRequestType(w, r, http.MethodPut) != nil {
    41  		metrics.Increment("FailedRequests")
    42  		return
    43  	}
    44  
    45  	var reqdata httputils.JSONTransportRequest

    50  	}

    63  // handleJSONTransportRequest handles any incoming JSON transport requests. First it validates the JWT, then it verifies if
    64  // the json data for the particular user exists, and finally it sends the data through the transport request map.
    65  func handleJSONTransportRequest(serverevent httputils.ServerRequest, transportRequestMap map[mandelboxtypes.MandelboxID]chan *httputils.JSONTransportRequest, transportMapLock *sync.Mutex) {
    66  	// Receive the value of the `json_transport request`
    67  	req := serverevent.(*httputils.JSONTransportRequest)
    68  
    69  	// Verify if we have already received the JSONData for this specific mandelbox. In case we have, we ignore the request.
    70  	// Otherwise we add the request to the map. By performing this validation, we make the host service resistant to DDOS
    71  	// attacks, and by keeping track of each mandelbox's json data, we avoid any race condition that might corrupt other mandelboxes.
    72  
    73  	// Acquire lock on transport requests map
    74  	transportMapLock.Lock()
    75  	defer transportMapLock.Unlock()
    76  
    77  	if transportRequestMap[req.MandelboxID] == nil {
    78  		transportRequestMap[req.MandelboxID] = make(chan *httputils.JSONTransportRequest, 1)
    79  	}
    80  
    81  	// Send the JSONTransportRequest data through the map's channel and close the channel to
    82  	// prevent more requests from being sent, this ensures we only receive the
    83  	// json transport request once per user.
    84  	transportRequestMap[req.MandelboxID] <- req
    85  	close(transportRequestMap[req.MandelboxID])
    86  }

    89  // in a safe way. It also creates the channel in case it doesn't exists for that particular mandelbox.
    90  func getJSONTransportRequestChannel(mandelboxID mandelboxtypes.MandelboxID,
    91  	transportRequestMap map[mandelboxtypes.MandelboxID]chan *httputils.JSONTransportRequest, transportMapLock *sync.Mutex) chan *httputils.JSONTransportRequest {
    92  	// Acquire lock on transport requests map

   101  
   102  	return transportRequestMap[mandelboxID]

   106  	transportMapLock *sync.Mutex) (*httputils.JSONTransportRequest, mandelboxtypes.AppName) {
   107  
   108  	var AppName mandelboxtypes.AppName
   109  	var req *httputils.JSONTransportRequest
   110  
   111  	if metadata.IsLocalEnvWithoutDB() {
   112  		// Receive the json transport request immediately when running on local env
   113  		jsonchan := getJSONTransportRequestChannel(mandelboxSubscription.ID, transportRequestMap, transportMapLock)
   114  
   115  		// We will wait 1 minute to get the JSON transport request
   116  		select {
   117  		case transportRequest := <-jsonchan:
   118  			req = transportRequest
   119  
   120  			if req.AppName == "" {
   121  				// If no app name is set, we default to using the `utils.MandelboxApp` image.
   122  				AppName = mandelboxtypes.AppName(utils.MandelboxApp)
   123  			} else {
   124  				AppName = req.AppName
   125  			}
   126  		case <-time.After(1 * time.Minute):
   127  			return nil, AppName
   128  		}
   129  
   130  	} else {
   131  		// If not on a local environment, we pull the app name from the database event.
   132  		// We need to append the "browsers/" prefix.
   133  		appName := utils.Sprintf("browsers/%s", strings.ToLower(mandelboxSubscription.App))
   134  		AppName = mandelboxtypes.AppName(appName)
   135  	}
   136  
   137  	return req, AppName

   140  // StartHTTPServer returns a channel of events from the HTTP server as its first return value
   141  func StartHTTPServer(globalCtx context.Context, globalCancel context.CancelFunc, goroutineTracker *sync.WaitGroup) (<-chan httputils.ServerRequest, error) {
   142  	logger.Info("Setting up HTTP server.")
   143  
   144  	err := httputils.InitializeTLS(privatekeyPath, certPath)
   145  	if err != nil {
   146  		return nil, utils.MakeError("Error starting HTTP Server: %v", err)
   147  	}
   148  
   149  	// Buffer up to 100 events so we don't block. This should be mostly
   150  	// unnecessary, but we want to be able to handle a burst without blocking.
   151  	events := make(chan httputils.ServerRequest, 100)
   152  
   153  	createHandler := func(f func(http.ResponseWriter, *http.Request, chan<- httputils.ServerRequest)) func(http.ResponseWriter, *http.Request) {
   154  		return func(w http.ResponseWriter, r *http.Request) {
   155  			f(w, r, events)
   156  		}
   157  	}
   158  
   159  	jsonTransportHandler := httputils.EnableCORS(createHandler(processJSONDataRequest))
   160  	// Create a custom HTTP Request Multiplexer
   161  	mux := http.NewServeMux()
   162  	mux.Handle("/", http.NotFoundHandler())
   163  	mux.HandleFunc("/json_transport", jsonTransportHandler)
   164  
   165  	// Create the server itself
   166  	server := &http.Server{
   167  		Addr:    utils.Sprintf("0.0.0.0:%v", PortToListen),
   168  		Handler: mux,
   169  	}
   170  
   171  	// Start goroutine that shuts down `server` if the global context gets
   172  	// cancelled.
   173  	goroutineTracker.Add(1)
   174  	go func() {
   175  		defer goroutineTracker.Done()
   176  
   177  		// Start goroutine that actually listens for requests
   178  		goroutineTracker.Add(1)
   179  		go func() {
   180  			defer goroutineTracker.Done()
   181  
   182  			if err := server.ListenAndServeTLS(certPath, privatekeyPath); err != nil && err.Error() != "http: Server closed" {
   183  				close(events)
   184  				logger.Panicf(globalCancel, "Error listening and serving in httpserver: %s", err)
   185  			}
   186  		}()
   187  
   188  		// Listen for global context cancellation
   189  		<-globalCtx.Done()
   190  
   191  		logger.Infof("Shutting down httpserver...")
   192  		shutdownCtx, shutdownCancel := context.WithTimeout(globalCtx, 30*time.Second)
   193  		defer shutdownCancel()
   194  
   195  		err := server.Shutdown(shutdownCtx)
   196  		if err != nil {
   197  			logger.Infof("Shut down httpserver with error %s", err)
   198  		} else {
   199  			logger.Info("Gracefully shut down httpserver.")
   200  		}
   201  	}()
   202  
   203  	return events, nil
   204  }
