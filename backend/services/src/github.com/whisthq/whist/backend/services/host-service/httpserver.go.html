    81  func processJSONDataRequest(w http.ResponseWriter, r *http.Request, queue chan<- httputils.ServerRequest) {
    82  	// Start timer to measure average time processing http requests.
    83  	start := time.Now()
    84  
    85  	// Verify that it is an PUT request
    86  	if httputils.VerifyRequestType(w, r, http.MethodPut) != nil {
    87  		metrics.Increment("FailedRequests")
    88  		return
    89  	}
    90  
    91  	// Verify authorization and unmarshal into the right object type
    92  	var reqdata JSONTransportRequest
    93  	if err := authenticateRequest(w, r, &reqdata, !metadata.IsLocalEnv()); err != nil {
    94  		logger.Errorf("Error authenticating and parsing %T: %s", reqdata, err)
    95  		metrics.Increment("FailedRequests")
    96  		metrics.Increment("ErrorRate")
    97  		return
    98  	}
    99  
   100  	// Send request to queue, then wait for result
   101  	queue <- &reqdata
   102  	res := <-reqdata.resultChan
   103  
   104  	res.Send(w)
   105  
   106  	// Measure elapsed milliseconds and send to metrics.
   107  	elapsed := time.Since(start)
   108  	metrics.Increment("SuccessfulRequests")
   109  	metrics.Add("TotalRequestTimeMS", elapsed.Milliseconds())
   110  }

   112  // handleJSONTransportRequest handles any incoming JSON transport requests. First it validates the JWT, then it verifies if
   113  // the json data for the particular user exists, and finally it sends the data through the transport request map.
   114  func handleJSONTransportRequest(serverevent httputils.ServerRequest, transportRequestMap map[mandelboxtypes.MandelboxID]chan *JSONTransportRequest, transportMapLock *sync.Mutex) {
   115  	// Receive the value of the `json_transport request`
   116  	req := serverevent.(*JSONTransportRequest)
   117  
   118  	// First we validate the JWT received from the `json_transport` endpoint
   119  	// Set up auth
   120  	claims := new(auth.WhistClaims)
   121  	parser := &jwt.Parser{SkipClaimsValidation: true}
   122  
   123  	// Only verify auth in non-local environments
   124  	if !metadata.IsLocalEnv() {
   125  		// Decode the access token without validating any of its claims or
   126  		// verifying its signature because we've already done that in
   127  		// `authenticateAndParseRequest`. All we want to know is the value of the
   128  		// sub (subject) claim.
   129  		if _, _, err := parser.ParseUnverified(string(req.JwtAccessToken), claims); err != nil {
   130  			logger.Errorf("There was a problem while parsing the access token for the second time: %s", err)
   131  			metrics.Increment("ErrorRate")
   132  			return
   133  		}
   134  	}
   135  
   136  	// If the jwt was valid, the next step is to verify if we have already received the JSONData from
   137  	// this specific user. In case we have, we ignore the request. Otherwise we add the request to the map.
   138  	// By performing this validation, we make the host service resistant to DDOS attacks, and by keeping track
   139  	// of each user's json data, we avoid any race condition that might corrupt other mandelboxes.
   140  
   141  	// Acquire lock on transport requests map
   142  	transportMapLock.Lock()
   143  	defer transportMapLock.Unlock()
   144  
   145  	if transportRequestMap[req.MandelboxID] == nil {
   146  		transportRequestMap[req.MandelboxID] = make(chan *JSONTransportRequest, 1)
   147  	}
   148  
   149  	// Send the JSONTransportRequest data through the map's channel and close the channel to
   150  	// prevent more requests from being sent, this ensures we only receive the
   151  	// json transport request once per user.
   152  	transportRequestMap[req.MandelboxID] <- req
   153  	close(transportRequestMap[req.MandelboxID])
   154  }

   173  func getAppName(mandelboxSubscription subscriptions.Mandelbox, transportRequestMap map[mandelboxtypes.MandelboxID]chan *JSONTransportRequest,
   174  	transportMapLock *sync.Mutex) (*JSONTransportRequest, mandelboxtypes.AppName) {
   175  
   176  	var AppName mandelboxtypes.AppName
   177  	var req *JSONTransportRequest
   178  
   179  	if metadata.IsLocalEnvWithoutDB() {
   180  		// Receive the json transport request immediately when running on local env
   181  		jsonchan := getJSONTransportRequestChannel(mandelboxSubscription.ID, transportRequestMap, transportMapLock)
   182  
   183  		// We will wait 1 minute to get the JSON transport request
   184  		select {
   185  		case transportRequest := <-jsonchan:
   186  			req = transportRequest
   187  
   188  			if req.AppName == "" {
   189  				// If no app name is set, we default to using the `browsers/chrome` image.
   190  				AppName = mandelboxtypes.AppName("browsers/chrome")
   191  			} else {
   192  				AppName = req.AppName
   193  			}
   194  		case <-time.After(1 * time.Minute):
   195  			return nil, AppName
   196  		}
   197  
   198  	} else {
   199  		// If not on a local environment, we pull the app name from the database event.
   200  		// We need to append the "browsers/" prefix.
   201  		appName := utils.Sprintf("browsers/%s", strings.ToLower(mandelboxSubscription.App))
   202  		AppName = mandelboxtypes.AppName(appName)
   203  	}
   204  
   205  	return req, AppName

   208  // Function to authenticate an incoming request. Splitting this into a separate
   209  // function has the following advantages:
   210  // 1. We factor out duplicated functionality among all the endpoints that need
   211  // authentication.
   212  // 2. Doing so allows us not to unmarshal the `jwt_access_token` field, and
   213  // therefore prevents needing to pass it to client code in other packages that
   214  // don't need to understand our authentication mechanism or read the secret
   215  // key. We could alternatively do this by creating two separate types of
   216  // structs containing the data required for the endpoint --- one without the
   217  // auth secret, and one with it --- but this requires duplication of struct
   218  // definitions and writing code to copy values from one struct to another.
   219  // 3. If we get a bad, unauthenticated request we can minimize the amount of
   220  // processsing power we devote to it. This is useful for being resistant to
   221  // Denial-of-Service attacks.
   222  func authenticateRequest(w http.ResponseWriter, r *http.Request, s httputils.ServerRequest, authorizeAsBackend bool) (err error) {
   223  	// Get the raw map of the request, and unmarshal into
   224  	// the corresponding struct
   225  	rawmap, err := httputils.ParseRequest(w, r, s)
   226  	if err != nil {
   227  		return utils.MakeError("Error while parsing request. Err: %v", err)
   228  	}
   229  
   230  	if authorizeAsBackend {
   231  		var requestAuthSecret string
   232  
   233  		err = func() error {
   234  			if value, ok := rawmap["jwt_access_token"]; ok {
   235  				return json.Unmarshal(*value, &requestAuthSecret)
   236  			}
   237  			return utils.MakeError("Request body had no \"jwt_access_token\" field.")
   238  		}()
   239  
   240  		if err != nil {
   241  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   242  			return utils.MakeError("Error getting jwt_access_token from JSON body sent on %s to URL %s: %s", r.Host, r.URL, err)
   243  		}
   244  
   245  		// Actually verify authentication. We check that the access token sent is a valid JWT signed by Auth0.
   246  		// Parses a raw access token string, verifies the token's signature, ensures that it is valid at the current moment in time.
   247  		claims, err := auth.ParseToken(requestAuthSecret)
   248  		if err != nil {
   249  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   250  			return utils.MakeError("Received an unpermissioned backend request on %s to URL %s. Error: %s", r.Host, r.URL, err)
   251  		}
   252  
   253  		if err := auth.Verify(claims); err != nil {
   254  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   255  			return utils.MakeError("Received an unpermissioned backend request on %s to URL %s. Error: %s", r.Host, r.URL, err)
   256  		}
   257  	}
   258  
   259  	return nil
   260  }

   262  // StartHTTPServer returns a channel of events from the HTTP server as its first return value
   263  func StartHTTPServer(globalCtx context.Context, globalCancel context.CancelFunc, goroutineTracker *sync.WaitGroup) (<-chan httputils.ServerRequest, error) {
   264  	logger.Info("Setting up HTTP server.")
   265  
   266  	err := httputils.InitializeTLS(privatekeyPath, certPath)
   267  	if err != nil {
   268  		return nil, utils.MakeError("Error starting HTTP Server: %v", err)
   269  	}
   270  
   271  	// Buffer up to 100 events so we don't block. This should be mostly
   272  	// unnecessary, but we want to be able to handle a burst without blocking.
   273  	events := make(chan httputils.ServerRequest, 100)
   274  
   275  	createHandler := func(f func(http.ResponseWriter, *http.Request, chan<- httputils.ServerRequest)) func(http.ResponseWriter, *http.Request) {
   276  		return func(w http.ResponseWriter, r *http.Request) {
   277  			f(w, r, events)
   278  		}
   279  	}
   280  
   281  	// Create a custom HTTP Request Multiplexer
   282  	mux := http.NewServeMux()
   283  	mux.Handle("/", http.NotFoundHandler())
   284  	mux.HandleFunc("/json_transport", createHandler(processJSONDataRequest))
   285  
   286  	// Create the server itself
   287  	server := &http.Server{
   288  		Addr:    utils.Sprintf("0.0.0.0:%v", PortToListen),
   289  		Handler: mux,
   290  	}
   291  
   292  	// Start goroutine that shuts down `server` if the global context gets
   293  	// cancelled.
   294  	goroutineTracker.Add(1)
   295  	go func() {
   296  		defer goroutineTracker.Done()
   297  
   298  		// Start goroutine that actually listens for requests
   299  		goroutineTracker.Add(1)
   300  		go func() {
   301  			defer goroutineTracker.Done()
   302  
   303  			if err := server.ListenAndServeTLS(certPath, privatekeyPath); err != nil && err.Error() != "http: Server closed" {
   304  				close(events)
   305  				logger.Panicf(globalCancel, "Error listening and serving in httpserver: %s", err)
   306  			}
   307  		}()
   308  
   309  		// Listen for global context cancellation
   310  		<-globalCtx.Done()
   311  
   312  		logger.Infof("Shutting down httpserver...")
   313  		shutdownCtx, shutdownCancel := context.WithTimeout(globalCtx, 30*time.Second)
   314  		defer shutdownCancel()
   315  
   316  		err := server.Shutdown(shutdownCtx)
   317  		if err != nil {
   318  			logger.Infof("Shut down httpserver with error %s", err)
   319  		} else {
   320  			logger.Info("Gracefully shut down httpserver.")
   321  		}
   322  	}()
   323  
   324  	return events, nil
   325  }
