   123  func processJSONDataRequest(w http.ResponseWriter, r *http.Request, queue chan<- ServerRequest) {
   124  	// Start timer to measure average time processing http requests.
   125  	start := time.Now()
   126  
   127  	// Verify that it is an PUT request
   128  	if verifyRequestType(w, r, http.MethodPut) != nil {
   129  		metrics.Increment("FailedRequests")
   130  		return
   131  	}
   132  
   133  	// Verify authorization and unmarshal into the right object type
   134  	var reqdata JSONTransportRequest
   135  	if err := authenticateAndParseRequest(w, r, &reqdata, !metadata.IsLocalEnv()); err != nil {
   136  		logger.Errorf("Error authenticating and parsing %T: %s", reqdata, err)
   137  		metrics.Increment("FailedRequests")
   138  		metrics.Increment("ErrorRate")
   139  		return
   140  	}
   141  
   142  	// Send request to queue, then wait for result
   143  	queue <- &reqdata
   144  	res := <-reqdata.resultChan
   145  
   146  	res.send(w)
   147  
   148  	// Measure elapsed milliseconds and send to metrics.
   149  	elapsed := time.Since(start)
   150  	metrics.Increment("SuccessfulRequests")
   151  	metrics.Add("TotalRequestTimeMS", elapsed.Milliseconds())
   152  }

   154  // handleJSONTransportRequest handles any incoming JSON transport requests. First it validates the JWT, then it verifies if
   155  // the json data for the particular user exists, and finally it sends the data through the transport request map.
   156  func handleJSONTransportRequest(serverevent ServerRequest, transportRequestMap map[mandelboxtypes.MandelboxID]chan *JSONTransportRequest, transportMapLock *sync.Mutex) {
   157  	// Receive the value of the `json_transport request`
   158  	req := serverevent.(*JSONTransportRequest)
   159  
   160  	// First we validate the JWT received from the `json_transport` endpoint
   161  	// Set up auth
   162  	claims := new(auth.WhistClaims)
   163  	parser := &jwt.Parser{SkipClaimsValidation: true}
   164  
   165  	// Only verify auth in non-local environments
   166  	if !metadata.IsLocalEnv() {
   167  		// Decode the access token without validating any of its claims or
   168  		// verifying its signature because we've already done that in
   169  		// `authenticateAndParseRequest`. All we want to know is the value of the
   170  		// sub (subject) claim.
   171  		if _, _, err := parser.ParseUnverified(string(req.JwtAccessToken), claims); err != nil {
   172  			logger.Errorf("There was a problem while parsing the access token for the second time: %s", err)
   173  			metrics.Increment("ErrorRate")
   174  			return
   175  		}
   176  	}
   177  
   178  	// If the jwt was valid, the next step is to verify if we have already received the JSONData from
   179  	// this specific user. In case we have, we ignore the request. Otherwise we add the request to the map.
   180  	// By performing this validation, we make the host service resistant to DDOS attacks, and by keeping track
   181  	// of each user's json data, we avoid any race condition that might corrupt other mandelboxes.
   182  
   183  	// Acquire lock on transport requests map
   184  	transportMapLock.Lock()
   185  	defer transportMapLock.Unlock()
   186  
   187  	if transportRequestMap[req.MandelboxID] == nil {
   188  		transportRequestMap[req.MandelboxID] = make(chan *JSONTransportRequest, 1)
   189  	}
   190  
   191  	// Send the JSONTransportRequest data through the map's channel and close the channel to
   192  	// prevent more requests from being sent, this ensures we only receive the
   193  	// json transport request once per user.
   194  	transportRequestMap[req.MandelboxID] <- req
   195  	close(transportRequestMap[req.MandelboxID])
   196  }

   272  // Function to get the body of a result, authenticate it, and unmarshal it into
   273  // a golang struct. Splitting this into a separate function has the following
   274  // advantages:
   275  // 1. We factor out duplicated functionality among all the endpoints that need
   276  // authentication.
   277  // 2. Doing so allows us not to unmarshal the `jwt_access_token` field, and
   278  // therefore prevents needing to pass it to client code in other packages that
   279  // don't need to understand our authentication mechanism or read the secret
   280  // key. We could alternatively do this by creating two separate types of
   281  // structs containing the data required for the endpoint --- one without the
   282  // auth secret, and one with it --- but this requires duplication of struct
   283  // definitions and writing code to copy values from one struct to another.
   284  // 3. If we get a bad, unauthenticated request we can minimize the amount of
   285  // processsing power we devote to it. This is useful for being resistant to
   286  // Denial-of-Service attacks.
   287  // Note: authenticateAndParseRequest is declared as a variable to allow
   288  // it to be mocked for testing.
   289  var authenticateAndParseRequest = func(w http.ResponseWriter, r *http.Request, s ServerRequest, authorizeAsBackend bool) (err error) {
   290  	// Get body of request
   291  	body, err := io.ReadAll(r.Body)
   292  	if err != nil {
   293  		http.Error(w, "Malformed body", http.StatusBadRequest)
   294  		return utils.MakeError("Error getting body from request on %s to URL %s: %s", r.Host, r.URL, err)
   295  	}
   296  
   297  	// Extract only the jwt_access_token field from a raw JSON unmarshalling that
   298  	// delays as much decoding as possible
   299  	var rawmap map[string]*json.RawMessage
   300  	err = json.Unmarshal(body, &rawmap)
   301  	if err != nil {
   302  		http.Error(w, "Malformed body", http.StatusBadRequest)
   303  		return utils.MakeError("Error raw-unmarshalling JSON body sent on %s to URL %s: %s", r.Host, r.URL, err)
   304  	}
   305  
   306  	if authorizeAsBackend {
   307  		var requestAuthSecret string
   308  
   309  		err = func() error {
   310  			if value, ok := rawmap["jwt_access_token"]; ok {
   311  				return json.Unmarshal(*value, &requestAuthSecret)
   312  			}
   313  			return utils.MakeError("Request body had no \"jwt_access_token\" field.")
   314  		}()
   315  
   316  		if err != nil {
   317  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   318  			return utils.MakeError("Error getting jwt_access_token from JSON body sent on %s to URL %s: %s", r.Host, r.URL, err)
   319  		}
   320  
   321  		// Actually verify authentication. We check that the access token sent is a valid JWT signed by Auth0.
   322  		// Parses a raw access token string, verifies the token's signature, ensures that it is valid at the current moment in time.
   323  		claims, err := auth.ParseToken(requestAuthSecret)
   324  		if err != nil {
   325  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   326  			return utils.MakeError("Received an unpermissioned backend request on %s to URL %s. Error: %s", r.Host, r.URL, err)
   327  		}
   328  
   329  		if err := auth.Verify(claims); err != nil {
   330  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   331  			return utils.MakeError("Received an unpermissioned backend request on %s to URL %s. Error: %s", r.Host, r.URL, err)
   332  		}
   333  	}
   334  
   335  	// Now, actually do the unmarshalling into the right object type
   336  	err = json.Unmarshal(body, s)
   337  	if err != nil {
   338  		http.Error(w, "Malformed body", http.StatusBadRequest)
   339  		return utils.MakeError("Could not fully unmarshal the body of a request sent on %s to URL %s: %s", r.Host, r.URL, err)
   340  	}
   341  
   342  	// Set up the result channel
   343  	s.createResultChan()
   344  
   345  	return nil
   346  }

   348  // StartHTTPServer returns a channel of events from the webserver as its first return value
   349  func StartHTTPServer(globalCtx context.Context, globalCancel context.CancelFunc, goroutineTracker *sync.WaitGroup) (<-chan ServerRequest, error) {
   350  	logger.Info("Setting up HTTP server.")
   351  
   352  	err := initializeTLS()
   353  	if err != nil {
   354  		return nil, utils.MakeError("Error starting HTTP Server: %v", err)
   355  	}
   356  
   357  	// Buffer up to 100 events so we don't block. This should be mostly
   358  	// unnecessary, but we want to be able to handle a burst without blocking.
   359  	events := make(chan ServerRequest, 100)
   360  
   361  	createHandler := func(f func(http.ResponseWriter, *http.Request, chan<- ServerRequest)) func(http.ResponseWriter, *http.Request) {
   362  		return func(w http.ResponseWriter, r *http.Request) {
   363  			f(w, r, events)
   364  		}
   365  	}
   366  
   367  	// Create a custom HTTP Request Multiplexer
   368  	mux := http.NewServeMux()
   369  	mux.Handle("/", http.NotFoundHandler())
   370  	mux.HandleFunc("/json_transport", createHandler(processJSONDataRequest))
   371  
   372  	// Create the server itself
   373  	server := &http.Server{
   374  		Addr:    utils.Sprintf("0.0.0.0:%v", PortToListen),
   375  		Handler: mux,
   376  	}
   377  
   378  	// Start goroutine that shuts down `server` if the global context gets
   379  	// cancelled.
   380  	goroutineTracker.Add(1)
   381  	go func() {
   382  		defer goroutineTracker.Done()
   383  
   384  		// Start goroutine that actually listens for requests
   385  		goroutineTracker.Add(1)
   386  		go func() {
   387  			defer goroutineTracker.Done()
   388  
   389  			if err := server.ListenAndServeTLS(certPath, privatekeyPath); err != nil && err.Error() != "http: Server closed" {
   390  				close(events)
   391  				logger.Panicf(globalCancel, "Error listening and serving in httpserver: %s", err)
   392  			}
   393  		}()
   394  
   395  		// Listen for global context cancellation
   396  		<-globalCtx.Done()
   397  
   398  		// This is only necessary since we don't have the ability to subscribe to
   399  		// database events. In particular, the webserver might mark this host
   400  		// service as draining after allocating a mandelbox on it. If we didn't
   401  		// have this sleep, we would stop accepting requests right away, and the
   402  		// SpinUpMandelbox request from the client app would error out. We don't
   403  		// want that, so we accept requests for another 30 seconds. This would be
   404  		// annoying in local development, so it's disabled in that case.
   405  		// TODO: get rid of this once we have pubsub
   406  		if !metadata.IsLocalEnv() {
   407  			logger.Infof("Global context cancelled. Starting 30 second grace period for requests before http server is shutdown...")
   408  			time.Sleep(30 * time.Second)
   409  		}
   410  
   411  		logger.Infof("Shutting down httpserver...")
   412  		shutdownCtx, shutdownCancel := context.WithTimeout(globalCtx, 30*time.Second)
   413  		defer shutdownCancel()
   414  
   415  		err := server.Shutdown(shutdownCtx)
   416  		if err != nil {
   417  			logger.Infof("Shut down httpserver with error %s", err)
   418  		} else {
   419  			logger.Info("Gracefully shut down httpserver.")
   420  		}
   421  	}()
   422  
   423  	return events, nil
   424  }

   426  // Creates a TLS certificate/private key pair for secure communication with the Whist webserver
   427  func initializeTLS() error {
   428  	// Create a self-signed passwordless certificate
   429  	// https://unix.stackexchange.com/questions/104171/create-ssl-certificate-non-interactively
   430  	cmd := exec.Command(
   431  		"/usr/bin/openssl",
   432  		"req",
   433  		"-new",
   434  		"-newkey",
   435  		"rsa:4096",
   436  		"-days",
   437  		"365",
   438  		"-nodes",
   439  		"-x509",
   440  		"-subj",
   441  		"/C=US/ST=./L=./O=./CN=.",
   442  		"-addext", "subjectAltName=IP:127.0.0.1",
   443  		"-keyout",
   444  		privatekeyPath,
   445  		"-out",
   446  		certPath,
   447  	)
   448  	output, err := cmd.CombinedOutput()

   450  		return utils.MakeError("Unable to create x509 private key/certificate pair. Error: %v, Command output: %s", err, output)
