    69  // handleJSONTransportRequest handles any incoming JSON transport requests. First it validates the JWT, then it verifies if
    70  // the json data for the particular user exists, and finally it sends the data through the transport request map.
    71  func handleJSONTransportRequest(serverevent httputils.ServerRequest, transportRequestMap map[mandelboxtypes.MandelboxID]chan *httputils.JSONTransportRequest, transportMapLock *sync.Mutex) {
    72  	// Receive the value of the `json_transport request`
    73  	req := serverevent.(*httputils.JSONTransportRequest)
    74  
    75  	// First we validate the JWT received from the `json_transport` endpoint
    76  	// Set up auth
    77  	claims := new(auth.WhistClaims)
    78  	parser := &jwt.Parser{SkipClaimsValidation: true}
    79  
    80  	// Only verify auth in non-local environments
    81  	if !metadata.IsLocalEnv() {
    82  		// Decode the access token without validating any of its claims or
    83  		// verifying its signature because we've already done that in
    84  		// `authenticateAndParseRequest`. All we want to know is the value of the
    85  		// sub (subject) claim.
    86  		if _, _, err := parser.ParseUnverified(string(req.JwtAccessToken), claims); err != nil {
    87  			logger.Errorf("There was a problem while parsing the access token for the second time: %s", err)
    88  			metrics.Increment("ErrorRate")
    89  			return
    90  		}
    91  	}
    92  
    93  	// If the jwt was valid, the next step is to verify if we have already received the JSONData from
    94  	// this specific user. In case we have, we ignore the request. Otherwise we add the request to the map.
    95  	// By performing this validation, we make the host service resistant to DDOS attacks, and by keeping track
    96  	// of each user's json data, we avoid any race condition that might corrupt other mandelboxes.
    97  
    98  	// Acquire lock on transport requests map
    99  	transportMapLock.Lock()
   100  	defer transportMapLock.Unlock()
   101  
   102  	if transportRequestMap[req.MandelboxID] == nil {
   103  		transportRequestMap[req.MandelboxID] = make(chan *httputils.JSONTransportRequest, 1)
   104  	}
   105  
   106  	// Send the JSONTransportRequest data through the map's channel and close the channel to
   107  	// prevent more requests from being sent, this ensures we only receive the
   108  	// json transport request once per user.
   109  	transportRequestMap[req.MandelboxID] <- req
   110  	close(transportRequestMap[req.MandelboxID])
   111  }

   131  	transportMapLock *sync.Mutex) (*httputils.JSONTransportRequest, mandelboxtypes.AppName) {

   146  				// If no app name is set, we default to using the `browsers/chrome` image.

   148  			} else {

   157  		// We need to append the "browsers/" prefix.

   165  // Function to authenticate an incoming request. Splitting this into a separate
   166  // function has the following advantages:
   167  // 1. We factor out duplicated functionality among all the endpoints that need
   168  // authentication.
   169  // 2. Doing so allows us not to unmarshal the `jwt_access_token` field, and
   170  // therefore prevents needing to pass it to client code in other packages that
   171  // don't need to understand our authentication mechanism or read the secret
   172  // key. We could alternatively do this by creating two separate types of
   173  // structs containing the data required for the endpoint --- one without the
   174  // auth secret, and one with it --- but this requires duplication of struct
   175  // definitions and writing code to copy values from one struct to another.
   176  // 3. If we get a bad, unauthenticated request we can minimize the amount of
   177  // processsing power we devote to it. This is useful for being resistant to
   178  // Denial-of-Service attacks.
   179  func authenticateRequest(w http.ResponseWriter, r *http.Request, s httputils.ServerRequest, authorizeAsBackend bool) (err error) {
   180  	// Get the raw map of the request, and unmarshal into
   181  	// the corresponding struct
   182  	rawmap, err := httputils.ParseRequest(w, r, s)
   183  	if err != nil {
   184  		return utils.MakeError("Error while parsing request. Err: %v", err)
   185  	}
   186  
   187  	if authorizeAsBackend {
   188  		var requestAuthSecret string
   189  
   190  		err = func() error {
   191  			if value, ok := rawmap["jwt_access_token"]; ok {
   192  				return json.Unmarshal(*value, &requestAuthSecret)
   193  			}
   194  			return utils.MakeError("Request body had no \"jwt_access_token\" field.")
   195  		}()
   196  
   197  		if err != nil {
   198  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   199  			return utils.MakeError("Error getting jwt_access_token from JSON body sent on %s to URL %s: %s", r.Host, r.URL, err)
   200  		}
   201  
   202  		// Actually verify authentication. We check that the access token sent is a valid JWT signed by Auth0.
   203  		// Parses a raw access token string, verifies the token's signature, ensures that it is valid at the current moment in time.
   204  		claims, err := auth.ParseToken(requestAuthSecret)
   205  		if err != nil {
   206  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   207  			return utils.MakeError("Received an unpermissioned backend request on %s to URL %s. Error: %s", r.Host, r.URL, err)
   208  		}
   209  
   210  		if err := auth.Verify(claims); err != nil {
   211  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   212  			return utils.MakeError("Received an unpermissioned backend request on %s to URL %s. Error: %s", r.Host, r.URL, err)
   213  		}
   214  	}
   215  
   216  	return nil
   217  }

   219  // StartHTTPServer returns a channel of events from the HTTP server as its first return value
   220  func StartHTTPServer(globalCtx context.Context, globalCancel context.CancelFunc, goroutineTracker *sync.WaitGroup) (<-chan httputils.ServerRequest, error) {
   221  	logger.Info("Setting up HTTP server.")
   222  
   223  	err := httputils.InitializeTLS(privatekeyPath, certPath)
   224  	if err != nil {
   225  		return nil, utils.MakeError("Error starting HTTP Server: %v", err)
   226  	}
   227  
   228  	// Buffer up to 100 events so we don't block. This should be mostly
   229  	// unnecessary, but we want to be able to handle a burst without blocking.
   230  	events := make(chan httputils.ServerRequest, 100)
   231  
   232  	createHandler := func(f func(http.ResponseWriter, *http.Request, chan<- httputils.ServerRequest)) func(http.ResponseWriter, *http.Request) {
   233  		return func(w http.ResponseWriter, r *http.Request) {
   234  			f(w, r, events)
   235  		}
   236  	}
   237  
   238  	jsonTransportHandler := httputils.EnableCORS(createHandler(processJSONDataRequest))
   239  	// Create a custom HTTP Request Multiplexer
   240  	mux := http.NewServeMux()
   241  	mux.Handle("/", http.NotFoundHandler())
   242  	mux.HandleFunc("/json_transport", jsonTransportHandler)
   243  
   244  	// Create the server itself
   245  	server := &http.Server{
   246  		Addr:    utils.Sprintf("0.0.0.0:%v", PortToListen),
   247  		Handler: mux,
   248  	}
   249  
   250  	// Start goroutine that shuts down `server` if the global context gets
   251  	// cancelled.
   252  	goroutineTracker.Add(1)
   253  	go func() {
   254  		defer goroutineTracker.Done()
   255  
   256  		// Start goroutine that actually listens for requests
   257  		goroutineTracker.Add(1)
   258  		go func() {
   259  			defer goroutineTracker.Done()
   260  
   261  			if err := server.ListenAndServeTLS(certPath, privatekeyPath); err != nil && err.Error() != "http: Server closed" {
   262  				close(events)
   263  				logger.Panicf(globalCancel, "Error listening and serving in httpserver: %s", err)
   264  			}
   265  		}()
   266  
   267  		// Listen for global context cancellation
   268  		<-globalCtx.Done()
   269  
   270  		logger.Infof("Shutting down httpserver...")
   271  		shutdownCtx, shutdownCancel := context.WithTimeout(globalCtx, 30*time.Second)
   272  		defer shutdownCancel()
   273  
   274  		err := server.Shutdown(shutdownCtx)
   275  		if err != nil {
   276  			logger.Infof("Shut down httpserver with error %s", err)
   277  		} else {
   278  			logger.Info("Gracefully shut down httpserver.")
   279  		}
   280  	}()
   281  
   282  	return events, nil
   283  }
