   786  // We need to use a custom compare function to compare protectedFromScaleDown maps
   787  // because the `UpdatedAt` is a timestamp set with `time.Now`.
   788  func compareProtectedMaps(a map[string]subscriptions.Image, b map[string]subscriptions.Image) bool {
   789  	var equal bool
   790  
   791  	if (len(a) == 0) && (len(b) == 0) {
   792  		equal = true
   793  	}
   794  
   795  	for k, v := range a {
   796  		for s, i := range b {
   797  			if k != s {
   798  				break
   799  			}
   800  			i.UpdatedAt = v.UpdatedAt
   801  			equal = reflect.DeepEqual(v, i)
   802  		}
   803  	}
   804  
   805  	return equal
   806  }
