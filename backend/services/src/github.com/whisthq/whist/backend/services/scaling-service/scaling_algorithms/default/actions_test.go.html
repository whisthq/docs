   622  // We need to use a custom compare function to compare protectedFromScaleDown maps
   623  // because the `UpdatedAt` is a timestamp set with `time.Now`.
   624  func compareProtectedMaps(a map[string]subscriptions.Image, b map[string]subscriptions.Image) bool {
   625  	var equal bool
   626  
   627  	if (len(a) == 0) && (len(b) == 0) {
   628  		equal = true
   629  	}
   630  
   631  	for k, v := range a {
   632  		for s, i := range b {
   633  			if k != s {
   634  				break
   635  			}
   636  			i.UpdatedAt = v.UpdatedAt
   637  			equal = reflect.DeepEqual(v, i)
   638  		}
   639  	}
   640  
   641  	return equal
   642  }
