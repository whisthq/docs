   859  // We need to use a custom compare function to compare protectedFromScaleDown maps
   860  // because the `UpdatedAt` is a timestamp set with `time.Now`.
   861  func compareProtectedMaps(a map[string]subscriptions.Image, b map[string]subscriptions.Image) bool {
   862  	var equal bool
   863  
   864  	if (len(a) == 0) && (len(b) == 0) {
   865  		equal = true
   866  	}
   867  
   868  	for k, v := range a {
   869  		for s, i := range b {
   870  			if k != s {
   871  				break
   872  			}
   873  			i.UpdatedAt = v.UpdatedAt
   874  			equal = reflect.DeepEqual(v, i)
   875  		}
   876  	}
   877  
   878  	return equal
   879  }
