    90  // generateInstanceCapacityMap uses the global instanceTypeToGPUNum and instanceTypeToVCPUNum maps
    91  // to generate the maximum mandelbox capacity for each instance type in the intersection
    92  // of their keys.
    93  func generateInstanceCapacityMap(instanceToGPUMap, instanceToVCPUMap map[string]int) map[string]int64 {
    94  	// Initialize the instance capacity map
    95  	capacityMap := map[string]int64{}
    96  	for instanceType, gpuNum := range instanceToGPUMap {
    97  		// Only populate for instances that are in both maps
    98  		vcpuNum, ok := instanceToVCPUMap[instanceType]
    99  		if !ok {
   100  			continue
   101  		}
   102  		min := utils.Min(gpuNum*constants.MaxMandelboxesPerGPU, vcpuNum/VCPUsPerMandelbox)
   103  		capacityMap[instanceType] = int64(min)
   104  	}
   105  	return capacityMap
   106  }

   138  // getFrontendVersion returns the current version of the frontend, which is initially
   139  // populated from the config database. This value is used by the scaling algorithm to

   143  func getFrontendVersion() string {
   144  	versionLock.Lock()
   145  	defer versionLock.Unlock()
   146  
   147  	var version string
   148  
   149  	if frontendVersion == nil {

   153  	switch metadata.GetAppEnvironment() {
   154  	case metadata.EnvDev:
   155  		version = utils.Sprintf("%v.%v.%v-dev-rc.%v", frontendVersion.Major, frontendVersion.Minor, frontendVersion.Micro, frontendVersion.DevRC)
   156  	case metadata.EnvStaging:
   157  		version = utils.Sprintf("%v.%v.%v-staging-rc.%v", frontendVersion.Major, frontendVersion.Minor, frontendVersion.Micro, frontendVersion.StagingRC)
   158  	default:
   159  		version = utils.Sprintf("%v.%v.%v", frontendVersion.Major, frontendVersion.Minor, frontendVersion.Micro)
   160  	}
   161  
   162  	return version
