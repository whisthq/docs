    77  // generateInstanceCapacityMap uses the global instanceTypeToGPUNum and instanceTypeToVCPUNum maps
    78  // to generate the maximum mandelbox capacity for each instance type in the intersection
    79  // of their keys.
    80  func generateInstanceCapacityMap(instanceToGPUMap, instanceToVCPUMap map[string]int) map[string]int {
    81  	// Initialize the instance capacity map
    82  	capacityMap := map[string]int{}
    83  	for instanceType, gpuNum := range instanceToGPUMap {
    84  		// Only populate for instances that are in both maps
    85  		vcpuNum, ok := instanceToVCPUMap[instanceType]
    86  		if !ok {
    87  			continue
    88  		}
    89  		capacityMap[instanceType] = utils.Min(gpuNum*constants.MaxMandelboxesPerGPU, vcpuNum/VCPUsPerMandelbox)
    90  	}
    91  	return capacityMap
    92  }
