   126  var getInstanceName = func(instanceID string, region string) (string, error) {
   127  	// Initialize general AWS config and EC2 client
   128  	cfg, err := config.LoadDefaultConfig(context.Background(), config.WithRegion(string(region)))
   129  	if err != nil {
   130  		return "", utils.MakeError("unable to load AWS SDK config: %s", err)
   131  	}
   132  	ec2Client := ec2.NewFromConfig(cfg)
   133  
   134  	var (
   135  		resourceTypeStr string = "resource-type"
   136  		resourceIDStr   string = "resource-id"
   137  	)
   138  
   139  	out, err := ec2Client.DescribeTags(context.TODO(), &ec2.DescribeTagsInput{
   140  		Filters: []ec2Types.Filter{
   141  			{
   142  				Name:   &resourceTypeStr,
   143  				Values: []string{"instance"},
   144  			},
   145  			{
   146  				Name:   &resourceIDStr,
   147  				Values: []string{string(instanceID)},
   148  			},
   149  		},
   150  	})
   151  
   152  	if err != nil {
   153  		return "", utils.MakeError("error describing tags: %s", err)
   154  	}
   155  
   156  	for _, t := range out.Tags {
   157  		if *(t.Key) == "Name" {
   158  			return *(t.Value), nil
   159  		}
   160  	}
   161  
   162  	// "Name" tag wasn't found, so we print out all the ones we did find.
   163  	printableTags := make(map[string]string)
   164  	for _, t := range out.Tags {
   165  		printableTags[*(t.Key)] = *(t.Value)
   166  	}
   167  
   168  	return "", utils.MakeError(`did not find a "Name" tag for this instance! Found these tags instead: %v`, printableTags)
   169  }

   171  func metadataRetriever(resource string) (string, error) {
   172  	httpClient := http.Client{
   173  		Timeout: 1 * time.Second,
   174  	}

   179  	}

   181  	resp, err := httpClient.Get(u.String())
   182  	if err != nil {
   183  		return "", utils.MakeError("error retrieving data from URL %s: %v. Is the service running on an AWS EC2 instance?", u.String(), err)
   184  	}
   185  	defer resp.Body.Close()
   186

   190  	}
   191  	if resp.StatusCode != http.StatusOK {
   192  		return "", utils.MakeError("got non-200 response from URL %s: %s", u.String(), resp.Status)
   193  	}
