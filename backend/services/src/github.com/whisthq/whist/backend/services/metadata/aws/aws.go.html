   165  // This helper function generates functions that retrieve metadata from the
   166  // internal AWS endpoint. We use this together with
   167  // utils.MemoizeStringWithError() to expose memoized functions that query
   168  // information about the host from AWS.  We use this setup so that we only have
   169  // to query the AWS endpoint successfully once to get information for our
   170  // startup, shutdown, and heartbeat messages. This architecture is compelling
   171  // because of reports online that the AWS endpoint to query this information
   172  // has transient failures. We want to minimize the impact of these transient
   173  // failures on our application --- with this approach, host startup will be
   174  // affected by these failures, but transient failures should not affect already
   175  // running hosts.
   176  func generateAWSMetadataRetriever(path string) func() (string, error) {
   177  	// Note that we could use the ec2imds package of the AWS SDK, but that's
   178  	// proven unnecessary over months of usage of the (simple) below code in
   179  	// production so far. In the future, we could make the migration at the price
   180  	// of a small increase in code complexity.
   181  
   182  	const AWSendpointBase = "http://169.254.169.254/latest/meta-data/"
   183  	httpClient := http.Client{
   184  		Timeout: 1 * time.Second,
   185  	}
   186  
   187  	url := AWSendpointBase + path
   188  	return func() (string, error) {
   189  		resp, err := httpClient.Get(url)
   190  		if err != nil {
   191  			return "", utils.MakeError("Error retrieving data from URL %s: %v. Is the service running on an AWS EC2 instance?", url, err)
   192  		}
   193  		defer resp.Body.Close()
   194  
   195  		body, err := io.ReadAll(resp.Body)
   196  		if err != nil {
   197  			return string(body), utils.MakeError("Error reading response body from URL %s: %v", url, err)
   198  		}
   199  		if resp.StatusCode != 200 {
   200  			return string(body), utils.MakeError("Got non-200 response from URL %s: %s", url, resp.Status)
   201  		}
   202  		return string(body), nil
   203  	}
   204  }
