   129  var getInstanceName = func(instanceID string, region string) (string, error) {
   130  	// Initialize general AWS config and EC2 client
   131  	cfg, err := config.LoadDefaultConfig(context.Background(), config.WithRegion(string(region)))
   132  	if err != nil {
   133  		return "", utils.MakeError("unable to load AWS SDK config: %s", err)
   134  	}
   135  	ec2Client := ec2.NewFromConfig(cfg)
   136  
   137  	var (
   138  		resourceTypeStr string = "resource-type"
   139  		resourceIDStr   string = "resource-id"
   140  	)
   141  
   142  	out, err := ec2Client.DescribeTags(context.TODO(), &ec2.DescribeTagsInput{
   143  		Filters: []ec2Types.Filter{
   144  			{
   145  				Name:   &resourceTypeStr,
   146  				Values: []string{"instance"},
   147  			},
   148  			{
   149  				Name:   &resourceIDStr,
   150  				Values: []string{string(instanceID)},
   151  			},
   152  		},
   153  	})
   154  
   155  	if err != nil {
   156  		return "", utils.MakeError("error describing tags: %s", err)
   157  	}
   158  
   159  	for _, t := range out.Tags {
   160  		if *(t.Key) == "Name" {
   161  			return *(t.Value), nil
   162  		}
   163  	}
   164  
   165  	// "Name" tag wasn't found, so we print out all the ones we did find.
   166  	printableTags := make(map[string]string)
   167  	for _, t := range out.Tags {
   168  		printableTags[*(t.Key)] = *(t.Value)
   169  	}
   170  
   171  	return "", utils.MakeError(`did not find a "Name" tag for this instance! Found these tags instead: %v`, printableTags)
   172  }

   175  	httpClient := http.Client{
   176  		Timeout: 1 * time.Second,
   177  	}

   179  	u, err := url.Parse(EndpointBase)

   183  	u.Path = path.Join("latest", "meta-data", resource)

   185  	if err != nil {

   187  	}
