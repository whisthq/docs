    20  // serviceConfig stores service-global configuration values.
    21  type serviceConfig struct {
    22  	// enabledRegions is the list of regions in which users are allowed to request
    23  	// Mandelboxes.

    30  	// frontendVersion represents the current version of the frontend
    31  	// (e.g. "2.6.13").
    32  	frontendVersion string
    33  
    34  	// targetFreeMandelboxes maps region names to integers, where the integer
    35  	// associated with each region is the number of Mandelboxes we want to have

    37  	// of Mandelboxes we have available at any given time, but that's what scaling
    38  	// algorithms are for.

    40  	// TODO(owen): It might make more sense to store a mapping from strings to
    41  	// uint32s because we can't have negative Mandelboxes available. I haven't

    43  	// doing a lot of type conversions.
    44  	targetFreeMandelboxes map[string]int
