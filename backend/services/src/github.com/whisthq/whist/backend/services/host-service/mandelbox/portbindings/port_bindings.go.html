    45  // Reserve lets us mark certain ports as "reserved" so they don't get
    46  // allocated for mandelboxes. This needs to be called at program initialization
    47  // (ideally in an `init` function), before any mandelboxes are started.
    48  func Reserve(num uint16, protocol TransportProtocol) {
    49  	if !isInAllowedRange(num) {
    50  		logger.Errorf("Could not reserve port %v/%s as port is not in allowed range.", num, protocol)
    51  		return
    52  	}
    53  
    54  	mapToUse, err := getProtocolSpecificHostPortMap(protocol)
    55  	if err != nil {
    56  		logger.Errorf("Could not reserve port %v/%s. Err: %s", num, protocol, err)
    57  		return
    58  	}
    59  
    60  	if status, exists := (*mapToUse)[num]; exists {
    61  		logger.Errorf("Could not reserve port %v/%s. Port exists on table and is set to %v", num, protocol, status)
    62  		return
    63  	}
    64  
    65  	(*mapToUse)[num] = reserved
    66  	logger.Infof("Marked Port %v/%s as reserved", num, protocol)
    67  }

    81  // allocateSinglePort allocates a single port given a desired binding. It
    82  // requires that `portMapsLock` is held throughout. Just like `Allocate()`, it
    83  // interprets a zero value for `desiredBind.HostPort` as a request to allocate
    84  // any host port, and a nonzero value as a request to allocate that specific
    85  // host port.
    86  func allocateSinglePort(desiredBind PortBinding) (PortBinding, error) {
    87  	mapToUse, err := getProtocolSpecificHostPortMap(desiredBind.Protocol)
    88  	if err != nil {
    89  		return PortBinding{}, utils.MakeError("allocateSinglePort failed. Error: %s", err)
    90  	}
    91  
    92  	// If the given HostPort is nonzero, we want to use that one specifically.
    93  	// Else, we have to randomly allocate a free one.
    94  	if desiredBind.HostPort != 0 {
    95  		// Check that the desired port is actually in the allowed range
    96  		if !isInAllowedRange(desiredBind.HostPort) {
    97  			return PortBinding{}, utils.MakeError("allocateSinglePort: received a request to allocate a disallowed port: %v/%s", desiredBind.HostPort, desiredBind.Protocol)
    98  		}
    99  
   100  		// Check that this port isn't already allocated to a mandelbox, or reserved
   101  		if _, exists := (*mapToUse)[desiredBind.HostPort]; exists {
   102  			return PortBinding{}, utils.MakeError("allocateSinglePort: Could not allocate HostPort %v/%v: already bound or reserved.", desiredBind.HostPort, desiredBind.Protocol)
   103  		}
   104  
   105  		// Mark it as allocated and return
   106  		(*mapToUse)[desiredBind.HostPort] = inUse
   107  		return desiredBind, nil
   108  	}
   109  
   110  	// Gotta allocate a port ourselves
   111  	var randomPort uint16 = 0
   112  	maxTries := 100
   113  	for numTries := 0; numTries < maxTries; numTries++ {
   114  		temporaryPort := randomPortInAllowedRange()
   115  		if _, exists := (*mapToUse)[temporaryPort]; !exists {
   116  			// Assign randomPort if available
   117  			randomPort = temporaryPort
   118  			break
   119  		}
   120  	}
   121  	if randomPort == uint16(0) {
   122  		return PortBinding{}, utils.MakeError("Tried %v times to allocate a host port for mandelbox port %v/%v. Breaking out to avoid spinning for too long.", maxTries, desiredBind.HostPort, desiredBind.Protocol)
   123  	}
   124  
   125  	// Mark it as allocated and return
   126  	(*mapToUse)[randomPort] = inUse
   127  	return PortBinding{
   128  		MandelboxPort: desiredBind.MandelboxPort,
   129  		HostPort:      randomPort,
   130  		Protocol:      desiredBind.Protocol,
   131  		BindIP:        desiredBind.BindIP,
   132  	}, nil
   133  }

   135  // freeSinglePort marks the given port as free in the correct
   136  // `protocolSpecificHostPortMap`. If the given port is reserved, it logs an
   137  // error. This function requires that `portMapsLock` is held throughout.
   138  func freeSinglePort(bind PortBinding) {
   139  	mapToUse, err := getProtocolSpecificHostPortMap(bind.Protocol)
   140  	if err != nil {
   141  		logger.Errorf("FreePortBindings: failed for bind %v. Error: %s", bind, err)
   142  		return
   143  	}
   144  
   145  	v, ok := (*mapToUse)[bind.HostPort]
   146  	if !ok {
   147  		logger.Errorf("FreePortBindings: tried to remove nonexistent binding for %v/%v.", bind.HostPort, bind.Protocol)
   148  	} else if v == reserved {
   149  		logger.Errorf("FreePortBindings: tried to remove reserved binding for %v/%v. Sorry, but I can't let you do that.", bind.HostPort, bind.Protocol)
   150  	} else {
   151  		delete(*mapToUse, bind.HostPort)
   152  	}
   153  }

   170  // Allocate allocates the desired port bindings atomically. In other words, it
   171  // will either successfully allocate all of them, or none of them. It
   172  // interprets a zero value for `desiredBind.HostPort` as a request to allocate
   173  // any host port, and a nonzero value as a request to allocate that specific
   174  // host port.
   175  func Allocate(desiredBinds []PortBinding) ([]PortBinding, error) {
   176  	portMapsLock.Lock()
   177  	defer portMapsLock.Unlock()
   178  
   179  	var result = make([]PortBinding, 0, len(desiredBinds))
   180  	var reterr error = nil
   181  
   182  	for _, v := range desiredBinds {
   183  		var b PortBinding
   184  		b, reterr = allocateSinglePort(v)
   185  		if reterr != nil {
   186  			break
   187  		} else {
   188  			result = append(result, b)
   189  		}
   190  	}
   191  
   192  	// If one allocation failed, we don't want to leak any ports.
   193  	if reterr != nil {
   194  		for _, v := range result {
   195  			freeSinglePort(v)
   196  		}
   197  		return nil, reterr
   198  	}
   199  
   200  	return result, nil
   201  }
