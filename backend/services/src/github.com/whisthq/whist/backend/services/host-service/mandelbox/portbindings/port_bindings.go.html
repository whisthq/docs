    44  // Reserve lets us mark certain ports as "reserved" so they don't get
    45  // allocated for mandelboxes. This needs to be called at program initialization
    46  // (ideally in an `init` function), before any mandelboxes are started.
    47  func Reserve(num uint16, protocol TransportProtocol) {
    48  	if !isInAllowedRange(num) {
    49  		logger.Errorf("Could not reserve port %v/%s as port is not in allowed range.", num, protocol)
    50  		return
    51  	}
    52  
    53  	mapToUse, err := getProtocolSpecificHostPortMap(protocol)
    54  	if err != nil {
    55  		logger.Errorf("Could not reserve port %v/%s. Err: %s", num, protocol, err)
    56  		return
    57  	}
    58  
    59  	if status, exists := (*mapToUse)[num]; exists {
    60  		logger.Errorf("Could not reserve port %v/%s. Port exists on table and is set to %v", num, protocol, status)
    61  		return
    62  	}
    63  
    64  	(*mapToUse)[num] = reserved
    65  	logger.Infof("Marked Port %v/%s as reserved", num, protocol)
    66  }

    80  // allocateSinglePort allocates a single port given a desired binding. It
    81  // requires that `portMapsLock` is held throughout. Just like `Allocate()`, it
    82  // interprets a zero value for `desiredBind.HostPort` as a request to allocate
    83  // any host port, and a nonzero value as a request to allocate that specific
    84  // host port.
    85  func allocateSinglePort(desiredBind PortBinding) (PortBinding, error) {
    86  	mapToUse, err := getProtocolSpecificHostPortMap(desiredBind.Protocol)
    87  	if err != nil {
    88  		return PortBinding{}, utils.MakeError("allocateSinglePort failed. Error: %s", err)
    89  	}
    90  
    91  	// If the given HostPort is nonzero, we want to use that one specifically.
    92  	// Else, we have to randomly allocate a free one.
    93  	if desiredBind.HostPort != 0 {
    94  		// Check that the desired port is actually in the allowed range
    95  		if !isInAllowedRange(desiredBind.HostPort) {
    96  			return PortBinding{}, utils.MakeError("allocateSinglePort: received a request to allocate a disallowed port: %v/%s", desiredBind.HostPort, desiredBind.Protocol)
    97  		}
    98  
    99  		// Check that this port isn't already allocated to a mandelbox, or reserved
   100  		if _, exists := (*mapToUse)[desiredBind.HostPort]; exists {
   101  			return PortBinding{}, utils.MakeError("allocateSinglePort: Could not allocate HostPort %v/%v: already bound or reserved.", desiredBind.HostPort, desiredBind.Protocol)
   102  		}
   103  
   104  		// Mark it as allocated and return
   105  		(*mapToUse)[desiredBind.HostPort] = inUse
   106  		return desiredBind, nil
   107  	}
   108  
   109  	// Gotta allocate a port ourselves
   110  	var randomPort uint16 = 0
   111  	maxTries := 100
   112  	for numTries := 0; numTries < maxTries; numTries++ {
   113  		temporaryPort := randomPortInAllowedRange()
   114  		if _, exists := (*mapToUse)[temporaryPort]; !exists {
   115  			// Assign randomPort if available
   116  			randomPort = temporaryPort
   117  			break
   118  		}
   119  	}
   120  	if randomPort == uint16(0) {
   121  		return PortBinding{}, utils.MakeError("Tried %v times to allocate a host port for mandelbox port %v/%v. Breaking out to avoid spinning for too long.", maxTries, desiredBind.HostPort, desiredBind.Protocol)
   122  	}
   123  
   124  	// Mark it as allocated and return
   125  	(*mapToUse)[randomPort] = inUse
   126  	return PortBinding{
   127  		MandelboxPort: desiredBind.MandelboxPort,
   128  		HostPort:      randomPort,
   129  		Protocol:      desiredBind.Protocol,
   130  		BindIP:        desiredBind.BindIP,
   131  	}, nil
   132  }

   134  // freeSinglePort marks the given port as free in the correct
   135  // `protocolSpecificHostPortMap`. If the given port is reserved, it logs an
   136  // error. This function requires that `portMapsLock` is held throughout.
   137  func freeSinglePort(bind PortBinding) {
   138  	mapToUse, err := getProtocolSpecificHostPortMap(bind.Protocol)
   139  	if err != nil {
   140  		logger.Errorf("FreePortBindings: failed for bind %v. Error: %s", bind, err)
   141  		return
   142  	}
   143  
   144  	v, ok := (*mapToUse)[bind.HostPort]
   145  	if !ok {
   146  		logger.Errorf("FreePortBindings: tried to remove nonexistent binding for %v/%v.", bind.HostPort, bind.Protocol)
   147  	} else if v == reserved {
   148  		logger.Errorf("FreePortBindings: tried to remove reserved binding for %v/%v. Sorry, but I can't let you do that.", bind.HostPort, bind.Protocol)
   149  	} else {
   150  		delete(*mapToUse, bind.HostPort)
   151  	}
   152  }

   154  // Free marks all provided, non-reserved host ports as free in `tcpPortMap` and

   156  func Free(binds []PortBinding) {
   157  	if len(binds) == 0 {
   158  		return
   159  	}
   160  
   161  	portMapsLock.Lock()
   162  	defer portMapsLock.Unlock()
   163  
   164  	for _, bind := range binds {
   165  		freeSinglePort(bind)
   166  	}

   169  // Allocate allocates the desired port bindings atomically. In other words, it
   170  // will either successfully allocate all of them, or none of them. It
   171  // interprets a zero value for `desiredBind.HostPort` as a request to allocate
   172  // any host port, and a nonzero value as a request to allocate that specific
   173  // host port.
   174  func Allocate(desiredBinds []PortBinding) ([]PortBinding, error) {
   175  	portMapsLock.Lock()
   176  	defer portMapsLock.Unlock()
   177  
   178  	var result = make([]PortBinding, 0, len(desiredBinds))
   179  	var reterr error = nil
   180  
   181  	for _, v := range desiredBinds {
   182  		var b PortBinding
   183  		b, reterr = allocateSinglePort(v)
   184  		if reterr != nil {
   185  			break
   186  		} else {
   187  			result = append(result, b)
   188  		}
   189  	}
   190  
   191  	// If one allocation failed, we don't want to leak any ports.
   192  	if reterr != nil {
   193  		for _, v := range result {
   194  			freeSinglePort(v)
   195  		}
   196  		return nil, reterr
   197  	}
   198  
   199  	return result, nil
   200  }
