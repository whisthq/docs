     4  This file contains the code related to starting the scaling service's HTTP server,

    44  // mandelboxAssignHandler is the http handler for any requests to the `/assign` endpoint. It authenticates
    45  // the request, verifies the type, and parses the body. After that it sends it to the server events channel.
    46  func mandelboxAssignHandler(w http.ResponseWriter, r *http.Request, events chan<- algos.ScalingEvent) {
    47  	// Verify that we got a POST request
    48  	err := verifyRequestType(w, r, http.MethodPost)
    49  	if err != nil {
    50  		logger.Errorf("error verifying request type: %s", err)
    51  		http.Error(w, http.StatusText(http.StatusBadRequest), http.StatusBadRequest)
    52  		return
    53  	}
    54  
    55  	var reqdata httputils.MandelboxAssignRequest
    56  	claims, err := httputils.AuthenticateRequest(w, r, &reqdata)
    57  	if err != nil {
    58  		logger.Error(err)
    59  		return
    60  	}
    61  
    62  	if !metadata.IsLocalEnv() {
    63  		// Add user id to the request. This way we don't expose the
    64  		// access token to other processes that don't need access to it.
    65  		reqdata.UserID = types.UserID(claims.Subject)
    66  	}
    67  
    68  	// Once we have authenticated and validated the request send it to the scaling
    69  	// algorithm for processing. Mandelbox assign is region-agnostic so we don't need
    70  	// to specify a region on the event. The event handler will select the default region automatically.
    71  	events <- algos.ScalingEvent{
    72  		ID:   uuid.NewString(),
    73  		Type: "SERVER_MANDELBOX_ASSIGN_EVENT",
    74  		Data: &reqdata,
    75  	}
    76  	res := <-reqdata.ResultChan
    77  	assignResult := res.Result.(httputils.MandelboxAssignRequestResult)
    78  
    79  	var (
    80  		buf    []byte
    81  		status int
    82  	)
    83  
    84  	if assignResult.Error != "" {
    85  		// Send a 503
    86  		status = http.StatusServiceUnavailable
    87  	} else {
    88  		// Send a 200 code
    89  		status = http.StatusOK
    90  	}
    91  
    92  	buf, err = json.Marshal(assignResult)
    93  	w.WriteHeader(status)
    94  	if err != nil {
    95  		logger.Errorf("error marshalling a %d HTTP Response body: %s", status, err)
    96  	}
    97  	_, _ = w.Write(buf)
    98  }

   100  // paymentSessionHandler handles a payment session request. It verifies the access token,
   101  // and then creates a StripeClient to get the URL of a payment session.
   102  func paymentSessionHandler(w http.ResponseWriter, r *http.Request) {
   103  	// Verify that we got a GET request
   104  	err := verifyRequestType(w, r, http.MethodGet)
   105  	if err != nil {
   106  		// err is already logged
   107  		http.Error(w, "", http.StatusMethodNotAllowed)
   108  		return
   109  	}
   110  
   111  	// Extract access token from request header
   112  	accessToken, err := httputils.GetAccessToken(r)
   113  	if err != nil {
   114  		logger.Error(err)
   115  		http.Error(w, "Did not receive an access token", http.StatusUnauthorized)
   116  		return
   117  	}
   118  
   119  	// Get claims from access token
   120  	claims, err := auth.ParseToken(accessToken)
   121  	if err != nil {
   122  		logger.Errorf("received an unpermissioned backend request on %s to URL %s: %s", r.Host, r.URL, err)
   123  		http.Error(w, "Invalid access token", http.StatusUnauthorized)
   124  		return
   125  	}
   126  
   127  	// Setup a config database GraphQL client to get the Stripe configurations.
   128  	// The client is simply passed to the payments client, no database operations
   129  	// are done here.
   130  	useConfigDB := true
   131  
   132  	configGraphqlClient := &subscriptions.GraphQLClient{}
   133  	err = configGraphqlClient.Initialize(useConfigDB)
   134  	if err != nil {
   135  		logger.Errorf("failed to setup config GraphQL client: %s", err)
   136  		http.Error(w, "", http.StatusInternalServerError)
   137  	}
   138  
   139  	// Create a new StripeClient to handle the customer
   140  	paymentsClient := &payments.PaymentsClient{}
   141  	stripeClient := &payments.StripeClient{}
   142  	err = paymentsClient.Initialize(claims.CustomerID, claims.SubscriptionStatus, configGraphqlClient, stripeClient)
   143  	if err != nil {
   144  		logger.Errorf("failed to Initialize Stripe Client: %s", err)
   145  		http.Error(w, "", http.StatusInternalServerError)
   146  		return
   147  	}
   148  
   149  	sessionUrl, err := paymentsClient.CreateSession()
   150  	if err != nil {
   151  		logger.Errorf("failed to create Stripe Session: %s", err)
   152  		http.Error(w, "", http.StatusInternalServerError)
   153  		return
   154  	}
   155  
   156  	// Marshal and send session URL
   157  	buf, err := json.Marshal(map[string]string{
   158  		"url": sessionUrl,
   159  	})
   160  	if err != nil {
   161  		logger.Errorf("error marshalling HTTP Response body: %s", err)
   162  		http.Error(w, "", http.StatusInternalServerError)
   163  		return
   164  	}
   165  
   166  	w.WriteHeader(http.StatusOK)
   167  	_, _ = w.Write(buf)
   168  }

   170  // processJSONTransportRequest processes an HTTP request to receive data
   171  // directly from the frontend.
   172  func processJSONTransportRequest(w http.ResponseWriter, r *http.Request) {
   173  	// Verify that it is an PUT request
   174  	if httputils.VerifyRequestType(w, r, http.MethodPut) != nil {
   175  		http.Error(w, "", http.StatusMethodNotAllowed)
   176  		return
   177  	}
   178  
   179  	// Extract access token from request header to pass to host service
   180  	accessToken, err := httputils.GetAccessToken(r)
   181  	if err != nil {
   182  		logger.Error(err)
   183  		http.Error(w, "Did not receive an access token", http.StatusUnauthorized)
   184  		return
   185  	}
   186  
   187  	// Verify authorization and unmarshal into the right object type
   188  	var reqdata httputils.JSONTransportRequest
   189  	if _, err := httputils.AuthenticateRequest(w, r, &reqdata); err != nil {
   190  		logger.Error(err)
   191  		return
   192  	}
   193  
   194  	// Send the request to the instance and then return the response
   195  	url := utils.Sprintf("https://%s:%v/json_transport", reqdata.IP, HostServicePort)
   196  
   197  	jsonBody, err := json.Marshal(reqdata)
   198  	if err != nil {
   199  		logger.Errorf("error marshalling body for host service: %s", err)
   200  		return
   201  	}
   202  
   203  	// Create a new request
   204  	bodyReader := bytes.NewReader(jsonBody)
   205  	hostReq, err := http.NewRequest("PUT", url, bodyReader)
   206  	if err != nil {
   207  		logger.Errorf("failed to create JSON transport request for host service: %s", err)
   208  		http.Error(w, "", http.StatusInternalServerError)
   209  		return
   210  	}
   211  
   212  	// Add necessary headers to host request
   213  	hostReq.Header.Add("content-type", "application/json")
   214  	hostReq.Header.Add("Authorization", utils.Sprintf("Bearer %s", accessToken))
   215  	hostReq.Close = true
   216  
   217  	// Instanciate a new http client with a custom transport
   218  	tr := &http.Transport{
   219  		TLSClientConfig: &tls.Config{
   220  			InsecureSkipVerify: true,
   221  		},
   222  	}
   223  	client := &http.Client{Transport: tr}
   224  
   225  	// Now that request is fully assembled and the client is initialized, send the request
   226  	res, err := client.Do(hostReq)
   227  	defer hostReq.Body.Close()
   228  
   229  	if err != nil {
   230  		logger.Errorf("failed to send JSON transport request to instance: %s", err)
   231  		http.Error(w, "", http.StatusInternalServerError)
   232  		return
   233  	}
   234  
   235  	hostBody, err := io.ReadAll(res.Body)
   236  	if err != nil {
   237  		logger.Errorf("could not read host response body: %s", err)
   238  		http.Error(w, "", http.StatusInternalServerError)
   239  		return
   240  	}
   241  
   242  	w.WriteHeader(http.StatusOK)
   243  	_, _ = w.Write(hostBody)
   244  }

   246  // throttleMiddleware will limit requests on the endpoint using the provided rate limiter.
   247  // It uses a token bucket algorithm, so that every interval of time the "bucket" will refill
   248  // and continue to serve tokens up to a maximum defined by the burst capacity. In case the
   249  // limit is exceeded, return a http 429 error (too many requests).
   250  func throttleMiddleware(limiter *rate.Limiter, f func(http.ResponseWriter, *http.Request)) func(http.ResponseWriter, *http.Request) {
   251  	return http.HandlerFunc(func(rw http.ResponseWriter, r *http.Request) {
   252  		if !limiter.Allow() {
   253  			http.Error(rw, http.StatusText(http.StatusTooManyRequests), http.StatusTooManyRequests)
   254  		}
   255  		f(rw, r)
   256  	})
   257  }

   259  // verifyPaymentMiddleware will verify if the user has an active subscription in Stripe. This
   260  // middleware should be used to restrict endpoint for paying users only.
   261  func VerifyPaymentMiddleware(f func(http.ResponseWriter, *http.Request)) http.Handler {
   262  	return http.HandlerFunc(func(rw http.ResponseWriter, r *http.Request) {
   263  		// Get and parse Authorization header from access token
   264  		accessToken, err := httputils.GetAccessToken(r)
   265  		if err != nil {
   266  			logger.Error(err)
   267  			http.Error(rw, http.StatusText(http.StatusUnauthorized), http.StatusUnauthorized)
   268  			return
   269  		}
   270  
   271  		// Verify if the user's subscription is valid
   272  		paymentValid, err := payments.VerifyPayment(accessToken)
   273  		if err != nil {
   274  			logger.Errorf("failed to validate Stripe payment from access token: %s", err)
   275  			http.Error(rw, http.StatusText(http.StatusInternalServerError), http.StatusInternalServerError)
   276  			return
   277  		}
   278  
   279  		if !paymentValid {
   280  			logger.Warningf("User does not have an active Stripe subscription.")
   281  			http.Error(rw, http.StatusText(http.StatusForbidden), http.StatusForbidden)
   282  			return
   283  		}
   284  		f(rw, r)
   285  	})
   286  }

   288  // Function to verify the type (method) of a request
   289  func verifyRequestType(w http.ResponseWriter, r *http.Request, method string) error {
   290  	if r == nil {
   291  		err := utils.MakeError("received a nil request expecting to be type %s", method)
   292  		logger.Error(err)
   293  
   294  		http.Error(w, utils.Sprintf("Bad request. Expected %s, got nil", method), http.StatusBadRequest)
   295  
   296  		return err
   297  	}
   298  
   299  	if r.Method != method {
   300  		err := utils.MakeError("received a request on %s to URL %s of type %s, but it should have been type %s", r.Host, r.URL, r.Method, method)
   301  		logger.Error(err)
   302  
   303  		http.Error(w, utils.Sprintf("Bad request type. Expected %s, got %s", method, r.Method), http.StatusBadRequest)
   304  
   305  		return err
   306  	}
   307  	return nil
   308  }

   310  // StartHTTPServer will start the server with the necessary configurations and
   311  // assign the handlers to each endpoint.
   312  func StartHTTPServer(events chan algos.ScalingEvent) {
   313  	logger.Infof("Starting HTTP server...")
   314  
   315  	createHandler := func(f func(http.ResponseWriter, *http.Request, chan<- algos.ScalingEvent)) func(http.ResponseWriter, *http.Request) {
   316  		return func(w http.ResponseWriter, r *http.Request) {
   317  			f(w, r, events)
   318  		}
   319  	}
   320  
   321  	// Start a new rate limiter. This will limit requests on an endpoint
   322  	// to every `interval` with a burst of up to `burst` requests. This
   323  	// will help mitigate Denial of Service attacks, or a rogue frontend.
   324  	interval := 1 * time.Second
   325  	burst := 10
   326  	limiter := rate.NewLimiter(rate.Every(interval), burst)
   327  
   328  	// Create the final handlers, with the necessary middleware
   329  	assignHandler := throttleMiddleware(limiter, httputils.EnableCORS(createHandler(mandelboxAssignHandler)))
   330  	jsonTransportHandler := httputils.EnableCORS(processJSONTransportRequest)
   331  	paymentsHandler := httputils.EnableCORS(paymentSessionHandler)
   332  
   333  	// Create a custom HTTP Request Multiplexer
   334  	mux := http.NewServeMux()
   335  	mux.Handle("/", http.NotFoundHandler())
   336  	mux.Handle("/mandelbox/assign", http.HandlerFunc(assignHandler))
   337  	mux.Handle("/json_transport", http.HandlerFunc(jsonTransportHandler))
   338  	mux.Handle("/payment_portal_url", http.HandlerFunc(paymentsHandler))
   339  
   340  	// The PORT env var will be automatically set by Heroku.
   341  	// If running on localdev, use default port.
   342  	port := os.Getenv("PORT")
   343  	if metadata.IsLocalEnv() || port == "" {
   344  		port = "7730"
   345  	}
   346  
   347  	// Set read/write timeouts to help mitigate potential rogue clients
   348  	// or DDOS attacks.
   349  	srv := &http.Server{
   350  		Addr:         utils.Sprintf("0.0.0.0:%v", port),
   351  		ReadTimeout:  10 * time.Second,
   352  		WriteTimeout: 10 * time.Second,
   353  		IdleTimeout:  30 * time.Second,
   354  		Handler:      mux,
   355  	}
   356  
   357  	go func() {
   358  		logger.Error(srv.ListenAndServe())
   359  	}()
   360  }
