   153  	// Verify that it is an PUT request
   154  	if httputils.VerifyRequestType(w, r, http.MethodPut) != nil {

   180  		return

   189

   191  	if err != nil {
   192  		logger.Errorf("Failed to send JSON transport request to instance. Err: %v", err)
   193  		http.Error(w, "", http.StatusInternalServerError)
   194  		return
   195  	}

   198  	if err != nil {

   233  func getAccessToken(r *http.Request) (string, error) {
   234  	authorization := r.Header.Get("Authorization")
   235  	bearer := strings.Split(authorization, "Bearer ")
   236  	if len(bearer) <= 1 {
   237  		return "", utils.MakeError("Bearer token is empty.")

   240  	return accessToken, nil

   243  // throttleMiddleware will limit requests on the endpoint using the provided rate limiter.
   244  // It uses a token bucket algorithm, so that every interval of time the "bucket" will refill
   245  // and continue to serve tokens up to a maximum defined by the burst capacity. In case the
   246  // limit is exceeded, return a http 429 error (too many requests).
   247  func throttleMiddleware(limiter *rate.Limiter, f func(http.ResponseWriter, *http.Request)) func(http.ResponseWriter, *http.Request) {
   248  	return http.HandlerFunc(func(rw http.ResponseWriter, r *http.Request) {
   249  		if !limiter.Allow() {
   250  			http.Error(rw, http.StatusText(http.StatusTooManyRequests), http.StatusTooManyRequests)
   251  		}
   252  		f(rw, r)
   253  	})
   254  }

   256  // verifyPaymentMiddleware will verify if the user has an active subscription in Stripe. This
   257  // middleware should be used to restrict endpoint for paying users only.
   258  func verifyPaymentMiddleware(f func(http.ResponseWriter, *http.Request)) http.Handler {
   259  	return http.HandlerFunc(func(rw http.ResponseWriter, r *http.Request) {
   260  		// Get and parse Authorization header from access token
   261  		accessToken, err := getAccessToken(r)
   262  		if err != nil {
   263  			logger.Error(err)
   264  			http.Error(rw, http.StatusText(http.StatusUnauthorized), http.StatusUnauthorized)
   265  			return
   266  		}
   267  
   268  		if len(strings.Split(accessToken, "Bearer")) <= 1 {
   269  			logger.Infof("Access token does not have Bearer header. Trying to parse token as is.")
   270  		} else {
   271  			accessToken = strings.Split(accessToken, "Bearer ")[1]
   272  		}
   273  
   274  		// Verify if the user's subscription is valid
   275  		paymentValid, err := payments.VerifyPayment(accessToken)
   276  		if err != nil {
   277  			logger.Errorf("Failed to validate Stripe payment from access token. Err: %v", err)
   278  			http.Error(rw, http.StatusText(http.StatusInternalServerError), http.StatusInternalServerError)
   279  			return
   280  		}
   281  
   282  		if !paymentValid {
   283  			logger.Warningf("User does not have an active Stripe subscription.")
   284  			http.Error(rw, http.StatusText(http.StatusForbidden), http.StatusForbidden)
   285  			return
   286  		}
   287  		f(rw, r)
   288  	})
   289  }

   292  func verifyRequestType(w http.ResponseWriter, r *http.Request, method string) error {
   293  	if r == nil {
   294  		err := utils.MakeError("Received a nil request expecting to be type %s", method)
   295  		logger.Error(err)
   296  
   297  		http.Error(w, utils.Sprintf("Bad request. Expected %s, got nil", method), http.StatusBadRequest)
   298  
   299  		return err
   300  	}
   301  
   302  	if r.Method != method {
   303  		err := utils.MakeError("Received a request on %s to URL %s of type %s, but it should have been type %s", r.Host, r.URL, r.Method, method)
   304  		logger.Error(err)
   305  
   306  		http.Error(w, utils.Sprintf("Bad request type. Expected %s, got %s", method, r.Method), http.StatusBadRequest)
   307  
   308  		return err
   309  	}
   310  	return nil
   311  }

   313  func StartHTTPServer(events chan algos.ScalingEvent) {
   314  	logger.Infof("Starting HTTP server...")
   315  
   316  	createHandler := func(f func(http.ResponseWriter, *http.Request, chan<- algos.ScalingEvent)) func(http.ResponseWriter, *http.Request) {
   317  		return func(w http.ResponseWriter, r *http.Request) {
   318  			f(w, r, events)
   319  		}
   320  	}
   321  
   322  	// Start a new rate limiter. This will limit requests on an endpoint
   323  	//  to every `interval` with a burst of up to `burst` requests. This
   324  	// will help mitigate Denial of Service attacks, or a client app
   325  	// spamming too many requests.
   326  	interval := 1 * time.Second
   327  	burst := 10
   328  	limiter := rate.NewLimiter(rate.Every(interval), burst)
   329  
   330  	// Create the final handlers, with the necessary middleware
   331  	assignHandler := verifyPaymentMiddleware(throttleMiddleware(limiter, httputils.EnableCORS(createHandler(mandelboxAssignHandler))))
   332  	jsonTransportHandler := httputils.EnableCORS(processJSONTransportRequest)
   333  	paymentsHandler := httputils.EnableCORS(paymentSessionHandler)
   334  
   335  	// Create a custom HTTP Request Multiplexer
   336  	mux := http.NewServeMux()
   337  	mux.Handle("/", http.NotFoundHandler())
   338  	mux.Handle("/mandelbox/assign", assignHandler)
   339  	mux.Handle("/json_transport", http.HandlerFunc(jsonTransportHandler))
   340  	mux.Handle("/payment_portal_url", http.HandlerFunc(paymentsHandler))
   341  
   342  	// The PORT env var will be automatically set by Heroku.
   343  	// If running on localdev, use default port.
   344  	port := os.Getenv("PORT")
   345  	if metadata.IsLocalEnv() || port == "" {
   346  		port = "7730"
   347  	}
   348  
   349  	// Set read/write timeouts to help mitigate potential rogue clients
   350  	// or DDOS attacks.
   351  	srv := &http.Server{
   352  		Addr:         utils.Sprintf("0.0.0.0:%v", port),
   353  		ReadTimeout:  5 * time.Second,
   354  		WriteTimeout: 5 * time.Second,
   355  		IdleTimeout:  120 * time.Second,
   356  		Handler:      mux,
   357  	}
   358  
   359  	go func() {
   360  		logger.Error(srv.ListenAndServe())
   361  	}()
   362  }
