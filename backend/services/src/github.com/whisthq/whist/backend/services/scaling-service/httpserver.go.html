   166  func getAccessToken(r *http.Request) (string, error) {
   167  	authorization := r.Header.Get("Authorization")
   168  	bearer := strings.Split(authorization, "Bearer ")
   169  	if len(bearer) <= 1 {
   170  		return "", utils.MakeError("Bearer token is empty.")

   173  	return accessToken, nil

   176  // throttleMiddleware will limit requests on the endpoint using the provided rate limiter.
   177  // It uses a token bucket algorithm, so that every interval of time the "bucket" will refill
   178  // and continue to serve tokens up to a maximum defined by the burst capacity. In case the
   179  // limit is exceeded, return a http 429 error (too many requests).
   180  func throttleMiddleware(limiter *rate.Limiter, f func(http.ResponseWriter, *http.Request)) func(http.ResponseWriter, *http.Request) {
   181  	return http.HandlerFunc(func(rw http.ResponseWriter, r *http.Request) {
   182  		if !limiter.Allow() {
   183  			http.Error(rw, http.StatusText(http.StatusTooManyRequests), http.StatusTooManyRequests)
   184  		}
   185  		f(rw, r)
   186  	})
   187  }

   191  func verifyPaymentMiddleware(f func(http.ResponseWriter, *http.Request)) http.Handler {
   192  	return http.HandlerFunc(func(rw http.ResponseWriter, r *http.Request) {
   193  		// Get and parse Authorization header from access token
   194  		accessToken, err := getAccessToken(r)
   195  		if err != nil {
   196  			logger.Error(err)
   197  			http.Error(rw, http.StatusText(http.StatusUnauthorized), http.StatusUnauthorized)
   198  			return
   199  		}
   200  
   201  		if len(strings.Split(accessToken, "Bearer")) <= 1 {
   202  			logger.Warningf("Access token does not have Bearer header. Trying to parse token as is.")
   203  		} else {
   204  			accessToken = strings.Split(accessToken, "Bearer ")[1]
   205  		}
   206  
   207  		// Verify if the user's subscription is valid
   208  		paymentValid, err := payments.VerifyPayment(accessToken)
   209  		if err != nil {
   210  			logger.Errorf("Failed to validate Stripe payment from access token. Err: %v", err)
   211  			http.Error(rw, http.StatusText(http.StatusInternalServerError), http.StatusInternalServerError)
   212  			return
   213  		}
   214  
   215  		if !paymentValid {
   216  			logger.Warningf("User does not have an active Stripe subscription.")
   217  			http.Error(rw, http.StatusText(http.StatusForbidden), http.StatusForbidden)
   218  			return
   219  		}
   220  		f(rw, r)
   221  	})
   222  }

   247  	logger.Infof("Starting HTTP server...")
   248  
   249  	createHandler := func(f func(http.ResponseWriter, *http.Request, chan<- algos.ScalingEvent)) func(http.ResponseWriter, *http.Request) {
   250  		return func(w http.ResponseWriter, r *http.Request) {
   251  			f(w, r, events)
   252  		}
   253  	}
   254  
   255  	// Start a new rate limiter. This will limit requests on an endpoint
   256  	//  to every `interval` with a burst of up to `burst` requests. This
   257  	// will help mitigate Denial of Service attacks, or a client app
   258  	// spamming too many requests.
   259  	interval := 1 * time.Second
   260  	burst := 10
   261  	limiter := rate.NewLimiter(rate.Every(interval), burst)
   262  
   263  	// Create the final assign handler, with the necessary middleware
   264  	assignHandler := verifyPaymentMiddleware(throttleMiddleware(limiter, createHandler(mandelboxAssignHandler)))
   265  
   266  	// Create a custom HTTP Request Multiplexer
   267  	mux := http.NewServeMux()
   268  	mux.Handle("/", http.NotFoundHandler())
   269  	mux.Handle("/mandelbox/assign", assignHandler)
   270  	mux.Handle("/payment_portal_url", http.HandlerFunc(paymentsHandler))
   271  
   272  	// The PORT env var will be automatically set by Heroku.
   273  	// If running on localdev, use default port.
   274  	port := os.Getenv("PORT")
   275  	if metadata.IsLocalEnv() || port == "" {
   276  		port = "7730"
   277  	}
   278  
   279  	// Set read/write timeouts to help mitigate potential rogue clients
   280  	// or DDOS attacks.
   281  	srv := &http.Server{
   282  		Addr:         utils.Sprintf("0.0.0.0:%v", port),
   283  		ReadTimeout:  5 * time.Second,
   284  		WriteTimeout: 5 * time.Second,
   285  		IdleTimeout:  120 * time.Second,
   286  		Handler:      mux,
   287  	}
