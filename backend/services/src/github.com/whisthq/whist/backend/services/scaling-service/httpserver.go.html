    68  	accessToken := r.Header.Get("Authorization")
    69  	accessToken = strings.Split(accessToken, "Bearer ")[1]

    72  	if err != nil {
    73  		http.Error(w, "Unauthorized", http.StatusUnauthorized)

    75  	}
    76  
    77  	_, err = httputils.ParseRequest(w, r, s)

    88  // throttleMiddleware will limit requests on the endpoint using the provided rate limiter.
    89  // It uses a token bucket algorithm, so that every interval of time the "bucket" will refill
    90  // and continue to serve tokens up to a maximum defined by the burst capacity. In case the
    91  // limit is exceeded, return a http 429 error (too many requests).
    92  func throttleMiddleware(limiter *rate.Limiter, f func(http.ResponseWriter, *http.Request)) func(http.ResponseWriter, *http.Request) {
    93  	return http.HandlerFunc(func(rw http.ResponseWriter, r *http.Request) {
    94  		if !limiter.Allow() {
    95  			http.Error(rw, http.StatusText(http.StatusTooManyRequests), http.StatusTooManyRequests)
    96  		}
    97  		f(rw, r)
    98  	})
    99  }

   102  // middleware should be used to restrict endpoint for paying users only.
   103  func verifyPaymentMiddleware(f func(http.ResponseWriter, *http.Request)) http.Handler {
   104  	return http.HandlerFunc(func(rw http.ResponseWriter, r *http.Request) {
   105  		// Get and parse Authorization header from access token
   106  		accessToken := r.Header.Get("Authorization")
   107  		accessToken = strings.Split(accessToken, "Bearer ")[1]
   108  
   109  		// Verify if the user's subscription is valid
   110  		paymentValid, err := payments.VerifyPayment(accessToken)
   111  		if err != nil {
   112  			logger.Errorf("Failed to validate Stripe payment from access token. Err: %v", err)
   113  			http.Error(rw, http.StatusText(http.StatusInternalServerError), http.StatusInternalServerError)
   114  		}
   115  
   116  		if !paymentValid {
   117  			logger.Warningf("User does not have an active Stripe subscription.")
   118  			http.Error(rw, http.StatusText(http.StatusForbidden), http.StatusForbidden)
   119  		}
   120  		f(rw, r)
   121  	})
   122  }

   175  		ReadTimeout:  5 * time.Second,
