    67  // GetAccessToken is a helper function that extracts the access token
    68  // from the request "Authorization" header. If it fails, fallback to
    69  // extracting the token from the request's body.
    70  func GetAccessToken(r *http.Request) (string, error) {
    71  	if metadata.IsLocalEnv() {
    72  		return "", nil
    73  	}
    74  
    75  	authorization := r.Header.Get("Authorization")
    76  	bearer := strings.Split(authorization, "Bearer ")
    77  
    78  	var (
    79  		accessToken string
    80  		bodyMap     map[string]interface{}
    81  	)
    82  	if len(bearer) <= 1 || bearer[1] == "" || bearer[1] == "undefined" {
    83  		logger.Warningf("Bearer token is empty. Trying to parse token from request body.")
    84  
    85  		// Read request body and replace for subsequent reads
    86  		body, err := ioutil.ReadAll(r.Body)
    87  		if err != nil {
    88  			return "", utils.MakeError("failed to read request body: %s", err)
    89  		}
    90  		r.Body.Close()
    91  		r.Body = io.NopCloser(bytes.NewBuffer(body))
    92  
    93  		// Here we unmarshal into a simple map because we are only interested
    94  		// in the access token.
    95  		err = json.Unmarshal(body, &bodyMap)
    96  		if err != nil {
    97  			return "", utils.MakeError("failed to umarshal request body: %s", err)
    98  		}
    99  
   100  		val, ok := bodyMap["jwt_access_token"]
   101  		if !ok {
   102  			return "", utils.MakeError("did not find jwt_access_token field in request body.")
   103  		}
   104  
   105  		// TODO: Once the client application is superseded by the Chromium extension, remove
   106  		// this logic to only obtain the token from the "jwt_access_token" key.
   107  		accessToken = val.(string)
   108  
   109  	} else {
   110  		accessToken = bearer[1]
   111  	}
   112  
   113  	return accessToken, nil
   114  }

   116  // AuthenticateRequest will verify that the access token is valid
   117  // and will parse the request body and try to unmarshal into a
   118  // `ServerRequest` type.
   119  func AuthenticateRequest(w http.ResponseWriter, r *http.Request, s ServerRequest) (*auth.WhistClaims, error) {
   120  	accessToken, err := GetAccessToken(r)
   121  	if err != nil {
   122  		return nil, err
   123  	}
   124  
   125  	var claims *auth.WhistClaims
   126  	// Skip token validation if running on local environment
   127  	if !metadata.IsLocalEnv() {
   128  		claims, err = auth.ParseToken(accessToken)
   129  		if err != nil {
   130  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   131  			return nil, utils.MakeError("Received an unpermissioned backend request on %s to URL %s. Error: %s", r.Host, r.URL, err)
   132  		}
   133  
   134  		if err := auth.Verify(claims); err != nil {
   135  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   136  			return nil, utils.MakeError("Received an unpermissioned backend request on %s to URL %s. Error: %s", r.Host, r.URL, err)
   137  		}
   138  	}
   139  
   140  	_, err = ParseRequest(w, r, s)
   141  	if err != nil {
   142  		return nil, utils.MakeError("Error while parsing request. Err: %v", err)
   143  	}
   144  
   145  	return claims, nil
   146  }

   148  // ParseRequest will split the request body, unmarshal into a raw JSON map, and then unmarshal
   149  // the remaining fields into the struct `s`. We unmarshal the raw JSON map and the rest of the
   150  // body so that we don't expose the authorization fields to code that does not handle authentication.
   151  func ParseRequest(w http.ResponseWriter, r *http.Request, s ServerRequest) (map[string]*json.RawMessage, error) {
   152  	// Get body of request
   153  	body, err := io.ReadAll(r.Body)
   154  	if err != nil {
   155  		http.Error(w, "Malformed body", http.StatusBadRequest)
   156  		return nil, utils.MakeError("Error getting body from request on %s to URL %s: %s", r.Host, r.URL, err)
   157  	}
   158  
   159  	// Extract only the jwt_access_token field from a raw JSON unmarshalling that
   160  	// delays as much decoding as possible
   161  	var rawmap map[string]*json.RawMessage
   162  	err = json.Unmarshal(body, &rawmap)
   163  	if err != nil {
   164  		http.Error(w, "Malformed body", http.StatusBadRequest)
   165  		return nil, utils.MakeError("Error raw-unmarshalling JSON body sent on %s to URL %s: %s", r.Host, r.URL, err)
   166  	}
   167  
   168  	// Now, actually do the unmarshalling into the right object type
   169  	err = json.Unmarshal(body, s)
   170  	if err != nil {
   171  		http.Error(w, "Malformed body", http.StatusBadRequest)
   172  		return nil, utils.MakeError("Could not fully unmarshal the body of a request sent on %s to URL %s: %s", r.Host, r.URL, err)
   173  	}
   174  
   175  	// Set up the result channel
   176  	s.CreateResultChan()
   177  
   178  	return rawmap, nil
   179  }

   182  func VerifyRequestType(w http.ResponseWriter, r *http.Request, method string) error {
   183  	if r == nil {
   184  		err := utils.MakeError("Received a nil request expecting to be type %s", method)
   185  		logger.Error(err)
   186  
   187  		http.Error(w, utils.Sprintf("Bad request. Expected %s, got nil", method), http.StatusBadRequest)
   188  
   189  		return err
   190  	}
   191  
   192  	if r.Method != method {
   193  		err := utils.MakeError("Received a request on %s to URL %s of type %s, but it should have been type %s", r.Host, r.URL, r.Method, method)
   194  		logger.Error(err)
   195  
   196  		http.Error(w, utils.Sprintf("Bad request type. Expected %s, got %s", method, r.Method), http.StatusBadRequest)
   197  
   198  		return err
   199  	}
   200  	return nil
   201  }

   203  // EnableCORS is a middleware that sets the Access control header to accept requests from all origins.
   204  func EnableCORS(f func(http.ResponseWriter, *http.Request)) func(http.ResponseWriter, *http.Request) {
   205  	return http.HandlerFunc(func(rw http.ResponseWriter, r *http.Request) {
   206  		rw.Header().Set("Access-Control-Allow-Origin", "*")
   207  		rw.Header().Set("Access-Control-Allow-Headers", "Origin Accept Content-Type X-Requested-With")
   208  		rw.Header().Set("Access-Control-Allow-Methods", "POST PUT OPTIONS")
   209  
   210  		if r.Method == http.MethodOptions {
   211  			http.Error(rw, "No Content", http.StatusNoContent)
   212  			return
   213  		}
   214  
   215  		f(rw, r)
   216  	})
   217  }
