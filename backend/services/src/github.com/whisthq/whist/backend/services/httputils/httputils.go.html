    30  func (r RequestResult) Send(w http.ResponseWriter) {

    32  	var err error
    33  	var status int
    34  
    35  	if r.Err != nil {
    36  		// Send a 406

    39  			struct {
    40  				Result interface{} `json:"result"`

    43  		)
    44  	} else {
    45  		// Send a 200 code
    46  		status = http.StatusOK
    47  		buf, err = json.Marshal(
    48  			struct {

    51  		)
    52  	}
    53

    56  		logger.Errorf("Error marshalling a %v HTTP Response body: %s", status, err)

    64  // ParseRequest will split the request body, unmarshal into a raw JSON map, and then unmarshal
    65  // the remaining fields into the struct `s`. We unmarshal the raw JSON map and the rest of the
    66  // body so that we don't expose the authorization fields to code that does not handle authentication.
    67  func ParseRequest(w http.ResponseWriter, r *http.Request, s ServerRequest) (map[string]*json.RawMessage, error) {
    68  	// Get body of request
    69  	body, err := io.ReadAll(r.Body)
    70  	if err != nil {
    71  		http.Error(w, "Malformed body", http.StatusBadRequest)
    72  		return nil, utils.MakeError("Error getting body from request on %s to URL %s: %s", r.Host, r.URL, err)
    73  	}
    74  
    75  	// Extract only the jwt_access_token field from a raw JSON unmarshalling that
    76  	// delays as much decoding as possible
    77  	var rawmap map[string]*json.RawMessage
    78  	err = json.Unmarshal(body, &rawmap)
    79  	if err != nil {
    80  		http.Error(w, "Malformed body", http.StatusBadRequest)
    81  		return nil, utils.MakeError("Error raw-unmarshalling JSON body sent on %s to URL %s: %s", r.Host, r.URL, err)
    82  	}
    83  
    84  	// Now, actually do the unmarshalling into the right object type
    85  	err = json.Unmarshal(body, s)
    86  	if err != nil {
    87  		http.Error(w, "Malformed body", http.StatusBadRequest)
    88  		return nil, utils.MakeError("Could not fully unmarshal the body of a request sent on %s to URL %s: %s", r.Host, r.URL, err)
    89  	}
    90  
    91  	// Set up the result channel
    92  	s.CreateResultChan()
    93  
    94  	return rawmap, nil
    95  }

    99  	if r == nil {
   100  		err := utils.MakeError("Received a nil request expecting to be type %s", method)
   101  		logger.Error(err)
   102  
   103  		http.Error(w, utils.Sprintf("Bad request. Expected %s, got nil", method), http.StatusBadRequest)
   104  
   105  		return err
   106  	}
   107  
   108  	if r.Method != method {
   109  		err := utils.MakeError("Received a request on %s to URL %s of type %s, but it should have been type %s", r.Host, r.URL, r.Method, method)
   110  		logger.Error(err)
   111  
   112  		http.Error(w, utils.Sprintf("Bad request type. Expected %s, got %s", method, r.Method), http.StatusBadRequest)
   113  
   114  		return err
   115  	}
   116  	return nil
