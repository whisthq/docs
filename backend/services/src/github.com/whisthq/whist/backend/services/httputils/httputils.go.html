    35  	var status int

    66  // from the request "Authorization" header.
    67  func GetAccessToken(r *http.Request) (string, error) {
    68  	if metadata.IsLocalEnv() {
    69  		return "", nil
    70  	}
    71  
    72  	authorization := r.Header.Get("Authorization")
    73  	bearer := strings.Split(authorization, "Bearer ")
    74  	if len(bearer) <= 1 || bearer[1] == "" || bearer[1] == "undefined" {
    75  		return "", utils.MakeError("couldn't find access token in Authorization header")
    76  	}
    77  
    78  	return bearer[1], nil
    79  }

    81  // AuthenticateRequest will verify that the access token is valid
    82  // and will parse the request body and try to unmarshal into a
    83  // `ServerRequest` type.
    84  func AuthenticateRequest(w http.ResponseWriter, r *http.Request, s ServerRequest) (*auth.WhistClaims, error) {
    85  	accessToken, err := GetAccessToken(r)
    86  	if err != nil {
    87  		return nil, err
    88  	}
    89  
    90  	var claims *auth.WhistClaims
    91  	// Skip token validation if running on local environment
    92  	if !metadata.IsLocalEnv() {
    93  		claims, err = auth.ParseToken(accessToken)
    94  		if err != nil {
    95  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
    96  			return nil, utils.MakeError("received an unpermissioned backend request on %s to URL %s: %s", r.Host, r.URL, err)
    97  		}
    98  
    99  		if err := auth.Verify(claims); err != nil {
   100  			http.Error(w, "Unauthorized", http.StatusUnauthorized)
   101  			return nil, utils.MakeError("received an unpermissioned backend request on %s to URL %s: %s", r.Host, r.URL, err)
   102  		}
   103  	}
   104  
   105  	_, err = ParseRequest(w, r, s)
   106  	if err != nil {
   107  		return nil, utils.MakeError("Error while parsing request. Err: %v", err)
   108  	}
   109  
   110  	return claims, nil
   111  }

   113  // ParseRequest will split the request body, unmarshal into a raw JSON map, and then unmarshal
   114  // the remaining fields into the struct `s`. We unmarshal the raw JSON map and the rest of the
   115  // body so that we don't expose the authorization fields to code that does not handle authentication.
   116  func ParseRequest(w http.ResponseWriter, r *http.Request, s ServerRequest) (map[string]*json.RawMessage, error) {
   117  	// Get body of request
   118  	body, err := io.ReadAll(r.Body)
   119  	defer r.Body.Close()
   120  
   121  	if err != nil {
   122  		http.Error(w, "Malformed body", http.StatusBadRequest)
   123  		return nil, utils.MakeError("error getting body from request on %s to URL %s: %s", r.Host, r.URL, err)
   124  	}
   125  
   126  	// Extract only the jwt_access_token field from a raw JSON unmarshalling that
   127  	// delays as much decoding as possible
   128  	var rawmap map[string]*json.RawMessage
   129  	err = json.Unmarshal(body, &rawmap)
   130  	if err != nil {
   131  		http.Error(w, "Malformed body", http.StatusBadRequest)
   132  		return nil, utils.MakeError("error raw-unmarshalling JSON body sent on %s to URL %s: %s", r.Host, r.URL, err)
   133  	}
   134  
   135  	// Now, actually do the unmarshalling into the right object type
   136  	err = json.Unmarshal(body, s)
   137  	if err != nil {
   138  		http.Error(w, "Malformed body", http.StatusBadRequest)
   139  		return nil, utils.MakeError("could not fully unmarshal the body of a request sent on %s to URL %s: %s", r.Host, r.URL, err)
   140  	}
   141  
   142  	// Set up the result channel
   143  	s.CreateResultChan()
   144  
   145  	return rawmap, nil
   146  }

   149  func VerifyRequestType(w http.ResponseWriter, r *http.Request, method string) error {
   150  	if r == nil {
   151  		err := utils.MakeError("received a nil request expecting to be type %s", method)
   152  		logger.Error(err)
   153  
   154  		http.Error(w, utils.Sprintf("Bad request. Expected %s, got nil", method), http.StatusBadRequest)
   155  
   156  		return err
   157  	}
   158  
   159  	if r.Method != method {
   160  		err := utils.MakeError("received a request on %s to URL %s of type %s, but it should have been type %s", r.Host, r.URL, r.Method, method)
   161  		logger.Error(err)
   162  
   163  		http.Error(w, utils.Sprintf("Bad request type. Expected %s, got %s", method, r.Method), http.StatusBadRequest)
   164  
   165  		return err
   166  	}
   167  	return nil
   168  }

   170  // EnableCORS is a middleware that sets the Access control header to accept requests from all origins.
   171  func EnableCORS(f func(http.ResponseWriter, *http.Request)) func(http.ResponseWriter, *http.Request) {
   172  	return http.HandlerFunc(func(rw http.ResponseWriter, r *http.Request) {
   173  		rw.Header().Set("Access-Control-Allow-Origin", "*")
   174  		rw.Header().Set("Access-Control-Allow-Headers", "Origin Accept Content-Type X-Requested-With")
   175  		rw.Header().Set("Access-Control-Allow-Methods", "POST PUT OPTIONS")
   176  
   177  		if r.Method == http.MethodOptions {
   178  			http.Error(rw, "No Content", http.StatusNoContent)
   179  			return
   180  		}
   181  
   182  		f(rw, r)
   183  	})
   184  }
