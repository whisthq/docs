   132  // StartSchedulerEvents starts the scheduler and its events without blocking the main thread.
   133  // `interval` sets the time when the event will run in minutes (i.e. every 10 minutes), and `start`
   134  // sets the time when the first event will happen (i.e. 10 minutes from now).
   135  func StartSchedulerEvents(scheduledEvents chan algos.ScalingEvent, interval interface{}, start time.Duration) {
   136  	s := gocron.NewScheduler(time.UTC)
   137  
   138  	// Schedule scale down routine every 10 minutes, start 10 minutes from now.
   139  	t := time.Now().Add(start)
   140  	s.Every(interval).Minutes().StartAt(t).Do(func() {
   141  		// Send into scheduling channel
   142  		scheduledEvents <- algos.ScalingEvent{
   143  			// Create a UUID so we can identify and search this event on our logs
   144  			ID: uuid.NewString(),
   145  			// We set the event type to SCHEDULED_SCALE_DOWN_EVENT
   146  			// here so that we have more information about the event.
   147  			// SCHEDULED means the source of the event is the scheduler
   148  			// and SCALE_DOWN means it will perform a scale down action.
   149  			Type: "SCHEDULED_SCALE_DOWN_EVENT",
   150  		}
   151  	})
   152  
   153  	s.StartAsync()
   154  }

   177  func getRegionImageMap() (map[string]interface{}, error) {
   178  	var regionImageMap map[string]interface{}
   179  
   180  	// Get current working directory to read images file.
   181  	currentWorkingDirectory, err := os.Getwd()
   182  	if err != nil {
   183  		return nil, utils.MakeError("Failed to get working directory. Err: %v", err)
   184  	}
   185  
   186  	// Read file which contains the region to image on JSON format. This file will
   187  	// be read by the binary generated during deploy, located in the `bin` directory.
   188  	// The file is also generated during deploy and lives in the scaling service directory.
   189  	content, err := os.ReadFile(path.Join(currentWorkingDirectory, "images.json"))
   190  	if err != nil {
   191  		return nil, utils.MakeError("Failed to read region to image map from file. Not performing image upgrade. Err: %v", err)
   192  	}
   193  
   194  	// Try to unmarshal contents of file into a map
   195  	err = json.Unmarshal(content, &regionImageMap)
   196  	if err != nil {
   197  		return nil, utils.MakeError("Failed to unmarshal region to image map. Not performing image upgrade. Err: %v", err)
   198  	}
   199  
   200  	return regionImageMap, nil
   201  }

   204  func getScalingAlgorithm(algorithmByRegion *sync.Map, scalingEvent algos.ScalingEvent) algos.ScalingAlgorithm {
   205  	// Try to get the scaling algorithm on the region the scaling event was requested.
   206  	// If no region is specified, use the default region.
   207  	// TODO: figure out how to get non-default scaling algorihtms.
   208  	var (
   209  		name   string
   210  		region string
   211  	)
   212  	const defaultRegion = "us-east-1"
   213  
   214  	region = scalingEvent.Region
   215  
   216  	if region == "" {
   217  		logger.Infof("No region found on scaling event. Getting scaling algorithm on default region %v.", defaultRegion)
   218  		name = utils.Sprintf("default-sa-%s", defaultRegion)
   219  	} else {
   220  		name = utils.Sprintf("default-sa-%s", scalingEvent.Region)
   221  	}
   222  
   223  	algorithm, ok := algorithmByRegion.Load(name)
   224  	if ok {
   225  		return algorithm.(algos.ScalingAlgorithm)
   226  	}
   227  
   228  	logger.Errorf("Failed to get scaling algorithm in %v", scalingEvent.Region)
   229  
   230  	return nil

   233  // eventLoop is the main loop of the scaling service which will receive events from different sources
   234  // and send them to the appropiate channels.
   235  func eventLoop(globalCtx context.Context, globalCancel context.CancelFunc, serverEvents <-chan algos.ScalingEvent, subscriptionEvents <-chan subscriptions.SubscriptionEvent,
   236  	scheduledEvents <-chan algos.ScalingEvent, algorithmByRegion *sync.Map, configClient *subscriptions.SubscriptionClient) {
   237  
   238  	for {
   239  		select {
   240  		case subscriptionEvent := <-subscriptionEvents:
   241  			// Since we are dealing with a database event, figure out what type it has
   242  			// and send it to the corresponding channel
   243  			switch subscriptionEvent := subscriptionEvent.(type) {
   244  
   245  			case *subscriptions.InstanceEvent:
   246  				var scalingEvent algos.ScalingEvent
   247  
   248  				// We set the event type to DATABASE_INSTANCE_EVENT
   249  				// here so that we have more information about the event.
   250  				// DATABASE means the source of the event is the database
   251  				// and INSTANCE_EVENT means it will perform an action with
   252  				// the instance information received from the database.
   253  				scalingEvent.Type = "DATABASE_INSTANCE_EVENT"
   254  
   255  				if len(subscriptionEvent.Instances) > 0 {
   256  					instance := subscriptionEvent.Instances[0]
   257  					scalingEvent.ID = uuid.NewString()
   258  					scalingEvent.Data = instance
   259  					scalingEvent.Region = instance.Region
   260  				}
   261  
   262  				// Start scaling algorithm based on region
   263  				logger.Infof("Received database event.")
   264  				algorithm := getScalingAlgorithm(algorithmByRegion, scalingEvent)
   265  
   266  				switch algorithm := algorithm.(type) {
   267  				case *algos.DefaultScalingAlgorithm:
   268  					algorithm.InstanceEventChan <- scalingEvent
   269  				}
   270  			case *subscriptions.ClientAppVersionEvent:
   271  				var (
   272  					scalingEvent algos.ScalingEvent
   273  					version      subscriptions.ClientAppVersion
   274  				)
   275  
   276  				// We set the event type to DATABASE_CLIENT_VERSION_EVENT
   277  				// here so that we have more information about the event.
   278  				// DATABASE means the source of the event is the database
   279  				// and CLIENT_VERSION means it will perform an action with
   280  				// the client version information received from the database.
   281  				scalingEvent.Type = "DATABASE_CLIENT_VERSION_EVENT"
   282  
   283  				// We only expect one version to come from the database subscription,
   284  				// anything more or less than it indicates an error in our backend.
   285  				if len(subscriptionEvent.ClientAppVersions) == 1 {
   286  					version = subscriptionEvent.ClientAppVersions[0]
   287  				} else {
   288  					logger.Errorf("Unexpected length of %v in version received from the config database.", len(subscriptionEvent.ClientAppVersions))
   289  				}
   290  
   291  				// TODO: once we keep track of client versions per region,
   292  				// get the region directly from the database. For now use
   293  				// the regionImageMap.
   294  				regionImageMap, err := getRegionImageMap()
   295  				if err != nil {
   296  					logger.Errorf("Error getting regionImageMap. Err: %v", err)
   297  					break
   298  				}
   299  
   300  				for region := range regionImageMap {
   301  					scalingEvent.Region = region
   302  					scalingEvent.Data = version
   303  
   304  					// Start scaling algorithm based on region
   305  					logger.Infof("Received database event.")
   306  					algorithm := getScalingAlgorithm(algorithmByRegion, scalingEvent)
   307  
   308  					switch algorithm := algorithm.(type) {
   309  					case *algos.DefaultScalingAlgorithm:
   310  						algorithm.ClientAppVersionChan <- scalingEvent
   311  					}
   312  				}
   313  
   314  				// Its necessary to close the config client since we only want to
   315  				// do the image swapover once, but Hasura will keep receiving
   316  				// the event after it has switched.
   317  				err = configClient.Close()
   318  				if err != nil {
   319  					// err is already wrapped here
   320  					logger.Error(err)
   321  				}
   322  			}
   323  
   324  		case scheduledEvent := <-scheduledEvents:
   325  			// Start scaling algorithm based on region
   326  			logger.Infof("Received scheduled event. %v", scheduledEvent)
   327  
   328  			for _, region := range algos.GetEnabledRegions() {
   329  				scheduledEvent.Region = region
   330  				algorithm := getScalingAlgorithm(algorithmByRegion, scheduledEvent)
   331  				switch algorithm := algorithm.(type) {
   332  				case *algos.DefaultScalingAlgorithm:
   333  					algorithm.ScheduledEventChan <- scheduledEvent
   334  				}
   335  			}
   336  		case serverEvent := <-serverEvents:
   337  			logger.Infof("Received server event. %v", serverEvent)
   338  
   339  			algorithm := getScalingAlgorithm(algorithmByRegion, serverEvent)
   340  			switch algorithm := algorithm.(type) {
   341  			case *algos.DefaultScalingAlgorithm:
   342  				algorithm.ServerEventChan <- serverEvent
   343  			}
   344  
   345  		case <-globalCtx.Done():
   346  			logger.Infof("Gloal context has been cancelled, exiting event loop...")
   347  			return
   348  		}
   349  	}
