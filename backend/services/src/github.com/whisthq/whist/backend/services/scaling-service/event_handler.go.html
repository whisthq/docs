   106  // `interval` sets the time when the event will run in minutes (i.e. every 10 minutes), and `start`
   107  // sets the time when the first event will happen (i.e. 10 minutes from now).
   108  func StartSchedulerEvents(scheduledEvents chan algos.ScalingEvent, interval interface{}, start time.Duration) {
   109  	s := gocron.NewScheduler(time.UTC)
   110  
   111  	// Schedule scale down routine every 10 minutes, start 10 minutes from now.
   112  	t := time.Now().Add(start)
   113  	s.Every(interval).Minutes().StartAt(t).Do(func() {
   114  		// Send into scheduling channel
   115  		scheduledEvents <- algos.ScalingEvent{
   116  			// Create a UUID so we can identify and search this event on our logs
   117  			ID: uuid.NewString(),
   118  			// We set the event type to SCHEDULED_SCALE_DOWN_EVENT
   119  			// here so that we have more information about the event.
   120  			// SCHEDULED means the source of the event is the scheduler
   121  			// and SCALE_DOWN means it will perform a scale down action.
   122  			Type: "SCHEDULED_SCALE_DOWN_EVENT",
   123  		}
   124  	})
   125  
   126  	s.StartAsync()
   127  }

   129  func StartDeploy(scheduledEvents chan algos.ScalingEvent) {
   130  	var regionImageMap map[string]interface{}
   131  
   132  	// Get current working directory to read images file.
   133  	currentWorkingDirectory, err := os.Getwd()
   134  	if err != nil {
   135  		logger.Errorf("Failed to get working directory. Err: %v", err)
   136  		return
   137  	}
   138  
   139  	// Read file which contains the region to image on JSON format. This file will
   140  	// be read by the binary generated during deploy, located in the `bin` directory.
   141  	// The file is also generated during deploy and lives in the scaling service directory.
   142  	content, err := os.ReadFile(path.Join(currentWorkingDirectory, "images.json"))
   143  	if err != nil {
   144  		logger.Errorf("Failed to read region to image map from file. Not performing image upgrade. Err: %v", err)
   145  		return
   146  	}
   147  
   148  	// Try to unmarshal contents of file into a map
   149  	err = json.Unmarshal(content, &regionImageMap)
   150  	if err != nil {
   151  		logger.Errorf("Failed to unmarshal region to image map. Not performing image upgrade. Err: %v", err)
   152  		return
   153  	}
   154  
   155  	// Send image upgrade event to scheduled chan.
   156  	scheduledEvents <- algos.ScalingEvent{
   157  		// Create a UUID so we can identify and search this event on our logs
   158  		ID: uuid.NewString(),
   159  		// We set the event type to SCHEDULED_IMAGE_UPGRADE_EVENT
   160  		// here so that we have more information about the event.
   161  		// SCHEDULED means the source of the event is the scheduler
   162  		// and IMAGE_UPGRADE means it will perform an image upgrade action.
   163  		Type: "SCHEDULED_IMAGE_UPGRADE_EVENT",
   164  		Data: regionImageMap,

   183  // eventLoop is the main loop of the scaling service which will receive events from different sources
   184  // and send them to the appropiate channels.
   185  func eventLoop(globalCtx context.Context, globalCancel context.CancelFunc, goroutineTracker *sync.WaitGroup,
   186  	subscriptionEvents <-chan subscriptions.SubscriptionEvent, scheduledEvents <-chan algos.ScalingEvent, algorithmByRegion *sync.Map) {
   187  
   188  	for {
   189  		select {
   190  		case subscriptionEvent := <-subscriptionEvents:
   191  			// Since we are dealing with a database event, figure out what type it has
   192  			// and send it to the corresponding channel
   193  			switch subscriptionEvent := subscriptionEvent.(type) {
   194  
   195  			case *subscriptions.InstanceEvent:
   196  				var scalingEvent algos.ScalingEvent
   197  
   198  				// We set the event type to DATABASE_INSTANCE_EVENT
   199  				// here so that we have more information about the event.
   200  				// DATABASE means the source of the event is the database
   201  				// and INSTANCE_EVENT means it will perform an action with
   202  				// the instance information received from the database.
   203  				scalingEvent.Type = "DATABASE_INSTANCE_EVENT"
   204  
   205  				if len(subscriptionEvent.Instances) > 0 {
   206  					instance := subscriptionEvent.Instances[0]
   207  					scalingEvent.ID = uuid.NewString()
   208  					scalingEvent.Data = instance
   209  					scalingEvent.Region = instance.Region
   210  				}
   211  
   212  				// Start scaling algorithm based on region
   213  				logger.Infof("Received database event.")
   214  				algorithm := getScalingAlgorithm(algorithmByRegion, scalingEvent)
   215  
   216  				switch algorithm := algorithm.(type) {
   217  				case *algos.DefaultScalingAlgorithm:
   218  					algorithm.InstanceEventChan <- scalingEvent
   219  				}
   220  			}
   221  		case scheduledEvent := <-scheduledEvents:
   222  			// Start scaling algorithm based on region
   223  			logger.Infof("Received scheduled event. %v", scheduledEvent)
   224  
   225  			for _, region := range algos.BundledRegions {
   226  				scheduledEvent.Region = region
   227  				algorithm := getScalingAlgorithm(algorithmByRegion, scheduledEvent)
   228  				switch algorithm := algorithm.(type) {
   229  				case *algos.DefaultScalingAlgorithm:
   230  					algorithm.ScheduledEventChan <- scheduledEvent
   231  				}
   232  			}
   233  		case <-globalCtx.Done():
   234  			logger.Infof("Gloal context has been cancelled, exiting event loop...")
   235  			return
   236  		}
   237  	}
   238  }
