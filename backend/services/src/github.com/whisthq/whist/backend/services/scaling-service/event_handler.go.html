     3  /*
     4  The scaling service is a server responsible of managing the state of the backend.
     5  It manages the number of instances running by starting and terminating according
     6  to demand. Additionally, it assigns users to available instances, and registers a
     7  mandelbox on the database so the host service running on the instance can start it.
     8  This component is also responsible of validating payments, as well as interacting
     9  with Stripe to create new subscriptions.
    10  
    11  The scaling service is an event-driven component, so most of its logic is handled using
    12  go channels, in reaction to external events. It has many layers, the bottom layer being
    13  the event handler which only passes the events to the corresponding channels. The layer
    14  on top of it includes the many scaling algorithms which are sub-processes for each region
    15  on the backend. The code is written in an extensible way so it is possible to write and
    16  integrate a new scaling algorithm with a different behaviour easily. As such, most of
    17  the code is inside the scaling algorithm implementations.
    18  
    19  The top-most layer has the "hosts handlers" which are a collection of methods responsible
    20  of calling a cloud provider (e.g. AWS, Azure, Google Cloud) API and starting an instance
    21  (cloud-compute resource configured to run a Whist container). This code is abstracted in
    22  an interface so that support for a new cloud provider is easily added. The code in the
    23  algorithms is provider-agnostic, tied only to the necessities of the backend.
    24  */

    49  	globalCtx, globalCancel := context.WithCancel(context.Background())
    50  	goroutineTracker := &sync.WaitGroup{}
    51  
    52  	// Add some additional fields for Logz.io
    53  	tags := make(map[string]string)

    55  	tags["sub-component"] = "scaling-service"
    56  	logger.AddLogzioFields(tags)

    58  	var (
    59  		dbClient            dbclient.WhistDBClient                // The client that abstracts database interactions
    60  		graphqlClient       subscriptions.WhistGraphQLClient      // The GraphQL client to query the Hasura server
    61  		configGraphqlClient subscriptions.WhistGraphQLClient      // The GraphQL client to query the config Hasura server
    62  		subscriptionClient  subscriptions.WhistSubscriptionClient // The subscription client to subscribe to the Hasura server
    63  		configClient        subscriptions.WhistSubscriptionClient // The subscriptions client to subscribe to the config Hasura server
    64  		subscriptionEvents  chan subscriptions.SubscriptionEvent  // Channel to process database subscription events
    65  		scheduledEvents     chan algos.ScalingEvent               // Channel to process scheduled events
    66  
    67  	)
    68  
    69  	// Start HTTP Server for assigning mandelboxes
    70  	serverEvents := make(chan algos.ScalingEvent, 100)
    71  	StartHTTPServer(serverEvents)
    72  
    73  	// Start GraphQL client for queries/mutations
    74  	useConfigDB := false
    75  	graphqlClient = &subscriptions.GraphQLClient{}
    76  	err := graphqlClient.Initialize(useConfigDB)
    77  	if err != nil {
    78  		logger.Errorf("failed to start GraphQL client: %s", err)
    79  	}
    80  
    81  	// Start GraphQL client for getting configuration from the config db
    82  	useConfigDB = true
    83  	configGraphqlClient = &subscriptions.GraphQLClient{}
    84  	err = configGraphqlClient.Initialize(useConfigDB)
    85  	if err != nil {
    86  		logger.Errorf("failed to start config GraphQL client: %s", err)
    87  	}
    88  
    89  	// Start database subscriptions
    90  	subscriptionEvents = make(chan subscriptions.SubscriptionEvent, 100)
    91  	subscriptionClient = &subscriptions.SubscriptionClient{}
    92  	configClient = &subscriptions.SubscriptionClient{}
    93  	StartDatabaseSubscriptions(globalCtx, goroutineTracker, subscriptionEvents, subscriptionClient, configClient)
    94  
    95  	// Initialize database client
    96  	dbClient = &dbclient.DBClient{}
    97  
    98  	// Start scheduler and setup scheduler event chan
    99  	scheduledEvents = make(chan algos.ScalingEvent, 100)
   100  
   101  	// Set to run every 10 minutes, starting 10 minutes from now
   102  	start := time.Duration(10 * time.Minute)
   103  	StartSchedulerEvents(scheduledEvents, 10, start)
   104  
   105  	// Start the deploy events once since we are starting the scaling service.
   106  	StartDeploy(scheduledEvents)
   107  
   108  	// algorithmByRegionMap holds all of the scaling algorithms mapped by region.
   109  	// Use a sync map since we only write the keys once but will be reading multiple
   110  	// times by different goroutines.
   111  	algorithmByRegionMap := &sync.Map{}
   112  
   113  	// Load default scaling algorithm for all enabled regions.
   114  	for _, region := range algos.GetEnabledRegions() {
   115  		name := utils.Sprintf("default-sa-%s", region)
   116  		algorithmByRegionMap.Store(name, &algos.DefaultScalingAlgorithm{
   117  			Region: region,
   118  		})
   119  	}
   120  
   121  	// Instantiate scaling algorithms on allowed regions
   122  	algorithmByRegionMap.Range(func(key, value interface{}) bool {
   123  		scalingAlgorithm := value.(algos.ScalingAlgorithm)
   124  		scalingAlgorithm.CreateEventChans()
   125  		scalingAlgorithm.CreateGraphQLClient(graphqlClient)
   126  		scalingAlgorithm.CreateDBClient(dbClient)
   127  		scalingAlgorithm.ProcessEvents(globalCtx, goroutineTracker)
   128  		scalingAlgorithm.GetConfig(configGraphqlClient)
   129  
   130  		return true
   131  	})
   132  
   133  	// Start main event loop
   134  	go eventLoop(globalCtx, globalCancel, serverEvents, subscriptionEvents, scheduledEvents, algorithmByRegionMap, configClient)
   135  
   136  	// Register a signal handler for Ctrl-C so that we cleanup if Ctrl-C is pressed.
   137  	sigChan := make(chan os.Signal, 2)
   138  	signal.Notify(sigChan, os.Interrupt, syscall.SIGTERM, syscall.SIGINT)
   139  
   140  	// Wait for either the global context to get cancelled by a worker goroutine,
   141  	// or for us to receive an interrupt. This needs to be the end of main().
   142  	select {
   143  	case <-sigChan:
   144  		logger.Infof("Got an interrupt or SIGTERM")
   145  		logger.Sync()
   146  	case <-globalCtx.Done():
   147  		logger.Infof("Global context cancelled!")
   148  		logger.Sync()
   149  	}

   176  // StartSchedulerEvents starts the scheduler and its events without blocking the main thread.
   177  // `interval` sets the time when the event will run in minutes (i.e. every 10 minutes), and `start`
   178  // sets the time when the first event will happen (i.e. 10 minutes from now).
   179  func StartSchedulerEvents(scheduledEvents chan algos.ScalingEvent, interval interface{}, start time.Duration) {
   180  	if metadata.IsLocalEnvWithoutDB() && !metadata.IsRunningInCI() {
   181  		return
   182  	}
   183  
   184  	s := gocron.NewScheduler(time.UTC)
   185  
   186  	// Schedule scale down routine every 10 minutes, start 10 minutes from now.
   187  	t := time.Now().Add(start)
   188  	s.Every(interval).Minutes().StartAt(t).Do(func() {
   189  		// Send into scheduling channel
   190  		scheduledEvents <- algos.ScalingEvent{
   191  			// Create a UUID so we can identify and search this event on our logs
   192  			ID: uuid.NewString(),
   193  			// We set the event type to SCHEDULED_SCALE_DOWN_EVENT
   194  			// here so that we have more information about the event.
   195  			// SCHEDULED means the source of the event is the scheduler
   196  			// and SCALE_DOWN means it will perform a scale down action.
   197  			Type: "SCHEDULED_SCALE_DOWN_EVENT",
   198  		}
   199  	})
   200  
   201  	s.StartAsync()
   202  }

   204  // StartDeploy reads the `images.json` file which is written by the Github
   205  // deploy workflow, and sends the event to the appropiate channel.
   206  func StartDeploy(scheduledEvents chan algos.ScalingEvent) {
   207  	if metadata.IsLocalEnv() && !metadata.IsRunningInCI() {
   208  		logger.Infof("Running in localenv so not performing deploy actions.")
   209  		return
   210  	}
   211  
   212  	regionImageMap, err := getRegionImageMap()
   213  	if err != nil {
   214  		logger.Error(err)
   215  		return
   216  	}
   217  
   218  	// Send image upgrade event to scheduled chan.
   219  	scheduledEvents <- algos.ScalingEvent{
   220  		// Create a UUID so we can identify and search this event on our logs
   221  		ID: uuid.NewString(),
   222  		// We set the event type to SCHEDULED_IMAGE_UPGRADE_EVENT
   223  		// here so that we have more information about the event.
   224  		// SCHEDULED means the source of the event is the scheduler
   225  		// and IMAGE_UPGRADE means it will perform an image upgrade action.
   226  		Type: "SCHEDULED_IMAGE_UPGRADE_EVENT",
   227  		Data: regionImageMap,
   228  	}
   229  }

   231  // getRegionImageMap tries to read and parse the contents of the `images.json` file
   232  // which is written by the Github deploy workflow.
   233  func getRegionImageMap() (map[string]interface{}, error) {
   234  	var regionImageMap map[string]interface{}
   235  
   236  	// Get current working directory to read images file.
   237  	currentWorkingDirectory, err := os.Getwd()
   238  	if err != nil {
   239  		return nil, utils.MakeError("failed to get working directory: %s", err)
   240  	}
   241  
   242  	// Read file which contains the region to image on JSON format. This file will
   243  	// be read by the binary generated during deploy, located in the `bin` directory.
   244  	// The file is also generated during deploy and lives in the scaling service directory.
   245  	content, err := os.ReadFile(path.Join(currentWorkingDirectory, "images.json"))
   246  	if err != nil {
   247  		return nil, utils.MakeError("failed to read region to image map from file: %s", err)
   248  	}
   249  
   250  	// Try to unmarshal contents of file into a map
   251  	err = json.Unmarshal(content, &regionImageMap)
   252  	if err != nil {
   253  		return nil, utils.MakeError("failed to unmarshal region to image map: %s", err)
   254  	}
   255  
   256  	return regionImageMap, nil
   257  }

   259  // getScalingAlgorithm is a helper function that returns the scaling algorithm from the sync map.
   260  func getScalingAlgorithm(algorithmByRegion *sync.Map, scalingEvent algos.ScalingEvent) algos.ScalingAlgorithm {
   261  	// Try to get the scaling algorithm on the region the scaling event was requested.
   262  	// If no region is specified, use the default region.
   263  	// TODO: figure out how to get non-default scaling algorihtms.
   264  	var (
   265  		name   string
   266  		region string
   267  	)
   268  	const defaultRegion = "us-east-1"
   269  
   270  	region = scalingEvent.Region
   271  
   272  	if region == "" {
   273  		logger.Infof("No region found in scaling event, using default region %s", defaultRegion)
   274  		name = utils.Sprintf("default-sa-%s", defaultRegion)
   275  	} else {
   276  		name = utils.Sprintf("default-sa-%s", scalingEvent.Region)
   277  	}
   278  
   279  	algorithm, ok := algorithmByRegion.Load(name)
   280  	if ok {
   281  		return algorithm.(algos.ScalingAlgorithm)
   282  	}
   283  
   284  	logger.Warningf("Failed to get scaling algorithm in %s", scalingEvent.Region)
   285  
   286  	return nil
   287  }

   289  // eventLoop is the main loop of the scaling service which will receive events from different sources
   290  // and send them to the appropiate channels.
   291  func eventLoop(globalCtx context.Context, globalCancel context.CancelFunc, serverEvents <-chan algos.ScalingEvent, subscriptionEvents <-chan subscriptions.SubscriptionEvent,
   292  	scheduledEvents <-chan algos.ScalingEvent, algorithmByRegion *sync.Map, configClient subscriptions.WhistSubscriptionClient) {
   293  
   294  	for {
   295  		select {
   296  		case subscriptionEvent := <-subscriptionEvents:
   297  			// Since we are dealing with a database event, figure out what type it has
   298  			// and send it to the corresponding channel
   299  			switch subscriptionEvent := subscriptionEvent.(type) {
   300  
   301  			case *subscriptions.InstanceEvent:
   302  				var scalingEvent algos.ScalingEvent
   303  
   304  				// We set the event type to DATABASE_INSTANCE_EVENT
   305  				// here so that we have more information about the event.
   306  				// DATABASE means the source of the event is the database
   307  				// and INSTANCE_EVENT means it will perform an action with
   308  				// the instance information received from the database.
   309  				scalingEvent.Type = "DATABASE_INSTANCE_EVENT"
   310  
   311  				if len(subscriptionEvent.Instances) > 0 {
   312  					instance := subscriptionEvent.Instances[0]
   313  					scalingEvent.ID = uuid.NewString()
   314  					scalingEvent.Data = instance
   315  					scalingEvent.Region = instance.Region
   316  				}
   317  
   318  				// Start scaling algorithm based on region
   319  				algorithm := getScalingAlgorithm(algorithmByRegion, scalingEvent)
   320  
   321  				switch algorithm := algorithm.(type) {
   322  				case *algos.DefaultScalingAlgorithm:
   323  					algorithm.InstanceEventChan <- scalingEvent
   324  				}
   325  			case *subscriptions.FrontendVersionEvent:
   326  				var (
   327  					scalingEvent algos.ScalingEvent
   328  					version      subscriptions.FrontendVersion
   329  				)
   330  
   331  				// We set the event type to DATABASE_FRONTEND_VERSION_EVENT
   332  				// here so that we have more information about the event.
   333  				// DATABASE means the source of the event is the database
   334  				// and FRONTEND_VERSION means it will perform an action with
   335  				// the frontend version information received from the database.
   336  				scalingEvent.Type = "DATABASE_FRONTEND_VERSION_EVENT"
   337  
   338  				// We only expect one version to come from the database subscription,
   339  				// anything more or less than it indicates an error in our backend.
   340  				if len(subscriptionEvent.FrontendVersions) == 1 {
   341  					version = subscriptionEvent.FrontendVersions[0]
   342  				} else {
   343  					logger.Errorf("unexpected length of %d in version received from the config database", len(subscriptionEvent.FrontendVersions))
   344  				}
   345  
   346  				regionImageMap, err := getRegionImageMap()
   347  				if err != nil {
   348  					logger.Error(err)
   349  					break
   350  				}
   351  
   352  				for region := range regionImageMap {
   353  					scalingEvent.Region = region
   354  					scalingEvent.Data = version
   355  
   356  					// Start scaling algorithm based on region
   357  					algorithm := getScalingAlgorithm(algorithmByRegion, scalingEvent)
   358  
   359  					switch algorithm := algorithm.(type) {
   360  					case *algos.DefaultScalingAlgorithm:
   361  						algorithm.FrontendVersionChan <- scalingEvent
   362  					}
   363  				}
   364  
   365  				// Its necessary to close the config client since we only want to
   366  				// do the image swapover once, but Hasura will keep receiving
   367  				// the event after it has switched.
   368  				err = configClient.Close()
   369  				if err != nil {
   370  					// err is already wrapped here
   371  					logger.Error(err)
   372  				}
   373  			}
   374  
   375  		case scheduledEvent := <-scheduledEvents:
   376  			// Start scaling algorithm based on region
   377  			for _, region := range algos.GetEnabledRegions() {
   378  				scheduledEvent.Region = region
   379  				algorithm := getScalingAlgorithm(algorithmByRegion, scheduledEvent)
   380  				switch algorithm := algorithm.(type) {
   381  				case *algos.DefaultScalingAlgorithm:
   382  					algorithm.ScheduledEventChan <- scheduledEvent
   383  				}
   384  			}
   385  		case serverEvent := <-serverEvents:
   386  			algorithm := getScalingAlgorithm(algorithmByRegion, serverEvent)
   387  			switch algorithm := algorithm.(type) {
   388  			case *algos.DefaultScalingAlgorithm:
   389  				algorithm.ServerEventChan <- serverEvent
   390  			}
   391  
   392  		case <-globalCtx.Done():
   393  			logger.Infof("Global context has been cancelled, exiting event loop...")
   394  			return
   395  		}
   396  	}
