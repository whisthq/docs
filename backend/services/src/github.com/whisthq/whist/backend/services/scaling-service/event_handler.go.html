   128  // StartSchedulerEvents starts the scheduler and its events without blocking the main thread.
   129  // `interval` sets the time when the event will run in minutes (i.e. every 10 minutes), and `start`
   130  // sets the time when the first event will happen (i.e. 10 minutes from now).
   131  func StartSchedulerEvents(scheduledEvents chan algos.ScalingEvent, interval interface{}, start time.Duration) {
   132  	s := gocron.NewScheduler(time.UTC)
   133  
   134  	// Schedule scale down routine every 10 minutes, start 10 minutes from now.
   135  	t := time.Now().Add(start)
   136  	s.Every(interval).Minutes().StartAt(t).Do(func() {
   137  		// Send into scheduling channel
   138  		scheduledEvents <- algos.ScalingEvent{
   139  			// Create a UUID so we can identify and search this event on our logs
   140  			ID: uuid.NewString(),
   141  			// We set the event type to SCHEDULED_SCALE_DOWN_EVENT
   142  			// here so that we have more information about the event.
   143  			// SCHEDULED means the source of the event is the scheduler
   144  			// and SCALE_DOWN means it will perform a scale down action.
   145  			Type: "SCHEDULED_SCALE_DOWN_EVENT",
   146  		}
   147  	})
   148  
   149  	s.StartAsync()
   150  }

   173  func getRegionImageMap() (map[string]interface{}, error) {
   174  	var regionImageMap map[string]interface{}
   175  
   176  	// Get current working directory to read images file.
   177  	currentWorkingDirectory, err := os.Getwd()
   178  	if err != nil {
   179  		return nil, utils.MakeError("Failed to get working directory. Err: %v", err)
   180  	}
   181  
   182  	// Read file which contains the region to image on JSON format. This file will
   183  	// be read by the binary generated during deploy, located in the `bin` directory.
   184  	// The file is also generated during deploy and lives in the scaling service directory.
   185  	content, err := os.ReadFile(path.Join(currentWorkingDirectory, "images.json"))
   186  	if err != nil {
   187  		return nil, utils.MakeError("Failed to read region to image map from file. Not performing image upgrade. Err: %v", err)
   188  	}
   189  
   190  	// Try to unmarshal contents of file into a map
   191  	err = json.Unmarshal(content, &regionImageMap)
   192  	if err != nil {
   193  		return nil, utils.MakeError("Failed to unmarshal region to image map. Not performing image upgrade. Err: %v", err)
   194  	}
   195  
   196  	return regionImageMap, nil
   197  }

   214  // eventLoop is the main loop of the scaling service which will receive events from different sources
   215  // and send them to the appropiate channels.
   216  func eventLoop(globalCtx context.Context, globalCancel context.CancelFunc, subscriptionEvents <-chan subscriptions.SubscriptionEvent,
   217  	scheduledEvents <-chan algos.ScalingEvent, algorithmByRegion *sync.Map, configClient *subscriptions.SubscriptionClient) {
   218  
   219  	for {
   220  		select {
   221  		case subscriptionEvent := <-subscriptionEvents:
   222  			// Since we are dealing with a database event, figure out what type it has
   223  			// and send it to the corresponding channel
   224  			switch subscriptionEvent := subscriptionEvent.(type) {
   225  
   226  			case *subscriptions.InstanceEvent:
   227  				var scalingEvent algos.ScalingEvent
   228  
   229  				// We set the event type to DATABASE_INSTANCE_EVENT
   230  				// here so that we have more information about the event.
   231  				// DATABASE means the source of the event is the database
   232  				// and INSTANCE_EVENT means it will perform an action with
   233  				// the instance information received from the database.
   234  				scalingEvent.Type = "DATABASE_INSTANCE_EVENT"
   235  
   236  				if len(subscriptionEvent.Instances) > 0 {
   237  					instance := subscriptionEvent.Instances[0]
   238  					scalingEvent.ID = uuid.NewString()
   239  					scalingEvent.Data = instance
   240  					scalingEvent.Region = instance.Region
   241  				}
   242  
   243  				// Start scaling algorithm based on region
   244  				logger.Infof("Received database event.")
   245  				algorithm := getScalingAlgorithm(algorithmByRegion, scalingEvent)
   246  
   247  				switch algorithm := algorithm.(type) {
   248  				case *algos.DefaultScalingAlgorithm:
   249  					algorithm.InstanceEventChan <- scalingEvent
   250  				}
   251  			case *subscriptions.ClientAppVersionEvent:
   252  				var (
   253  					scalingEvent algos.ScalingEvent
   254  					version      subscriptions.ClientAppVersion
   255  				)
   256  
   257  				// We set the event type to DATABASE_CLIENT_VERSION_EVENT
   258  				// here so that we have more information about the event.
   259  				// DATABASE means the source of the event is the database
   260  				// and CLIENT_VERSION means it will perform an action with
   261  				// the client version information received from the database.
   262  				scalingEvent.Type = "DATABASE_CLIENT_VERSION_EVENT"
   263  
   264  				// We only expect one version to come from the database subscription,
   265  				// anything more or less than it indicates an error in our backend.
   266  				if len(subscriptionEvent.ClientAppVersions) == 1 {
   267  					version = subscriptionEvent.ClientAppVersions[0]
   268  				} else {
   269  					logger.Errorf("Unexpected length of %v in version received from the config database.", len(subscriptionEvent.ClientAppVersions))
   270  				}
   271  
   272  				// TODO: once we keep track of client versions per region,
   273  				// get the region directly from the database. For now use
   274  				// the regionImageMap.
   275  				regionImageMap, err := getRegionImageMap()
   276  				if err != nil {
   277  					logger.Errorf("Error getting regionImageMap. Err: %v", err)
   278  					break
   279  				}
   280  
   281  				for region := range regionImageMap {
   282  					scalingEvent.Region = region
   283  					scalingEvent.Data = version
   284  
   285  					// Start scaling algorithm based on region
   286  					logger.Infof("Received database event.")
   287  					algorithm := getScalingAlgorithm(algorithmByRegion, scalingEvent)
   288  
   289  					switch algorithm := algorithm.(type) {
   290  					case *algos.DefaultScalingAlgorithm:
   291  						algorithm.ClientAppVersionChan <- scalingEvent
   292  					}
   293  				}
   294  
   295  				// Its necessary to close the config client since we only want to
   296  				// do the image swapover once, but Hasura will keep receiving
   297  				// the event after it has switched.
   298  				err = configClient.Close()
   299  				if err != nil {
   300  					// err is already wrapped here
   301  					logger.Error(err)
   302  				}
   303  			}
   304  
   305  		case scheduledEvent := <-scheduledEvents:
   306  			// Start scaling algorithm based on region
   307  			logger.Infof("Received scheduled event. %v", scheduledEvent)
   308  
   309  			for _, region := range algos.BundledRegions {
   310  				scheduledEvent.Region = region
   311  				algorithm := getScalingAlgorithm(algorithmByRegion, scheduledEvent)
   312  				switch algorithm := algorithm.(type) {
   313  				case *algos.DefaultScalingAlgorithm:
   314  					algorithm.ScheduledEventChan <- scheduledEvent
   315  				}
   316  			}
   317  		case <-globalCtx.Done():
   318  			logger.Infof("Gloal context has been cancelled, exiting event loop...")
   319  			return
   320  		}
   321  	}
