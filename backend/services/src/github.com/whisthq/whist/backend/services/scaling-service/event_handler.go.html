     3  /*
     4  The scaling service is a server responsible of managing the state of the backend.
     5  It manages the number of instances running by starting and terminating according
     6  to demand. Additionally, it assigns users to available instances, and registers a
     7  mandelbox on the database so the host service running on the instance can start it.
     8  This component is also responsible of validating payments, as well as interacting
     9  with Stripe to create new subscriptions.
    10  
    11  The scaling service is an event-driven component, so most of its logic is handled using
    12  go channels, in reaction to external events. It has many layers, the bottom layer being
    13  the event handler which only passes the events to the corresponding channels. The layer
    14  on top of it includes the many scaling algorithms which are sub-processes for each region
    15  on the backend. The code is written in an extensible way so it is possible to write and
    16  integrate a new scaling algorithm with a different behaviour easily. As such, most of
    17  the code is inside the scaling algorithm implementations.
    18  
    19  The top-most layer has the "hosts handlers" which are a collection of methods responsible
    20  of calling a cloud provider (e.g. AWS, Azure, Google Cloud) API and starting an instance
    21  (cloud-compute resource configured to run a Whist container). This code is abstracted in
    22  an interface so that support for a new cloud provider is easily added. The code in the
    23  algorithms is provider-agnostic, tied only to the necessities of the backend.
    24  */

    51  	globalCtx, globalCancel := context.WithCancel(context.Background())
    52  	goroutineTracker := &sync.WaitGroup{}
    53  
    54  	// Add some additional fields for Logz.io
    55  	tags := make(map[string]string)
    56  	tags["component"] = "backend"
    57  	tags["sub-component"] = "scaling-service"
    58  	logger.AddLogzioFields(tags)

    60  	var (
    61  		dbClient            dbclient.WhistDBClient                // The client that abstracts database interactions
    62  		graphqlClient       subscriptions.WhistGraphQLClient      // The GraphQL client to query the Hasura server
    63  		configGraphqlClient subscriptions.WhistGraphQLClient      // The GraphQL client to query the config Hasura server
    64  		subscriptionClient  subscriptions.WhistSubscriptionClient // The subscription client to subscribe to the Hasura server
    65  		configClient        subscriptions.WhistSubscriptionClient // The subscriptions client to subscribe to the config Hasura server

    67  		scheduledEvents     chan algos.ScalingEvent               // Channel to process scheduled events
    68  
    69  	)
    70  
    71  	// Start HTTP Server for assigning mandelboxes
    72  	serverEvents := make(chan algos.ScalingEvent, 100)
    73  	StartHTTPServer(serverEvents)
    74  
    75  	// Start GraphQL client for queries/mutations
    76  	useConfigDB := false
    77  	graphqlClient = &subscriptions.GraphQLClient{}
    78  	err := graphqlClient.Initialize(useConfigDB)
    79  	if err != nil {
    80  		logger.Errorf("failed to start GraphQL client: %s", err)
    81  	}
    82  
    83  	// Start GraphQL client for getting configuration from the config db
    84  	useConfigDB = true
    85  	configGraphqlClient = &subscriptions.GraphQLClient{}
    86  	err = configGraphqlClient.Initialize(useConfigDB)
    87  	if err != nil {
    88  		logger.Errorf("failed to start config GraphQL client: %s", err)
    89  	}
    90  
    91  	if err := config.Initialize(globalCtx, configGraphqlClient); err != nil {
    92  		logger.Errorf("Failed to retrieve configuration values: %s", err)
    93  		return
    94  	}
    95  
    96  	// Start database subscriptions
    97  	subscriptionEvents = make(chan subscriptions.SubscriptionEvent, 100)
    98  	subscriptionClient = &subscriptions.SubscriptionClient{}
    99  	configClient = &subscriptions.SubscriptionClient{}
   100  	StartDatabaseSubscriptions(globalCtx, goroutineTracker, subscriptionEvents, subscriptionClient, configClient)
   101  
   102  	// Initialize database client
   103  	dbClient = &dbclient.DBClient{}
   104  
   105  	// Start scheduler and setup scheduler event chan
   106  	scheduledEvents = make(chan algos.ScalingEvent, 100)
   107  
   108  	// Set to run every 10 minutes, starting 10 minutes from now
   109  	start := time.Duration(10 * time.Minute)
   110  	StartSchedulerEvents(scheduledEvents, 10, start)
   111  
   112  	// Start the deploy events once since we are starting the scaling service.
   113  	StartDeploy(scheduledEvents)
   114  
   115  	// algorithmByRegionMap holds all of the scaling algorithms mapped by region.
   116  	// Use a sync map since we only write the keys once but will be reading multiple
   117  	// times by different goroutines.
   118  	algorithmByRegionMap := &sync.Map{}
   119  
   120  	// Load default scaling algorithm for all enabled regions.
   121  	for _, region := range config.GetEnabledRegions() {
   122  		logger.Infof("There should be as close as possible to %d unassigned "+
   123  			"Mandelboxes available at all times in %s",
   124  			config.GetTargetFreeMandelboxes(region), region)
   125  		name := utils.Sprintf("default-sa-%s", region)
   126  		algorithmByRegionMap.Store(name, &algos.DefaultScalingAlgorithm{
   127  			Region: region,
   128  		})
   129  	}
   130  
   131  	// Instantiate scaling algorithms on allowed regions
   132  	algorithmByRegionMap.Range(func(key, value interface{}) bool {
   133  		scalingAlgorithm := value.(algos.ScalingAlgorithm)
   134  		scalingAlgorithm.CreateEventChans()
   135  		scalingAlgorithm.CreateGraphQLClient(graphqlClient)
   136  		scalingAlgorithm.CreateDBClient(dbClient)
   137  		scalingAlgorithm.ProcessEvents(globalCtx, goroutineTracker)
   138  
   139  		return true
   140  	})
   141  
   142  	// Start main event loop
   143  	go eventLoop(globalCtx, globalCancel, serverEvents, subscriptionEvents, scheduledEvents, algorithmByRegionMap, configClient)
   144  
   145  	// Register a signal handler for Ctrl-C so that we cleanup if Ctrl-C is pressed.
   146  	sigChan := make(chan os.Signal, 2)
   147  	signal.Notify(sigChan, os.Interrupt, syscall.SIGTERM, syscall.SIGINT)
   148  
   149  	// Wait for either the global context to get cancelled by a worker goroutine,
   150  	// or for us to receive an interrupt. This needs to be the end of main().
   151  	select {
   152  	case <-sigChan:
   153  		logger.Infof("Got an interrupt or SIGTERM")
   154  		logger.Sync()
   155  	case <-globalCtx.Done():
   156  		logger.Infof("Global context cancelled!")
   157  		logger.Sync()
   158  	}

   185  // StartSchedulerEvents starts the scheduler and its events without blocking the main thread.
   186  // `interval` sets the time when the event will run in minutes (i.e. every 10 minutes), and `start`
   187  // sets the time when the first event will happen (i.e. 10 minutes from now).
   188  func StartSchedulerEvents(scheduledEvents chan algos.ScalingEvent, interval interface{}, start time.Duration) {
   189  	if metadata.IsLocalEnvWithoutDB() && !metadata.IsRunningInCI() {
   190  		return
   191  	}
   192  
   193  	s := gocron.NewScheduler(time.UTC)
   194  
   195  	// Schedule scale down routine every 10 minutes, start 10 minutes from now.
   196  	t := time.Now().Add(start)
   197  	s.Every(interval).Minutes().StartAt(t).Do(func() {
   198  		// Send into scheduling channel
   199  		scheduledEvents <- algos.ScalingEvent{
   200  			// Create a UUID so we can identify and search this event on our logs
   201  			ID: uuid.NewString(),
   202  			// We set the event type to SCHEDULED_SCALE_DOWN_EVENT
   203  			// here so that we have more information about the event.
   204  			// SCHEDULED means the source of the event is the scheduler
   205  			// and SCALE_DOWN means it will perform a scale down action.
   206  			Type: "SCHEDULED_SCALE_DOWN_EVENT",
   207  		}
   208  	})
   209  
   210  	s.StartAsync()
   211  }

   213  // StartDeploy reads the `images.json` file which is written by the Github
   214  // deploy workflow, and sends the event to the appropiate channel.
   215  func StartDeploy(scheduledEvents chan algos.ScalingEvent) {
   216  	if metadata.IsLocalEnv() && !metadata.IsRunningInCI() {
   217  		logger.Infof("Running on localdev so not performing deploy actions.")
   218  		return
   219  	}
   220  
   221  	regionImageMap, err := getRegionImageMap()
   222  	if err != nil {
   223  		logger.Error(err)
   224  		return
   225  	}
   226  
   227  	// Send image upgrade event to scheduled chan.
   228  	scheduledEvents <- algos.ScalingEvent{
   229  		// Create a UUID so we can identify and search this event on our logs
   230  		ID: uuid.NewString(),
   231  		// We set the event type to SCHEDULED_IMAGE_UPGRADE_EVENT
   232  		// here so that we have more information about the event.
   233  		// SCHEDULED means the source of the event is the scheduler
   234  		// and IMAGE_UPGRADE means it will perform an image upgrade action.
   235  		Type: "SCHEDULED_IMAGE_UPGRADE_EVENT",
   236  		Data: regionImageMap,
   237  	}
   238  }

   240  // getRegionImageMap tries to read and parse the contents of the `images.json` file
   241  // which is written by the Github deploy workflow.
   242  func getRegionImageMap() (map[string]interface{}, error) {
   243  	var (
   244  		regionImageMap map[string]interface{}
   245  		filename       string
   246  	)
   247  
   248  	// Read file which contains the region to image on JSON format. This file will
   249  	// be read by the binary generated during deploy, located in the `bin` directory.
   250  	// The file is written to /etc/whist/images.json when building the Docker container.
   251  	if metadata.IsLocalEnv() {
   252  		// Get current working directory to read images file.
   253  		currentWorkingDirectory, err := os.Getwd()
   254  		if err != nil {
   255  			return nil, utils.MakeError("failed to get working directory: %s", err)
   256  		}
   257  		filename = path.Join(currentWorkingDirectory, "images.json")
   258  	} else {
   259  		filename = "/etc/whist/images.json"
   260  	}
   261  	content, err := os.ReadFile(filename)
   262  	if err != nil {
   263  		return nil, utils.MakeError("failed to read region to image map from file: %s", err)
   264  	}
   265  
   266  	// Try to unmarshal contents of file into a map
   267  	err = json.Unmarshal(content, &regionImageMap)
   268  	if err != nil {
   269  		return nil, utils.MakeError("failed to unmarshal region to image map: %s", err)
   270  	}
   271  
   272  	return regionImageMap, nil
   273  }

   275  // getScalingAlgorithm is a helper function that returns the scaling algorithm from the sync map.
   276  func getScalingAlgorithm(algorithmByRegion *sync.Map, scalingEvent algos.ScalingEvent) algos.ScalingAlgorithm {
   277  	// Try to get the scaling algorithm on the region the scaling event was requested.
   278  	// If no region is specified, use the default region.
   279  	// TODO: figure out how to get non-default scaling algorihtms.
   280  	var (
   281  		name   string
   282  		region string
   283  	)
   284  	const defaultRegion = "us-east-1"
   285  
   286  	region = scalingEvent.Region
   287  
   288  	if region == "" {
   289  		logger.Infof("No region found in scaling event, using default region %s", defaultRegion)
   290  		name = utils.Sprintf("default-sa-%s", defaultRegion)
   291  	} else {
   292  		name = utils.Sprintf("default-sa-%s", scalingEvent.Region)
   293  	}
   294  
   295  	algorithm, ok := algorithmByRegion.Load(name)
   296  	if ok {
   297  		return algorithm.(algos.ScalingAlgorithm)
   298  	}
   299  
   300  	logger.Warningf("Failed to get scaling algorithm in %s", scalingEvent.Region)
   301  
   302  	return nil
   303  }

   305  // eventLoop is the main loop of the scaling service which will receive events from different sources
   306  // and send them to the appropiate channels.
   307  func eventLoop(globalCtx context.Context, globalCancel context.CancelFunc, serverEvents <-chan algos.ScalingEvent, subscriptionEvents <-chan subscriptions.SubscriptionEvent,
   308  	scheduledEvents <-chan algos.ScalingEvent, algorithmByRegion *sync.Map, configClient subscriptions.WhistSubscriptionClient) {
   309  
   310  	for {
   311  		select {
   312  		case subscriptionEvent := <-subscriptionEvents:
   313  			// Since we are dealing with a database event, figure out what type it has
   314  			// and send it to the corresponding channel
   315  			switch subscriptionEvent := subscriptionEvent.(type) {
   316  
   317  			case *subscriptions.InstanceEvent:
   318  				var scalingEvent algos.ScalingEvent
   319  
   320  				// We set the event type to DATABASE_INSTANCE_EVENT
   321  				// here so that we have more information about the event.
   322  				// DATABASE means the source of the event is the database
   323  				// and INSTANCE_EVENT means it will perform an action with
   324  				// the instance information received from the database.
   325  				scalingEvent.Type = "DATABASE_INSTANCE_EVENT"
   326  
   327  				if len(subscriptionEvent.Instances) > 0 {
   328  					instance := subscriptionEvent.Instances[0]
   329  					scalingEvent.ID = uuid.NewString()
   330  					scalingEvent.Data = instance
   331  					scalingEvent.Region = instance.Region
   332  				}
   333  
   334  				// Start scaling algorithm based on region
   335  				algorithm := getScalingAlgorithm(algorithmByRegion, scalingEvent)
   336  
   337  				switch algorithm := algorithm.(type) {
   338  				case *algos.DefaultScalingAlgorithm:
   339  					algorithm.InstanceEventChan <- scalingEvent
   340  				}
   341  			case *subscriptions.FrontendVersionEvent:
   342  				var (
   343  					scalingEvent algos.ScalingEvent
   344  					version      subscriptions.FrontendVersion
   345  				)
   346  
   347  				// We set the event type to DATABASE_FRONTEND_VERSION_EVENT
   348  				// here so that we have more information about the event.
   349  				// DATABASE means the source of the event is the database
   350  				// and FRONTEND_VERSION means it will perform an action with
   351  				// the frontend version information received from the database.
   352  				scalingEvent.Type = "DATABASE_FRONTEND_VERSION_EVENT"
   353  
   354  				// We only expect one version to come from the database subscription,
   355  				// anything more or less than it indicates an error in our backend.
   356  				if len(subscriptionEvent.FrontendVersions) == 1 {
   357  					version = subscriptionEvent.FrontendVersions[0]
   358  				} else {
   359  					logger.Errorf("unexpected length of %d in version received from the config database", len(subscriptionEvent.FrontendVersions))
   360  				}
   361  
   362  				regionImageMap, err := getRegionImageMap()
   363  				if err != nil {
   364  					logger.Error(err)
   365  					break
   366  				}
   367  
   368  				for region := range regionImageMap {
   369  					scalingEvent.Region = region
   370  					scalingEvent.Data = version
   371  
   372  					// Start scaling algorithm based on region
   373  					algorithm := getScalingAlgorithm(algorithmByRegion, scalingEvent)
   374  
   375  					switch algorithm := algorithm.(type) {
   376  					case *algos.DefaultScalingAlgorithm:
   377  						algorithm.FrontendVersionChan <- scalingEvent
   378  					}
   379  				}
   380  
   381  				// Its necessary to close the config client since we only want to
   382  				// do the image swapover once, but Hasura will keep receiving
   383  				// the event after it has switched.
   384  				err = configClient.Close()
   385  				if err != nil {
   386  					// err is already wrapped here
   387  					logger.Error(err)
   388  				}
   389  			}
   390  
   391  		case scheduledEvent := <-scheduledEvents:
   392  			// Start scaling algorithm based on region
   393  			for _, region := range config.GetEnabledRegions() {
   394  				scheduledEvent.Region = region
   395  				algorithm := getScalingAlgorithm(algorithmByRegion, scheduledEvent)
   396  				switch algorithm := algorithm.(type) {
   397  				case *algos.DefaultScalingAlgorithm:
   398  					algorithm.ScheduledEventChan <- scheduledEvent
   399  				}
   400  			}
   401  		case serverEvent := <-serverEvents:
   402  			algorithm := getScalingAlgorithm(algorithmByRegion, serverEvent)
   403  			switch algorithm := algorithm.(type) {
   404  			case *algos.DefaultScalingAlgorithm:
   405  				algorithm.ServerEventChan <- serverEvent
   406  			}
   407  
   408  		case <-globalCtx.Done():
   409  			logger.Infof("Global context has been cancelled, exiting event loop...")
   410  			return
   411  		}
   412  	}
