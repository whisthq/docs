    76  // detects changes on the config database client app version to the current.
    77  func ClientAppVersionHandler(event SubscriptionEvent, variables map[string]interface{}) bool {
    78  	result := event.(ClientAppVersionEvent)
    79  
    80  	var (
    81  		version    ClientAppVersion
    82  		commitHash string
    83  	)
    84  
    85  	if len(result.ClientAppVersions) > 0 {
    86  		version = result.ClientAppVersions[0]
    87  	}
    88  
    89  	if version == (ClientAppVersion{}) {
    90  		return false
    91  	}
    92  
    93  	// get the commit hash of the environment we are running in
    94  	switch metadata.GetAppEnvironmentLowercase() {
    95  	case string(metadata.EnvDev):
    96  		commitHash = version.DevCommitHash
    97  	case string(metadata.EnvStaging):
    98  		commitHash = version.StagingCommitHash
    99  	case string(metadata.EnvProd):
   100  		commitHash = version.ProdCommitHash
   101  	default:
   102  		commitHash = version.DevCommitHash
   103  	}
   104  
   105  	return metadata.GetGitCommit() == commitHash
   106  }

   170  // Start is the main function in the subscriptions package. It initializes a client, sets up the received subscriptions,
   171  // and starts a goroutine for the client. It also has a goroutine to close the client and subscriptions when the global
   172  // context gets cancelled. It's possible to subscribe to the config database instead of the dev/staging/prod database by
   173  // setting the `useConfigDatabase` argument.
   174  func Start(whistClient WhistSubscriptionClient, globalCtx context.Context, goroutineTracker *sync.WaitGroup, subscriptionEvents chan SubscriptionEvent, useConfigDB bool) error {
   175  	if !Enabled {
   176  		logger.Infof("Running in app environment %s so not enabling Subscription client code.", metadata.GetAppEnvironment())
   177  		return nil
   178  	}
   179  
   180  	// Slice to hold subscription IDs, necessary to properly unsubscribe when we are done.
   181  	var subscriptionIDs []string
   182  
   183  	// Initialize subscription client
   184  	whistClient.Initialize(useConfigDB)
   185  
   186  	// Start goroutine that shuts down the client if the global context gets
   187  	// cancelled.
   188  	goroutineTracker.Add(1)
   189  	go func() {
   190  		defer goroutineTracker.Done()
   191  
   192  		// Listen for global context cancellation
   193  		<-globalCtx.Done()
   194  		whistClient.Close()
   195  	}()
   196  
   197  	// Send subscriptions to the client
   198  	for _, subscriptionParams := range whistClient.GetSubscriptions() {
   199  		query := subscriptionParams.Query
   200  		variables := subscriptionParams.Variables
   201  		result := subscriptionParams.Result
   202  		handler := subscriptionParams.Handler
   203  
   204  		id, err := whistClient.Subscribe(query, variables, result, handler, subscriptionEvents)
   205  		if err != nil {
   206  			return err
   207  		}
   208  		subscriptionIDs = append(subscriptionIDs, id)
   209  	}
   210  
   211  	// Run the client
   212  	whistClient.SetSubscriptionsIDs(subscriptionIDs)
   213  	whistClient.Run(goroutineTracker)
   214  
   215  	return nil
   216  }
