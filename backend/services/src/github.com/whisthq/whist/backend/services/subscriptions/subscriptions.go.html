    17  // InstanceStatusHandler handles events from the hasura subscription which
    18  // detects changes on instance instanceName to the given status in the database.
    19  func InstanceStatusHandler(event SubscriptionEvent, variables map[string]interface{}) bool {
    20  	result := event.(InstanceEvent)
    21  
    22  	var instance Instance
    23  	if len(result.Instances) > 0 {
    24  		instance = result.Instances[0]
    25  	}
    26  
    27  	if (variables["id"] == nil) || (variables["status"] == nil) {
    28  		return false
    29  	}
    30  
    31  	id := string(variables["id"].(graphql.String))
    32  	status := string(variables["status"].(graphql.String))
    33  
    34  	return (instance.ID == id) && (instance.Status == status)

   118  // Start is the main function in the subscriptions package. It initializes a client, sets up the received subscriptions,
   119  // and starts a goroutine for the client. It also has a goroutine to close the client and subscriptions when the global
   120  // context gets cancelled.
   121  func Start(whistClient WhistSubscriptionClient, globalCtx context.Context, goroutineTracker *sync.WaitGroup, subscriptionEvents chan SubscriptionEvent) error {
   122  	if !Enabled {
   123  		logger.Infof("Running in app environment %s so not enabling Subscription client code.", metadata.GetAppEnvironment())
   124  		return nil
   125  	}
   126  
   127  	// Slice to hold subscription IDs, necessary to properly unsubscribe when we are done.
   128  	var subscriptionIDs []string
   129  
   130  	// Initialize subscription client
   131  	whistClient.Initialize()
   132  
   133  	// Start goroutine that shuts down the client if the global context gets
   134  	// cancelled.
   135  	goroutineTracker.Add(1)
   136  	go func() {
   137  		defer goroutineTracker.Done()
   138  
   139  		// Listen for global context cancellation
   140  		<-globalCtx.Done()
   141  		whistClient.Close(whistClient.GetSubscriptionIDs())
   142  	}()
   143  
   144  	// Send subscriptions to the client
   145  	for _, subscriptionParams := range whistClient.GetSubscriptions() {
   146  		query := subscriptionParams.Query
   147  		variables := subscriptionParams.Variables
   148  		result := subscriptionParams.Result
   149  		handler := subscriptionParams.Handler
   150  
   151  		id, err := whistClient.Subscribe(query, variables, result, handler, subscriptionEvents)
   152  		if err != nil {
   153  			return err
   154  		}
   155  		subscriptionIDs = append(subscriptionIDs, id)
   156  	}
   157  
   158  	// Run the client
   159  	whistClient.SetSubscriptionsIDs(subscriptionIDs)
   160  	whistClient.Run(goroutineTracker)
   161  
   162  	return nil
   163  }
