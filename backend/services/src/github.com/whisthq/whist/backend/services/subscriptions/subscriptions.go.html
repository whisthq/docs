    76  // ClientAppVersionHandler handles events from the hasura subscription which
    77  // detects changes on the config database client app version to the current.
    78  func ClientAppVersionHandler(event SubscriptionEvent, variables map[string]interface{}) bool {
    79  	result := event.(ClientAppVersionEvent)
    80  
    81  	var (
    82  		version    ClientAppVersion
    83  		commitHash string
    84  	)
    85  
    86  	if len(result.ClientAppVersions) > 0 {
    87  		version = result.ClientAppVersions[0]
    88  	}
    89  
    90  	if version == (ClientAppVersion{}) {
    91  		return false
    92  	}
    93  
    94  	// get the commit hash of the environment we are running in
    95  	switch metadata.GetAppEnvironmentLowercase() {
    96  	case string(metadata.EnvDev):
    97  		commitHash = version.DevCommitHash
    98  	case string(metadata.EnvStaging):
    99  		commitHash = version.StagingCommitHash
   100  	case string(metadata.EnvProd):
   101  		commitHash = version.ProdCommitHash
   102  	default:
   103  		commitHash = version.DevCommitHash
   104  	}
   105  
   106  	return metadata.GetGitCommit() == commitHash
   107  }

   172  // Start is the main function in the subscriptions package. It initializes a client, sets up the received subscriptions,
   173  // and starts a goroutine for the client. It also has a goroutine to close the client and subscriptions when the global
   174  // context gets cancelled. It's possible to subscribe to the config database instead of the dev/staging/prod database by
   175  // setting the `useConfigDatabase` argument.
   176  func Start(whistClient WhistSubscriptionClient, globalCtx context.Context, goroutineTracker *sync.WaitGroup, subscriptionEvents chan SubscriptionEvent, useConfigDB bool) error {
   177  	if !Enabled {
   178  		logger.Infof("Running in app environment %s so not enabling Subscription client code.", metadata.GetAppEnvironment())
   179  		return nil
   180  	}
   181  
   182  	// Slice to hold subscription IDs, necessary to properly unsubscribe when we are done.
   183  	var subscriptionIDs []string
   184  
   185  	// Initialize subscription client
   186  	whistClient.Initialize(useConfigDB)
   187  
   188  	// Start goroutine that shuts down the client if the global context gets
   189  	// cancelled.
   190  	goroutineTracker.Add(1)
   191  	go func() {
   192  		defer goroutineTracker.Done()
   193  
   194  		// Listen for global context cancellation
   195  		<-globalCtx.Done()
   196  		whistClient.Close()
   197  	}()
   198  
   199  	// Send subscriptions to the client
   200  	for _, subscriptionParams := range whistClient.GetSubscriptions() {
   201  		query := subscriptionParams.Query
   202  		variables := subscriptionParams.Variables
   203  		result := subscriptionParams.Result
   204  		handler := subscriptionParams.Handler
   205  
   206  		id, err := whistClient.Subscribe(query, variables, result, handler, subscriptionEvents)
   207  		if err != nil {
   208  			return err
   209  		}
   210  		subscriptionIDs = append(subscriptionIDs, id)
   211  	}
   212  
   213  	// Run the client
   214  	whistClient.SetSubscriptionsIDs(subscriptionIDs)
   215  	whistClient.Run(goroutineTracker)
   216  
   217  	return nil
   218  }
