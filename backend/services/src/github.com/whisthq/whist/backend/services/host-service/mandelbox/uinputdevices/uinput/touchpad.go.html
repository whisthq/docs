     9  // A TouchPad is an input device that uses absolute axis events, meaning that you can specify

   109  func createTouchPad(path string, name []byte, minX int32, maxX int32, minY int32, maxY int32) (fd *os.File, err error) {
   110  	deviceFile, err := createDeviceFile(path)
   111  	if err != nil {
   112  		return nil, fmt.Errorf("could not create absolute axis input device: %v", err)
   113  	}
   114  
   115  	err = registerDevice(deviceFile, uintptr(evKey))
   116  	if err != nil {
   117  		deviceFile.Close()
   118  		return nil, fmt.Errorf("failed to register key device: %v", err)
   119  	}
   120  	// register button events
   121  	for _, event := range []int{evBtnTouch, evBtnToolPen, evBtnStylus} {
   122  		err = ioctl(deviceFile, uiSetKeyBit, uintptr(event))
   123  		if err != nil {
   124  			deviceFile.Close()
   125  			return nil, fmt.Errorf("failed to register button event %v: %v", event, err)
   126  		}
   127  	}
   128  
   129  	err = registerDevice(deviceFile, uintptr(evAbs))
   130  	if err != nil {
   131  		deviceFile.Close()
   132  		return nil, fmt.Errorf("failed to register absolute axis input device: %v", err)
   133  	}
   134  
   135  	// register x and y axis events
   136  	for _, event := range []int{absX, absY} {
   137  		err = ioctl(deviceFile, uiSetAbsBit, uintptr(event))
   138  		if err != nil {
   139  			deviceFile.Close()
   140  			return nil, fmt.Errorf("failed to register absolute axis event %v: %v", event, err)
   141  		}
   142  	}
   143  
   144  	err = ioctl(deviceFile, uiSetPropBit, inputPropPointer)
   145  
   146  	if err != nil {
   147  		return nil, fmt.Errorf("failed during ioctl: %v", err)
   148  	}
   149  
   150  	var absMin [absSize]int32
   151  	absMin[absX] = minX
   152  	absMin[absY] = minY
   153  
   154  	var absMax [absSize]int32
   155  	absMax[absX] = maxX
   156  	absMax[absY] = maxY
   157  
   158  	return createUsbDevice(deviceFile,
   159  		uinputUserDev{
   160  			Name: toUinputName(name),
   161  			ID: inputID{
   162  				Bustype: busUsb,
   163  				Vendor:  0xf4c1,
   164  				Product: 0x1124,
   165  				Version: 1},
   166  			Absmin: absMin,
   167  			Absmax: absMax})
   168  }

   170  func sendAbsEvent(deviceFile *os.File, xPos int32, yPos int32) error {
   171  	var ev [2]inputEvent
   172  	ev[0].Type = evAbs
   173  	ev[0].Code = absX
   174  	ev[0].Value = xPos
   175  
   176  	// Various tests (using evtest) have shown that positioning on x=0;y=0 doesn't trigger any event and will not move
   177  	// the cursor as expected. Setting at least one of the coordinates to -1 will however have the desired effect of
   178  	// moving the cursor to the upper left corner. Interestingly, the same is true for equivalent code in C, which rules
   179  	// out issues related to Go's data type representation or the like. This will need to be investigated further...
   180  	if xPos == 0 && yPos == 0 {
   181  		yPos--
   182  	}
   183  
   184  	ev[1].Type = evAbs
   185  	ev[1].Code = absY
   186  	ev[1].Value = yPos
   187  
   188  	for _, iev := range ev {
   189  		buf, err := inputEventToBuffer(iev)
   190  		if err != nil {
   191  			return fmt.Errorf("writing abs event failed: %v", err)
   192  		}
   193  
   194  		_, err = deviceFile.Write(buf)
   195  		if err != nil {
   196  			return fmt.Errorf("failed to write abs event to device file: %v", err)
   197  		}
   198  	}
   199  
   200  	return syncEvents(deviceFile)
   201  }
