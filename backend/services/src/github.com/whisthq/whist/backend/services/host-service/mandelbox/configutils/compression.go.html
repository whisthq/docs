    18  // ExtractTarLz4 extracts a tar.lz4 file in memory to the specified directory.
    19  // Returns the total number of bytes extracted or an error.
    20  func ExtractTarLz4(file []byte, dir string) (int64, error) {
    21  	// Create the destination directory if it doesn't exist.
    22  	_, err := os.Stat(dir)
    23  	if err != nil {
    24  		if os.IsNotExist(err) {
    25  			if err := os.MkdirAll(dir, 0755); err != nil {
    26  				return 0, utils.MakeError("failed to create directory %s: %v", dir, err)
    27  			}
    28  		} else {
    29  			return 0, utils.MakeError("failed to stat directory %s: %v", dir, err)
    30  		}
    31  	}
    32  
    33  	// Track the total number of bytes extracted
    34  	var totalBytes int64
    35  
    36  	// Extracting the tar.lz4 file involves:
    37  	// 1. Decompress the the lz4 file into a tar
    38  	// 2. Untar the config to the target directory
    39  	dataReader := bytes.NewReader(file)
    40  	lz4Reader := lz4.NewReader(dataReader)
    41  	tarReader := tar.NewReader(lz4Reader)
    42  
    43  	for {
    44  		// Read the header for the next file in the tar
    45  		header, err := tarReader.Next()
    46  
    47  		// If we reach EOF, that means we are done untaring
    48  		if err != nil {
    49  			if err == io.EOF {
    50  				break
    51  			}
    52  			return totalBytes, utils.MakeError("error reading tar header: %v", err)
    53  		}
    54  
    55  		// This really should not happen
    56  		if header == nil {
    57  			break
    58  		}
    59  
    60  		path := filepath.Join(dir, header.Name)
    61  
    62  		// Check that the path is within the destination directory
    63  		// to avoid the zip slip vulnerability
    64  		// See: https://snyk.io/research/zip-slip-vulnerability
    65  		if !strings.HasPrefix(path, dir) {
    66  			return totalBytes, utils.MakeError("destination path %s is illegal", path)
    67  		}
    68  
    69  		info := header.FileInfo()
    70  
    71  		// Create directory if it doesn't exist, otherwise go next
    72  		if info.IsDir() {
    73  			if err = os.MkdirAll(path, info.Mode()); err != nil {
    74  				return totalBytes, utils.MakeError("error creating directory: %v", err)
    75  			}
    76  			continue
    77  		}
    78  
    79  		// Create file and copy contents
    80  		file, err := os.OpenFile(path, os.O_CREATE|os.O_TRUNC|os.O_WRONLY, info.Mode())
    81  		if err != nil {
    82  			return totalBytes, utils.MakeError("error opening file: %v", err)
    83  		}
    84  
    85  		numBytes, err := io.Copy(file, tarReader)
    86  		if err != nil {
    87  			file.Close()
    88  			return totalBytes, utils.MakeError("error copying data to file: %v", err)
    89  		}
    90  
    91  		totalBytes += numBytes
    92  
    93  		// Manually close file instead of defer otherwise files are only
    94  		// closed when ALL files are done unpacking
    95  		if err := file.Close(); err != nil {
    96  			logger.Warningf("Failed to close file %s: %v", path, err)
    97  		}
    98  	}
    99  
   100  	return totalBytes, nil
   101  }

   103  // CompressTarLz4 takes a directory and compresses it into a tar.lz4 file in memory.
   104  func CompressTarLz4(dir string) ([]byte, error) {
   105  	// Check that the directory exists and is a directory
   106  	info, err := os.Stat(dir)
   107  	if err != nil {
   108  		if os.IsNotExist(err) {
   109  			return nil, utils.MakeError("directory %s does not exist", dir)
   110  		}
   111  		return nil, utils.MakeError("failed to stat directory %s: %v", dir, err)
   112  	}
   113  	if !info.IsDir() {
   114  		return nil, utils.MakeError("%s is not a directory", dir)
   115  	}
   116  
   117  	// Create a buffer to write the tar.lz4 file to
   118  	buf := new(bytes.Buffer)
   119  
   120  	// Compressing the tar.lz4 file involves:
   121  	// 1. Create a tar of the directory
   122  	// 2. Compress the tar into a lz4 file
   123  	lz4Writer := lz4.NewWriter(buf)
   124  	tarWriter := tar.NewWriter(lz4Writer)
   125  
   126  	// Walk the directory and add all files to the tar
   127  	err = filepath.Walk(dir, func(path string, info os.FileInfo, err error) error {
   128  		if err != nil {
   129  			return utils.MakeError("error walking directory: %v", err)
   130  		}
   131  
   132  		// Create header for the file
   133  		header, err := tar.FileInfoHeader(info, info.Name())
   134  		if err != nil {
   135  			return utils.MakeError("error creating tar header for file %s: %v", path, err)
   136  		}
   137  
   138  		// Convert the path to a relative path from the base directory.
   139  		// This is required because we need the header name to be the
   140  		// full relative path from the base directory, info.Name()
   141  		// only gives the base name of the file.
   142  		// See: https://pkg.go.dev/archive/tar#FileInfoHeader
   143  		relPath, err := filepath.Rel(dir, path)
   144  		if err != nil {
   145  			return utils.MakeError("error converting path %s to relative path: %v", path, err)
   146  		}
   147  		header.Name = relPath
   148  
   149  		// Write the header to the tar
   150  		if err := tarWriter.WriteHeader(header); err != nil {
   151  			return utils.MakeError("error writing tar header for file %s: %v", path, err)
   152  		}
   153  
   154  		// Skip directories and symlinks since there is no data to tar
   155  		// source: https://medium.com/@komuw/just-like-you-did-fbdd7df829d3
   156  		if !info.Mode().IsRegular() {
   157  			return nil
   158  		}
   159  
   160  		// Open the file and copy its contents to the tar
   161  		file, err := os.Open(path)
   162  		if err != nil {
   163  			return utils.MakeError("error opening file %s: %v", path, err)
   164  		}
   165  
   166  		if _, err := io.Copy(tarWriter, file); err != nil {
   167  			file.Close()
   168  			return utils.MakeError("error copying data from file %s: %v", path, err)
   169  		}
   170  
   171  		// Manually close file instead of defer otherwise files are only
   172  		// closed when ALL files are done packing
   173  		if err := file.Close(); err != nil {
   174  			logger.Warningf("Failed to close file %s: %v", path, err)
   175  		}
   176  
   177  		return nil
   178  	})
   179  	if err != nil {
   180  		return nil, utils.MakeError("error tarring directory: %v", err)
   181  	}
   182  
   183  	// Close writers manually instead of defer to ensure that all
   184  	// contents are written to buffer before returning.
   185  	tarWriter.Close()
   186  	lz4Writer.Close()
   187  
   188  	return buf.Bytes(), nil
   189  }

   235  // ValidateDirectoryContents checks if all directories and files in the
   236  // old directory are present in the new directory and have the same contents.
   237  func ValidateDirectoryContents(oldDir, newDir string) error {
   238  	return filepath.Walk(oldDir, func(filePath string, info os.FileInfo, err error) error {
   239  		if err != nil {
   240  			return err
   241  		}
   242  
   243  		relativePath, err := filepath.Rel(oldDir, filePath)
   244  		if err != nil {
   245  			return utils.MakeError("error getting relative path for %s: %v", filePath, err)
   246  		}
   247  
   248  		newFilePath := filepath.Join(newDir, relativePath)
   249  		matchingFile, err := os.Open(newFilePath)
   250  		if err != nil {
   251  			return utils.MakeError("error opening matching file %s: %v", newFilePath, err)
   252  		}
   253  		defer matchingFile.Close()
   254  
   255  		matchingFileInfo, err := matchingFile.Stat()
   256  		if err != nil {
   257  			return utils.MakeError("error reading stat for file: %s: %v", newFilePath, err)
   258  		}
   259  
   260  		// If one is a directory, both should be
   261  		if info.IsDir() {
   262  			if !matchingFileInfo.IsDir() {
   263  				return utils.MakeError("expected %s to be a directory", newFilePath)
   264  			}
   265  		} else {
   266  			testFileContents, err := ioutil.ReadFile(filePath)
   267  			if err != nil {
   268  				return utils.MakeError("error reading test file %s: %v", filePath, err)
   269  			}
   270  
   271  			matchingFileBuf := bytes.NewBuffer(nil)
   272  			_, err = matchingFileBuf.ReadFrom(matchingFile)
   273  			if err != nil {
   274  				return utils.MakeError("error reading matching file %s: %v", newFilePath, err)
   275  			}
   276  
   277  			// Check contents match
   278  			if string(testFileContents) != matchingFileBuf.String() {
   279  				return utils.MakeError("file contents don't match for file %s: '%s' vs '%s'", filePath, testFileContents, matchingFileBuf.Bytes())
   280  			}
   281  		}
   282  
   283  		return nil
   284  	})
