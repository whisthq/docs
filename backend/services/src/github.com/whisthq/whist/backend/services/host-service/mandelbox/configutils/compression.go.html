    20  // ExtractTarLz4 extracts a tar.lz4 file in memory to the specified directory.
    21  // Returns the total number of bytes extracted or an error.
    22  func ExtractTarLz4(file []byte, dir string) (int64, error) {
    23  	// Create the destination directory if it doesn't exist.
    24  	_, err := os.Stat(dir)
    25  	if err != nil {
    26  		if os.IsNotExist(err) {
    27  			if err := os.MkdirAll(dir, 0755); err != nil {
    28  				return 0, utils.MakeError("failed to create directory %s: %s", dir, err)
    29  			}
    30  		} else {
    31  			return 0, utils.MakeError("failed to stat directory %s: %s", dir, err)
    32  		}
    33  	}
    34  
    35  	// Track the total number of bytes extracted
    36  	var totalBytes int64
    37  
    38  	// Extracting the tar.lz4 file involves:
    39  	// 1. Decompress the the lz4 file into a tar
    40  	// 2. Untar the config to the target directory
    41  	dataReader := bytes.NewReader(file)
    42  	lz4Reader := lz4.NewReader(dataReader)
    43  	tarReader := tar.NewReader(lz4Reader)
    44  
    45  	for {
    46  		// Read the header for the next file in the tar
    47  		header, err := tarReader.Next()
    48  
    49  		// If we reach EOF, that means we are done untaring
    50  		if err != nil {
    51  			if err == io.EOF {
    52  				break
    53  			}
    54  			return totalBytes, utils.MakeError("error reading tar header: %s", err)
    55  		}
    56  
    57  		// This really should not happen
    58  		if header == nil {
    59  			break
    60  		}
    61  
    62  		path := filepath.Join(dir, header.Name)
    63  
    64  		// Check that the path is within the destination directory
    65  		// to avoid the zip slip vulnerability
    66  		// See: https://snyk.io/research/zip-slip-vulnerability
    67  		if !strings.HasPrefix(path, dir) {
    68  			return totalBytes, utils.MakeError("destination path %s is illegal", path)
    69  		}
    70  
    71  		info := header.FileInfo()
    72  
    73  		// Create directory if it doesn't exist, otherwise go next
    74  		if info.IsDir() {
    75  			if err = os.MkdirAll(path, info.Mode()); err != nil {
    76  				return totalBytes, utils.MakeError("error creating directory: %s", err)
    77  			}
    78  			continue
    79  		}
    80  
    81  		// Create file and copy contents
    82  		file, err := os.OpenFile(path, os.O_CREATE|os.O_TRUNC|os.O_WRONLY, info.Mode())
    83  		if err != nil {
    84  			return totalBytes, utils.MakeError("error opening file: %s", err)
    85  		}
    86  
    87  		numBytes, err := io.Copy(file, tarReader)
    88  		if err != nil {
    89  			file.Close()
    90  			return totalBytes, utils.MakeError("error copying data to file: %s", err)
    91  		}
    92  
    93  		totalBytes += numBytes
    94  
    95  		// Manually close file instead of defer otherwise files are only
    96  		// closed when ALL files are done unpacking
    97  		if err := file.Close(); err != nil {
    98  			logger.Warningf("Failed to close file %s: %s", path, err)
    99  		}
   100  	}
   101  
   102  	return totalBytes, nil
   103  }

   105  // CompressTarLz4 takes a directory and compresses it into a tar.lz4 file in memory.
   106  func CompressTarLz4(dir string) ([]byte, error) {
   107  	// Check that the directory exists and is a directory
   108  	info, err := os.Stat(dir)
   109  	if err != nil {
   110  		if os.IsNotExist(err) {
   111  			return nil, utils.MakeError("directory %s does not exist", dir)
   112  		}
   113  		return nil, utils.MakeError("failed to stat directory %s: %s", dir, err)
   114  	}
   115  	if !info.IsDir() {
   116  		return nil, utils.MakeError("%s is not a directory", dir)
   117  	}
   118  
   119  	// Create a buffer to write the tar.lz4 file to
   120  	buf := new(bytes.Buffer)
   121  
   122  	// Compressing the tar.lz4 file involves:
   123  	// 1. Create a tar of the directory
   124  	// 2. Compress the tar into a lz4 file
   125  	lz4Writer := lz4.NewWriter(buf)
   126  	tarWriter := tar.NewWriter(lz4Writer)
   127  
   128  	// Walk the directory and add all files to the tar
   129  	err = filepath.Walk(dir, func(path string, info os.FileInfo, err error) error {
   130  		if err != nil {
   131  			return utils.MakeError("error walking directory: %s", err)
   132  		}
   133  
   134  		// Create header for the file
   135  		header, err := tar.FileInfoHeader(info, info.Name())
   136  		if err != nil {
   137  			return utils.MakeError("error creating tar header for file %s: %s", path, err)
   138  		}
   139  
   140  		// Convert the path to a relative path from the base directory.
   141  		// This is required because we need the header name to be the
   142  		// full relative path from the base directory, info.Name()
   143  		// only gives the base name of the file.
   144  		// See: https://pkg.go.dev/archive/tar#FileInfoHeader
   145  		relPath, err := filepath.Rel(dir, path)
   146  		if err != nil {
   147  			return utils.MakeError("error converting path %s to relative path: %s", path, err)
   148  		}
   149  		header.Name = relPath
   150  
   151  		// Write the header to the tar
   152  		if err := tarWriter.WriteHeader(header); err != nil {
   153  			return utils.MakeError("error writing tar header for file %s: %s", path, err)
   154  		}
   155  
   156  		// Skip directories and symlinks since there is no data to tar
   157  		// source: https://medium.com/@komuw/just-like-you-did-fbdd7df829d3
   158  		if !info.Mode().IsRegular() {
   159  			return nil
   160  		}
   161  
   162  		// Open the file and copy its contents to the tar
   163  		file, err := os.Open(path)
   164  		if err != nil {
   165  			return utils.MakeError("error opening file %s: %s", path, err)
   166  		}
   167  
   168  		if _, err := io.Copy(tarWriter, file); err != nil {
   169  			file.Close()
   170  			return utils.MakeError("error copying data from file %s: %s", path, err)
   171  		}
   172  
   173  		// Manually close file instead of defer otherwise files are only
   174  		// closed when ALL files are done packing
   175  		if err := file.Close(); err != nil {
   176  			logger.Warningf("Failed to close file %s: %s", path, err)
   177  		}
   178  
   179  		return nil
   180  	})
   181  	if err != nil {
   182  		return nil, utils.MakeError("error tarring directory: %s", err)
   183  	}
   184  
   185  	// Close writers manually instead of defer to ensure that all
   186  	// contents are written to buffer before returning.
   187  	tarWriter.Close()
   188  	lz4Writer.Close()
   189  
   190  	return buf.Bytes(), nil
   191  }

   238  // old directory are present in the new directory and have the same contents.
   239  func ValidateDirectoryContents(oldDir, newDir string) error {
   240  	return filepath.Walk(oldDir, func(filePath string, info os.FileInfo, err error) error {
   241  		if err != nil {
   242  			return err
   243  		}
   244  
   245  		relativePath, err := filepath.Rel(oldDir, filePath)
   246  		if err != nil {
   247  			return utils.MakeError("error getting relative path for %s: %s", filePath, err)
   248  		}
   249  
   250  		newFilePath := filepath.Join(newDir, relativePath)
   251  		matchingFile, err := os.Open(newFilePath)
   252  		if err != nil {
   253  			return utils.MakeError("error opening matching file %s: %s", newFilePath, err)
   254  		}
   255  		defer matchingFile.Close()
   256  
   257  		matchingFileInfo, err := matchingFile.Stat()
   258  		if err != nil {
   259  			return utils.MakeError("error reading stat for file: %s: %s", newFilePath, err)
   260  		}
   261  
   262  		// If one is a directory, both should be
   263  		if info.IsDir() {
   264  			if !matchingFileInfo.IsDir() {
   265  				return utils.MakeError("expected %s to be a directory", newFilePath)
   266  			}
   267  		} else {
   268  			testFileContents, err := ioutil.ReadFile(filePath)
   269  			if err != nil {
   270  				return utils.MakeError("error reading test file %s: %s", filePath, err)
   271  			}
   272  
   273  			matchingFileBuf := bytes.NewBuffer(nil)
   274  			_, err = matchingFileBuf.ReadFrom(matchingFile)
   275  			if err != nil {
   276  				return utils.MakeError("error reading matching file %s: %s", newFilePath, err)
   277  			}
   278  
   279  			// Check contents match
   280  			if string(testFileContents) != matchingFileBuf.String() {
   281  				return utils.MakeError("file contents don't match for file %s: '%s' vs '%s'", filePath, testFileContents, matchingFileBuf.Bytes())
   282  			}
   283  		}
   284  
   285  		return nil
   286  	})

   290  func inflateGzip(w io.Writer, data []byte) error {
   291  	gr, err := gzip.NewReader(bytes.NewBuffer(data))
   292  	if err != nil {
   293  		return utils.MakeError("couldn't create gzip reader: %s", err)
   294  	}
   295  	defer gr.Close()
   296  
   297  	data, err = ioutil.ReadAll(gr)
   298  	if err != nil {
   299  		return utils.MakeError("couldn't extract inflated string from gzip reader: %s", err)
   300  	}
   301  
   302  	w.Write(data)
   303  
   304  	return nil
   305  }

   325  func GzipInflateString(compressedGzipString string) (string, error) {

   331  		var gzipDecodedDataBuffer bytes.Buffer

   335  		}
   336  		return gzipDecodedDataBuffer.String(), nil
