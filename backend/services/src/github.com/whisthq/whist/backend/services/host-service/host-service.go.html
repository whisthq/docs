     1  /*
     2  The Whist Host-Service is responsible for orchestrating mandelboxes (i.e.
     3  Whist-enabled containers) on EC2 instances (referred to as "hosts" throughout
     4  the codebase). The host-service is responsible for making Docker calls to start
     5  and stop mandelboxes, for enabling multiple mandelboxes to run concurrently on
     6  the same host (by dynamically allocating and assigning resources), and for
     7  passing startup data to the mandelboxes, both from the rest of the backend and
     8  from the user's frontend application.

    11  (i.e. for frontend development), check out the file `httpserver.go`.

    14  comments to explain the design decisions of the host service. It also contains
    15  an HTTPS server that exposes the necessary endpoints and sets up the necessary
    16  infrastructure for concurrent handlers, etc.

   158  		metrics.Increment("ErrorRate")
   159  	}

   161  	mandelbox.Close()

   194  func initializeFilesystem(globalCancel context.CancelFunc) {
   195  	// Check if the instance has ephemeral storage. If it does, set the WhistDir path to the
   196  	// ephemeral volume, which should already be mounted by the userdata script. We use the
   197  	// command below to check if the instance has an ephemeral device present.
   198  	// See: https://stackoverflow.com/questions/10781516/how-to-pipe-several-commands-in-go
   199  	ephemeralDeviceCmd := "nvme list -o json | jq -r '.Devices | map(select(.ModelNumber == \"Amazon EC2 NVMe Instance Storage\")) | max_by(.PhysicalSize) | .DevicePath'"
   200  	out, err := exec.Command("bash", "-c", ephemeralDeviceCmd).CombinedOutput()
   201  	if err != nil {
   202  		logger.Errorf("Error while getting ephemeral device path, not using ephemeral storage.")
   203  	}
   204  
   205  	ephemeralDevicePath := string(out)
   206  
   207  	// We check if the command exited successfully, and if the ephemeral device exists.
   208  	// The string will contain "bash" if something went wrong. Also take into account
   209  	// if the WhistDir already contains the ephemeral path.
   210  	if !metadata.IsLocalEnv() &&
   211  		ephemeralDevicePath != "" && ephemeralDevicePath != "null" &&
   212  		!strings.Contains(ephemeralDevicePath, "bash") && !strings.Contains(utils.WhistDir, utils.WhistEphemeralFSPath) {
   213  		logger.Infof("Creating Whist directory on ephemeral device.")
   214  		utils.WhistDir = path.Join(utils.WhistEphemeralFSPath, utils.WhistDir)
   215  		utils.TempDir = path.Join(utils.WhistDir, "temp/")
   216  	}
   217  
   218  	// check if "/whist" already exists --- if so, panic, since
   219  	// we don't know why it's there or if it's valid. The host-service shutting down
   220  	// from this panic will clean up the directory and the next run will work properly.
   221  	if _, err := os.Lstat(utils.WhistDir); !os.IsNotExist(err) {
   222  		if err == nil {
   223  			logger.Panicf(globalCancel, "Directory %s already exists!", utils.WhistDir)
   224  		} else {
   225  			logger.Panicf(globalCancel, "Could not make directory %s because of error %v", utils.WhistDir, err)
   226  		}
   227  	}
   228  
   229  	// Create the whist directory and make it non-root user owned so that
   230  	// non-root users in mandelboxes can access files within (especially user
   231  	// configs).
   232  	err = os.MkdirAll(utils.WhistDir, 0777)
   233  	if err != nil {
   234  		logger.Panicf(globalCancel, "Failed to create directory %s: error: %s\n", utils.WhistDir, err)
   235  	}
   236  	cmd := exec.Command("chown", "-R", "ubuntu", utils.WhistDir)
   237  	cmd.Run()
   238  
   239  	// Create whist-private directory
   240  	err = os.MkdirAll(utils.WhistPrivateDir, 0777)
   241  	if err != nil {
   242  		logger.Panicf(globalCancel, "Failed to create directory %s: error: %s\n", utils.WhistPrivateDir, err)
   243  	}
   244  
   245  	// Create whist temp directory (only let root read and write this, since it
   246  	// contains logs and uinput sockets).
   247  	err = os.MkdirAll(utils.TempDir, 0600)
   248  	if err != nil {
   249  		logger.Panicf(globalCancel, "Could not mkdir path %s. Error: %s", utils.TempDir, err)
   250  	}
   251  }

   283  func main() {
   284  	// The first thing we want to do is to initialize Logz.io and Sentry so that
   285  	// we can catch any errors that might occur, or logs if we print them.
   286  	logger.InitHostLogging()
   287  
   288  	// We create a global context (i.e. for the entire host service) that can be
   289  	// cancelled if the entire program needs to terminate. We also create a
   290  	// WaitGroup for all goroutines to tell us when they've stopped (if the
   291  	// context gets cancelled). Finally, we defer a function which cancels the
   292  	// global context if necessary, logs any panic we might be recovering from,
   293  	// and cleans up after the entire host service (although some aspects of
   294  	// cleanup may not seem necessary on a production instance, they are
   295  	// immensely helpful to prevent the host service from clogging up the local
   296  	// filesystem during development). After the permissions check, the creation
   297  	// of this context and WaitGroup, and the following defer must be the first
   298  	// statements in main().
   299  	globalCtx, globalCancel := context.WithCancel(context.Background())
   300  	goroutineTracker := sync.WaitGroup{}
   301  	defer func() {
   302  		// This function cleanly shuts down the Whist Host-Service. Note that
   303  		// besides the host machine itself being forcefully shut down, this
   304  		// deferred function from main() should be the _only_ way that the host
   305  		// service exits. In particular, it should be as a result of a panic() in
   306  		// main, the global context being cancelled, or a Ctrl+C interrupt.
   307  
   308  		// Note that this function, while nontrivial, has intentionally been left
   309  		// as part of main() for the following reasons:
   310  		//   1. To prevent it being called from anywhere else accidentally
   311  		//   2. To keep the entire program control flow clearly in main().
   312  
   313  		// Catch any panics that might have originated in main() or one of its
   314  		// direct children.
   315  		r := recover()
   316  		if r != nil {
   317  			logger.Infof("Shutting down host service after caught panic in main(): %v", r)
   318  		} else {
   319  			logger.Infof("Beginning host service shutdown procedure...")
   320  		}
   321  
   322  		// Cancel the global context, if it hasn't already been cancelled.
   323  		globalCancel()
   324  
   325  		// Clean all the waiting mandelboxes so they don't block the shut down.
   326  		mandelboxData.StopWaitingMandelboxes()
   327  
   328  		// Wait for all goroutines to stop, so we can run the rest of the cleanup
   329  		// process.
   330  		utils.WaitWithDebugPrints(&goroutineTracker, 2*time.Minute, 2)
   331  
   332  		// Stop processing new events
   333  		close(eventLoopKeepalive)
   334  
   335  		uninitializeFilesystem()
   336  
   337  		// Remove our row from the database and close out the database driver.
   338  		dbdriver.Close()
   339  
   340  		// Close out our metrics collection.
   341  		metrics.Close()
   342  
   343  		// Drain to our remote logging providers, but don't yet stop recording new
   344  		// events, in case the shutdown fails.
   345  		logger.FlushLogzio()
   346  		logger.FlushSentry()
   347  
   348  		logger.Info("Finished host service shutdown procedure. Finally exiting...")
   349  		if shutdownInstanceOnExit {
   350  			if err := exec.Command("shutdown", "now").Run(); err != nil {
   351  				if strings.TrimSpace(err.Error()) == "signal: terminated" {
   352  					// The instance seems to shut down even when this error is fired, so
   353  					// we'll just ignore it. We still `logger.Info()` it just in case.
   354  					logger.Infof("Shutdown command returned 'signal: terminated' error. Ignoring it.")
   355  				} else {
   356  					logger.Errorf("Couldn't shut down instance: %s", err)
   357  					metrics.Increment("ErrorRate")
   358  				}
   359  			}
   360  		}
   361  
   362  		// Shut down the logging infrastructure (including re-draining the queues).
   363  		logger.Close()
   364  
   365  		os.Exit(0)
   366  	}()
   367  
   368  	// Log the Git commit of the running executable
   369  	logger.Info("Host Service Version: %s", metadata.GetGitCommit())
   370  
   371  	// Initialize the database driver, if necessary (the `dbdriver` package
   372  	// takes care of the "if necessary" part).
   373  	if err := dbdriver.Initialize(globalCtx, globalCancel, &goroutineTracker); err != nil {
   374  		logger.Panic(globalCancel, err)
   375  	}
   376  
   377  	// Log the instance name we're running on
   378  	instanceName, err := aws.GetInstanceName()
   379  	if err != nil {
   380  		logger.Panic(globalCancel, err)
   381  	}
   382  	logger.Infof("Running on instance name: %s", instanceName)
   383  
   384  	initializeAppArmor(globalCancel)
   385  
   386  	initializeFilesystem(globalCancel)
   387  
   388  	// Start Docker
   389  	dockerClient, err := createDockerClient()
   390  	if err != nil {
   391  		logger.Panic(globalCancel, err)
   392  	}
   393  
   394  	if err := dbdriver.RegisterInstance(); err != nil {
   395  		// If the instance starts up and sees its status as unresponsive or
   396  		// draining, the backend doesn't want it anymore so we should shut down.
   397  
   398  		// TODO: make this a bit more robust
   399  		if !metadata.IsLocalEnv() && strings.Contains(err.Error(), string(dbdriver.InstanceStatusDraining)) {
   400  			logger.Infof("Instance wasn't registered in database because we found ourselves already marked draining. Shutting down.... Error: %s", err)
   401  			shutdownInstanceOnExit = true
   402  			globalCancel()
   403  		} else {
   404  			logger.Panic(globalCancel, err)
   405  		}
   406  	}
   407  
   408  	// Start database subscription client
   409  	instanceID, err := aws.GetInstanceID()
   410  	if err != nil {
   411  		logger.Errorf("Can't get AWS Instance Name. Error: %s", err)
   412  		metrics.Increment("ErrorRate")
   413  	}
   414  
   415  	capacity, err := dbdriver.GetInstanceCapacity(string(instanceID))
   416  	if err != nil {
   417  		logger.Errorf("Failed to get capacity of instance %v. Err: %s", instanceID, err)
   418  	}
   419  
   420  	// Now we start all the goroutines that actually do work.
   421  
   422  	// Start the HTTP server and listen for events
   423  	httpServerEvents, err := StartHTTPServer(globalCtx, globalCancel, &goroutineTracker)
   424  	if err != nil {
   425  		logger.Panic(globalCancel, err)
   426  	}
   427  
   428  	subscriptionEvents := make(chan subscriptions.SubscriptionEvent, 100)
   429  	// It's not necessary to subscribe to the config database
   430  	// in the host service
   431  	useConfigDB := false
   432  
   433  	subscriptionClient := &subscriptions.SubscriptionClient{}
   434  	subscriptions.SetupHostSubscriptions(string(instanceID), subscriptionClient)
   435  	subscriptions.Start(subscriptionClient, globalCtx, &goroutineTracker, subscriptionEvents, useConfigDB)
   436  	if err != nil {
   437  		logger.Errorf("Failed to start database subscriptions. Error: %s", err)
   438  	}
   439  
   440  	// Start warming up as many instances as we have capacity for. This will effectively create
   441  	// mandelboxes up to the point where we need a config token, and register them to the database.
   442  	// The scaling service will handling assigning users to this instance and will update the
   443  	// database row to assign the user to a waiting mandelbox.
   444  	SpinUpMandelboxes(globalCtx, globalCancel, &goroutineTracker, dockerClient, string(instanceID), capacity)
   445  
   446  	// Start main event loop. Note that we don't track this goroutine, but
   447  	// instead control its lifetime with `eventLoopKeepAlive`. This is because it
   448  	// needs to stay alive after the global context is cancelled, so we can
   449  	// process mandelbox death events.
   450  	go eventLoopGoroutine(globalCtx, globalCancel, &goroutineTracker, dockerClient, httpServerEvents, subscriptionEvents)
   451  
   452  	// Register a signal handler for Ctrl-C so that we cleanup if Ctrl-C is pressed.
   453  	sigChan := make(chan os.Signal, 2)
   454  	signal.Notify(sigChan, os.Interrupt, syscall.SIGTERM, syscall.SIGINT)
   455  
   456  	// Wait for either the global context to get cancelled by a worker goroutine,
   457  	// or for us to receive an interrupt. This needs to be the end of main().
   458  	select {
   459  	case <-sigChan:
   460  		logger.Infof("Got an interrupt or SIGTERM")
   461  	case <-globalCtx.Done():
   462  		logger.Infof("Global context cancelled!")
   463  	}
   464  }

   466  // As long as this channel is blocking, we continue processing events
   467  // (including Docker events).
   468  var eventLoopKeepalive = make(chan interface{}, 1)

   470  func eventLoopGoroutine(globalCtx context.Context, globalCancel context.CancelFunc,
   471  	goroutineTracker *sync.WaitGroup, dockerClient dockerclient.CommonAPIClient,
   472  	httpServerEvents <-chan httputils.ServerRequest, subscriptionEvents <-chan subscriptions.SubscriptionEvent) {
   473  	// Note that we don't use globalCtx for the Docker Context, since we still
   474  	// wish to process Docker events after the global context is cancelled.
   475  	dockerContext, dockerContextCancel := context.WithCancel(context.Background())
   476  	defer dockerContextCancel()
   477  
   478  	// Create filter for which docker events we care about
   479  	filters := dockerfilters.NewArgs()
   480  	filters.Add("type", dockerevents.ContainerEventType)
   481  	eventOptions := dockertypes.EventsOptions{
   482  		Filters: filters,
   483  	}
   484  
   485  	// We use this lock to protect the transportRequestMap
   486  	transportMapLock := &sync.Mutex{}
   487  
   488  	// Note: If a mandelbox suffers a bug, or fails to start correctly
   489  	// these channels will become a memory leak.
   490  	transportRequestMap := make(map[mandelboxtypes.MandelboxID]chan *JSONTransportRequest)
   491  
   492  	// In the following loop, this var determines whether to re-initialize the
   493  	// Docker event stream. This is necessary because the Docker event stream
   494  	// needs to be reopened after any error is sent over the error channel.
   495  	needToReinitDockerEventStream := false
   496  	dockerevents, dockererrs := dockerClient.Events(dockerContext, eventOptions)
   497  	logger.Info("Initialized docker event stream.")
   498  
   499  	logger.Info("Entering event loop...")
   500  
   501  	// The actual event loop
   502  	for {
   503  		if needToReinitDockerEventStream {
   504  			dockerevents, dockererrs = dockerClient.Events(dockerContext, eventOptions)
   505  			needToReinitDockerEventStream = false
   506  			logger.Info("Re-initialized docker event stream.")
   507  		}
   508  
   509  		select {
   510  		case <-eventLoopKeepalive:
   511  			logger.Infof("Leaving main event loop...")
   512  			return
   513  
   514  		case err := <-dockererrs:
   515  			needToReinitDockerEventStream = true
   516  			switch {
   517  			case err == nil:
   518  				logger.Info("We got a nil error over the Docker event stream. This might indicate an error inside the docker go library. Ignoring it and proceeding normally...")
   519  				continue
   520  			case err == io.EOF:
   521  				logger.Panicf(globalCancel, "Docker event stream has been completely read.")
   522  			case dockerclient.IsErrConnectionFailed(err):
   523  				// This means "Cannot connect to the Docker daemon..."
   524  				logger.Panicf(globalCancel, "Got error \"%v\". Could not connect to the Docker daemon.", err)
   525  			default:
   526  				if !strings.HasSuffix(strings.ToLower(err.Error()), "context canceled") {
   527  					logger.Panicf(globalCancel, "Got an unknown error from the Docker event stream: %v", err)
   528  				}
   529  				return
   530  			}
   531  
   532  		case dockerevent := <-dockerevents:
   533  			logger.Info("dockerevent: %s for %s %s\n", dockerevent.Action, dockerevent.Type, dockerevent.ID)
   534  			if dockerevent.Action == "die" {
   535  				mandelboxDieHandler(dockerevent.ID, transportRequestMap, transportMapLock, dockerClient)
   536  			}
   537  
   538  		// It may seem silly to just launch goroutines to handle these
   539  		// serverevents, but we aim to keep the high-level flow control and handling
   540  		// in this package, and the low-level authentication, parsing, etc. of
   541  		// requests in `httpserver`.
   542  		case serverevent := <-httpServerEvents:
   543  			switch serverevent := serverevent.(type) {
   544  			// TODO: actually handle panics in these goroutines
   545  			case *JSONTransportRequest:
   546  				if !metadata.IsLocalEnvWithoutDB() {
   547  					// Handle JSON transport validation on a separate goroutine
   548  					go handleJSONTransportRequest(serverevent, transportRequestMap, transportMapLock)
   549  				} else {
   550  					// If running on a local environment, disable any pubsub logic. We have to create a subscription request
   551  					// that mocks the Hasura subscription event. Doing this avoids the need of setting up a Hasura server and
   552  					// postgres database on the development instance.
   553  
   554  					// Start spinning up a mandelbox so developers can connect to it. Since
   555  					// the mandelbox ID is not known beforehand, modify the server event
   556  					// to use the ID from the newly created mandelbox.
   557  					zygote := StartMandelboxSpinUp(globalCtx, globalCancel, goroutineTracker, dockerClient)
   558  					serverevent.MandelboxID = zygote.GetID()
   559  
   560  					userID, err := metadata.GetUserID()
   561  					if err != nil {
   562  						logger.Errorf("Error getting userID, %v", err)
   563  						metrics.Increment("ErrorRate")
   564  					}
   565  
   566  					instanceID, err := aws.GetInstanceID()
   567  					if err != nil {
   568  						logger.Errorf("Error getting instance name from AWS, %v", err)
   569  						metrics.Increment("ErrorRate")
   570  					}
   571  
   572  					// Create a mandelbox object as would be received from a Hasura subscription.
   573  					mandelbox := subscriptions.Mandelbox{
   574  						InstanceID: string(instanceID),
   575  						ID:         zygote.GetID(),
   576  						SessionID:  utils.Sprintf("%v", time.Now().UnixMilli()),
   577  						UserID:     userID,
   578  					}
   579  					subscriptionEvent := subscriptions.MandelboxEvent{
   580  						Mandelboxes: []subscriptions.Mandelbox{mandelbox},
   581  					}
   582  
   583  					// Launch both the JSON transport handler and the SpinUpMandelbox goroutines.
   584  					go handleJSONTransportRequest(serverevent, transportRequestMap, transportMapLock)
   585  					go FinishMandelboxSpinUp(globalCtx, globalCancel, goroutineTracker, dockerClient, &subscriptionEvent, transportRequestMap, transportMapLock)
   586  				}
   587  			default:
   588  				if serverevent != nil {
   589  					err := utils.MakeError("unimplemented handling of server event [type: %T]: %v", serverevent, serverevent)
   590  					logger.Error(err)
   591  					serverevent.ReturnResult("", err)
   592  				}
   593  			}
   594  
   595  		case subscriptionEvent := <-subscriptionEvents:
   596  			switch subscriptionEvent := subscriptionEvent.(type) {
   597  			// TODO: actually handle panics in these goroutines
   598  			case *subscriptions.MandelboxEvent:
   599  				go FinishMandelboxSpinUp(globalCtx, globalCancel, goroutineTracker, dockerClient,
   600  					subscriptionEvent, transportRequestMap, transportMapLock)
   601  
   602  			case *subscriptions.InstanceEvent:
   603  				// Don't do this in a separate goroutine, since there's no reason to.
   604  				drainAndShutdown(globalCtx, globalCancel, goroutineTracker)
   605  
   606  			default:
   607  				if subscriptionEvent != nil {
   608  					err := utils.MakeError("Unimplemented handling of subscription event [type: %T]: %v", subscriptionEvent, subscriptionEvent)
   609  					logger.Error(err)
   610  				}
   611  			}
   612  		}
   613  	}
