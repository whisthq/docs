     1  /*
     2  The Whist Host Service is responsible for orchestrating mandelboxes (i.e.
     3  Whist-enabled containers) on EC2 instances (referred to as "hosts" throughout
     4  the codebase). The host service is responsible for making Docker calls to start
     5  and stop mandelboxes, for enabling multiple mandelboxes to run concurrently on
     6  the same host (by dynamically allocating and assigning resources), and for
     7  passing startup data to the mandelboxes, both from the rest of the backend and
     8  from the user's client application.
     9  
    10  If you are just interested in seeing what endpoints the host service exposes
    11  (i.e. for client-app development), check out the file `httpserver.go`.
    12  
    13  The main package of the host service contains the main logic and the most
    14  comments to explain the design decisions of the host service. It also contains
    15  an HTTPS server that exposes the necessary endpoints and sets up the necessary
    16  infrastructure for concurrent handlers, etc.
    17  */

    99  	imageFilters := dockerfilters.NewArgs(
   100  		dockerfilters.KeyValuePair{Key: "dangling", Value: "false"},
   101  	)
   102  	images, err := dockerClient.ImageList(globalCtx, dockertypes.ImageListOptions{All: false, Filters: imageFilters})
   103  	if err != nil {
   104  		return ""
   105  	}
   106  
   107  	for _, regex := range regexes {
   108  		rgx := regexp.MustCompile(regex)
   109  		for _, img := range images {
   110  			for _, tag := range img.RepoTags {
   111  				if rgx.MatchString(tag) {
   112  					return tag
   113  				}
   114  			}
   115  		}
   116  	}
   117  	return ""
   118  }

   516  	// shell into the container immediately, not conditional on everything

   523  		}
   524  
   525  		logger.Infof("SpinUpMandelbox(): Finished waiting for mandelbox %s whist application to start up", mandelboxSubscription.ID)
   526  	}

   528  	err = dbdriver.WriteMandelboxStatus(mandelboxSubscription.ID, dbdriver.MandelboxStatusRunning)

   530  		logAndReturnError("Error marking mandelbox running: %s", err)
   531  		return
   532  	}
   533  	logger.Infof("SpinUpMandelbox(): Successfully marked mandelbox %s as running", mandelboxSubscription.ID)

   535  	// Mark mandelbox creation as successful, preventing cleanup on function
   536  	// termination.
   537  	createFailed = false

   539  	result := JSONTransportRequestResult{
   540  		HostPortForTCP32262: hostPortForTCP32262,

   542  		HostPortForTCP32273: hostPortForTCP32273,
   543  		AesKey:              aesKey,
   544  	}
   545  	req.ReturnResult(result, nil)

   551  	// Exit if we are not dealing with a Whist mandelbox, or if it has already

   571  		metrics.Increment("ErrorRate")
   572  	}

   672  func main() {
   673  	// The first thing we want to do is to initialize logzio and Sentry so that
   674  	// we can catch any errors that might occur, or logs if we print them.
   675  	logger.InitHostLogging()
   676  
   677  	// We create a global context (i.e. for the entire host service) that can be
   678  	// cancelled if the entire program needs to terminate. We also create a
   679  	// WaitGroup for all goroutines to tell us when they've stopped (if the
   680  	// context gets cancelled). Finally, we defer a function which cancels the
   681  	// global context if necessary, logs any panic we might be recovering from,
   682  	// and cleans up after the entire host service (although some aspects of
   683  	// cleanup may not seem necessary on a production instance, they are
   684  	// immensely helpful to prevent the host service from clogging up the local
   685  	// filesystem during development). After the permissions check, the creation
   686  	// of this context and WaitGroup, and the following defer must be the first
   687  	// statements in main().
   688  	globalCtx, globalCancel := context.WithCancel(context.Background())
   689  	goroutineTracker := sync.WaitGroup{}
   690  	defer func() {
   691  		// This function cleanly shuts down the Whist Host Service. Note that
   692  		// besides the host machine itself being forcefully shut down, this
   693  		// deferred function from main() should be the _only_ way that the host
   694  		// service exits. In particular, it should be as a result of a panic() in
   695  		// main, the global context being cancelled, or a Ctrl+C interrupt.
   696  
   697  		// Note that this function, while nontrivial, has intentionally been left
   698  		// as part of main() for the following reasons:
   699  		//   1. To prevent it being called from anywhere else accidentally
   700  		//   2. To keep the entire program control flow clearly in main().
   701  
   702  		// Catch any panics that might have originated in main() or one of its
   703  		// direct children.
   704  		r := recover()
   705  		if r != nil {
   706  			logger.Infof("Shutting down host service after caught panic in main(): %v", r)
   707  		} else {
   708  			logger.Infof("Beginning host service shutdown procedure...")
   709  		}
   710  
   711  		// Cancel the global context, if it hasn't already been cancelled.
   712  		globalCancel()
   713  
   714  		// Wait for all goroutines to stop, so we can run the rest of the cleanup
   715  		// process.
   716  		utils.WaitWithDebugPrints(&goroutineTracker, 2*time.Minute, 2)
   717  
   718  		// Stop processing new events
   719  		close(eventLoopKeepalive)
   720  
   721  		uninitializeFilesystem()
   722  
   723  		// Remove our row from the database and close out the database driver.
   724  		dbdriver.Close()
   725  
   726  		// Close out our metrics collection.
   727  		metrics.Close()
   728  
   729  		// Drain to our remote logging providers, but don't yet stop recording new
   730  		// events, in case the shutdown fails.
   731  		logger.FlushLogzio()
   732  		logger.FlushSentry()
   733  
   734  		logger.Info("Finished host service shutdown procedure. Finally exiting...")
   735  		if shutdownInstanceOnExit {
   736  			if err := exec.Command("shutdown", "now").Run(); err != nil {
   737  				if strings.TrimSpace(err.Error()) == "signal: terminated" {
   738  					// The instance seems to shut down even when this error is fired, so
   739  					// we'll just ignore it. We still `logger.Info()` it just in case.
   740  					logger.Infof("Shutdown command returned 'signal: terminated' error. Ignoring it.")
   741  				} else {
   742  					logger.Errorf("Couldn't shut down instance: %s", err)
   743  					metrics.Increment("ErrorRate")
   744  				}
   745  			}
   746  		}
   747  
   748  		// Shut down the logging infrastructure (including re-draining the queues).
   749  		logger.Close()
   750  
   751  		os.Exit(0)
   752  	}()
   753  
   754  	// Log the Git commit of the running executable
   755  	logger.Info("Host Service Version: %s", metadata.GetGitCommit())
   756  
   757  	// Initialize the database driver, if necessary (the `dbdriver` package
   758  	// takes care of the "if necessary" part).
   759  	if err := dbdriver.Initialize(globalCtx, globalCancel, &goroutineTracker); err != nil {
   760  		logger.Panic(globalCancel, err)
   761  	}
   762  
   763  	// Log the instance name we're running on
   764  	instanceName, err := aws.GetInstanceName()
   765  	if err != nil {
   766  		logger.Panic(globalCancel, err)
   767  	}
   768  	logger.Infof("Running on instance name: %s", instanceName)
   769  
   770  	initializeAppArmor(globalCancel)
   771  
   772  	initializeFilesystem(globalCancel)
   773  
   774  	// Start Docker
   775  	dockerClient, err := createDockerClient()
   776  	if err != nil {
   777  		logger.Panic(globalCancel, err)
   778  	}
   779  
   780  	if err := dbdriver.RegisterInstance(); err != nil {
   781  		// If the instance starts up and sees its status as unresponsive or
   782  		// draining, the webserver doesn't want it anymore so we should shut down.
   783  
   784  		// TODO: make this a bit more robust
   785  		if !metadata.IsLocalEnv() && strings.Contains(err.Error(), string(dbdriver.InstanceStatusDraining)) {
   786  			logger.Infof("Instance wasn't registered in database because we found ourselves already marked draining. Shutting down.... Error: %s", err)
   787  			shutdownInstanceOnExit = true
   788  			globalCancel()
   789  		} else {
   790  			logger.Panic(globalCancel, err)
   791  		}
   792  	}
   793  
   794  	// Now we start all the goroutines that actually do work.
   795  
   796  	// Start the HTTP server and listen for events
   797  	httpServerEvents, err := StartHTTPServer(globalCtx, globalCancel, &goroutineTracker)
   798  	if err != nil {
   799  		logger.Panic(globalCancel, err)
   800  	}
   801  
   802  	// Start database subscription client
   803  	instanceID, err := aws.GetInstanceID()
   804  	if err != nil {
   805  		logger.Errorf("Can't get AWS Instance Name to start database subscriptions. Error: %s", err)
   806  		metrics.Increment("ErrorRate")
   807  	}
   808  	subscriptionEvents := make(chan subscriptions.SubscriptionEvent, 100)
   809  
   810  	subscriptionClient := &subscriptions.SubscriptionClient{}
   811  	subscriptions.SetupHostSubscriptions(string(instanceID), subscriptionClient)
   812  	subscriptions.Start(subscriptionClient, globalCtx, &goroutineTracker, subscriptionEvents)
   813  	if err != nil {
   814  		logger.Errorf("Failed to start database subscriptions. Error: %s", err)
   815  	}
   816  
   817  	// Start main event loop. Note that we don't track this goroutine, but
   818  	// instead control its lifetime with `eventLoopKeepAlive`. This is because it
   819  	// needs to stay alive after the global context is cancelled, so we can
   820  	// process mandelbox death events.
   821  	go eventLoopGoroutine(globalCtx, globalCancel, &goroutineTracker, dockerClient, httpServerEvents, subscriptionEvents)
   822  
   823  	// Register a signal handler for Ctrl-C so that we cleanup if Ctrl-C is pressed.
   824  	sigChan := make(chan os.Signal, 2)
   825  	signal.Notify(sigChan, os.Interrupt, syscall.SIGTERM, syscall.SIGINT)
   826  
   827  	// Wait for either the global context to get cancelled by a worker goroutine,
   828  	// or for us to receive an interrupt. This needs to be the end of main().
   829  	select {
   830  	case <-sigChan:
   831  		logger.Infof("Got an interrupt or SIGTERM")
   832  	case <-globalCtx.Done():
   833  		logger.Infof("Global context cancelled!")
   834  	}
   835  }

   838  // (including Docker events).
   839  var eventLoopKeepalive = make(chan interface{}, 1)

   841  func eventLoopGoroutine(globalCtx context.Context, globalCancel context.CancelFunc,
   842  	goroutineTracker *sync.WaitGroup, dockerClient dockerclient.CommonAPIClient,
   843  	httpServerEvents <-chan ServerRequest, subscriptionEvents <-chan subscriptions.SubscriptionEvent) {
   844  	// Note that we don't use globalCtx for the docker Context, since we still
   845  	// wish to process Docker events after the global context is cancelled.
   846  	dockerContext, dockerContextCancel := context.WithCancel(context.Background())
   847  	defer dockerContextCancel()
   848  
   849  	// Create filter for which docker events we care about
   850  	filters := dockerfilters.NewArgs()
   851  	filters.Add("type", dockerevents.ContainerEventType)
   852  	eventOptions := dockertypes.EventsOptions{
   853  		Filters: filters,
   854  	}
   855  
   856  	// We use this lock to protect the transportRequestMap
   857  	transportMapLock := &sync.Mutex{}
   858  
   859  	// Note: If a mandelbox suffers a bug, or fails to start correctly
   860  	// these channels will become a memory leak.
   861  	transportRequestMap := make(map[mandelboxtypes.MandelboxID]chan *JSONTransportRequest)
   862  
   863  	// In the following loop, this var determines whether to re-initialize the
   864  	// Docker event stream. This is necessary because the Docker event stream
   865  	// needs to be reopened after any error is sent over the error channel.
   866  	needToReinitDockerEventStream := false
   867  	dockerevents, dockererrs := dockerClient.Events(dockerContext, eventOptions)
   868  	logger.Info("Initialized docker event stream.")
   869  	logger.Info("Entering event loop...")
   870  
   871  	// The actual event loop
   872  	for {
   873  		if needToReinitDockerEventStream {
   874  			dockerevents, dockererrs = dockerClient.Events(dockerContext, eventOptions)
   875  			needToReinitDockerEventStream = false
   876  			logger.Info("Re-initialized docker event stream.")
   877  		}
   878  
   879  		select {
   880  		case <-eventLoopKeepalive:
   881  			logger.Infof("Leaving main event loop...")
   882  			return
   883  
   884  		case err := <-dockererrs:
   885  			needToReinitDockerEventStream = true
   886  			switch {
   887  			case err == nil:
   888  				logger.Info("We got a nil error over the Docker event stream. This might indicate an error inside the docker go library. Ignoring it and proceeding normally...")
   889  				continue
   890  			case err == io.EOF:
   891  				logger.Panicf(globalCancel, "Docker event stream has been completely read.")
   892  			case dockerclient.IsErrConnectionFailed(err):
   893  				// This means "Cannot connect to the Docker daemon..."
   894  				logger.Panicf(globalCancel, "Got error \"%v\". Could not connect to the Docker daemon.", err)
   895  			default:
   896  				if !strings.HasSuffix(strings.ToLower(err.Error()), "context canceled") {
   897  					logger.Panicf(globalCancel, "Got an unknown error from the Docker event stream: %v", err)
   898  				}
   899  				return
   900  			}
   901  
   902  		case dockerevent := <-dockerevents:
   903  			logger.Info("dockerevent: %s for %s %s\n", dockerevent.Action, dockerevent.Type, dockerevent.ID)
   904  			if dockerevent.Action == "die" {
   905  				mandelboxDieHandler(dockerevent.ID, transportRequestMap, transportMapLock, dockerClient)
   906  			}
   907  
   908  		// It may seem silly to just launch goroutines to handle these
   909  		// serverevents, but we aim to keep the high-level flow control and handling
   910  		// in this package, and the low-level authentication, parsing, etc. of
   911  		// requests in `httpserver`.
   912  		case serverevent := <-httpServerEvents:
   913  			switch serverevent := serverevent.(type) {
   914  			// TODO: actually handle panics in these goroutines
   915  			case *JSONTransportRequest:
   916  				if !metadata.IsLocalEnvWithoutDB() {
   917  					// Handle JSON transport validation on a separate goroutine
   918  					go handleJSONTransportRequest(serverevent, transportRequestMap, transportMapLock)
   919  				} else {
   920  					// If running on a local environment, disable any pubsub logic. We have to create a subscription request
   921  					// that mocks the Hasura subscription event. Doing this avoids the need of setting up a Hasura server and
   922  					// postgres database on the development instance.
   923  					jsonReq := serverevent
   924  
   925  					userID, err := metadata.GetUserID()
   926  					if err != nil {
   927  						logger.Errorf("Error getting userID, %v", err)
   928  						metrics.Increment("ErrorRate")
   929  					}
   930  
   931  					instanceID, err := aws.GetInstanceID()
   932  					if err != nil {
   933  						logger.Errorf("Error getting instance name from AWS, %v", err)
   934  						metrics.Increment("ErrorRate")
   935  					}
   936  					// Create a mandelbox object as would be received from a Hasura subscription.
   937  					mandelbox := subscriptions.Mandelbox{
   938  						InstanceID: string(instanceID),
   939  						ID:         jsonReq.MandelboxID,
   940  						SessionID:  "1234",
   941  						UserID:     userID,
   942  					}
   943  					subscriptionEvent := subscriptions.MandelboxEvent{
   944  						Mandelboxes: []subscriptions.Mandelbox{mandelbox},
   945  					}
   946  
   947  					// Launch both the JSON transport handler and the SpinUpMandelbox goroutines.
   948  					go handleJSONTransportRequest(serverevent, transportRequestMap, transportMapLock)
   949  					go SpinUpMandelbox(globalCtx, globalCancel, goroutineTracker, dockerClient, &subscriptionEvent, transportRequestMap, transportMapLock)
   950  				}
   951  			default:
   952  				if serverevent != nil {
   953  					err := utils.MakeError("unimplemented handling of server event [type: %T]: %v", serverevent, serverevent)
   954  					logger.Error(err)
   955  					serverevent.ReturnResult("", err)
   956  				}
   957  			}
   958  
   959  		case subscriptionEvent := <-subscriptionEvents:
   960  			switch subscriptionEvent := subscriptionEvent.(type) {
   961  			// TODO: actually handle panics in these goroutines
   962  			case *subscriptions.MandelboxEvent:
   963  				go SpinUpMandelbox(globalCtx, globalCancel, goroutineTracker, dockerClient,
   964  					subscriptionEvent, transportRequestMap, transportMapLock)
   965  
   966  			case *subscriptions.InstanceEvent:
   967  				// Don't do this in a separate goroutine, since there's no reason to.
   968  				drainAndShutdown(globalCtx, globalCancel, goroutineTracker)
   969  
   970  			default:
   971  				if subscriptionEvent != nil {
   972  					err := utils.MakeError("Unimplemented handling of subscription event [type: %T]: %v", subscriptionEvent, subscriptionEvent)
   973  					logger.Error(err)
   974  				}
   975  			}
   976  		}
   977  	}
