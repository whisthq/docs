     1  /*
     2  The Whist Host Service is responsible for orchestrating mandelboxes (i.e.
     3  Whist-enabled containers) on EC2 instances (referred to as "hosts" throughout
     4  the codebase). The host service is responsible for making Docker calls to start
     5  and stop mandelboxes, for enabling multiple mandelboxes to run concurrently on
     6  the same host (by dynamically allocating and assigning resources), and for
     7  passing startup data to the mandelboxes, both from the rest of the backend and
     8  from the user's client application.
     9  
    10  If you are just interested in seeing what endpoints the host service exposes
    11  (i.e. for client-app development), check out the file `httpserver.go`.
    12  
    13  The main package of the host service contains the main logic and the most
    14  comments to explain the design decisions of the host service. It also contains
    15  an HTTPS server that exposes the necessary endpoints and sets up the necessary
    16  infrastructure for concurrent handlers, etc.
    17  */

    96  // Given a list of regexes, find a docker image whose name matches the earliest

    98  func dockerImageFromRegexes(globalCtx context.Context, dockerClient dockerclient.CommonAPIClient, regexes []string) string {
    99  	imageFilters := dockerfilters.NewArgs(
   100  		dockerfilters.KeyValuePair{Key: "dangling", Value: "false"},
   101  	)
   102  	images, err := dockerClient.ImageList(globalCtx, dockertypes.ImageListOptions{All: false, Filters: imageFilters})
   103  	if err != nil {
   104  		return ""
   105  	}
   106  
   107  	for _, regex := range regexes {
   108  		rgx := regexp.MustCompile(regex)
   109  		for _, img := range images {
   110  			for _, tag := range img.RepoTags {
   111  				if rgx.MatchString(tag) {
   112  					return tag
   113  				}
   114  			}
   115  		}
   116  	}
   117  	return ""
   118  }

   150  		`ghcr.io/whisthq/.*/browsers/brave:current-build`,

   153  	}

   157  		return utils.MakeError("Couldn't find a suitable image")
   158  	}

   161  	// applicaiton) is a bit slow still. Therefore, we do this warmup _twice_.
   162  	warmupCount := 2
   163  	for iter := 1; iter <= warmupCount; iter++ {

   174  		// parallel, stopping at the first error encountered

   176

   190  		})

   196  				return utils.MakeError("Error initializing uinput devices: %s", err)

   215  			return nil
   216  		})
   217

   221  		}

   224  		config := dockercontainer.Config{

   275  				// Don't need to set CgroupParent, since each mandelbox is its own task.

   278  				Devices:            devices,

   294  		}
   295  		logger.Infof("warmUpDockerClient(): Created container %s with image %s", containerName, image)
   296  
   297  		postCreateGroup, _ := errgroup.WithContext(mandelbox.GetContext())

   301  				return utils.MakeError("Error writing parameters for mandelbox: %s", err)
   302  			}
   303  			return nil

   311  
   312  			logger.Infof("warmUpDockerClient(): Started container %s with image %s", containerName, image)
   313  			return nil
   314  		})
   315

   318  			return utils.MakeError("Error warming up host service: %v", err)
   319  		}
   320  
   321  		// Unblocks whist-startup.sh to start symlink loaded user configs
   322  		err = mandelbox.MarkParamsReady()
   323  		if err != nil {
   324  			return utils.MakeError("Error marking mandelbox params as ready: %s", err)
   325  		}
   326  		err = mandelbox.MarkConfigReady()
   327  		if err != nil {
   328  			return utils.MakeError("Error marking mandelbox configs as ready: %s", err)
   329  		}
   330  
   331  		logger.Infof("Waiting for whist application to warm up...")
   332  
   333  		if err = utils.WaitForFileCreation(utils.Sprintf("/whist/%s/mandelboxResourceMappings/", containerName), "done_sleeping_until_X_clients", time.Minute*5, nil); err != nil {
   334  			return utils.MakeError("Error warming up whist application: %s", err)
   335  		}
   336  
   337  		logger.Infof("Finished waiting for whist application to warm up.")
   338  
   339  		time.Sleep(5 * time.Second)
   340  
   341  		stopTimeout := 30 * time.Second
   342  		_ = client.ContainerStop(globalCtx, createBody.ID, &stopTimeout)
   343  		err = client.ContainerRemove(globalCtx, createBody.ID, dockertypes.ContainerRemoveOptions{Force: true})
   344  		if err != nil {
   345  			return utils.MakeError("Error running `remove` for %s:\n%s", containerName, err)
   346  		}
   347  		logger.Infof("warmUpDockerClient(): Removed container %s with image %s", containerName, image)
   348  
   349  		// Close the mandelbox, and wait 5 seconds so we don't pollute the logs with
   350  		// cleanup after the event loop starts.
   351  		mandelbox.Close()
   352  		time.Sleep(5 * time.Second)
   353  
   354  		logger.Infof("Finished warmup iteration %v of %v", iter, warmupCount)
   355  	}
   356  
   357  	return nil

   371  // SpinUpMandelbox is the request used to create a mandelbox on this host.
   372  func SpinUpMandelbox(globalCtx context.Context, globalCancel context.CancelFunc, goroutineTracker *sync.WaitGroup, dockerClient dockerclient.CommonAPIClient,
   373  	sub *subscriptions.MandelboxEvent, transportRequestMap map[mandelboxtypes.MandelboxID]chan *JSONTransportRequest, transportMapLock *sync.Mutex) {
   374  
   375  	logAndReturnError := func(fmt string, v ...interface{}) {
   376  		err := utils.MakeError("SpinUpMandelbox(): "+fmt, v...)
   377  		logger.Error(err)
   378  		metrics.Increment("ErrorRate")
   379  	}
   380  
   381  	// mandelboxSubscription is the pubsub event received from Hasura.
   382  	mandelboxSubscription := sub.MandelboxInfo[0]
   383  	req, AppName := getAppName(mandelboxSubscription.ID, transportRequestMap, transportMapLock)
   384  
   385  	logger.Infof("SpinUpMandelbox(): spinup started for mandelbox %s", mandelboxSubscription.ID)
   386  
   387  	// Then, verify that we are expecting this user to request a mandelbox.
   388  	err := dbdriver.VerifyAllocatedMandelbox(mandelboxSubscription.UserID, mandelboxSubscription.ID)
   389  
   390  	if err != nil {
   391  		logAndReturnError("Unable to spin up mandelbox: %s", err)
   392  		return
   393  	}
   394  
   395  	// If so, create the mandelbox object.
   396  	mandelbox := mandelboxData.New(context.Background(), goroutineTracker, mandelboxSubscription.ID)
   397  	logger.Infof("SpinUpMandelbox(): created Mandelbox object %s", mandelbox.GetID())
   398  
   399  	// If the creation of the mandelbox fails, we want to clean up after it. We
   400  	// do this by setting `createFailed` to true until all steps are done, and
   401  	// closing the mandelbox's context on function exit if `createFailed` is
   402  	// still set to true.
   403  	var createFailed bool = true
   404  	defer func() {
   405  		if createFailed {
   406  			mandelbox.Close()
   407  		}
   408  	}()
   409  
   410  	mandelbox.AssignToUser(mandelboxSubscription.UserID)
   411  	mandelbox.SetAppName(AppName)
   412  	logger.Infof("SpinUpMandelbox(): Successfully assigned mandelbox %s to user %s", mandelboxSubscription.ID, mandelboxSubscription.UserID)
   413  
   414  	// Begin downloading user configs at the same time as other setup is being done.
   415  	// This is done separately from the rest of the startup goroutines since the user
   416  	// config download is a potentially long-running process that other pieces do not
   417  	// depend on.
   418  	userConfigDownloadComplete := make(chan bool)
   419  	go func() {
   420  		// User config errors aren't fatal --- we still want to spin up a mandelbox,
   421  		// and we will just use the provided encryption token to save configs when
   422  		// the mandelbox dies.
   423  		logger.Infof("SpinUpMandelbox(): Beginning user config download for mandelbox %s", mandelboxSubscription.ID)
   424  		err := mandelbox.DownloadUserConfigs()
   425  		if err != nil {
   426  			logger.Warningf("Error downloading user configs for mandelbox %s: %v", mandelboxSubscription.ID, err)
   427  			userConfigDownloadComplete <- true
   428  			return
   429  		}
   430  
   431  		userConfigDownloadComplete <- true
   432  		logger.Infof("SpinUpMandelbox(): Successfully downloaded user configs for mandelbox %s", mandelboxSubscription.ID)
   433  	}()
   434  
   435  	// Do all startup tasks that can be done before Docker container creation in
   436  	// parallel, stopping at the first error encountered
   437  	preCreateGroup, _ := errgroup.WithContext(mandelbox.GetContext())
   438  
   439  	// Request port bindings for the mandelbox.
   440  	var hostPortForTCP32262, hostPortForUDP32263, hostPortForTCP32273 uint16
   441  	preCreateGroup.Go(func() error {
   442  		if err := mandelbox.AssignPortBindings([]portbindings.PortBinding{
   443  			{MandelboxPort: 32262, HostPort: 0, BindIP: "", Protocol: "tcp"},
   444  			{MandelboxPort: 32263, HostPort: 0, BindIP: "", Protocol: "udp"},
   445  			{MandelboxPort: 32273, HostPort: 0, BindIP: "", Protocol: "tcp"},
   446  		}); err != nil {
   447  			return utils.MakeError("Error assigning port bindings: %s", err)
   448  		}
   449  		logger.Infof("SpinUpMandelbox(): successfully assigned port bindings %v", mandelbox.GetPortBindings())
   450  
   451  		var err32262, err32263, err32273 error
   452  		hostPortForTCP32262, err32262 = mandelbox.GetHostPort(32262, portbindings.TransportProtocolTCP)
   453  		hostPortForUDP32263, err32263 = mandelbox.GetHostPort(32263, portbindings.TransportProtocolUDP)
   454  		hostPortForTCP32273, err32273 = mandelbox.GetHostPort(32273, portbindings.TransportProtocolTCP)
   455  		if err32262 != nil || err32263 != nil || err32273 != nil {
   456  			return utils.MakeError("Couldn't return host port bindings.")
   457  		}
   458  
   459  		return nil
   460  	})
   461  
   462  	// Initialize Uinput devices for the mandelbox
   463  	var devices []dockercontainer.DeviceMapping
   464  	preCreateGroup.Go(func() error {
   465  		if err := mandelbox.InitializeUinputDevices(goroutineTracker); err != nil {
   466  			return utils.MakeError("Error initializing uinput devices: %s", err)
   467  		}
   468  
   469  		logger.Infof("SpinUpMandelbox(): successfully initialized uinput devices.")
   470  		devices = mandelbox.GetDeviceMappings()
   471  		return nil
   472  	})
   473  
   474  	// Allocate a TTY and GPU for the mandelbox
   475  	preCreateGroup.Go(func() error {
   476  		if err := mandelbox.InitializeTTY(); err != nil {
   477  			return utils.MakeError("Error initializing TTY: %s", err)
   478  		}
   479  
   480  		// CI does not have GPUs
   481  		if !metadata.IsRunningInCI() {
   482  			if err := mandelbox.AssignGPU(); err != nil {
   483  				return utils.MakeError("Error assigning GPU: %s", err)
   484  			}
   485  		}
   486  
   487  		logger.Infof("SpinUpMandelbox(): successfully allocated TTY %v and assigned GPU %v for mandelbox %s", mandelbox.GetTTY(), mandelbox.GetGPU(), mandelboxSubscription.ID)
   488  		return nil
   489  	})
   490  
   491  	err = preCreateGroup.Wait()
   492  	if err != nil {
   493  		logAndReturnError(err.Error())
   494  		return
   495  	}
   496  
   497  	// We need to compute the docker image to use for this mandelbox. In local
   498  	// development, we want to accept any string so that we can run arbitrary
   499  	// containers as mandelboxes, including custom-tagged ones. However, in
   500  	// deployments we only want to accept the specific apps that are enabled, and
   501  	// don't want to leak the fact that we're using containers as the technology
   502  	// for mandelboxes, so we construct the image string ourselves.
   503  	var image string
   504  	if metadata.IsLocalEnv() {
   505  
   506  		regexes := []string{
   507  			string(AppName) + ":current-build",
   508  			string(AppName),
   509  		}
   510  
   511  		image = dockerImageFromRegexes(globalCtx, dockerClient, regexes)
   512  	} else {
   513  		image = utils.Sprintf("ghcr.io/whisthq/%s/%s:current-build", metadata.GetAppEnvironmentLowercase(), AppName)
   514  	}
   515  
   516  	// We now create the underlying docker container for this mandelbox.
   517  	exposedPorts := make(dockernat.PortSet)
   518  	exposedPorts[dockernat.Port("32262/tcp")] = struct{}{}
   519  	exposedPorts[dockernat.Port("32263/udp")] = struct{}{}
   520  	exposedPorts[dockernat.Port("32273/tcp")] = struct{}{}
   521  
   522  	aesKey := utils.RandHex(16)
   523  	envs := []string{
   524  		utils.Sprintf("WHIST_AES_KEY=%s", aesKey),
   525  		utils.Sprintf("NVIDIA_VISIBLE_DEVICES=%v", "all"),
   526  		"NVIDIA_DRIVER_CAPABILITIES=all",
   527  		utils.Sprintf("SENTRY_ENV=%s", metadata.GetAppEnvironment()),
   528  		utils.Sprintf("WHIST_INITIAL_USER_DATA_FILE=%v%v", mandelboxData.UserInitialBrowserDir, mandelboxData.UserInitialBrowserFile),
   529  	}
   530  	config := dockercontainer.Config{
   531  		ExposedPorts: exposedPorts,
   532  		Env:          envs,
   533  		Image:        image,
   534  		AttachStdin:  true,
   535  		AttachStdout: true,
   536  		AttachStderr: true,
   537  		Tty:          true,
   538  	}
   539  	natPortBindings := make(dockernat.PortMap)
   540  	natPortBindings[dockernat.Port("32262/tcp")] = []dockernat.PortBinding{{HostPort: utils.Sprintf("%v", hostPortForTCP32262)}}
   541  	natPortBindings[dockernat.Port("32263/udp")] = []dockernat.PortBinding{{HostPort: utils.Sprintf("%v", hostPortForUDP32263)}}
   542  	natPortBindings[dockernat.Port("32273/tcp")] = []dockernat.PortBinding{{HostPort: utils.Sprintf("%v", hostPortForTCP32273)}}
   543  
   544  	tmpfs := make(map[string]string)
   545  	tmpfs["/run"] = "size=52428800"
   546  	tmpfs["/run/lock"] = "size=52428800"
   547  
   548  	// Sanitize the received session ID. First, verify that the sessionID is a valid timestamp string.
   549  	// If not, use a new timestamp.
   550  	_, err = strconv.ParseInt(mandelboxSubscription.SessionID, 10, 64)
   551  	if err != nil {
   552  		sessionID := strconv.FormatInt(time.Now().UnixMilli(), 10)
   553  		logger.Warningf("Malformed session ID: %v Using new session ID: %v", mandelboxSubscription.SessionID, sessionID)
   554  		mandelboxSubscription.SessionID = sessionID
   555  	}
   556  
   557  	hostConfig := dockercontainer.HostConfig{
   558  		Binds: []string{
   559  			"/sys/fs/cgroup:/sys/fs/cgroup:ro",
   560  			utils.Sprintf("/whist/%s/mandelboxResourceMappings:/whist/resourceMappings", mandelbox.GetID()),
   561  			utils.Sprintf("%s%s/sockets:/tmp/sockets", utils.TempDir, mandelbox.GetID()),
   562  			utils.Sprintf("%slogs/%s/%s:/var/log/whist", utils.TempDir, mandelbox.GetID(), mandelboxSubscription.SessionID),
   563  			"/run/udev/data:/run/udev/data:ro",
   564  			utils.Sprintf("/whist/%s/userConfigs/unpacked_configs:/whist/userConfigs:rshared", mandelbox.GetID()),
   565  		},
   566  		PortBindings: natPortBindings,
   567  		CapDrop:      strslice.StrSlice{"ALL"},
   568  		CapAdd: strslice.StrSlice([]string{
   569  			"SETPCAP",
   570  			"MKNOD",
   571  			"AUDIT_WRITE",
   572  			"CHOWN",
   573  			"NET_RAW",
   574  			"DAC_OVERRIDE",
   575  			"FOWNER",
   576  			"FSETID",
   577  			"KILL",
   578  			"SETGID",
   579  			"SETUID",
   580  			"NET_BIND_SERVICE",
   581  			"SYS_CHROOT",
   582  			"SETFCAP",
   583  			// NOTE THAT CAP_SYS_NICE IS NOT ENABLED BY DEFAULT BY DOCKER --- THIS IS OUR DOING
   584  			"SYS_NICE",
   585  		}),
   586  		ShmSize: 2147483648,
   587  		Tmpfs:   tmpfs,
   588  		Resources: dockercontainer.Resources{
   589  			CPUShares: 2,
   590  			Memory:    6552550944,
   591  			NanoCPUs:  0,
   592  			// Don't need to set CgroupParent, since each mandelbox is its own task.
   593  			// We're not using anything like AWS services, where we'd want to put
   594  			// several mandelboxes under one limit.
   595  			Devices:            devices,
   596  			KernelMemory:       0,
   597  			KernelMemoryTCP:    0,
   598  			MemoryReservation:  0,
   599  			MemorySwap:         0,
   600  			Ulimits:            []*dockerunits.Ulimit{},
   601  			CPUCount:           0,
   602  			CPUPercent:         0,
   603  			IOMaximumIOps:      0,
   604  			IOMaximumBandwidth: 0,
   605  		},
   606  		SecurityOpt: []string{
   607  			"apparmor:mandelbox-apparmor-profile",
   608  		},
   609  	}
   610  	mandelboxName := utils.Sprintf("%s-%s", AppName, mandelboxSubscription.ID)
   611  	re := regexp.MustCompile(`[^a-zA-Z0-9_.-]`)
   612  	mandelboxName = re.ReplaceAllString(mandelboxName, "-")
   613  
   614  	dockerBody, err := dockerClient.ContainerCreate(mandelbox.GetContext(), &config, &hostConfig, nil, &v1.Platform{Architecture: "amd64", OS: "linux"}, mandelboxName)
   615  	if err != nil {
   616  		logAndReturnError("Error running `create` for %s:\n%s", mandelbox.GetID(), err)
   617  		return
   618  	}
   619  
   620  	logger.Infof("SpinUpMandelbox(): Value returned from ContainerCreate: %#v", dockerBody)
   621  	dockerID := mandelboxtypes.DockerID(dockerBody.ID)
   622  
   623  	logger.Infof("SpinUpMandelbox(): Successfully ran `create` command and got back runtime ID %s", dockerID)
   624  
   625  	postCreateGroup, _ := errgroup.WithContext(mandelbox.GetContext())
   626  
   627  	// Register docker ID in mandelbox object
   628  	postCreateGroup.Go(func() error {
   629  		err := mandelbox.RegisterCreation(dockerID)
   630  		if err != nil {
   631  			return utils.MakeError("Error registering mandelbox creation with runtime ID %s: %s", dockerID, err)
   632  		}
   633  
   634  		logger.Infof("SpinUpMandelbox(): Successfully registered mandelbox creation with runtime ID %s and AppName %s", dockerID, AppName)
   635  		return nil
   636  	})
   637  
   638  	// Write mandelbox parameters to file
   639  	postCreateGroup.Go(func() error {
   640  		if err := mandelbox.WriteMandelboxParams(); err != nil {
   641  			return utils.MakeError("Error writing mandelbox params: %s", err)
   642  		}
   643  
   644  		// Let server protocol wait 30 seconds by default before client connects.
   645  		// However, in a local environment, we set the default to an effectively
   646  		// infinite value.
   647  		protocolTimeout := 30
   648  		if metadata.IsLocalEnv() {
   649  			protocolTimeout = -1
   650  		}
   651  
   652  		if err := mandelbox.WriteProtocolTimeout(protocolTimeout); err != nil {
   653  			return utils.MakeError("Error writing protocol timeout: %s", err)
   654  		}
   655  
   656  		logger.Infof("SpinUpMandelbox(): Successfully wrote parameters for mandelbox %s", mandelboxSubscription.ID)
   657  		return nil
   658  	})
   659  
   660  	// Start docker container
   661  	postCreateGroup.Go(func() error {
   662  		err = dockerClient.ContainerStart(mandelbox.GetContext(), string(dockerID), dockertypes.ContainerStartOptions{})
   663  		if err != nil {
   664  			return utils.MakeError("Error starting mandelbox with runtime ID %s and MandelboxID %s: %s", dockerID, mandelboxSubscription.ID, err)
   665  		}
   666  
   667  		logger.Infof("SpinUpMandelbox(): Successfully started mandelbox %s with ID %s", mandelboxName, mandelboxSubscription.ID)
   668  		return nil
   669  	})
   670  
   671  	// Mark as ready only when all startup tasks have completed
   672  	err = postCreateGroup.Wait()
   673  	if err != nil {
   674  		logAndReturnError(err.Error())
   675  		return
   676  	}
   677  
   678  	err = mandelbox.MarkParamsReady()
   679  	if err != nil {
   680  		logAndReturnError("Error marking mandelbox %s as ready to start A/V + display: %s", mandelboxSubscription.ID, err)
   681  		return
   682  	}
   683  	logger.Infof("SpinUpMandelbox(): Successfully marked mandelbox %s params as ready. A/V and display services can soon start.", mandelboxSubscription.ID)
   684  
   685  	<-userConfigDownloadComplete
   686  
   687  	logger.Infof("SpinUpMandelbox(): Waiting for config encryption token from client...")
   688  
   689  	if req == nil {
   690  		// Receive the json transport request from the client via the httpserver.
   691  		jsonchan := getJSONTransportRequestChannel(mandelboxSubscription.ID, transportRequestMap, transportMapLock)
   692  
   693  		// Set a timeout for the json transport request to prevent the mandelbox from waiting forever.
   694  		select {
   695  		case transportRequest := <-jsonchan:
   696  			req = transportRequest
   697  		case <-time.After(1 * time.Minute):
   698  			// Clean up the mandelbox if the time out limit is reached.
   699  			mandelboxDieHandler(string(dockerID), transportRequestMap, transportMapLock, dockerClient)
   700  			logAndReturnError("Timed out waiting for config encryption token.")
   701  			return
   702  		}
   703  	}
   704  
   705  	// Verify that this user sent in a (nontrivial) config encryption token
   706  	if len(req.ConfigEncryptionToken) < 10 {
   707  		logAndReturnError("Unable to spin up mandelbox: trivial config encryption token received.", err)
   708  		return
   709  	}
   710  	mandelbox.SetConfigEncryptionToken(req.ConfigEncryptionToken)
   711  	mandelbox.SetJSONData(req.JSONData)
   712  
   713  	// If the config token wasn't reset, decrypt the previously downloaded user configs
   714  	if !req.IsNewConfigToken {
   715  		err = mandelbox.DecryptUserConfigs()
   716  		if err != nil {
   717  			logger.Errorf("Error decrypting user configs for mandelbox %s: %v", mandelboxSubscription.ID, err)
   718  			metrics.Increment("ErrorRate")
   719  		}
   720  	} else {
   721  		// We still want to setup the directories so the new configs can be populated.
   722  		logger.Infof("SpinUpMandelbox(): This is a new config encryption token for mandelbox %s, skipping config decryption", mandelboxSubscription.ID)
   723  		err = mandelbox.SetupUserConfigDirs()
   724  		if err != nil {
   725  			logger.Errorf("Error setting up user config directories for mandelbox %s: %v", mandelboxSubscription.ID, err)
   726  			metrics.Increment("ErrorRate")
   727  		}
   728  	}
   729  
   730  	// Write user initial browser data at the same time as writeJSONData.
   731  	// This is done separately since both functions are independent of each other and we can save time.
   732  	browserDataGroup, _ := errgroup.WithContext(mandelbox.GetContext())
   733  	browserDataGroup.Go(func() error {
   734  		logger.Infof("SpinUpMandelbox(): Beginning storing user initial browser data for mandelbox %s", mandelboxSubscription.ID)
   735  
   736  		// Create browser data
   737  		userInitialBrowserData := mandelboxData.BrowserData{
   738  			CookiesJSON:   req.CookiesJSON,
   739  			BookmarksJSON: req.BookmarksJSON,
   740  			Extensions:    req.Extensions,
   741  		}
   742  
   743  		destDir := path.Join(mandelbox.GetUserConfigDir(), mandelboxData.GetUnpackedConfigsDirectoryName())
   744  		err := mandelboxData.WriteUserInitialBrowserData(userInitialBrowserData, destDir)
   745  		if err != nil {
   746  			logger.Warningf("Error writing user initial browser data for mandelbox %s: %v", mandelboxSubscription.ID, err)
   747  			return nil
   748  		}
   749  
   750  		logger.Infof("SpinUpMandelbox(): Successfully wrote user initial browser data for mandelbox %s", mandelboxSubscription.ID)
   751  		return nil
   752  	})
   753  
   754  	// Write the config.json file with the data received from JSON transport
   755  	err = mandelbox.WriteJSONData()
   756  	if err != nil {
   757  		logger.Errorf("Error writing config.json file for protocol: %v", err)
   758  		metrics.Increment("ErrorRate")
   759  	}
   760  
   761  	browserDataGroup.Wait()
   762  
   763  	// Unblocks whist-startup.sh to start symlink loaded user configs
   764  	err = mandelbox.MarkConfigReady()
   765  	if err != nil {
   766  		logAndReturnError("Error marking mandelbox %s configs as ready: %s", mandelboxSubscription.ID, err)
   767  		return
   768  	}
   769  	logger.Infof("SpinUpMandelbox(): Successfully marked mandelbox %s as ready", mandelboxSubscription.ID)
   770  
   771  	// Don't wait for whist application to start up in local environment
   772  	if !metadata.IsLocalEnv() {
   773  		logger.Infof("SpinUpMandelbox(): Waiting for mandelbox %s whist application to start up...", mandelboxSubscription.ID)
   774  		if err = utils.WaitForFileCreation(utils.Sprintf("/whist/%s/mandelboxResourceMappings/", mandelboxSubscription.ID), "done_sleeping_until_X_clients", time.Second*20, nil); err != nil {
   775  			logAndReturnError("Error waiting for mandelbox %s whist application to start: %s", mandelboxSubscription.ID, err)
   776  			return
   777  		}
   778  
   779  		logger.Infof("SpinUpMandelbox(): Finished waiting for mandelbox %s whist application to start up", mandelboxSubscription.ID)
   780  	}
   781  
   782  	err = dbdriver.WriteMandelboxStatus(mandelboxSubscription.ID, dbdriver.MandelboxStatusRunning)
   783  	if err != nil {
   784  		logAndReturnError("Error marking mandelbox running: %s", err)
   785  		return
   786  	}
   787  	logger.Infof("SpinUpMandelbox(): Successfully marked mandelbox %s as running", mandelboxSubscription.ID)
   788  
   789  	// Mark mandelbox creation as successful, preventing cleanup on function
   790  	// termination.
   791  	createFailed = false
   792  
   793  	result := JSONTransportRequestResult{
   794  		HostPortForTCP32262: hostPortForTCP32262,
   795  		HostPortForUDP32263: hostPortForUDP32263,
   796  		HostPortForTCP32273: hostPortForTCP32273,
   797  		AesKey:              aesKey,
   798  	}
   799  	req.ReturnResult(result, nil)
   800  	logger.Infof("SpinUpMandelbox(): Finished starting up mandelbox %s", mandelbox.GetID())
   801  }

   803  // Handle tasks to be completed when a mandelbox dies
   804  func mandelboxDieHandler(id string, transportRequestMap map[mandelboxtypes.MandelboxID]chan *JSONTransportRequest, transportMapLock *sync.Mutex, dockerClient dockerclient.CommonAPIClient) {
   805  	// Exit if we are not dealing with a Whist mandelbox, or if it has already
   806  	// been closed (via a call to Close() or a context cancellation).
   807  	mandelbox, err := mandelboxData.LookUpByDockerID(mandelboxtypes.DockerID(id))
   808  	if err != nil {
   809  		logger.Infof("mandelboxDieHandler(): %s", err)
   810  		return
   811  	}
   812  
   813  	// Clean up this user from the JSON transport request map.
   814  	mandelboxID := mandelbox.GetID()
   815  	transportMapLock.Lock()
   816  	transportRequestMap[mandelboxID] = nil
   817  	transportMapLock.Unlock()
   818  
   819  	// Gracefully shut down the mandelbox docker container
   820  	stopTimeout := 30 * time.Second
   821  	err = dockerClient.ContainerStop(mandelbox.GetContext(), id, &stopTimeout)
   822  
   823  	if err != nil {
   824  		logger.Errorf("Failed to gracefully stop mandelbox docker container.")
   825  		metrics.Increment("ErrorRate")
   826  	}
   827  
   828  	mandelbox.Close()
   829  }

   859  // Create the directory used to store the mandelbox resource allocations
   860  // (e.g. TTYs) on disk
   861  func initializeFilesystem(globalCancel context.CancelFunc) {
   862  	// check if "/whist" already exists --- if so, panic, since
   863  	// we don't know why it's there or if it's valid
   864  	if _, err := os.Lstat(utils.WhistDir); !os.IsNotExist(err) {
   865  		if err == nil {
   866  			logger.Panicf(globalCancel, "Directory %s already exists!", utils.WhistDir)
   867  		} else {
   868  			logger.Panicf(globalCancel, "Could not make directory %s because of error %v", utils.WhistDir, err)
   869  		}
   870  	}
   871  
   872  	// Create the whist directory and make it non-root user owned so that
   873  	// non-root users in mandelboxes can access files within (especially user
   874  	// configs).
   875  	err := os.MkdirAll(utils.WhistDir, 0777)
   876  	if err != nil {
   877  		logger.Panicf(globalCancel, "Failed to create directory %s: error: %s\n", utils.WhistDir, err)
   878  	}
   879  	cmd := exec.Command("chown", "-R", "ubuntu", utils.WhistDir)
   880  	cmd.Run()
   881  
   882  	// Create whist-private directory
   883  	err = os.MkdirAll(utils.WhistPrivateDir, 0777)
   884  	if err != nil {
   885  		logger.Panicf(globalCancel, "Failed to create directory %s: error: %s\n", utils.WhistPrivateDir, err)
   886  	}
   887  
   888  	// Create whist temp directory (only let root read and write this, since it
   889  	// contains logs and uinput sockets).
   890  	err = os.MkdirAll(utils.TempDir, 0600)
   891  	if err != nil {
   892  		logger.Panicf(globalCancel, "Could not mkdir path %s. Error: %s", utils.TempDir, err)
   893  	}
   894  }
   895  
   896  // Delete the directory used to store the mandelbox resource allocations (e.g.

   926  func main() {
   927  	// We create a global context (i.e. for the entire host service) that can be
   928  	// cancelled if the entire program needs to terminate. We also create a
   929  	// WaitGroup for all goroutines to tell us when they've stopped (if the
   930  	// context gets cancelled). Finally, we defer a function which cancels the
   931  	// global context if necessary, logs any panic we might be recovering from,
   932  	// and cleans up after the entire host service (although some aspects of
   933  	// cleanup may not seem necessary on a production instance, they are
   934  	// immensely helpful to prevent the host service from clogging up the local
   935  	// filesystem during development). After the permissions check, the creation
   936  	// of this context and WaitGroup, and the following defer must be the first
   937  	// statements in main().
   938  	globalCtx, globalCancel := context.WithCancel(context.Background())
   939  	goroutineTracker := sync.WaitGroup{}
   940  	defer func() {
   941  		// This function cleanly shuts down the Whist Host Service. Note that
   942  		// besides the host machine itself being forcefully shut down, this
   943  		// deferred function from main() should be the _only_ way that the host
   944  		// service exits. In particular, it should be as a result of a panic() in
   945  		// main, the global context being cancelled, or a Ctrl+C interrupt.
   946  
   947  		// Note that this function, while nontrivial, has intentionally been left
   948  		// as part of main() for the following reasons:
   949  		//   1. To prevent it being called from anywhere else accidentally
   950  		//   2. To keep the entire program control flow clearly in main().
   951  
   952  		// Catch any panics that might have originated in main() or one of its
   953  		// direct children.
   954  		r := recover()
   955  		if r != nil {
   956  			logger.Infof("Shutting down host service after caught panic in main(): %v", r)
   957  		} else {
   958  			logger.Infof("Beginning host service shutdown procedure...")
   959  		}
   960  
   961  		// Cancel the global context, if it hasn't already been cancelled.
   962  		globalCancel()
   963  
   964  		// Wait for all goroutines to stop, so we can run the rest of the cleanup
   965  		// process.
   966  		utils.WaitWithDebugPrints(&goroutineTracker, 2*time.Minute, 2)
   967  
   968  		// Stop processing new events
   969  		close(eventLoopKeepalive)
   970  
   971  		uninitializeFilesystem()
   972  
   973  		// Remove our row from the database and close out the database driver.
   974  		dbdriver.Close()
   975  
   976  		// Close out our metrics collection.
   977  		metrics.Close()
   978  
   979  		// Drain to our remote logging providers, but don't yet stop recording new
   980  		// events, in case the shutdown fails.
   981  		logger.FlushLogzio()
   982  		logger.FlushSentry()
   983  
   984  		logger.Info("Finished host service shutdown procedure. Finally exiting...")
   985  		if shutdownInstanceOnExit {
   986  			if err := exec.Command("shutdown", "now").Run(); err != nil {
   987  				if strings.TrimSpace(err.Error()) == "signal: terminated" {
   988  					// The instance seems to shut down even when this error is fired, so
   989  					// we'll just ignore it. We still `logger.Info()` it just in case.
   990  					logger.Infof("Shutdown command returned 'signal: terminated' error. Ignoring it.")
   991  				} else {
   992  					logger.Errorf("Couldn't shut down instance: %s", err)
   993  					metrics.Increment("ErrorRate")
   994  				}
   995  			}
   996  		}
   997  
   998  		// Shut down the logging infrastructure (including re-draining the queues).
   999  		logger.Close()
  1000  
  1001  		os.Exit(0)
  1002  	}()
  1003  
  1004  	// Log the Git commit of the running executable
  1005  	logger.Info("Host Service Version: %s", metadata.GetGitCommit())
  1006  
  1007  	// Initialize the database driver, if necessary (the `dbdriver` package
  1008  	// takes care of the "if necessary" part).
  1009  	if err := dbdriver.Initialize(globalCtx, globalCancel, &goroutineTracker); err != nil {
  1010  		logger.Panic(globalCancel, err)
  1011  	}
  1012  
  1013  	// Log the instance name we're running on
  1014  	instanceName, err := aws.GetInstanceName()
  1015  	if err != nil {
  1016  		logger.Panic(globalCancel, err)
  1017  	}
  1018  	logger.Infof("Running on instance name: %s", instanceName)
  1019  
  1020  	initializeAppArmor(globalCancel)
  1021  
  1022  	initializeFilesystem(globalCancel)
  1023  
  1024  	// Start Docker
  1025  	dockerClient, err := createDockerClient()
  1026  	if err != nil {
  1027  		logger.Panic(globalCancel, err)
  1028  	}
  1029  
  1030  	if !metadata.IsLocalEnv() {
  1031  		if err := warmUpDockerClient(globalCtx, globalCancel, &goroutineTracker, dockerClient); err != nil {
  1032  			logger.Panicf(globalCancel, "Error warming up docker client: %s", err)
  1033  		}
  1034  	}
  1035  
  1036  	if err := dbdriver.RegisterInstance(); err != nil {
  1037  		// If the instance starts up and sees its status as unresponsive or
  1038  		// draining, the webserver doesn't want it anymore so we should shut down.
  1039  
  1040  		// TODO: make this a bit more robust
  1041  		if !metadata.IsLocalEnv() && (strings.Contains(err.Error(), string(dbdriver.InstanceStatusUnresponsive)) ||
  1042  			strings.Contains(err.Error(), string(dbdriver.InstanceStatusDraining))) {
  1043  			logger.Infof("Instance wasn't registered in database because we found ourselves already marked draining or unresponsive. Shutting down.... Error: %s", err)
  1044  			shutdownInstanceOnExit = true
  1045  			globalCancel()
  1046  		} else {
  1047  			logger.Panic(globalCancel, err)
  1048  		}
  1049  	}
  1050  
  1051  	// Now we start all the goroutines that actually do work.
  1052  
  1053  	// Start the HTTP server and listen for events
  1054  	httpServerEvents, err := StartHTTPServer(globalCtx, globalCancel, &goroutineTracker)
  1055  	if err != nil {
  1056  		logger.Panic(globalCancel, err)
  1057  	}
  1058  
  1059  	// Start database subscription client
  1060  	instanceName, err = aws.GetInstanceName()
  1061  	if err != nil {
  1062  		logger.Errorf("Can't get AWS Instance Name to start database subscriptions. Error: %s", err)
  1063  		metrics.Increment("ErrorRate")
  1064  	}
  1065  	subscriptionEvents := make(chan subscriptions.SubscriptionEvent, 100)
  1066  
  1067  	subscriptionClient := &subscriptions.SubscriptionClient{}
  1068  	subscriptions.SetupHostSubscriptions(string(instanceName), subscriptionClient)
  1069  	subscriptions.Start(subscriptionClient, globalCtx, &goroutineTracker, subscriptionEvents)
  1070  	if err != nil {
  1071  		logger.Errorf("Failed to start database subscriptions. Error: %s", err)
  1072  	}
  1073  
  1074  	// Start main event loop. Note that we don't track this goroutine, but
  1075  	// instead control its lifetime with `eventLoopKeepAlive`. This is because it
  1076  	// needs to stay alive after the global context is cancelled, so we can
  1077  	// process mandelbox death events.
  1078  	go eventLoopGoroutine(globalCtx, globalCancel, &goroutineTracker, dockerClient, httpServerEvents, subscriptionEvents)
  1079  
  1080  	// Register a signal handler for Ctrl-C so that we cleanup if Ctrl-C is pressed.
  1081  	sigChan := make(chan os.Signal, 2)
  1082  	signal.Notify(sigChan, os.Interrupt, syscall.SIGTERM, syscall.SIGINT)
  1083  
  1084  	// Wait for either the global context to get cancelled by a worker goroutine,
  1085  	// or for us to receive an interrupt. This needs to be the end of main().
  1086  	select {
  1087  	case <-sigChan:
  1088  		logger.Infof("Got an interrupt or SIGTERM")
  1089  	case <-globalCtx.Done():
  1090  		logger.Infof("Global context cancelled!")
  1091  	}

  1095  // (including Docker events).
  1096  var eventLoopKeepalive = make(chan interface{}, 1)
  1097  
  1098  func eventLoopGoroutine(globalCtx context.Context, globalCancel context.CancelFunc,
  1099  	goroutineTracker *sync.WaitGroup, dockerClient dockerclient.CommonAPIClient,
  1100  	httpServerEvents <-chan ServerRequest, subscriptionEvents <-chan subscriptions.SubscriptionEvent) {
  1101  	// Note that we don't use globalCtx for the docker Context, since we still
  1102  	// wish to process Docker events after the global context is cancelled.
  1103  	dockerContext, dockerContextCancel := context.WithCancel(context.Background())
  1104  	defer dockerContextCancel()
  1105  
  1106  	// Create filter for which docker events we care about
  1107  	filters := dockerfilters.NewArgs()
  1108  	filters.Add("type", dockerevents.ContainerEventType)
  1109  	eventOptions := dockertypes.EventsOptions{
  1110  		Filters: filters,
  1111  	}
  1112  
  1113  	// We use this lock to protect the transportRequestMap
  1114  	transportMapLock := &sync.Mutex{}
  1115  
  1116  	// Note: If a mandelbox suffers a bug, or fails to start correctly
  1117  	// these channels will become a memory leak.
  1118  	transportRequestMap := make(map[mandelboxtypes.MandelboxID]chan *JSONTransportRequest)
  1119  
  1120  	// In the following loop, this var determines whether to re-initialize the
  1121  	// Docker event stream. This is necessary because the Docker event stream
  1122  	// needs to be reopened after any error is sent over the error channel.
  1123  	needToReinitDockerEventStream := false
  1124  	dockerevents, dockererrs := dockerClient.Events(dockerContext, eventOptions)
  1125  	logger.Info("Initialized docker event stream.")
  1126  	logger.Info("Entering event loop...")
  1127  
  1128  	// The actual event loop
  1129  	for {
  1130  		if needToReinitDockerEventStream {
  1131  			dockerevents, dockererrs = dockerClient.Events(dockerContext, eventOptions)
  1132  			needToReinitDockerEventStream = false
  1133  			logger.Info("Re-initialized docker event stream.")
  1134  		}
  1135  
  1136  		select {
  1137  		case <-eventLoopKeepalive:
  1138  			logger.Infof("Leaving main event loop...")
  1139  			return
  1140  
  1141  		case err := <-dockererrs:
  1142  			needToReinitDockerEventStream = true
  1143  			switch {
  1144  			case err == nil:
  1145  				logger.Info("We got a nil error over the Docker event stream. This might indicate an error inside the docker go library. Ignoring it and proceeding normally...")
  1146  				continue
  1147  			case err == io.EOF:
  1148  				logger.Panicf(globalCancel, "Docker event stream has been completely read.")
  1149  			case dockerclient.IsErrConnectionFailed(err):
  1150  				// This means "Cannot connect to the Docker daemon..."
  1151  				logger.Panicf(globalCancel, "Got error \"%v\". Could not connect to the Docker daemon.", err)
  1152  			default:
  1153  				if !strings.HasSuffix(strings.ToLower(err.Error()), "context canceled") {
  1154  					logger.Panicf(globalCancel, "Got an unknown error from the Docker event stream: %v", err)
  1155  				}
  1156  				return
  1157  			}
  1158  
  1159  		case dockerevent := <-dockerevents:
  1160  			logger.Info("dockerevent: %s for %s %s\n", dockerevent.Action, dockerevent.Type, dockerevent.ID)
  1161  			if dockerevent.Action == "die" {
  1162  				mandelboxDieHandler(dockerevent.ID, transportRequestMap, transportMapLock, dockerClient)
  1163  			}
  1164  
  1165  		// It may seem silly to just launch goroutines to handle these
  1166  		// serverevents, but we aim to keep the high-level flow control and handling
  1167  		// in this package, and the low-level authentication, parsing, etc. of
  1168  		// requests in `httpserver`.
  1169  		case serverevent := <-httpServerEvents:
  1170  			switch serverevent := serverevent.(type) {
  1171  			// TODO: actually handle panics in these goroutines
  1172  			case *JSONTransportRequest:
  1173  				if !metadata.IsLocalEnvWithoutDB() {
  1174  					// Handle JSON transport validation on a separate goroutine
  1175  					go handleJSONTransportRequest(serverevent, transportRequestMap, transportMapLock)
  1176  				} else {
  1177  					// If running on a local environment, disable any pubsub logic. We have to create a subscription request
  1178  					// that mocks the Hasura subscription event. Doing this avoids the need of setting up a Hasura server and
  1179  					// postgres database on the development instance.
  1180  					jsonReq := serverevent
  1181  
  1182  					userID, err := metadata.GetUserID()
  1183  					if err != nil {
  1184  						logger.Errorf("Error getting userID, %v", err)
  1185  						metrics.Increment("ErrorRate")
  1186  					}
  1187  
  1188  					instanceName, err := aws.GetInstanceName()
  1189  					if err != nil {
  1190  						logger.Errorf("Error getting instance name from AWS, %v", err)
  1191  						metrics.Increment("ErrorRate")
  1192  					}
  1193  					// Create a mandelbox object as would be received from a Hasura subscription.
  1194  					mandelbox := subscriptions.Mandelbox{
  1195  						InstanceName: string(instanceName),
  1196  						ID:           jsonReq.MandelboxID,
  1197  						SessionID:    "1234",
  1198  						UserID:       userID,
  1199  					}
  1200  					subscriptionEvent := subscriptions.MandelboxEvent{
  1201  						MandelboxInfo: []subscriptions.Mandelbox{mandelbox},
  1202  					}
  1203  
  1204  					// Launch both the JSON transport handler and the SpinUpMandelbox goroutines.
  1205  					go handleJSONTransportRequest(serverevent, transportRequestMap, transportMapLock)
  1206  					go SpinUpMandelbox(globalCtx, globalCancel, goroutineTracker, dockerClient, &subscriptionEvent, transportRequestMap, transportMapLock)
  1207  				}
  1208  			default:
  1209  				if serverevent != nil {
  1210  					err := utils.MakeError("unimplemented handling of server event [type: %T]: %v", serverevent, serverevent)
  1211  					logger.Error(err)
  1212  					serverevent.ReturnResult("", err)
  1213  				}
  1214  			}
  1215  
  1216  		case subscriptionEvent := <-subscriptionEvents:
  1217  			switch subscriptionEvent := subscriptionEvent.(type) {
  1218  			// TODO: actually handle panics in these goroutines
  1219  			case *subscriptions.MandelboxEvent:
  1220  				go SpinUpMandelbox(globalCtx, globalCancel, goroutineTracker, dockerClient,
  1221  					subscriptionEvent, transportRequestMap, transportMapLock)
  1222  
  1223  			case *subscriptions.InstanceEvent:
  1224  				// Don't do this in a separate goroutine, since there's no reason to.
  1225  				drainAndShutdown(globalCtx, globalCancel, goroutineTracker)
  1226  
  1227  			default:
  1228  				if subscriptionEvent != nil {
  1229  					err := utils.MakeError("Unimplemented handling of subscription event [type: %T]: %v", subscriptionEvent, subscriptionEvent)
  1230  					logger.Error(err)
  1231  				}
  1232  			}
  1233  		}
  1234  	}
  1235  }
