466 /* Note that memset() is a byte assignment and this is a 32-bit assignment, so they're not directly equivalent. */
467 SDL_FORCE_INLINE void SDL_memset4(void *dst, Uint32 val, size_t dwords)
468 {
469 #ifdef __APPLE__
470  memset_pattern4(dst, &val, dwords * 4);
471 #elif defined(__GNUC__) && defined(__i386__)
472  int u0, u1, u2;
473  __asm__ __volatile__ (
474  "cld \n\t"
475  "rep ; stosl \n\t"
476  : "=&D" (u0), "=&a" (u1), "=&c" (u2)
477  : "0" (dst), "1" (val), "2" (SDL_static_cast(Uint32, dwords))
478  : "memory"
479  );
480 #else
481  size_t _n = (dwords + 3) / 4;
482  Uint32 *_p = SDL_static_cast(Uint32 *, dst);
483  Uint32 _val = (val);
484  if (dwords == 0)
485  return;
486  switch (dwords % 4)
487  {
488  case 0: do { *_p++ = _val; /* fallthrough */
489  case 3: *_p++ = _val; /* fallthrough */
490  case 2: *_p++ = _val; /* fallthrough */
491  case 1: *_p++ = _val; /* fallthrough */
492  } while ( --_n );
493  }
494 #endif
495 }
