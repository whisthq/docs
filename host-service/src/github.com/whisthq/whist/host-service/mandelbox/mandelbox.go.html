     1  /*
     2  Package mandelbox provides the abstractions of mandelboxes managed by Whist.
     3  It defines the types and implements lots of functionality associated with
     4  mandelboxes. In `tracker.go`, it also keeps a list of all mandelboxes on this
     5  host at any given point.
     6  
     7  This package, and its children, are meant to be low-level enough that it can
     8  be imported by higher-level host service packages.
     9  
    10  Note: Many of the methods that access the mandelbox data use a lock, please follow the
    11  guidelines outlined below when writing code that directly uses the struct fields:
    12  
    13  1. Are you accessing a mandelbox struct field directly? If so, lock.
    14  2. Are you calling a method? If so, do not lock or you may cause a deadlock.
    15  3. Are you inside a method? If so, assume the lock is unlocked on method entry.
    16  
    17  Additionally, consider the following rules:
    18  
    19  - All locking should be done when and where the actual data access occurs.
    20    This means at the point of accessing the actual struct field
    21  - No method should assume that the lock is locked before entry
    22  - Getters and setters should always be used for struct field access when possible
    23  */

   131  func New(baseCtx context.Context, goroutineTracker *sync.WaitGroup, fid types.MandelboxID) Mandelbox {
   132  	// We create a context for this mandelbox specifically.
   133  	ctx, cancel := context.WithCancel(baseCtx)
   134  
   135  	mandelbox := &mandelboxData{
   136  		ctx:                  ctx,
   137  		cancel:               cancel,
   138  		ID:                   fid,
   139  		uinputDeviceMappings: []dockercontainer.DeviceMapping{},
   140  		otherDeviceMappings:  []dockercontainer.DeviceMapping{},
   141  	}
   142  
   143  	mandelbox.createResourceMappingDir()
   144  
   145  	trackMandelbox(mandelbox)
   146  
   147  	// Mount /dev/fuse for in-mandelbox FUSE filesystems to work
   148  	mandelbox.otherDeviceMappings = append(mandelbox.otherDeviceMappings, dockercontainer.DeviceMapping{
   149  		PathOnHost:        "/dev/fuse",
   150  		PathInContainer:   "/dev/fuse",
   151  		CgroupPermissions: "rwm", // read, write, mknod (the default)
   152  	})
   153  
   154  	// We start a goroutine that cleans up this Mandelbox as soon as its
   155  	// context is cancelled. This ensures that we automatically clean up when
   156  	// either `Close()` is called for this mandelbox, or the global context is
   157  	// cancelled.
   158  	goroutineTracker.Add(1)
   159  	go func() {
   160  		defer goroutineTracker.Done()
   161  
   162  		<-ctx.Done()
   163  
   164  		// Mark mandelbox as dying in the database, but only if it's not a warmup
   165  		if fid != types.MandelboxID(utils.PlaceholderWarmupUUID()) {
   166  			if err := dbdriver.WriteMandelboxStatus(mandelbox.GetID(), dbdriver.MandelboxStatusDying); err != nil {
   167  				logger.Error(err)
   168  			}
   169  		}
   170  
   171  		untrackMandelbox(mandelbox)
   172  		logger.Infof("Successfully untracked mandelbox %s", mandelbox.GetID())
   173  
   174  		mandelbox.rwlock.Lock()
   175  
   176  		// Free port bindings
   177  		portbindings.Free(mandelbox.portBindings)
   178  		mandelbox.portBindings = nil
   179  		logger.Infof("Successfully freed port bindings for mandelbox %s", mandelbox.GetID())
   180  
   181  		// Free uinput devices
   182  		mandelbox.uinputDevices.Close()
   183  		mandelbox.uinputDevices = nil
   184  		mandelbox.uinputDeviceMappings = []dockercontainer.DeviceMapping{}
   185  		logger.Infof("Successfully freed uinput devices for mandelbox %s", mandelbox.GetID())
   186  
   187  		// Free TTY
   188  		ttys.Free(mandelbox.tty)
   189  		logger.Infof("Successfully freed TTY %v for mandelbox %s", mandelbox.tty, mandelbox.GetID())
   190  		mandelbox.tty = 0
   191  
   192  		// CI does not have GPUs
   193  		if !metadata.IsRunningInCI() {
   194  			if err := gpus.Free(mandelbox.gpuIndex, mandelbox.GetID()); err != nil {
   195  				logger.Errorf("Error freeing GPU %v for mandelbox %s: %s", mandelbox.gpuIndex, mandelbox.GetID(), err)
   196  			} else {
   197  				logger.Infof("Successfully freed GPU %v for mandelbox %s", mandelbox.gpuIndex, mandelbox.GetID())
   198  			}
   199  		}
   200  
   201  		mandelbox.rwlock.Unlock()
   202  
   203  		// Clean resource mappings
   204  		mandelbox.cleanResourceMappingDir()
   205  		logger.Infof("Successfully cleaned resource mapping dir for mandelbox %s", mandelbox.GetID())
   206  
   207  		// Backup and clean user config directory.
   208  		err := mandelbox.BackupUserConfigs()
   209  		if err != nil {
   210  			logger.Errorf("Error backing up user configs for MandelboxID %s. Error: %s", mandelbox.GetID(), err)
   211  		} else {
   212  			logger.Infof("Successfully backed up user configs for MandelboxID %s", mandelbox.GetID())
   213  		}
   214  		mandelbox.cleanUserConfigDir()
   215  
   216  		// Remove mandelbox from the database altogether, once again excluding warmups
   217  		if fid != types.MandelboxID(utils.PlaceholderWarmupUUID()) {
   218  			if err := dbdriver.RemoveMandelbox(mandelbox.GetID()); err != nil {
   219  				logger.Error(err)
   220  			}
   221  		}
   222  
   223  		logger.Infof("Cleaned up after Mandelbox %s", mandelbox.GetID())
   224  	}()
   225  
   226  	return mandelbox

   331  	binds := mandelbox.GetPortBindings()

   456  // InitializeUinputDevices tries to assign uinput devices to the mandelbox.
   457  func (mandelbox *mandelboxData) InitializeUinputDevices(goroutineTracker *sync.WaitGroup) error {
   458  	devices, mappings, err := uinputdevices.Allocate()
   459  	if err != nil {
   460  		return utils.MakeError("Couldn't allocate uinput devices: %s", err)
   461  	}
   462  
   463  	mandelbox.rwlock.Lock()
   464  	defer mandelbox.rwlock.Unlock()
   465  
   466  	mandelbox.uinputDevices = devices
   467  	mandelbox.uinputDeviceMappings = mappings
   468  
   469  	goroutineTracker.Add(1)
   470  	go func() {
   471  		defer goroutineTracker.Done()
   472  
   473  		err := uinputdevices.SendDeviceFDsOverSocket(mandelbox.ctx, goroutineTracker, devices, utils.TempDir+mandelbox.GetID().String()+"/sockets/uinput.sock")
   474  		if err != nil {
   475  			placeholderUUID := types.MandelboxID(utils.PlaceholderWarmupUUID())
   476  			if mandelbox.GetID() == placeholderUUID && strings.Contains(err.Error(), "use of closed network connection") {
   477  				logger.Warningf("SendDeviceFDsOverSocket returned for MandelboxID %s with error: %s", mandelbox.GetID(), err)
   478  			} else {
   479  				logger.Errorf("SendDeviceFDsOverSocket returned for MandelboxID %s with error: %s", mandelbox.GetID(), err)
   480  			}
   481  		} else {
   482  			logger.Infof("SendDeviceFDsOverSocket returned successfully for MandelboxID %s", mandelbox.GetID())
   483  		}
   484  	}()
   485  
   486  	return nil
   487  }
