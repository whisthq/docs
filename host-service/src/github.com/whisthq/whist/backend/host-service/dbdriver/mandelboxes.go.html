    39  func VerifyAllocatedMandelbox(userID types.UserID, mandelboxID types.MandelboxID) error {
    40  	if !enabled {
    41  		return nil
    42  	}
    43  	if dbpool == nil {
    44  		return utils.MakeError("VerifyAllocatedMandelbox() called but dbdriver is not initialized!")
    45  	}
    46  
    47  	// Create a transaction to register the instance, since we are querying and
    48  	// writing separately.
    49  	tx, err := dbpool.BeginTx(context.Background(), pgx.TxOptions{IsoLevel: pgx.ReadCommitted})
    50  	if err != nil {
    51  		return utils.MakeError("Couldn't register instance: unable to begin transaction: %s", err)
    52  	}
    53  	// Safe to do even if committed -- see tx.Rollback() docs.
    54  	defer tx.Rollback(context.Background())
    55  
    56  	instanceName, err := aws.GetInstanceName()
    57  	if err != nil {
    58  		return utils.MakeError("Couldn't verify mandelbox %s for user %s: couldn't get instance name: %s", mandelboxID, userID, err)
    59  	}
    60  
    61  	q := queries.NewQuerier(tx)
    62  	rows, err := q.FindMandelboxByID(context.Background(), mandelboxID.String())
    63  	if err != nil {
    64  		return utils.MakeError("Couldn't verify mandelbox %s for user %s: error running query: %s", mandelboxID, userID, err)
    65  	}
    66  
    67  	// We expect that `rows` should contain exactly one, allocated (but not
    68  	// running) mandelbox with a matching mandelbox ID. Any other state is an
    69  	// error.
    70  	if len(rows) == 0 {
    71  		return utils.MakeError("Couldn't verify mandelbox %s for user %s: didn't find matching row in the database.", mandelboxID, userID)
    72  	} else if rows[0].Status.String != string(MandelboxStatusAllocated) {
    73  		return utils.MakeError(`Couldn't verify mandelbox %s for user %s: found a mandelbox row in the database for this instance, but it's in the wrong state. Expected "%s", got "%s".`, mandelboxID, userID, MandelboxStatusAllocated, rows[0].Status.String)
    74  	} else if rows[0].InstanceName.String != string(instanceName) {
    75  		return utils.MakeError(`Couldn't verify mandelbox %s for user %s: found an allocated mandelbox row in the database, but it has the wrong instanceName. Expected "%s", got "%s".`, mandelboxID, userID, rows[0].InstanceName.String, instanceName)
    76  	}
    77  
    78  	// Mark the container as connecting. We can't just use WriteMandelboxStatus
    79  	// since we want to do it in a single transaction.
    80  	result, err := q.WriteMandelboxStatus(context.Background(), pgtype.Varchar{
    81  		String: string(MandelboxStatusConnecting),
    82  		Status: pgtype.Present,
    83  	}, mandelboxID.String())
    84  	if err != nil {
    85  		return utils.MakeError("Couldn't write status %s for mandelbox %s: error updating existing row in table `cloud.mandelbox_info`: %s", MandelboxStatusConnecting, mandelboxID, err)
    86  	} else if result.RowsAffected() == 0 {
    87  		return utils.MakeError("Couldn't write status %s for mandelbox %s: row in database missing!", MandelboxStatusConnecting, mandelboxID)
    88  	}
    89  	logger.Infof("Updated status in database for mandelbox %s to %s: %s", mandelboxID, MandelboxStatusConnecting, result)
    90  
    91  	// Finish the transaction
    92  	tx.Commit(context.Background())
    93  
    94  	return nil
    95  }

   146  	if !enabled {

   168  	}
   169  	if result.RowsAffected() != 0 {
   170  		// We avoid logging this every time to avoid polluting the logs.
   171  		logger.Infof("Removed %v stale mandelboxes", result.RowsAffected())
   172  		metrics.Add("CleanedStaleMandelboxes", result.RowsAffected())
   173  	}
   174  	return nil

   177  func removeStaleMandelboxesGoroutine(globalCtx context.Context) {
   178  	defer logger.Infof("Finished removeStaleMandelboxes goroutine.")
   179  	timerChan := make(chan interface{})
   180  
   181  	// Instead of running exactly every 10 seconds, we choose a random time in
   182  	// the range [9.5, 10.5] seconds to prevent waves of hosts repeatedly crowding
   183  	// the database.
   184  	for {
   185  		sleepTime := 10000 - rand.Intn(1001)
   186  		timer := time.AfterFunc(time.Duration(sleepTime)*time.Millisecond, func() { timerChan <- struct{}{} })
   187  
   188  		select {
   189  		case <-globalCtx.Done():
   190  			// Remove allocated stale mandelboxes one last time
   191  			if err := removeStaleMandelboxes(10*time.Second, 90*time.Second); err != nil {
   192  				logger.Error(err)
   193  			}
   194  
   195  			utils.StopAndDrainTimer(timer)
   196  			return
   197  
   198  		case <-timerChan:
   199  			if err := removeStaleMandelboxes(10*time.Second, 90*time.Second); err != nil {
   200  				logger.Error(err)
   201  			}
   202  		}
   203  	}
   204  }
