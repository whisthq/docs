    34  // RegisterInstance registers the instance in the database. If the expected row
    35  // is not found, then it returns an error. This function also starts the
    36  // heartbeat goroutine.
    37  func RegisterInstance() error {
    38  	if !enabled {
    39  		return nil
    40  	}
    41  	if dbpool == nil {
    42  		return utils.MakeError("registerInstance() called but dbdriver is not initialized!")
    43  	}
    44  
    45  	instanceName, err := aws.GetInstanceName()
    46  	if err != nil {
    47  		return utils.MakeError("Couldn't register instance: couldn't get instance name: %s", err)
    48  	}
    49  	publicIP4, err := aws.GetPublicIpv4()
    50  	if err != nil {
    51  		return utils.MakeError("Couldn't register instance: couldn't get public IPv4: %s", err)
    52  	}
    53  	amiID, err := aws.GetAmiID()
    54  	if err != nil {
    55  		return utils.MakeError("Couldn't register instance: couldn't get AMI ID: %s", err)
    56  	}
    57  	region, err := aws.GetPlacementRegion()
    58  	if err != nil {
    59  		return utils.MakeError("Couldn't register instance: couldn't get AWS Placement Region: %s", err)
    60  	}
    61  	instanceType, err := aws.GetInstanceType()
    62  	if err != nil {
    63  		return utils.MakeError("Couldn't register instance: couldn't get AWS Instance type: %s", err)
    64  	}
    65  	instanceID, err := aws.GetInstanceID()
    66  	if err != nil {
    67  		return utils.MakeError("Couldn't register instance: couldn't get AWS Instance id: %s", err)
    68  	}
    69  
    70  	latestMetrics, errs := metrics.GetLatest()
    71  	if len(errs) != 0 {
    72  		return utils.MakeError("Couldn't register instance: errors getting metrics: %+v", errs)
    73  	}
    74  
    75  	// Create a transaction to register the instance, since we are querying and
    76  	// writing separately.
    77  	tx, err := dbpool.BeginTx(context.Background(), pgx.TxOptions{IsoLevel: pgx.ReadCommitted})
    78  	if err != nil {
    79  		return utils.MakeError("Couldn't register instance: unable to begin transaction: %s", err)
    80  	}
    81  	// Safe to do even if committed -- see tx.Rollback() docs.
    82  	defer tx.Rollback(context.Background())
    83  
    84  	// Check if there's a row for us in the database already
    85  	q := queries.NewQuerier(tx)
    86  	rows, err := q.FindInstanceByName(context.Background(), string(instanceName))
    87  	if err != nil {
    88  		return utils.MakeError("RegisterInstance(): Error running query: %s", err)
    89  	}
    90  
    91  	// Since the `instance_name` is the primary key of `cloud.instance_info`,
    92  	// we know that `rows` ought to contain either 0 or 1 results.
    93  	if len(rows) == 0 {
    94  		return utils.MakeError("RegisterInstance(): Existing row for this instance not found in the database.")
    95  	}
    96  
    97  	// Verify that the properties of the existing row are actually as we expect
    98  	if rows[0].AwsAmiID.String != string(amiID) {
    99  		return utils.MakeError(`RegisterInstance(): Existing database row found, but AMI differs. Expected "%s", Got "%s"`, amiID, rows[0].AwsAmiID.String)
   100  	}
   101  	if rows[0].Location.String != string(region) {
   102  		return utils.MakeError(`RegisterInstance(): Existing database row found, but location differs. Expected "%s", Got "%s"`, region, rows[0].Location.String)
   103  	}
   104  	if !(rows[0].CommitHash.Status == pgtype.Present && strings.HasPrefix(metadata.GetGitCommit(), rows[0].CommitHash.String)) {
   105  		// This is the only string where we have to check status, since an empty string is a prefix for anything.
   106  		return utils.MakeError(`RegisterInstance(): Existing database row found, but commit hash differs. Expected "%s", Got "%s"`, metadata.GetGitCommit(), rows[0].CommitHash.String)
   107  	}
   108  	if rows[0].AwsInstanceType.String != string(instanceType) {
   109  		return utils.MakeError(`RegisterInstance(): Existing database row found, but AWS instance type differs. Expected "%s", Got "%s"`, instanceType, rows[0].AwsInstanceType.String)
   110  	}
   111  	if rows[0].Status.String != string(InstanceStatusPreConnection) {
   112  		return utils.MakeError(`RegisterInstance(): Existing database row found, but status differs. Expected "%s", Got "%s"`, InstanceStatusPreConnection, rows[0].Status.String)
   113  	}
   114  
   115  	// There is an existing row in the database for this instance --- we now "take over" and update it with the correct information.
   116  	result, err := q.RegisterInstance(context.Background(), queries.RegisterInstanceParams{
   117  		CloudProviderID: pgtype.Varchar{
   118  			String: "aws-" + string(instanceID),
   119  			Status: pgtype.Present,
   120  		},
   121  		MemoryRemainingKB:    int(latestMetrics.AvailableMemoryKB),
   122  		NanoCPUsRemainingKB:  int(latestMetrics.NanoCPUsRemaining),
   123  		GpuVramRemainingKb:   int(latestMetrics.FreeVideoMemoryKB),
   124  		MandelboxCapacity:    latestMetrics.NumberOfGPUs,
   125  		LastUpdatedUtcUnixMs: int(time.Now().UnixNano() / 1_000_000),
   126  		Ip: pgtype.Varchar{
   127  			String: publicIP4.String(),
   128  			Status: pgtype.Present,
   129  		},
   130  		Status: pgtype.Varchar{
   131  			String: string(InstanceStatusActive),
   132  			Status: pgtype.Present,
   133  		},
   134  		InstanceName: string(instanceName),
   135  	})
   136  	if err != nil {
   137  		return utils.MakeError("Couldn't register instance: error updating existing row in table `cloud.instance_info`: %s", err)
   138  	} else if result.RowsAffected() == 0 {
   139  		return utils.MakeError("Couldn't register instance in database: row went missing!")
   140  	}
   141  	tx.Commit(context.Background())
   142  
   143  	logger.Infof("Successfully registered %s instance in database.", instanceName)
   144  
   145  	// Initialize the heartbeat goroutine. Note that this goroutine does not
   146  	// listen to the global context, and is not tracked by the global
   147  	// goroutineTracker. This is because we want the heartbeat goroutine to
   148  	// outlive the global context. Instead, we have the explicit Close(), which
   149  	// ends up finishing off the heartbeat goroutine as well.
   150  	go heartbeatGoroutine()
   151  
   152  	return nil
   153  }

   186  // unregisterInstance removes the row for the instance from the
   187  // `cloud.instance_info` table. Note that due to the `delete cascade`
   188  // constraint on `cloud.mandelbox_info` this automatically removes all the
   189  // mandelboxes for the instance as well.
   190  func unregisterInstance() error {
   191  	if !enabled {
   192  		return nil
   193  	}
   194  	if dbpool == nil {
   195  		return utils.MakeError("UnregisterInstance() called but dbdriver is not initialized!")
   196  	}
   197

   199  	if err != nil {
   200  		return utils.MakeError("Couldn't unregister instance: couldn't get instance name: %s", err)
   201  	}
   202

   204  	result, err := q.DeleteInstance(context.Background(), string(instanceName))
   205  	if err != nil {

   207  	} else if result.RowsAffected() == 0 {
   208  		return utils.MakeError("UnregisterInstance(): row went missing before we could delete it!")
   209  	}
   210  	logger.Infof("UnregisterInstance(): delete command successful")
   211  
   212  	return nil
   213  }
