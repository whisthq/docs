    21  func init() {
    22  	// Set some options for the Go logging package.
    23  	log.Default().SetFlags(log.Ldate | log.Lmicroseconds | log.LUTC)
    24  
    25  	// The first thing we want to do is to initialize logzio and Sentry so that
    26  	// we can catch any errors that might occur, or logs if we print them.
    27  	// The credentials for Sentry are hardcoded into the host-service,
    28  	// and the credentials for logzio are written to the environment file used
    29  	// by host-service systemd service.
    30  	// It is more likely that logzio initialization would fail, So we initialize
    31  	// Sentry first.
    32  
    33  	// We declare error separately to avoid shadowing sentryTransport.
    34  	var err error
    35  
    36  	// Note that according to the Sentry Go documentation, if we run
    37  	// sentry.CaptureMessage or sentry.CaptureError on separate goroutines, they
    38  	// can overwrite each other's tags. The thread-safe solution is to use
    39  	// goroutine-local hubs, but the way to do that would be to use contexts and
    40  	// add an additional argument to each logging method taking in the current
    41  	// context. This seems like a lot of work, so we just use a set of global
    42  	// tags instead, initializing them in InitializeSentry() and not touching
    43  	// them afterwards. Any mandelbox-specific information (which is what I
    44  	// imagine we would use local tags for) we just add in the text of the
    45  	// respective error message sent to Sentry. Alternatively, we might just be
    46  	// able to use sentry.WithScope(), but that is future work.
    47  	sentryTransport, err = initializeSentry()
    48  	if err != nil {
    49  		// Error, don't Panic, since a Sentry outage should not bring down our
    50  		// entire service.
    51  		Errorf("Failed to initialize Sentry! Error: %s", err)
    52  	}
    53  
    54  	logzioTransport, err = initializeLogzIO()
    55  	if err != nil {
    56  		// Error, don't Panic, since a logzio outage should not bring down our
    57  		// entire service.
    58  		Errorf("Failed to initialize LogzIO! Error: %s", err)
    59  	}
    60  
    61  	// Disable output to stdout when running on a nonn local environment.
    62  	if !metadata.IsLocalEnv() {
    63  		// Set the output of the `log` package to the whistBuffer. This captures
    64  		// any output to avoid clogging stdout and only send it to Logz and Sentry.
    65  		log.SetOutput(&whistBuffer)
    66  	}
    67  }

   125  // Panic sends an error to Sentry and "pretends" to panic on it by printing the
   126  // stack trace and calling the provided global context-cancelling function.
   127  // This causes all the goroutines in the program to kill themselves (cleanly).
   128  // This function should not be used except to initiate termination of the
   129  // entire host service. Note that passing in a nil first argument would cause
   130  // this function to _actually_ panic, and if we're gonna panic we might as well
   131  // do so in a useful way. Therefore, passing in a nil `globalCancel` parameter
   132  // will just panic on `err` instead.
   133  func Panic(globalCancel context.CancelFunc, err error) {
   134  	errstr := fmt.Sprintf("PANIC: %s", err)
   135  	if logzioTransport != nil {
   136  		logzioTransport.send(errstr, logzioTypeError)
   137  	}
   138  	if sentryTransport != nil {
   139  		sentryTransport.send(err)
   140  	}
   141  	PrintStackTrace()
   142  
   143  	if globalCancel != nil {
   144  		Error(err)
   145  		globalCancel()
   146  	} else {
   147  		// If we're truly trying to panic, let's at least flush our logging queues
   148  		// first so this error actually gets sent.
   149  		FlushLogzio()
   150  		FlushSentry()
   151  		log.Panicf(utils.ColorRed(errstr))
   152  	}
   153  }
