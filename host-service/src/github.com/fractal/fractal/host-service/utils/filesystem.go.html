    12  // WaitForFileCreation blocks until the provided filename is created in the
    13  // provided directory, or the timeout duration elapses. If the target file is
    14  // created in time, a nil error is returned. If the timeout elapses, a
    15  // context.DeadlineExceeded error is returned. In any other case, a non-nil
    16  // error is returned explaining what went wrong.
    17  //
    18  // For maximum correctness, we require that any paths passed in are absolute.
    19  // This may not be strictly necessary, but the documentation for `fsnotify` and
    20  // `path/filepath` are just vague enough for me to enforce this rule for the
    21  // callers. This rule may be subject to relaxation in the future.
    22  //
    23  // NOTE: Each invocation of this function creates an `inotify` instance and
    24  // holds onto it for the duration of this function call. By default, our
    25  // instances have a limit of 128 watchers per user. Therefore, we bump this
    26  // limit in host-setup to prevent it from being a limiting factor for our
    27  // mandelbox launches.
    28  func WaitForFileCreation(absParentDirectory, fileName string, timeout time.Duration) error {
    29  	if !path.IsAbs(absParentDirectory) {
    30  		return MakeError("Can't pass non-absolute paths into WaitForFileCreation")
    31  	}
    32  	targetFileName := path.Join(absParentDirectory, fileName)
    33  
    34  	watcher, err := fsnotify.NewWatcher()
    35  	if err != nil {
    36  		return MakeError("Couldn't create new fsnotify.Watcher: %s", err)
    37  	}
    38  	defer watcher.Close()
    39  
    40  	err = watcher.Add(absParentDirectory)
    41  	if err != nil {
    42  		return MakeError("Error adding dir %s to fsnotify.Watcher: %s", absParentDirectory, err)
    43  	}
    44  
    45  	timer := time.NewTimer(timeout)
    46  
    47  	for {
    48  		select {
    49  		case <-timer.C:
    50  			return context.DeadlineExceeded
    51  
    52  		case err, ok := <-watcher.Errors:
    53  			if !ok {
    54  				return MakeError("fsnotify.Watcher error channel closed.")
    55  			}
    56  			// Note that for us, dropped events _are_ errors, since we should not be
    57  			// having nearly enough filesystem activity to drop any events.
    58  			return MakeError("fsnotify returned error: %s", err)
    59  
    60  		case ev, ok := <-watcher.Events:
    61  			if !ok {
    62  				return MakeError("fsnotify.Watcher events channel closed.")
    63  			}
    64  			// TODO: remove this log once we're confident this part of the code works
    65  			log.Printf("Watched filesystem event: %+v", ev)
    66  			// Check if it's a creation event that matches the filename we expect
    67  			if ev.Op&fsnotify.Create == fsnotify.Create && ev.Name == targetFileName {
    68  				return nil
    69  			}
    70  		}
    71  	}
    72  
    73  }
