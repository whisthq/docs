    32  // instanceStatusHandler handles events from the hasura subscription which
    33  // detects changes on instance instanceName to the given status in the database.
    34  func instanceStatusHandler(instanceName string, status string, client *graphql.SubscriptionClient, subscriptionEvents chan<- SubscriptionEvent) (string, error) {
    35  	// variables holds the values needed to run the graphql subscription
    36  	variables := map[string]interface{}{
    37  		"instance_name": graphql.String(instanceName),
    38  		"status":        graphql.String(status),
    39  	}
    40  	// This subscriptions fires when the running instance status changes to draining on the database
    41  	id, err := client.Subscribe(InstanceStatusSubscription, variables, func(data *json.RawMessage, err error) error {
    42  
    43  		if err != nil {
    44  			return utils.MakeError("Error receiving subscription event from Hasura: %v", err)
    45  		}
    46  
    47  		var result InstanceStatusEvent
    48  		err = json.Unmarshal(*data, &result)
    49  
    50  		if err != nil {
    51  			return utils.MakeError("Failed to unmarshal subscription event: %v", err)
    52  		}
    53  
    54  		var instance Instance
    55  		// If the result array returned by Hasura is not empty, it means
    56  		// there was a change in the database row
    57  		if len(result.InstanceInfo) > 0 {
    58  			instance = result.InstanceInfo[0]
    59  		} else {
    60  			return nil
    61  		}
    62  
    63  		if (instance.InstanceName == instanceName) && (instance.Status == status) {
    64  			// We notify via the subscriptionsEvent channel to start the drain and shutdown process
    65  			subscriptionEvents <- &result
    66  		}
    67  
    68  		return nil
    69  	})
    70  
    71  	if err != nil {
    72  		return "", err
    73  	}
    74  
    75  	return id, nil
    76  }

   130  // Run is responsible for starting the subscription client, as well as
   131  // subscribing to the subscriptions defined in queries.go
   132  func Run(globalCtx context.Context, globalCancel context.CancelFunc, goroutineTracker *sync.WaitGroup, instanceName string, subscriptionEvents chan SubscriptionEvent) error {
   133  	if !enabled {
   134  		logger.Infof("Running in app environment %s so not enabling Hasura code.", metadata.GetAppEnvironment())
   135  		return nil
   136  	}
   137  	logger.Infof("Setting up Hasura subscriptions...")
   138  	// We set up the client with auth and logging parameters
   139  	client, err := SetUpHasuraClient()
   140  
   141  	if err != nil {
   142  		// Error creating the hasura client
   143  		logger.Errorf("Error starting Hasura client: %v", err)
   144  		return err
   145  	}
   146  
   147  	// Slice to hold subscription IDs, necessary to properly unsubscribe when we are done.
   148  	var subscriptionIDs []string
   149  
   150  	// Here we run all subscriptions
   151  	id, err := instanceStatusHandler(instanceName, "DRAINING", client, subscriptionEvents)
   152  	if err != nil {
   153  		// handle subscription error
   154  		return err
   155  	}
   156  	subscriptionIDs = append(subscriptionIDs, id)
   157  
   158  	id, err = mandelboxInfoHandler(instanceName, "ALLOCATED", client, subscriptionEvents)
   159  	if err != nil {
   160  		// handle subscription error
   161  		return err
   162  	}
   163  	subscriptionIDs = append(subscriptionIDs, id)
   164  
   165  	// Start goroutine that shuts down hasura client if the global context gets
   166  	// cancelled.
   167  	goroutineTracker.Add(1)
   168  	go func() {
   169  		defer goroutineTracker.Done()
   170  
   171  		// Listen for global context cancellation
   172  		<-globalCtx.Done()
   173  		Close(client, subscriptionIDs)
   174  	}()
   175  
   176  	// Run the client on a go routine to make sure it closes properly when we are done
   177  	go client.Run()
   178  
   179  	return nil
   180  }
