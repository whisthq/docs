    46  // instanceStatusHandler handles events from the hasura subscription which
    47  // detects changes on instance instanceName to the given status in the database.
    48  func instanceStatusHandler(instanceName string, status string, client *graphql.SubscriptionClient, subscriptionEvents chan<- SubscriptionEvent) (string, error) {
    49  	// variables holds the values needed to run the graphql subscription
    50  	variables := map[string]interface{}{
    51  		"instance_name": graphql.String(instanceName),
    52  		"status":        graphql.String(status),
    53  	}
    54  	// This subscriptions fires when the running instance status changes to draining on the database
    55  	id, err := client.Subscribe(InstanceStatusSubscription, variables, func(data *json.RawMessage, err error) error {
    56  
    57  		if err != nil {
    58  			return err
    59  		}
    60  
    61  		var result InstanceStatusEvent
    62  		json.Unmarshal(*data, &result)
    63  		result.createResultChan()
    64  
    65  		var instance Instance
    66  		// If the result array returned by Hasura is not empty, it means
    67  		// there was a change in the database row
    68  		if len(result.InstanceInfo) > 0 {
    69  			instance = result.InstanceInfo[0]
    70  		} else {
    71  			return nil
    72  		}
    73  
    74  		if (instance.InstanceName == instanceName) && (instance.Status == status) {
    75  			// We notify via the subscriptionsEvent channel to start the drain and shutdown process
    76  			subscriptionEvents <- &result
    77  		}
    78  
    79  		return nil
    80  	})
    81  
    82  	if err != nil {
    83  		return "", err
    84  	}
    85  
    86  	return id, nil
    87  }

   153  // Run is responsible for starting the subscription client, as well as
   154  // subscribing to the subscriptions defined in queries.go
   155  func Run(globalCtx context.Context, globalCancel context.CancelFunc, goroutineTracker *sync.WaitGroup, instanceName string, subscriptionEvents chan SubscriptionEvent) error {
   156  	if !enabled {
   157  		logger.Infof("Running in app environment %s so not enabling Hasura code.", metadata.GetAppEnvironment())
   158  		return nil
   159  	}
   160  	logger.Infof("Setting up Hasura subscriptions...")
   161  	// We set up the client with auth and logging parameters
   162  	client, err := SetUpHasuraClient()
   163  
   164  	if err != nil {
   165  		// Error creating the hasura client
   166  		logger.Errorf("Error starting Hasura client: %v", err)
   167  		return err
   168  	}
   169  
   170  	// Slice to hold subscription IDs, necessary to properly unsubscribe when we are done.
   171  	var subscriptionIDs []string
   172  
   173  	// Here we run all subscriptions
   174  	id, err := instanceStatusHandler(instanceName, "DRAINING", client, subscriptionEvents)
   175  	if err != nil {
   176  		// handle subscription error
   177  		return err
   178  	}
   179  	subscriptionIDs = append(subscriptionIDs, id)
   180  
   181  	id, err = mandelboxInfoHandler(instanceName, "ALLOCATED", client, subscriptionEvents)
   182  	if err != nil {
   183  		// handle subscription error
   184  		return err
   185  	}
   186  	subscriptionIDs = append(subscriptionIDs, id)
   187  
   188  	// Start goroutine that shuts down hasura client if the global context gets
   189  	// cancelled.
   190  	goroutineTracker.Add(1)
   191  	go func() {
   192  		defer goroutineTracker.Done()
   193  
   194  		// Listen for global context cancellation
   195  		<-globalCtx.Done()
   196  		Close(client, subscriptionIDs)
   197  	}()
   198  
   199  	// Run the client on a go routine to make sure it closes properly when we are done
   200  	go client.Run()
   201  
   202  	return nil
   203  }
