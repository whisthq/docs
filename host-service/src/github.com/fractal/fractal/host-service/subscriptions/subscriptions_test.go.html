   120  // For testing the subscriptions, we need to create a mock socket server that
   121  // is able to communicate with the Hasura client as if it were the Hasura server.
   122  func mockServer(t *testing.T, message string) int {
   123  	mockserver := http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
   124  		// Accept handshake from client
   125  		c, err := websocket.Accept(w, r, nil)
   126  		if err != nil {
   127  			t.Logf("Error accepting handshake from client: %v", err)
   128  		}
   129  		defer c.Close(websocket.StatusInternalError, "closing mock socket server")
   130  
   131  		ctx, cancel := context.WithCancel(r.Context())
   132  		defer cancel()
   133  
   134  		// Create ticker to receive messages from the client
   135  		time := time.NewTicker(time.Second)
   136  		defer time.Stop()
   137  
   138  		for {
   139  			// Server loop
   140  			select {
   141  			case <-ctx.Done():
   142  				t.Log("Closing mock server due to context cancelled...")
   143  				c.Close(websocket.StatusNormalClosure, "")
   144  				return
   145  			case <-time.C:
   146  				// Read message from client
   147  				var v interface{}
   148  				err = wsjson.Read(ctx, c, &v)
   149  
   150  				if err != nil {
   151  					t.Log("received nil message on server")
   152  					break
   153  				}
   154  
   155  				operation := v.(map[string]interface{})
   156  				// If the message matches the `start` operation
   157  				// it means the client successfully started the
   158  				// subscription.
   159  				if operation["type"] == "start" {
   160  					t.Log("Client successfully subscribed")
   161  					jsonMessage := json.RawMessage(message)
   162  
   163  					// Create a mock message to send to the client
   164  					// We send the "database event" as a raw json message
   165  					operationMessage := graphql.OperationMessage{
   166  						ID:      operation["id"].(string),
   167  						Type:    graphql.GQL_DATA,
   168  						Payload: jsonMessage,
   169  					}
   170  					// Write the message once and cancel the context
   171  					wsjson.Write(ctx, c, operationMessage)
   172  					cancel()
   173  				}
   174  			}
   175  		}
   176  	})
   177  
   178  	// Start the server on a goroutine to avoid blocking
   179  	// Use the portchan to return the port it is running on.
   180  	portchan := make(chan int)
   181  	go func(portchan chan int) {
   182  		// We use random available ports to ensure the server always starts.,
   183  		listener, err := net.Listen("tcp", ":0")
   184  		if err != nil {
   185  			panic(err)
   186  		}
   187  
   188  		port := listener.Addr().(*net.TCPAddr).Port
   189  		t.Log("Using port:", port)
   190  		portchan <- port
   191  		t.Error(http.Serve(listener, mockserver))
   192  	}(portchan)
   193  
   194  	return <-portchan
   195  }
