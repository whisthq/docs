    31  // FractalClaims is a struct type that models the claims that must be present
    32  // in an Auth0-issued Fractal access token.
    33  type FractalClaims struct {
    34  	jwt.StandardClaims
    35  
    36  	// Audience stores the value of the access token's "aud" claim. Inside the
    37  	// token's payload, the value of the "aud" claim can be either a JSON
    38  	// string or list of strings. We implement custom deserialization on the
    39  	// Audience type to handle both of these cases.
    40  	Audience Audience `json:"aud"`
    41  
    42  	// Scopes stores the value of the access token's "scope" claim. The value

    44  	// The *Scopes type implements the encoding.TextUnmarshaler interface such

    47  	Scopes Scopes `json:"scope"`

    72  // UnmarshalJSON unmarshals a JSON string or list of strings into an *Audience
    73  // type. It overwrites the contents of *audience with the unmarshalled data.
    74  func (audience *Audience) UnmarshalJSON(data []byte) error {
    75  	var aud string
    76  
    77  	// Try to unmarshal the input data into a string slice.
    78  	err := json.Unmarshal(data, (*[]string)(audience))
    79  
    80  	switch v := err.(type) {
    81  	case *json.UnmarshalTypeError:
    82  		// Ignore *json.UnmarshalTypeErrors, which are returned when the input
    83  		// data cannot be unmarshalled into a string slice. Instead, we will
    84  		// try to unmarshal the data into a string type below.
    85  	default:
    86  		// Return an error if err was non-nil or nil otherwise.
    87  		return v
    88  	}
    89  
    90  	// Try to unmarshal the input data into a string.
    91  	if err := json.Unmarshal(data, &aud); err != nil {
    92  		return err
    93  	}
    94  
    95  	// Overwrite any pre-existing data in *audience. Avoid creating a new
    96  	// allocation if possible by truncating and reusing the existing slice.
    97  	*audience = append((*audience)[0:0], aud)
    98  
    99  	return nil
   100  }

   121  // Verify parses an raw access token string, verifies the token's signature,
   122  // ensures that it is valid at the current moment in time, and checks that it
   123  // was issued by the proper issuer for the proper audience. It returns a
   124  // pointer to a FractalClaims type containing the values of its claims if all
   125  // checks are successful.
   126  func Verify(tokenString string) (*FractalClaims, error) {
   127  	claims := new(FractalClaims)
   128  	_, err := jwt.ParseWithClaims(tokenString, claims, jwks.KeyFunc)
   129  
   130  	if err != nil {
   131  		return nil, err
   132  	}
   133  
   134  	if !claims.VerifyAudience(config.Aud, true) {
   135  		return nil, jwt.NewValidationError(
   136  			utils.Sprintf("Bad audience %s", claims.Audience),
   137  			jwt.ValidationErrorAudience,
   138  		)
   139  	}
   140  
   141  	if !claims.VerifyIssuer(config.Iss, true) {
   142  		return nil, jwt.NewValidationError(
   143  			utils.Sprintf("Bad issuer %s", claims.Issuer),
   144  			jwt.ValidationErrorIssuer,
   145  		)
   146  	}
   147  
   148  	return claims, nil
   149  }
