    96  func New(baseCtx context.Context, goroutineTracker *sync.WaitGroup, fid types.MandelboxID) Mandelbox {
    97  	// We create a context for this mandelbox specifically.
    98  	ctx, cancel := context.WithCancel(baseCtx)
    99  
   100  	c := &mandelboxData{
   101  		ctx:                  ctx,
   102  		cancel:               cancel,
   103  		mandelboxID:          fid,
   104  		uinputDeviceMappings: []dockercontainer.DeviceMapping{},
   105  		otherDeviceMappings:  []dockercontainer.DeviceMapping{},
   106  	}
   107  
   108  	c.createResourceMappingDir()
   109  
   110  	trackMandelbox(c)
   111  
   112  	// Mount /dev/fuse for in-mandelbox FUSE filesystems to work
   113  	c.otherDeviceMappings = append(c.otherDeviceMappings, dockercontainer.DeviceMapping{
   114  		PathOnHost:        "/dev/fuse",
   115  		PathInContainer:   "/dev/fuse",
   116  		CgroupPermissions: "rwm", // read, write, mknod (the default)
   117  	})
   118  
   119  	// We start a goroutine that cleans up this Mandelbox as soon as its
   120  	// context is cancelled. This ensures that we automatically clean up when
   121  	// either `Close()` is called for this mandelbox, or the global context is
   122  	// cancelled.
   123  	goroutineTracker.Add(1)
   124  	go func() {
   125  		defer goroutineTracker.Done()
   126  
   127  		<-ctx.Done()
   128  
   129  		// Mark mandelbox as dying in the database, but only if it's not a warmup
   130  		if fid != "host-service-warmup" {
   131  			if err := dbdriver.WriteMandelboxStatus(c.mandelboxID, dbdriver.MandelboxStatusDying); err != nil {
   132  				logger.Error(err)
   133  			}
   134  		}
   135  
   136  		untrackMandelbox(c)
   137  		logger.Infof("Successfully untracked mandelbox %s", c.mandelboxID)
   138  
   139  		c.rwlock.Lock()
   140  		defer c.rwlock.Unlock()
   141  
   142  		// Free port bindings
   143  		portbindings.Free(c.portBindings)
   144  		c.portBindings = nil
   145  		logger.Infof("Successfully freed port bindings for mandelbox %s", c.mandelboxID)
   146  
   147  		// Free uinput devices
   148  		c.uinputDevices.Close()
   149  		c.uinputDevices = nil
   150  		c.uinputDeviceMappings = []dockercontainer.DeviceMapping{}
   151  		logger.Infof("Successfully freed uinput devices for mandelbox %s", c.mandelboxID)
   152  
   153  		// Free TTY
   154  		ttys.Free(c.tty)
   155  		logger.Infof("Successfully freed TTY %v for mandelbox %s", c.tty, c.mandelboxID)
   156  		c.tty = 0
   157  
   158  		if err := gpus.Free(c.gpuIndex); err != nil {
   159  			logger.Errorf("Error freeing GPU %v for mandelbox %s: %s", c.gpuIndex, c.mandelboxID, err)
   160  		} else {
   161  			logger.Infof("Successfully freed GPU %v for mandelbox %s", c.gpuIndex, c.mandelboxID)
   162  		}
   163  
   164  		// Clean resource mappings
   165  		c.cleanResourceMappingDir()
   166  		logger.Infof("Successfully cleaned resource mapping dir for mandelbox %s", c.mandelboxID)
   167  
   168  		// Backup and clean user config directory.
   169  		err := c.backupUserConfigs()
   170  		if err != nil {
   171  			logger.Errorf("Error backing up user configs for MandelboxID %s. Error: %s", c.mandelboxID, err)
   172  		} else {
   173  			logger.Infof("Successfully backed up user configs for MandelboxID %s", c.mandelboxID)
   174  		}
   175  		c.cleanUserConfigDir()
   176  
   177  		// Remove mandelbox from the database altogether, once again excluding warmups
   178  		if fid != "host-service-warmup" {
   179  			if err := dbdriver.RemoveMandelbox(c.mandelboxID); err != nil {
   180  				logger.Error(err)
   181  			}
   182  		}
   183  
   184  		logger.Infof("Cleaned up after Mandelbox %s", c.mandelboxID)
   185  	}()
   186  
   187  	return c

   262  func (c *mandelboxData) GetHostPort(mandelboxPort uint16, protocol portbindings.TransportProtocol) (uint16, error) {
   263  	// Don't lock ourselves, since `c.GetPortBindings()` will lock for us.
   264  
   265  	binds := c.GetPortBindings()
   266  	for _, b := range binds {
   267  		if b.Protocol == protocol && b.MandelboxPort == mandelboxPort {

   271  
   272  	return 0, utils.MakeError("Couldn't GetHostPort(%v, %v) for mandelbox with MandelboxID %s", mandelboxPort, protocol, c.GetMandelboxID())
   273  }

   371  func (c *mandelboxData) InitializeUinputDevices(goroutineTracker *sync.WaitGroup) error {
   372  	devices, mappings, err := uinputdevices.Allocate()
   373  	if err != nil {
   374  		return utils.MakeError("Couldn't allocate uinput devices: %s", err)
   375  	}
   376  
   377  	c.rwlock.Lock()
   378  	c.uinputDevices = devices
   379  	c.uinputDeviceMappings = mappings
   380  	defer c.rwlock.Unlock()
   381  
   382  	goroutineTracker.Add(1)
   383  	go func() {
   384  		defer goroutineTracker.Done()
   385  
   386  		err := uinputdevices.SendDeviceFDsOverSocket(c.ctx, goroutineTracker, devices, utils.TempDir+string(c.mandelboxID)+"/sockets/uinput.sock")
   387  		if err != nil {
   388  			if c.mandelboxID == "host-service-warmup" && strings.Contains(err.Error(), "use of closed network connection") {
   389  				logger.Warningf("SendDeviceFDsOverSocket returned for MandelboxID %s with error: %s", c.mandelboxID, err)
   390  			} else {
   391  				logger.Errorf("SendDeviceFDsOverSocket returned for MandelboxID %s with error: %s", c.mandelboxID, err)
   392  			}
   393  		} else {
   394  			logger.Infof("SendDeviceFDsOverSocket returned successfully for MandelboxID %s", c.mandelboxID)
   395  		}
   396  	}()
   397  
   398  	return nil
   399  }
