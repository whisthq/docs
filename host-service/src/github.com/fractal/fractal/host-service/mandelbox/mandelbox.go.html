   106  func New(baseCtx context.Context, goroutineTracker *sync.WaitGroup, fid types.MandelboxID) Mandelbox {
   107  	// We create a context for this mandelbox specifically.
   108  	ctx, cancel := context.WithCancel(baseCtx)
   109  
   110  	mandelbox := &mandelboxData{
   111  		ctx:                  ctx,
   112  		cancel:               cancel,
   113  		ID:                   fid,
   114  		uinputDeviceMappings: []dockercontainer.DeviceMapping{},
   115  		otherDeviceMappings:  []dockercontainer.DeviceMapping{},
   116  	}
   117  
   118  	mandelbox.createResourceMappingDir()
   119  
   120  	trackMandelbox(mandelbox)
   121  
   122  	// Mount /dev/fuse for in-mandelbox FUSE filesystems to work
   123  	mandelbox.otherDeviceMappings = append(mandelbox.otherDeviceMappings, dockercontainer.DeviceMapping{
   124  		PathOnHost:        "/dev/fuse",
   125  		PathInContainer:   "/dev/fuse",
   126  		CgroupPermissions: "rwm", // read, write, mknod (the default)
   127  	})
   128  
   129  	// We start a goroutine that cleans up this Mandelbox as soon as its
   130  	// context is cancelled. This ensures that we automatically clean up when
   131  	// either `Close()` is called for this mandelbox, or the global context is
   132  	// cancelled.
   133  	goroutineTracker.Add(1)
   134  	go func() {
   135  		defer goroutineTracker.Done()
   136  
   137  		<-ctx.Done()
   138  
   139  		// Mark mandelbox as dying in the database, but only if it's not a warmup
   140  		if fid != "host-service-warmup" {
   141  			if err := dbdriver.WriteMandelboxStatus(mandelbox.ID, dbdriver.MandelboxStatusDying); err != nil {
   142  				logger.Error(err)
   143  			}
   144  		}
   145  
   146  		untrackMandelbox(mandelbox)
   147  		logger.Infof("Successfully untracked mandelbox %s", mandelbox.ID)
   148  
   149  		mandelbox.rwlock.Lock()
   150  		defer mandelbox.rwlock.Unlock()
   151  
   152  		// Free port bindings
   153  		portbindings.Free(mandelbox.portBindings)
   154  		mandelbox.portBindings = nil
   155  		logger.Infof("Successfully freed port bindings for mandelbox %s", mandelbox.ID)
   156  
   157  		// Free uinput devices
   158  		mandelbox.uinputDevices.Close()
   159  		mandelbox.uinputDevices = nil
   160  		mandelbox.uinputDeviceMappings = []dockercontainer.DeviceMapping{}
   161  		logger.Infof("Successfully freed uinput devices for mandelbox %s", mandelbox.ID)
   162  
   163  		// Free TTY
   164  		ttys.Free(mandelbox.tty)
   165  		logger.Infof("Successfully freed TTY %v for mandelbox %s", mandelbox.tty, mandelbox.ID)
   166  		mandelbox.tty = 0
   167  
   168  		// CI does not have GPUs
   169  		if !metadata.IsRunningInCI() {
   170  			if err := gpus.Free(mandelbox.gpuIndex, mandelbox.ID); err != nil {
   171  				logger.Errorf("Error freeing GPU %v for mandelbox %s: %s", mandelbox.gpuIndex, mandelbox.ID, err)
   172  			} else {
   173  				logger.Infof("Successfully freed GPU %v for mandelbox %s", mandelbox.gpuIndex, mandelbox.ID)
   174  			}
   175  		}
   176  
   177  		// Clean resource mappings
   178  		mandelbox.cleanResourceMappingDir()
   179  		logger.Infof("Successfully cleaned resource mapping dir for mandelbox %s", mandelbox.ID)
   180  
   181  		// Backup and clean user config directory.
   182  		err := mandelbox.backupUserConfigs()
   183  		if err != nil {
   184  			logger.Errorf("Error backing up user configs for MandelboxID %s. Error: %s", mandelbox.ID, err)
   185  		} else {
   186  			logger.Infof("Successfully backed up user configs for MandelboxID %s", mandelbox.ID)
   187  		}
   188  		mandelbox.cleanUserConfigDir()
   189  
   190  		// Remove mandelbox from the database altogether, once again excluding warmups
   191  		if fid != "host-service-warmup" {
   192  			if err := dbdriver.RemoveMandelbox(mandelbox.ID); err != nil {
   193  				logger.Error(err)
   194  			}
   195  		}
   196  
   197  		logger.Infof("Cleaned up after Mandelbox %s", mandelbox.ID)
   198  	}()
   199  
   200  	return mandelbox
   201  }

   297  func (mandelbox *mandelboxData) GetHostPort(mandelboxPort uint16, protocol portbindings.TransportProtocol) (uint16, error) {
   298  	// Don't lock ourselves, since `c.GetPortBindings()` will lock for us.
   299  
   300  	binds := mandelbox.GetPortBindings()
   301  	for _, b := range binds {
   302  		if b.Protocol == protocol && b.MandelboxPort == mandelboxPort {
   303  			return b.HostPort, nil
   304  		}
   305  	}
   306  
   307  	return 0, utils.MakeError("Couldn't GetHostPort(%v, %v) for mandelbox with MandelboxID %s", mandelboxPort, protocol, mandelbox.GetID())
   308  }

   416  func (mandelbox *mandelboxData) InitializeUinputDevices(goroutineTracker *sync.WaitGroup) error {
   417  	devices, mappings, err := uinputdevices.Allocate()
   418  	if err != nil {
   419  		return utils.MakeError("Couldn't allocate uinput devices: %s", err)
   420  	}
   421  
   422  	mandelbox.rwlock.Lock()
   423  	mandelbox.uinputDevices = devices
   424  	mandelbox.uinputDeviceMappings = mappings
   425  	defer mandelbox.rwlock.Unlock()
   426  
   427  	goroutineTracker.Add(1)
   428  	go func() {
   429  		defer goroutineTracker.Done()
   430  
   431  		err := uinputdevices.SendDeviceFDsOverSocket(mandelbox.ctx, goroutineTracker, devices, utils.TempDir+string(mandelbox.ID)+"/sockets/uinput.sock")
   432  		if err != nil {
   433  			if mandelbox.ID == "host-service-warmup" && strings.Contains(err.Error(), "use of closed network connection") {
   434  				logger.Warningf("SendDeviceFDsOverSocket returned for MandelboxID %s with error: %s", mandelbox.ID, err)
   435  			} else {
   436  				logger.Errorf("SendDeviceFDsOverSocket returned for MandelboxID %s with error: %s", mandelbox.ID, err)
   437  			}
   438  		} else {
   439  			logger.Infof("SendDeviceFDsOverSocket returned successfully for MandelboxID %s", mandelbox.ID)
   440  		}
   441  	}()
   442  
   443  	return nil
   444  }
