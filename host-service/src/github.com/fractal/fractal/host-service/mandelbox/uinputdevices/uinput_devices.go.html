   118  // SendDeviceFDsOverSocket blocks until it can send the provided devices' file
   119  // descriptors over a socket at the provided path. If the given context is
   120  // cancelled, it also aborts (therefore letting us not leak goroutines if a
   121  // mandelbox dies or fails during creation after this function is called).
   122  func SendDeviceFDsOverSocket(baseCtx context.Context, goroutineTracker *sync.WaitGroup, devices *UinputDevices, socketPath string) error {
   123  	// Create our own context so we can safely cancel it.
   124  	ctx, cancel := context.WithCancel(baseCtx)
   125  	defer cancel()
   126  
   127  	dir, err := filepath.Abs(filepath.Dir(socketPath))
   128  	if err != nil {
   129  		return utils.MakeError("Unable to calculate parent directory of given socketPath %s. Error: %s", socketPath, err)
   130  	}
   131  
   132  	os.MkdirAll(dir, 0777)
   133  	os.RemoveAll(socketPath)
   134  
   135  	var lc net.ListenConfig
   136  	listener, err := lc.Listen(ctx, "unix", socketPath)
   137  	if err != nil {
   138  		return utils.MakeError("Could not create unix socket at %s: %s", socketPath, err)
   139  	}
   140  	// Instead of running `defer listener.Close()` we ran `defer cancel()` above,
   141  	// which will cause the following goroutine to always close `listener`.
   142  	// Normally, listener.Accept() blocks until the protocol connects. If the
   143  	// context is closed, though, we want to unblock.
   144  	goroutineTracker.Add(1)
   145  	go func() {
   146  		defer goroutineTracker.Done()
   147  		<-ctx.Done()
   148  		listener.Close()
   149  	}()
   150  
   151  	logger.Infof("Successfully created unix socket at: %s", socketPath)
   152  
   153  	// listener.Accept() blocks until the protocol connects
   154  	client, err := listener.Accept()
   155  	if err != nil {
   156  		return utils.MakeError("Could not connect to client over unix socket at %s: %s", socketPath, err)
   157  	}
   158  	defer client.Close()
   159  
   160  	connf, err := client.(*net.UnixConn).File()
   161  	if err != nil {
   162  		return utils.MakeError("Could not get file corresponding to client connection for unix socket at %s: %s", socketPath, err)
   163  	}
   164  	defer connf.Close()
   165  
   166  	connfd := int(connf.Fd())
   167  	fds := [3]int{
   168  		int(devices.absmouse.DeviceFile().Fd()),
   169  		int(devices.relmouse.DeviceFile().Fd()),
   170  		int(devices.keyboard.DeviceFile().Fd()),
   171  	}
   172  	rights := syscall.UnixRights(fds[:]...)
   173  	err = syscall.Sendmsg(connfd, nil, rights, nil, 0)
   174  	if err != nil {
   175  		return utils.MakeError("Error sending uinput file descriptors over socket: %s", err)
   176  	}
   177  
   178  	logger.Infof("Sent uinput file descriptors to socket at %s", socketPath)
   179  	return nil
   180  }

   182  // getDeviceFilePath returns the file path (e.g. /dev/input/event3) given a
   183  // file descriptor corresponding to a virtual device created with uinput
   184  func getDeviceFilePath(fd *os.File) (string, error) {
   185  	const maxlen uintptr = 32
   186  	bsysname := make([]byte, maxlen)
   187  	_, _, errno := syscall.Syscall(syscall.SYS_IOCTL, uintptr(fd.Fd()), linuxUIGetSysName(maxlen), uintptr(unsafe.Pointer(&bsysname[0])))
   188  	if errno != 0 {
   189  		return "", utils.MakeError("ioctl to get sysname failed with errno %d", errno)
   190  	}
   191  	sysname := string(bytes.Trim(bsysname, "\x00"))
   192  	syspath := utils.Sprintf("/sys/devices/virtual/input/%s", sysname)
   193  	sysdir, err := os.Open(syspath)
   194  	if err != nil {
   195  		return "", utils.MakeError("could not open directory %s: %s", syspath, err)
   196  	}
   197  	names, err := sysdir.Readdirnames(0)
   198  	if err != nil {
   199  		return "", utils.MakeError("Readdirnames for directory %s failed: %s", syspath, err)
   200  	}
   201  	for _, name := range names {
   202  		if strings.HasPrefix(name, "event") {
   203  			return "/dev/input/" + name, nil
   204  		}
   205  	}
   206  	return "", utils.MakeError("did not find file in %s with prefix 'event'", syspath)
   207  }

   209  // In summary, this function computes the argument that gets passed to the
   210  // ioctl system call. For more details, and an explanation of the argument
   211  // names to linuxIoc, see:
   212  // https://www.kernel.org/doc/html/latest/driver-api/ioctl.html
   213  func linuxUIGetSysName(len uintptr) uintptr {
   214  	// from ioctl.h and uinput.h
   215  	const (
   216  		iocDirshift  = 30
   217  		iocSizeshift = 16
   218  		iocTypeshift = 8
   219  		iocNrshift   = 0
   220  		iocRead      = 2
   221  		uiIoctlBase  = 85 // 'U'
   222  	)
   223  
   224  	linuxIoc := func(dir uintptr, requestType uintptr, nr uintptr, size uintptr) uintptr {
   225  		return (dir << iocDirshift) + (requestType << iocTypeshift) + (nr << iocNrshift) + (size << iocSizeshift)
   226  	}
   227  
   228  	return linuxIoc(iocRead, uiIoctlBase, 44, len)
   229  }
