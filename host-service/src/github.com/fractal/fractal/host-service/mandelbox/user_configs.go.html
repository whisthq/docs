    29  // DownloadUserConfigs downloads user configs from S3 and saves them to an in-memory buffer.
    30  func (mandelbox *mandelboxData) DownloadUserConfigs() error {
    31  	// If userID is not set, then we don't retrieve configs from s3
    32  	if len(mandelbox.GetUserID()) == 0 {
    33  		logger.Warningf("User ID is not set for mandelbox %s. Skipping config download.", mandelbox.GetID())
    34  		return nil
    35  	}
    36  
    37  	s3ConfigKey := mandelbox.getS3ConfigKey()
    38  
    39  	logger.Infof("Starting S3 config download for mandelbox %s", mandelbox.GetID())
    40  
    41  	s3Client, err := configutils.NewS3Client("us-east-1")
    42  	if err != nil {
    43  		return utils.MakeError("failed to create s3 client: %v", err)
    44  	}
    45  
    46  	// Fetch the HeadObject first to see how much memory we need to allocate
    47  	logger.Infof("Fetching head object for bucket: %s, key: %s", userConfigS3Bucket, s3ConfigKey)
    48  	headObject, err := configutils.GetHeadObject(s3Client, userConfigS3Bucket, s3ConfigKey)
    49  
    50  	var apiErr smithy.APIError
    51  	if err != nil {
    52  		// If aws s3 errors out due to the file not existing, don't log an error because
    53  		// this means that it's the user's first run and they don't have any settings
    54  		// stored for this application yet.
    55  		if errors.As(err, &apiErr) && apiErr.ErrorCode() == "NotFound" {
    56  			logger.Infof("Could not get head object because config does not exist for user %s", mandelbox.GetUserID())
    57  			return nil
    58  		}
    59  
    60  		return utils.MakeError("Failed to download head object from s3: %v", err)
    61  	}
    62  
    63  	// Log config version
    64  	logger.Infof("Using user config version %s for mandelbox %s", *headObject.VersionId, mandelbox.GetID())
    65  
    66  	// Download file into a pre-allocated in-memory buffer
    67  	// This should be okay as we don't expect configs to be very large
    68  	configBuffer := manager.NewWriteAtBuffer(make([]byte, headObject.ContentLength))
    69  	numBytes, err := configutils.DownloadObjectToBuffer(s3Client, userConfigS3Bucket, s3ConfigKey, configBuffer)
    70  
    71  	var noSuchKeyErr *types.NoSuchKey
    72  	if err != nil {
    73  		if errors.As(err, &noSuchKeyErr) {
    74  			logger.Infof("Could not download user config because config does not exist for user %s", mandelbox.GetUserID())
    75  			return nil
    76  		}
    77  
    78  		return utils.MakeError("Failed to download user configuration from s3: %v", err)
    79  	}
    80  
    81  	mandelbox.SetConfigBuffer(configBuffer)
    82  	logger.Infof("Downloaded %d bytes from s3 for mandelbox %s", numBytes, mandelbox.GetID())
    83  
    84  	return nil
    85  }

    87  // DecryptUserConfigs decrypts and unpacks the previously downloaded
    88  // s3 config using the encryption token received through JSON transport.
    89  func (mandelbox *mandelboxData) DecryptUserConfigs() error {
    90  	if len(mandelbox.GetConfigEncryptionToken()) == 0 {
    91  		return utils.MakeError("Cannot get user configs for MandelboxID %s since ConfigEncryptionToken is empty", mandelbox.GetID())
    92  	}
    93  
    94  	if mandelbox.GetConfigBuffer() == nil {
    95  		logger.Infof("Not decrypting user configs because the config buffer is empty.")
    96  		return nil
    97  	}
    98  
    99  	logger.Infof("Decrypting user config for mandelbox %s", mandelbox.GetID())
   100  	logger.Infof("Using (hashed) decryption token %s for mandelbox %s", getTokenHash(string(mandelbox.GetConfigEncryptionToken())), mandelbox.GetID())
   101  
   102  	// Decrypt the downloaded archive directly from memory
   103  	encryptedFile := mandelbox.GetConfigBuffer().Bytes()
   104  	decryptedData, err := configutils.DecryptAES256GCM(string(mandelbox.GetConfigEncryptionToken()), encryptedFile)
   105  	if err != nil {
   106  		return utils.MakeError("Failed to decrypt user configs: %v", err)
   107  	}
   108  
   109  	logger.Infof("Finished decrypting user config for mandelbox %s", mandelbox.GetID())
   110  	logger.Infof("Decompressing user config for mandelbox %s", mandelbox.GetID())
   111  
   112  	// Make directory for user configs
   113  	configDir := mandelbox.getUserConfigDir()
   114  	unpackedConfigDir := path.Join(configDir, mandelbox.getUnpackedConfigsDirectoryName())
   115  	err = mandelbox.SetupUserConfigDirs()
   116  	if err != nil {
   117  		return utils.MakeError("failed to setup user config directories: %v", err)
   118  	}
   119  
   120  	// Once we've extracted everything, we open up permissions for the user
   121  	// config directory so it's accessible by the non-root user in the
   122  	// mandelboxes. We also are okay with setting executable bits here, since
   123  	// it's possible that some apps store executable scripts in their config.
   124  	defer func() {
   125  		cmd := exec.Command("chown", "-R", "ubuntu", configDir)
   126  		cmd.Run()
   127  		cmd = exec.Command("chmod", "-R", "777", configDir)
   128  		cmd.Run()
   129  	}()
   130  
   131  	totalFileSize, err := configutils.ExtractTarLz4(decryptedData, unpackedConfigDir)
   132  	if err != nil {
   133  		logger.Errorf("Error extracting tar lz4 file: %v", err)
   134  	}
   135  
   136  	logger.Infof("Untarred config to: %s, total size was %d bytes", unpackedConfigDir, totalFileSize)
   137  
   138  	// Set the mandelbox config buffer to nil after we're done
   139  	mandelbox.SetConfigBuffer(nil)
   140  
   141  	return nil
   142  }

   144  // BackupUserConfigs compresses, encrypts, and then uploads user config files to S3.
   145  func (mandelbox *mandelboxData) BackupUserConfigs() error {
   146  	userID := mandelbox.GetUserID()
   147  	configToken := mandelbox.GetConfigEncryptionToken()
   148  	mandelboxID := mandelbox.GetID()
   149  
   150  	if len(userID) == 0 {
   151  		logger.Infof("Cannot save user configs for MandelboxID %s since UserID is empty.", mandelboxID)
   152  		return nil
   153  	}
   154  
   155  	if len(configToken) == 0 {
   156  		return utils.MakeError("Cannot save user configs for MandelboxID %s since ConfigEncryptionToken is empty", mandelboxID)
   157  	}
   158  
   159  	configDir := mandelbox.getUserConfigDir()
   160  	unpackedConfigPath := path.Join(configDir, mandelbox.getUnpackedConfigsDirectoryName())
   161  
   162  	// Compress the user configs into a tar.lz4 file
   163  	compressedConfig, err := configutils.CompressTarLz4(unpackedConfigPath)
   164  	if err != nil {
   165  		return utils.MakeError("Failed to compress user configs: %v", err)
   166  	}
   167  
   168  	// Encrypt the compressed config
   169  	logger.Infof("Using (hashed) encryption token %s for mandelbox %s", getTokenHash(string(configToken)), mandelboxID)
   170  	encryptedConfig, err := configutils.EncryptAES256GCM(string(configToken), compressedConfig)
   171  	if err != nil {
   172  		return utils.MakeError("Failed to encrypt user configs: %v", err)
   173  	}
   174  	encryptedConfigBuffer := bytes.NewBuffer(encryptedConfig)
   175  
   176  	// Upload encrypted config to S3
   177  	s3Client, err := configutils.NewS3Client("us-east-1")
   178  	if err != nil {
   179  		return utils.MakeError("error creating s3 client: %v", err)
   180  	}
   181  
   182  	uploadResult, err := configutils.UploadFileToBucket(s3Client, userConfigS3Bucket, mandelbox.getS3ConfigKey(), encryptedConfigBuffer)
   183  	if err != nil {
   184  		return utils.MakeError("error uploading encrypted config to s3: %v", err)
   185  	}
   186  
   187  	logger.Infof("Saved user config for mandelbox %s, version %s", mandelboxID, *uploadResult.VersionID)
   188  
   189  	return nil
   190  }

   194  func (mandelbox *mandelboxData) WriteUserInitialBrowserData(cookieJSON string, bookmarksJSON string) error {
   195  	// Avoid doing work for empty string/array string
   196  	if len(cookieJSON) <= 2 && len(bookmarksJSON) <= 2 {
   197  		logger.Infof("Not writing to file as user initial browser data is empty.")
   198  		// the browser data can be empty
   199  		return nil
   200  	}
   201  
   202  	// The initial browser cookies will use the same directory as the user config
   203  	// Make directories for user configs if not exists
   204  	configDir := mandelbox.getUserConfigDir()
   205  	if _, err := os.Stat(configDir); os.IsNotExist(err) {
   206  		if err := os.MkdirAll(configDir, 0777); err != nil {
   207  			return utils.MakeError("Could not make dir %s. Error: %s", configDir, err)
   208  		}
   209  		defer func() {
   210  			cmd := exec.Command("chown", "-R", "ubuntu", configDir)
   211  			cmd.Run()
   212  			cmd = exec.Command("chmod", "-R", "777", configDir)
   213  			cmd.Run()
   214  		}()
   215  	}
   216  
   217  	unpackedConfigDir := path.Join(configDir, mandelbox.getUnpackedConfigsDirectoryName())
   218  	if _, err := os.Stat(unpackedConfigDir); os.IsNotExist(err) {
   219  		if err := os.MkdirAll(unpackedConfigDir, 0777); err != nil {
   220  			return utils.MakeError("Could not make dir %s. Error: %s", unpackedConfigDir, err)
   221  		}
   222  	}
   223  
   224  	// Begin writing user initial browser data
   225  	cookieFilePath := path.Join(unpackedConfigDir, utils.UserInitialCookiesFile)
   226  
   227  	if err := utils.WriteToNewFile(cookieFilePath, cookieJSON); err != nil {
   228  		return utils.MakeError("error creating cookies file. Error: %v", err)
   229  	}
   230  
   231  	bookmarkFilePath := path.Join(unpackedConfigDir, utils.UserInitialBookmarksFile)
   232  
   233  	if err := utils.WriteToNewFile(bookmarkFilePath, bookmarksJSON); err != nil {
   234  		return utils.MakeError("error creating bookmarks file. Error: %v", err)
   235  	}
   236  
   237  	logger.Infof("Finished storing user initial browser data.")
   238  
   239  	return nil
   240  }
