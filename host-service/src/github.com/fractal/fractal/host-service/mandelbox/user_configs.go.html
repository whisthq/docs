   109  // s3 config using the encryption token received through JSON transport.
   110  func (mandelbox *mandelboxData) DecryptUserConfigs() error {
   111  
   112  	if len(mandelbox.configEncryptionToken) == 0 {
   113  		return utils.MakeError("Cannot get user configs for MandelboxID %s since ConfigEncryptionToken is empty", mandelbox.ID)
   114  	}
   115  
   116  	if mandelbox.configBuffer == nil {
   117  		logger.Infof("Not decrypting user configs because the config buffer is empty.")
   118  		return nil
   119  	}
   120  
   121  	logger.Infof("Decrypting user config for mandelbox %s", mandelbox.ID)
   122  
   123  	// Decrypt the downloaded archive directly from memory
   124  	encryptedFile := mandelbox.configBuffer.Bytes()
   125  	salt, data, err := getSaltAndDataFromOpenSSLEncryptedFile(encryptedFile)
   126  	if err != nil {
   127  		return utils.MakeError("failed to get salt and data from encrypted file: %v", err)
   128  	}
   129  
   130  	key, iv := getKeyAndIVFromPassAndSalt([]byte(mandelbox.GetConfigEncryptionToken()), salt)
   131  	err = decryptAES256CBC(key, iv, data)
   132  	if err != nil {
   133  		return utils.MakeError("failed to decrypt user config: %v", err)
   134  	}
   135  
   136  	logger.Infof("Finished decrypting user config for mandelbox %s", mandelbox.ID)
   137  	logger.Infof("Decompressing user config for mandelbox %s", mandelbox.ID)
   138  
   139  	// Lock directory to avoid cleanup
   140  	mandelbox.rwlock.Lock()
   141  	defer mandelbox.rwlock.Unlock()
   142  
   143  	// Make directory for user configs
   144  	configDir := mandelbox.getUserConfigDir()
   145  	if err := os.MkdirAll(configDir, 0777); err != nil {
   146  		return utils.MakeError("Could not make dir %s. Error: %s", configDir, err)
   147  	}
   148  
   149  	unpackedConfigDir := path.Join(configDir, mandelbox.getUnpackedConfigsDirectoryName())
   150  	if err := os.MkdirAll(unpackedConfigDir, 0777); err != nil {
   151  		return utils.MakeError("Could not make dir %s. Error: %s", unpackedConfigDir, err)
   152  	}
   153  
   154  	// Once we've extracted everything, we open up permissions for the user
   155  	// config directory so it's accessible by the non-root user in the
   156  	// mandelboxes. We also are okay with setting executable bits here, since
   157  	// it's possible that some apps store executable scripts in their config.
   158  	defer func() {
   159  		cmd := exec.Command("chown", "-R", "ubuntu", configDir)
   160  		cmd.Run()
   161  		cmd = exec.Command("chmod", "-R", "777", configDir)
   162  		cmd.Run()
   163  	}()
   164  
   165  	// Unpacking the decrypted data involves:
   166  	// 1. Decompress the the lz4 file into a tar
   167  	// 2. Untar the config to the target directory
   168  	dataReader := bytes.NewReader(data)
   169  	lz4Reader := lz4.NewReader(dataReader)
   170  	tarReader := tar.NewReader(lz4Reader)
   171  
   172  	// Track total size of files in the tar
   173  	totalFileSize := int64(0)
   174  
   175  	for {
   176  		// Read the header for the next file in the tar
   177  		header, err := tarReader.Next()
   178  
   179  		// If we reach EOF, that means we are done untaring
   180  		switch {
   181  		case err == io.EOF:
   182  			break
   183  		case err != nil:
   184  			return utils.MakeError("error reading tar file: %v", err)
   185  		}
   186  
   187  		// Not certain why this case happens, but causes segfaults if removed
   188  		if header == nil {
   189  			break
   190  		}
   191  
   192  		path := filepath.Join(unpackedConfigDir, header.Name)
   193  		info := header.FileInfo()
   194  
   195  		// Create directory if it doesn't exist, otherwise go next
   196  		if info.IsDir() {
   197  			if err = os.MkdirAll(path, info.Mode()); err != nil {
   198  				return utils.MakeError("error creating directory: %v", err)
   199  			}
   200  			continue
   201  		}
   202  
   203  		// Create file and copy contents
   204  		file, err := os.OpenFile(path, os.O_CREATE|os.O_TRUNC|os.O_WRONLY, info.Mode())
   205  		if err != nil {
   206  			return utils.MakeError("error opening file: %v", err)
   207  		}
   208  
   209  		numBytesWritten, err := io.Copy(file, tarReader)
   210  		if err != nil {
   211  			file.Close()
   212  			return utils.MakeError("error copying data to file: %v", err)
   213  		}
   214  
   215  		// Add written bytes to total size
   216  		totalFileSize += numBytesWritten
   217  
   218  		// Manually close file instead of defer otherwise files are only
   219  		// closed when ALL files are done unpacking
   220  		if err := file.Close(); err != nil {
   221  			logger.Warningf("Failed to close file %s: %v", path, err)
   222  		}
   223  	}
   224  
   225  	logger.Infof("Untarred config to: %s, total size was %d bytes", unpackedConfigDir, totalFileSize)
   226  
   227  	return nil
   228  }

   232  func (mandelbox *mandelboxData) backupUserConfigs() error {
   233  	if len(mandelbox.userID) == 0 {
   234  		logger.Infof("Cannot save user configs for MandelboxID %s since UserID is empty.", mandelbox.ID)
   235  		return nil
   236  	}
   237  
   238  	if len(mandelbox.configEncryptionToken) == 0 {
   239  		return utils.MakeError("Cannot save user configs for MandelboxID %s since ConfigEncryptionToken is empty", mandelbox.ID)
   240  	}
   241  
   242  	configDir := mandelbox.getUserConfigDir()
   243  	encTarPath := path.Join(configDir, mandelbox.getEncryptedArchiveFilename())
   244  	decTarPath := path.Join(configDir, mandelbox.getDecryptedArchiveFilename())
   245  	unpackedConfigPath := path.Join(configDir, mandelbox.getUnpackedConfigsDirectoryName())
   246  
   247  	tarConfigCmd := exec.Command(
   248  		"/usr/bin/tar", "-I", "lz4", "-C", unpackedConfigPath, "-cf", decTarPath,
   249  		".")
   250  	tarConfigOutput, err := tarConfigCmd.CombinedOutput()
   251  	// tar is only fatal when exit status is 2 -
   252  	//    exit status 1 just means that some files have changed while tarring,
   253  	//    which is an ignorable error
   254  	if err != nil && !strings.Contains(string(tarConfigOutput), "file changed") {
   255  		return utils.MakeError("Could not tar config directory: %s. Output: %s", err, tarConfigOutput)
   256  	}
   257  	logger.Infof("Tar config directory output: %s", tarConfigOutput)
   258  
   259  	// At this point, config archive must exist: encrypt app config
   260  	encryptConfigCmd := exec.Command(
   261  		"/usr/bin/openssl", "aes-256-cbc", "-e",
   262  		"-in", decTarPath,
   263  		"-out", encTarPath,
   264  		"-pass", "pass:"+string(mandelbox.configEncryptionToken), "-pbkdf2")
   265  	encryptConfigOutput, err := encryptConfigCmd.CombinedOutput()
   266  	if err != nil {
   267  		// If the config could not be encrypted, don't upload
   268  		return utils.MakeError("Could not encrypt config: %s. Output: %s", err, encryptConfigOutput)
   269  	}
   270  	logger.Infof("Encrypted config to %s", encTarPath)
   271  
   272  	cfg, err := config.LoadDefaultConfig(context.Background())
   273  	if err != nil {
   274  		return utils.MakeError("failed to load aws config: %v", err)
   275  	}
   276  
   277  	s3Client := s3.NewFromConfig(cfg, func(o *s3.Options) {
   278  		o.Region = "us-east-1"
   279  	})
   280  	uploader := manager.NewUploader(s3Client)
   281  
   282  	encryptedConfig, err := os.Open(encTarPath)
   283  	if err != nil {
   284  		return utils.MakeError("failed to open encrypted config: %v", err)
   285  	}
   286  	defer encryptedConfig.Close()
   287  
   288  	// Find size of config we are uploading to S3
   289  	fileInfo, err := encryptedConfig.Stat()
   290  	if err != nil {
   291  		// We don't want to stop uploading configs because we can't get file stats
   292  		logger.Warningf("error opening file stats for encrypted user config: %s", encTarPath)
   293  	} else {
   294  		// Log encrypted user config size before S3 upload
   295  		logger.Infof("%s is %d bytes", encTarPath, fileInfo.Size())
   296  	}
   297  
   298  	uploadResult, err := uploader.Upload(context.Background(), &s3.PutObjectInput{
   299  		Bucket: aws.String(userConfigS3Bucket),
   300  		Key:    aws.String(mandelbox.getS3ConfigKeyWithoutLocking()),
   301  		Body:   encryptedConfig,
   302  	})
   303  	if err != nil {
   304  		return utils.MakeError("error uploading encrypted config to s3: %v", err)
   305  	}
   306  
   307  	logger.Infof("Saved user config for mandelbox %s, version %s", mandelbox.ID, *uploadResult.VersionID)
   308  
   309  	return nil
   310  }

   367  // getSaltAndDataFromOpenSSLEncryptedFile takes OpenSSL encrypted data
   368  // and returns the salt used to encrypt and the encrypted data itself.
   369  func getSaltAndDataFromOpenSSLEncryptedFile(file []byte) (salt, data []byte, err error) {
   370  	if len(file) < aes.BlockSize {
   371  		return nil, nil, utils.MakeError("file is too short to have been encrypted by openssl")
   372  	}
   373  
   374  	// All OpenSSL encrypted files that use a salt have a 16-byte header
   375  	// The first 8 bytes are the string "Salted__"
   376  	// The next 8 bytes are the salt itself
   377  	// See: https://security.stackexchange.com/questions/29106/openssl-recover-key-and-iv-by-passphrase
   378  	saltHeader := file[:aes.BlockSize]
   379  	byteHeaderLength := len([]byte(opensslSaltHeader))
   380  	if string(saltHeader[:byteHeaderLength]) != opensslSaltHeader {
   381  		return nil, nil, utils.MakeError("file does not appear to have been encrypted by openssl: wrong salt header")
   382  	}
   383  
   384  	salt = saltHeader[byteHeaderLength:]
   385  	data = file[aes.BlockSize:]
   386  
   387  	return
   388  }

   390  // getKeyAndIVFromPassAndSalt uses pbkdf2 to generate a key, IV pair
   391  // given a known password and salt.
   392  func getKeyAndIVFromPassAndSalt(password, salt []byte) (key, iv []byte) {
   393  	derivedKey := pbkdf2.Key(password, salt, pbkdf2Iterations, aes256KeyLength+aes256IVLength, sha256.New)
   394  
   395  	// First 32 bytes of the derived key are the key and the next 16 bytes are the IV
   396  	key = derivedKey[:aes256KeyLength]
   397  	iv = derivedKey[aes256KeyLength : aes256KeyLength+aes256IVLength]
   398  
   399  	return
   400  }

   402  // decryptAES256CBC takes a key, IV, and AES-256 CBC encrypted data and decrypts
   403  // the data in place.
   404  // Based on example code from: https://golang.org/src/crypto/cipher/example_test.go.
   405  func decryptAES256CBC(key, iv, data []byte) error {
   406  	block, err := aes.NewCipher(key)
   407  	if err != nil {
   408  		return utils.MakeError("could not create aes cipher block: %v", err)
   409  	}
   410  
   411  	// Validate that encrypted data follows the correct format
   412  	// This is because CBC encryption always works in whole blocks
   413  	if len(data)%aes.BlockSize != 0 {
   414  		return utils.MakeError("given data is not a multiple of the block size")
   415  	}
   416  
   417  	mode := cipher.NewCBCDecrypter(block, iv)
   418  	mode.CryptBlocks(data, data)
   419  
   420  	return nil
   421  }
