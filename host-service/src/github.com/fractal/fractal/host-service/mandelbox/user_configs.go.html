    13  func (c *mandelboxData) PopulateUserConfigs() error {
    14  	// We (write) lock here so that the resourceMappingDir doesn't get cleaned up
    15  	// from under us.
    16  	c.rwlock.Lock()
    17  	defer c.rwlock.Unlock()
    18  
    19  	// Make directory for user configs
    20  	configDir := c.getUserConfigDir()
    21  	if err := os.MkdirAll(configDir, 0777); err != nil {
    22  		return utils.MakeError("Could not make dir %s. Error: %s", configDir, err)
    23  	}
    24  
    25  	unpackedConfigDir := configDir + c.getUnpackedConfigsDirectoryName()
    26  	if err := os.MkdirAll(unpackedConfigDir, 0777); err != nil {
    27  		return utils.MakeError("Could not make dir %s. Error: %s", unpackedConfigDir, err)
    28  	}
    29  
    30  	// Once we've extracted everything, we open up permissions for the user
    31  	// config directory so it's accessible by the non-root user in the
    32  	// mandelboxes. We also are okay with setting executable bits here, since
    33  	// it's possible that some apps store executable scripts in their config.
    34  	defer func() {
    35  		cmd := exec.Command("chown", "-R", "ubuntu", configDir)
    36  		cmd.Run()
    37  		cmd = exec.Command("chmod", "-R", "777", configDir)
    38  		cmd.Run()
    39  	}()
    40  
    41  	// If userID is not set, then we don't retrieve configs from s3
    42  	if len(c.userID) == 0 {
    43  		return nil
    44  	}
    45  
    46  	if len(c.configEncryptionToken) == 0 {
    47  		return utils.MakeError("Cannot get user configs for MandelboxID %s since ConfigEncryptionToken is empty", c.mandelboxID)
    48  	}
    49  
    50  	// Retrieve config from s3
    51  	s3ConfigPath := c.getS3ConfigPath()
    52  	getConfigCmd := exec.Command("/usr/bin/aws", "s3", "cp", s3ConfigPath, configDir)
    53  	getConfigOutput, err := getConfigCmd.CombinedOutput()
    54  	// If aws s3 cp errors out due to the file not existing, don't log an error because
    55  	//    this means that it's the user's first run and they don't have any settings
    56  	//    stored for this application yet.
    57  	if err != nil {
    58  		if strings.Contains(string(getConfigOutput), "does not exist") {
    59  			logger.Infof("Ran \"aws s3 cp\" and config does not exist")
    60  			return nil
    61  		}
    62  		return utils.MakeError("Could not run \"aws s3 cp\" get config command: %s. Output: %s", err, getConfigOutput)
    63  	}
    64  	logger.Infof("Ran \"aws s3 cp\" get config command with output: %s", getConfigOutput)
    65  
    66  	encTarPath := configDir + c.getEncryptedArchiveFilename()
    67  	decTarPath := configDir + c.getDecryptedArchiveFilename()
    68  	unpackedConfigPath := configDir + c.getUnpackedConfigsDirectoryName()
    69  
    70  	// At this point, config archive must exist: decrypt app config
    71  	decryptConfigCmd := exec.Command(
    72  		"/usr/bin/openssl", "aes-256-cbc", "-d",
    73  		"-in", encTarPath,
    74  		"-out", decTarPath,
    75  		"-pass", "pass:"+string(c.configEncryptionToken), "-pbkdf2")
    76  	decryptConfigOutput, err := decryptConfigCmd.CombinedOutput()
    77  	if err != nil {
    78  		return utils.MakeError("Could not decrypt config for userID %s and mandelboxID %s: %s. Output: %s", c.userID, c.GetMandelboxID(), err, decryptConfigOutput)
    79  	}
    80  	logger.Infof("Decrypted config to %s", decTarPath)
    81  
    82  	// Delete original encrypted config
    83  	err = os.Remove(encTarPath)
    84  	if err != nil {
    85  		logger.Errorf("getUserConfig(): Failed to delete user config encrypted archive %s", encTarPath)
    86  	}
    87  
    88  	// Extract the config archive to the user config directory
    89  	untarConfigCmd := exec.Command("/usr/bin/tar", "-I", "lz4", "-xf", decTarPath, "-C", unpackedConfigPath)
    90  	untarConfigOutput, err := untarConfigCmd.CombinedOutput()
    91  	if err != nil {
    92  		return utils.MakeError("Could not untar config archive: %s. Output: %s", err, untarConfigOutput)
    93  	}
    94  	logger.Infof("Untar config directory output: %s", untarConfigOutput)
    95  
    96  	return nil
    97  }

   101  func (c *mandelboxData) backupUserConfigs() error {
   102  	if len(c.userID) == 0 {
   103  		logger.Infof("Cannot save user configs for MandelboxID %s since UserID is empty.", c.mandelboxID)
   104  		return nil
   105  	}
   106  
   107  	if len(c.configEncryptionToken) == 0 {
   108  		return utils.MakeError("Cannot save user configs for MandelboxID %s since ConfigEncryptionToken is empty", c.mandelboxID)
   109  	}
   110  
   111  	configDir := c.getUserConfigDir()
   112  	encTarPath := configDir + c.getEncryptedArchiveFilename()
   113  	decTarPath := configDir + c.getDecryptedArchiveFilename()
   114  	unpackedConfigPath := configDir + c.getUnpackedConfigsDirectoryName()
   115  	s3ConfigPath := c.getS3ConfigPath()
   116  
   117  	tarConfigCmd := exec.Command(
   118  		"/usr/bin/tar", "-I", "lz4", "-C", unpackedConfigPath, "-cf", decTarPath,
   119  		".")
   120  	tarConfigOutput, err := tarConfigCmd.CombinedOutput()
   121  	// tar is only fatal when exit status is 2 -
   122  	//    exit status 1 just means that some files have changed while tarring,
   123  	//    which is an ignorable error
   124  	if err != nil && !strings.Contains(string(tarConfigOutput), "file changed") {
   125  		return utils.MakeError("Could not tar config directory: %s. Output: %s", err, tarConfigOutput)
   126  	}
   127  	logger.Infof("Tar config directory output: %s", tarConfigOutput)
   128  
   129  	// At this point, config archive must exist: encrypt app config
   130  	encryptConfigCmd := exec.Command(
   131  		"/usr/bin/openssl", "aes-256-cbc", "-e",
   132  		"-in", decTarPath,
   133  		"-out", encTarPath,
   134  		"-pass", "pass:"+string(c.configEncryptionToken), "-pbkdf2")
   135  	encryptConfigOutput, err := encryptConfigCmd.CombinedOutput()
   136  	if err != nil {
   137  		// If the config could not be encrypted, don't upload
   138  		return utils.MakeError("Could not encrypt config: %s. Output: %s", err, encryptConfigOutput)
   139  	}
   140  	logger.Infof("Encrypted config to %s", encTarPath)
   141  
   142  	saveConfigCmd := exec.Command("/usr/bin/aws", "s3", "cp", encTarPath, s3ConfigPath)
   143  	saveConfigOutput, err := saveConfigCmd.CombinedOutput()
   144  	if err != nil {
   145  		return utils.MakeError("Could not run \"aws s3 cp\" save config command: %s. Output: %s", err, saveConfigOutput)
   146  	}
   147  	logger.Infof("Ran \"aws s3 cp\" save config command with output: %s", saveConfigOutput)
   148  
   149  	return nil
   150  }
