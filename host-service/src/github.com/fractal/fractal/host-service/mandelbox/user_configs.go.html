    66  		cmd.Run()
    67  		cmd = exec.Command("chmod", "-R", "777", configDir)

    73  		return nil
    74  	}
    75  
    76  	if len(c.configEncryptionToken) == 0 {
    77  		return utils.MakeError("Cannot get user configs for MandelboxID %s since ConfigEncryptionToken is empty", c.mandelboxID)
    78  	}

    83  	if err != nil {
    84  		return utils.MakeError("failed to load aws config: %v", err)
    85  	}

    88  		o.Region = "us-east-1"
    89  	})

    91  
    92  	// Fetch the HeadObject first to see how much memory we need to allocate
    93  	headObject, err := s3Client.HeadObject(context.Background(), &s3.HeadObjectInput{
    94  		Bucket: aws.String(userConfigS3Bucket),
    95  		Key:    aws.String(c.getS3ConfigKey()),
    96  	})
    97  
    98  	var noSuchKeyErr *types.NoSuchKey
    99  	if err != nil {
   100  		// If aws s3 errors out due to the file not existing, don't log an error because
   101  		// this means that it's the user's first run and they don't have any settings
   102  		// stored for this application yet.
   103  		if errors.As(err, &noSuchKeyErr) {
   104  			logger.Infof("Could not get head object because config does not exist for user %s", c.userID)
   105  			return nil
   106  		}
   107  
   108  		return utils.MakeError("Failed to download head object from s3: %v", err)
   109  	}

   112  	// This should be okay as we don't expect configs to be very large

   115  		Bucket: aws.String(userConfigS3Bucket),

   147  	// 1. Decompress the the lz4 file into a tar
   148  	// 2. Untar the config to the target directory
   149  	dataReader := bytes.NewReader(data)

   167  			break

   176  				return utils.MakeError("error creating directory: %v", err)
   177  			}
   178  			continue
   179  		}

   182  		file, err := os.OpenFile(path, os.O_CREATE|os.O_TRUNC|os.O_WRONLY, info.Mode())
   183  		if err != nil {
   184  			return utils.MakeError("error opening file: %v", err)
   185  		}

   187  		_, err = io.Copy(file, tarReader)
   188  		if err != nil {

   191  		}

   206  	if len(c.userID) == 0 {
   207  		logger.Infof("Cannot save user configs for MandelboxID %s since UserID is empty.", c.mandelboxID)
   208  		return nil
   209  	}
   210  
   211  	if len(c.configEncryptionToken) == 0 {
   212  		return utils.MakeError("Cannot save user configs for MandelboxID %s since ConfigEncryptionToken is empty", c.mandelboxID)
   213  	}
   214  
   215  	configDir := c.getUserConfigDir()
   216  	encTarPath := configDir + c.getEncryptedArchiveFilename()
   217  	decTarPath := configDir + c.getDecryptedArchiveFilename()
   218  	unpackedConfigPath := configDir + c.getUnpackedConfigsDirectoryName()
   219  
   220  	tarConfigCmd := exec.Command(
   221  		"/usr/bin/tar", "-I", "lz4", "-C", unpackedConfigPath, "-cf", decTarPath,
   222  		".")
   223  	tarConfigOutput, err := tarConfigCmd.CombinedOutput()
   224  	// tar is only fatal when exit status is 2 -
   225  	//    exit status 1 just means that some files have changed while tarring,
   226  	//    which is an ignorable error
   227  	if err != nil && !strings.Contains(string(tarConfigOutput), "file changed") {
   228  		return utils.MakeError("Could not tar config directory: %s. Output: %s", err, tarConfigOutput)
   229  	}
   230  	logger.Infof("Tar config directory output: %s", tarConfigOutput)
   231  
   232  	// At this point, config archive must exist: encrypt app config
   233  	encryptConfigCmd := exec.Command(
   234  		"/usr/bin/openssl", "aes-256-cbc", "-e",
   235  		"-in", decTarPath,
   236  		"-out", encTarPath,
   237  		"-pass", "pass:"+string(c.configEncryptionToken), "-pbkdf2")
   238  	encryptConfigOutput, err := encryptConfigCmd.CombinedOutput()
   239  	if err != nil {
   240  		// If the config could not be encrypted, don't upload
   241  		return utils.MakeError("Could not encrypt config: %s. Output: %s", err, encryptConfigOutput)
   242  	}
   243  	logger.Infof("Encrypted config to %s", encTarPath)
   244  
   245  	cfg, err := config.LoadDefaultConfig(context.Background())
   246  	if err != nil {
   247  		return utils.MakeError("failed to load aws config: %v", err)
   248  	}
   249  
   250  	s3Client := s3.NewFromConfig(cfg, func(o *s3.Options) {
   251  		o.Region = "us-east-1"
   252  	})
   253  	uploader := manager.NewUploader(s3Client)
   254  
   255  	encryptedConfig, err := os.Open(encTarPath)
   256  	if err != nil {
   257  		return utils.MakeError("failed to open encrypted config: %v", err)
   258  	}
   259  
   260  	_, err = uploader.Upload(context.Background(), &s3.PutObjectInput{
   261  		Bucket: aws.String(userConfigS3Bucket),
   262  		Key:    aws.String(c.getS3ConfigKey()),
   263  		Body:   encryptedConfig,
   264  	})
   265  	if err != nil {
   266  		return utils.MakeError("error uploading encrypted config to s3: %v", err)
   267  	}
   268  
   269  	logger.Infof("Ran \"aws s3 cp\" save config command with output")
   270  
   271  	return nil

   315  // getSaltAndDataFromOpenSSLEncryptedFile takes OpenSSL encrypted data
   316  // and returns the salt used to encrypt and the encrypted data itself.
   317  func getSaltAndDataFromOpenSSLEncryptedFile(file []byte) (salt, data []byte, err error) {
   318  	if len(file) < aes.BlockSize {
   319  		return nil, nil, utils.MakeError("file is too short to have been encrypted by openssl")
   320  	}
   321  
   322  	// All OpenSSL encrypted files that use a salt have a 16-byte header
   323  	// The first 8 bytes are the string "Salted__"
   324  	// The next 8 bytes are the salt itself
   325  	// See: https://security.stackexchange.com/questions/29106/openssl-recover-key-and-iv-by-passphrase
   326  	saltHeader := file[:aes.BlockSize]
   327  	byteHeaderLength := len([]byte(opensslSaltHeader))
   328  	if string(saltHeader[:byteHeaderLength]) != opensslSaltHeader {
   329  		return nil, nil, utils.MakeError("file does not appear to have been encrypted by openssl: wrong salt header")
   330  	}
   331  
   332  	salt = saltHeader[byteHeaderLength:]
   333  	data = file[aes.BlockSize:]
   334  
   335  	return
   336  }

   338  // getKeyAndIVFromPassAndSalt uses pbkdf2 to generate a key, IV pair
   339  // given a known password and salt.
   340  func getKeyAndIVFromPassAndSalt(password, salt []byte) (key, iv []byte) {
   341  	derivedKey := pbkdf2.Key(password, salt, pbkdf2Iterations, aes256KeyLength+aes256IVLength, sha256.New)
   342  
   343  	// First 32 bytes of the derived key are the key and the next 16 bytes are the IV
   344  	key = derivedKey[:aes256KeyLength]
   345  	iv = derivedKey[aes256KeyLength : aes256KeyLength+aes256IVLength]
   346  
   347  	return
   348  }

   350  // decryptAES256CBC takes a key, IV, and AES-256 CBC encrypted data and decrypts
   351  // the data in place.
   352  // Based on example code from: https://golang.org/src/crypto/cipher/example_test.go.
   353  func decryptAES256CBC(key, iv, data []byte) error {
   354  	block, err := aes.NewCipher(key)
   355  	if err != nil {
   356  		return utils.MakeError("could not create aes cipher block: %v", err)
   357  	}
   358  
   359  	// Validate that encrypted data follows the correct format
   360  	// This is because CBC encryption always works in whole blocks
   361  	if len(data)%aes.BlockSize != 0 {
   362  		return utils.MakeError("given data is not a multiple of the block size")
   363  	}
   364  
   365  	mode := cipher.NewCBCDecrypter(block, iv)
   366  	mode.CryptBlocks(data, data)
   367  
   368  	return nil
   369  }
