185 // LOG_FATAL implies that the protocol cannot recover from this (usually OS-related) error,
186 // This is generally used for OS errors (the host being out of RAM, SDL failing to initialize),
187 // If an OS-related error code is recoverable, but exceedingly unlikely or unheard of thusfar
188 // (e.g. mutex creation failing), LOG_FATAL is highly preferred over attempting a recovery.
189 // TLDR ~ Problems with *the OS* that are unexpected and not easily recoverable from.
190 #define LOG_FATAL(message, ...) \
191  do { \
192  LOG_MESSAGE(FATAL_ERROR, message, ##__VA_ARGS__); \
193  terminate_protocol(WHIST_EXIT_FAILURE); \
194  } while (0)
