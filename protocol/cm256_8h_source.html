96 /*
97  * Cauchy MDS GF(256) encode
98  *
99  * This produces a set of recovery blocks that should be transmitted after the
100  * original data blocks.
101  *
102  * It takes in 'originalCount' equal-sized blocks and produces 'recoveryCount'
103  * equally-sized recovery blocks.
104  *
105  * The input 'originals' array allows more natural usage of the library.
106  * The output recovery blocks are stored end-to-end in 'recoveryBlocks'.
107  * 'recoveryBlocks' should have recoveryCount * blockBytes bytes available.
108  *
109  * Precondition: originalCount + recoveryCount <= 256
110  *
111  * When transmitting the data, the block index of the data should be sent,
112  * and the recovery block index is also needed. The decoder should also
113  * be provided with the values of originalCount, recoveryCount and blockBytes.
114  *
115  * Example wire format:
116  * [originalCount(1 byte)] [recoveryCount(1 byte)]
117  * [blockIndex(1 byte)] [blockData(blockBytes bytes)]
118  *
119  * Be careful not to mix blocks from different encoders.
120  *
121  * It is possible to support variable-length data by including the original
122  * data length at the front of each message in 2 bytes, such that when it is
123  * recovered after a loss the data length is available in the block data and
124  * the remaining bytes of padding can be neglected.
125  *
126  * Returns 0 on success, and any other code indicates failure.
127  */
128 extern int cm256_encode(
129  cm256_encoder_params params, // Encoder parameters
130  cm256_block* originals, // Array of pointers to original blocks
131  void* recoveryBlocks); // Output recovery blocks end-to-end

141 /*
142  * Cauchy MDS GF(256) decode
143  *
144  * This recovers the original data from the recovery data in the provided
145  * blocks. There should be 'originalCount' blocks in the provided array.
146  * Recovery will always be possible if that many blocks are received.
147  *
148  * Provide the same values for 'originalCount', 'recoveryCount', and
149  * 'blockBytes' used by the encoder.
150  *
151  * The block Index should be set to the block index of the original data,
152  * as described in the cm256_block struct comments above.
153  *
154  * Recovery blocks will be replaced with original data and the Index
155  * will be updated to indicate the original block that was recovered.
156  *
157  * Returns 0 on success, and any other code indicates failure.
158  */
159 extern int cm256_decode(
160  cm256_encoder_params params, // Encoder parameters
161  cm256_block* blocks); // Array of 'originalCount' blocks as described above
